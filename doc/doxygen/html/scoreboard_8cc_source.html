83 void Scoreboard::reserveRegisters(const class warp_inst_t* inst)
84 {
85  for( unsigned r=0; r < 4; r++) {
86  if(inst->out[r] > 0) {
87  reserveRegister(inst->warp_id(), inst->out[r]);
88  SHADER_DPRINTF( SCOREBOARD,
89  "Reserved register - warp:%d, reg: %d\n",
90  inst->warp_id(),
91  inst->out[r] );
92  }
93  }
94 
95  //Keep track of long operations
96  if (inst->is_load() &&
97  ( inst->space.get_type() == global_space ||
98  inst->space.get_type() == local_space ||
99  inst->space.get_type() == param_space_kernel ||
100  inst->space.get_type() == param_space_local ||
101  inst->space.get_type() == param_space_unclassified ||
102  inst->space.get_type() == tex_space)){
103  for ( unsigned r=0; r<4; r++) {
104  if(inst->out[r] > 0) {
105  SHADER_DPRINTF( SCOREBOARD,
106  "New longopreg marked - warp:%d, reg: %d\n",
107  inst->warp_id(),
108  inst->out[r] );
109  longopregs[inst->warp_id()].insert(inst->out[r]);
110  }
111  }
112  }
113 }

136 bool Scoreboard::checkCollision( unsigned wid, const class inst_t *inst ) const
137 {
138  // Get list of all input and output registers
139  std::set<int> inst_regs;
140 
141  if(inst->out[0] > 0) inst_regs.insert(inst->out[0]);
142  if(inst->out[1] > 0) inst_regs.insert(inst->out[1]);
143  if(inst->out[2] > 0) inst_regs.insert(inst->out[2]);
144  if(inst->out[3] > 0) inst_regs.insert(inst->out[3]);
145  if(inst->in[0] > 0) inst_regs.insert(inst->in[0]);
146  if(inst->in[1] > 0) inst_regs.insert(inst->in[1]);
147  if(inst->in[2] > 0) inst_regs.insert(inst->in[2]);
148  if(inst->in[3] > 0) inst_regs.insert(inst->in[3]);
149  if(inst->pred > 0) inst_regs.insert(inst->pred);
150  if(inst->ar1 > 0) inst_regs.insert(inst->ar1);
151  if(inst->ar2 > 0) inst_regs.insert(inst->ar2);
152 
153  // Check for collision, get the intersection of reserved registers and instruction registers
154  std::set<int>::const_iterator it2;
155  for ( it2=inst_regs.begin() ; it2 != inst_regs.end(); it2++ )
156  if(reg_table[wid].find(*it2) != reg_table[wid].end()) {
157  return true;
158  }
159  return false;
160 }
