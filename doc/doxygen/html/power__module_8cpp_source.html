145 wire const & Power_Module::wireOptimize(double L){
146  map<double, wire>::iterator iter = wire_map.find(L);
147  if(iter == wire_map.end()){
148 
149  double W = 64;
150  double bestMetric = 100000000 ;
151  double bestK = -1;
152  double bestM = -1;
153  double bestN = -1;
154  for (double K = 1.0 ; K < 10 ; K+=0.1 ) {
155  for (double N = 1.0 ; N < 40 ; N += 1.0 ) {
156  for (double M = 1.0 ; M < 40.0 ; M +=1.0 ) {
157  double l = 1.0 * L/( N * M) ;
158 
159  double k0 = R * (Co_delay + Ci_delay) ;
160  double k1 = R/K * Cw + K * Rw * Ci_delay ;
161  double k2 = 0.5 * Rw * Cw ;
162  double Tw = k0 + (k1 * l) + k2 * (l * l) ;
163  double alpha = 0.2 ;
164  double power = alpha * W * powerRepeatedWire( L, K, M, N) + powerWireDFF( M, W, alpha ) ;
165  double metric = M * M * M * M * power ;
166  if ( (N*Tw) < (0.8 * tCLK) ) {
167  if ( metric < bestMetric ) {
168  bestMetric = metric ;
169  bestK = K ;
170  bestM = M ;
171  bestN = N ;
172  }
173  }
174  }
175  }
176  }
177  cout<<"L = "<<L<<" K = "<<bestK<<" M = "<<bestM<<" N = "<<bestN<<endl;
178 
179  wire const temp = {L, bestK, bestM, bestN};
180  iter = wire_map.insert(make_pair(L, temp)).first;
181  }

212 double Power_Module::powerWireDFF(double M, double W, double alpha){
213  double Cdin = 2 * 0.8 * (Ci + Co) + 2 * ( 2.0/3.0 * 0.8 * Co ) ;
214  double Cclk = 2 * 0.8 * (Ci + Co) + 2 * ( 2.0/3.0 * 0.8 * Cg_pwr) ;
215  double Cint = (alpha * 0.5) * Cdin + alpha * Cclk ;
216 
217  return Cint * M * W * (Vdd*Vdd) * fCLK ;

250 double Power_Module::powerWordLine(double memoryWidth, double memoryDepth){
251  // wordline capacitance
252  double Ccell = 2 * ( 4.0 * LAMBDA ) * Cg_pwr + 6 * MetalPitch * Cw ;
253  double Cwl = memoryWidth * Ccell ;
254 
255  // wordline circuits
256  double Warray = 8 * MetalPitch + memoryDepth ;
257  double x = 1.0 + (5.0/16.0) * (1 + Co/Ci) ;
258  double Cpredecode = x * (Cw * Warray * Ci) ;
259  double Cdecode = x * Cwl ;
260 
261  // bitline circuits
262  double Harray = 6 * memoryWidth * MetalPitch ;
263  double y = (1 + 0.25) * (1 + Co/Ci) ;
264  double Cprecharge = y * ( Cw * Harray + 3 * channel_width * Ci ) ;
265  double Cwren = y * ( Cw * Harray + 2 * channel_width * Ci ) ;
266 
267  double Cbd = Cprecharge + Cwren ;
268  double Cwd = 2 * Cpredecode + Cdecode ;
269 
270  return ( Cbd + Cwd ) * Vdd * Vdd * fCLK ;
271 
272 }

337 double Power_Module::powerCrossbar(double width, double inputs, double outputs, double from, double to){
338  // datapath traversal power
339  double Wxbar = width * outputs * CrossbarPitch ;
340  double Hxbar = width * inputs * CrossbarPitch ;
341 
342  // wires
343  double CwIn = Wxbar * Cw ;
344  double CwOut = Hxbar * Cw ;
345 
346  // cross-points
347  double Cxi = (1.0/16.0) * CwOut ;
348  double Cxo = 4.0 * Cxi * (Co_delay/Ci_delay) ;
349 
350  // drivers
351  double Cti = (1.0/16.0) * CwIn ;
352  double Cto = 4.0 * Cti * (Co_delay/Ci_delay) ;
353 
354  double CinputDriver = 5.0/16.0 * (1 + Co_delay/Ci_delay) * (0.5 * Cw * Wxbar + Cti) ;
355 
356  // total switched capacitance
357 
358  //this maybe missing +Cto
359  double Cin = CinputDriver + CwIn + Cti + (outputs * Cxi) ;
360  if ( to < outputs/2 ) {
361  Cin -= ( 0.5 * CwIn + outputs/2 * Cxi) ;
362  }
363  //this maybe missing +cti
364  double Cout = CwOut + Cto + (inputs * Cxo) ;
365  if ( from < inputs/2) {
366  Cout -= ( 0.5 * CwOut + (inputs/2 * Cxo)) ;
367  }
368  return 0.5 * (Cin + Cout) * (Vdd * Vdd * fCLK) ;
369 }
