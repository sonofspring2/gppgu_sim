56  for ( int iter = 0; iter < _iSLIP_iter; ++iter ) {

58 
59  vector<int> grants(_outputs, -1);
60 
61  for ( output = 0; output < _outputs; ++output ) {
62 
63  // Skip loop if there are no requests
64  // or the output is already matched
65  if ( ( _out_req[output].empty( ) ) ||
66  ( _outmatch[output] != -1 ) ) {
67  continue;
68  }
69 
70  // A round-robin arbiter between input requests
71  input_offset = _gptrs[output];
72 
73  p = _out_req[output].begin( );
74  while( ( p != _out_req[output].end( ) ) &&
75  ( p->second.port < input_offset ) ) {
76  p++;
77  }
78 
79  wrapped = false;
80  while( (!wrapped) ||
81  ( ( p != _out_req[output].end( ) ) &&
82  ( p->second.port < input_offset ) ) ) {
83  if ( p == _out_req[output].end( ) ) {
84  if ( wrapped ) { break; }
85  // p is valid here because empty lists
86  // are skipped (above)
87  p = _out_req[output].begin( );
88  wrapped = true;
89  }
90 
91  input = p->second.port;
92 
93  // we know the output is free (above) and
94  // if the input is free, grant request
95  if ( _inmatch[input] == -1 ) {
96  grants[output] = input;
97  break;
98  }
99 
100  p++;
101  }
102  }
