59 void ptx_cta_info::check_cta_thread_status_and_reset()
60 {
61  bool fail = false;
62  if ( m_threads_that_have_exited.size() != m_threads_in_cta.size() ) {
63  printf("\n\n");
64  printf("Execution error: Some threads still running in CTA during CTA reallocation! (1)\n");
65  printf(" CTA uid = %Lu (sm_idx = %u) : %lu running out of %zu total\n",
66  m_uid,
67  m_sm_idx,
68  (m_threads_in_cta.size() - m_threads_that_have_exited.size()), m_threads_in_cta.size() );
69  printf(" These are the threads that are still running:\n");
70  std::set<ptx_thread_info*>::iterator t_iter;
71  for ( t_iter=m_threads_in_cta.begin(); t_iter != m_threads_in_cta.end(); ++t_iter ) {
72  ptx_thread_info *t = *t_iter;
73  if ( m_threads_that_have_exited.find(t) == m_threads_that_have_exited.end() ) {
74  if ( m_dangling_pointers.find(t) != m_dangling_pointers.end() ) {
75  printf(" <thread deleted>\n");
76  } else {
77  printf(" [done=%c] : ", (t->is_done()?'Y':'N') );
78  t->print_insn( t->get_pc(), stdout );
79  printf("\n");
80  }
81  }
82  }
83  printf("\n\n");
84  fail = true;
85  }
86  if ( fail ) {
87  abort();
88  }
89 
90  bool fail2 = false;
91  std::set<ptx_thread_info*>::iterator t_iter;
92  for ( t_iter=m_threads_in_cta.begin(); t_iter != m_threads_in_cta.end(); ++t_iter ) {
93  ptx_thread_info *t = *t_iter;
94  if ( m_dangling_pointers.find(t) == m_dangling_pointers.end() ) {
95  if ( !t->is_done() ) {
96  if ( !fail2 ) {
97  printf("Execution error: Some threads still running in CTA during CTA reallocation! (2)\n");
98  printf(" CTA uid = %Lu (sm_idx = %u) :\n", m_uid, m_sm_idx );
99  fail2 = true;
100  }
101  printf(" ");
102  t->print_insn( t->get_pc(), stdout );
103  printf("\n");
104  }
105  }
106  }
107  if ( fail2 ) {
108  abort();
109  }
110  m_threads_in_cta.clear();
111  m_threads_that_have_exited.clear();
112  m_dangling_pointers.clear();
113 }

190 unsigned ptx_thread_info::get_builtin( int builtin_id, unsigned dim_mod )
191 {
192  assert( m_valid );
193  switch ((builtin_id&0xFFFF)) {
194  case CLOCK_REG:
195  return (unsigned)(gpu_sim_cycle + gpu_tot_sim_cycle);
196  case CLOCK64_REG:
197  abort(); // change return value to unsigned long long?
198  // GPGPUSim clock is 4 times slower - multiply by 4
199  return (gpu_sim_cycle + gpu_tot_sim_cycle)*4;
200  case HALFCLOCK_ID:
201  // GPGPUSim clock is 4 times slower - multiply by 4
202  // Hardware clock counter is incremented at half the shader clock frequency - divide by 2 (Henry '10)
203  return (gpu_sim_cycle + gpu_tot_sim_cycle)*2;
204  case CTAID_REG:
205  assert( dim_mod < 3 );
206  if( dim_mod == 0 ) return m_ctaid.x;
207  if( dim_mod == 1 ) return m_ctaid.y;
208  if( dim_mod == 2 ) return m_ctaid.z;
209  abort();
210  break;
211  case ENVREG_REG:{
212  int index = builtin_id >> 16;
213  dim3 gdim = this->get_core()->get_kernel_info()->get_grid_dim();
214  switch(index){
215  case 0:
216  case 1:
217  case 2:
218  case 3:
219  case 4:
220  case 5:
221  return 0;
222  break;
223  case 6:
224  return gdim.x;
225  case 7:
226  return gdim.y;
227  case 8:
228  return gdim.z;
229  case 9:
230  if(gdim.z == 1 && gdim.y == 1)
231  return 1;
232  else if(gdim.z == 1)
233  return 2;
234  else
235  return 3;
236  break;
237  default:
238  break;
239  }
240  }
241  case GRIDID_REG:
242  return m_gridid;
243  case LANEID_REG: feature_not_implemented( "%laneid" ); return 0;
244  case LANEMASK_EQ_REG: feature_not_implemented( "%lanemask_eq" ); return 0;
245  case LANEMASK_LE_REG: feature_not_implemented( "%lanemask_le" ); return 0;
246  case LANEMASK_LT_REG: feature_not_implemented( "%lanemask_lt" ); return 0;
247  case LANEMASK_GE_REG: feature_not_implemented( "%lanemask_ge" ); return 0;
248  case LANEMASK_GT_REG: feature_not_implemented( "%lanemask_gt" ); return 0;
249  case NCTAID_REG:
250  assert( dim_mod < 3 );
251  if( dim_mod == 0 ) return m_nctaid.x;
252  if( dim_mod == 1 ) return m_nctaid.y;
253  if( dim_mod == 2 ) return m_nctaid.z;
254  abort();
255  break;
256  case NTID_REG:
257  assert( dim_mod < 3 );
258  if( dim_mod == 0 ) return m_ntid.x;
259  if( dim_mod == 1 ) return m_ntid.y;
260  if( dim_mod == 2 ) return m_ntid.z;
261  abort();
262  break;
263  case NWARPID_REG: feature_not_implemented( "%nwarpid" ); return 0;
264  case PM_REG: feature_not_implemented( "%pm" ); return 0;
265  case SMID_REG: feature_not_implemented( "%smid" ); return 0;
266  case TID_REG:
267  assert( dim_mod < 3 );
268  if( dim_mod == 0 ) return m_tid.x;
269  if( dim_mod == 1 ) return m_tid.y;
270  if( dim_mod == 2 ) return m_tid.z;
271  abort();
272  break;
273  case WARPSZ_REG: return m_core->get_warp_size() ;
274  default:
275  assert(0);
276  }
277  return 0;
278 }

307 static void print_reg( FILE *fp, std::string name, ptx_reg_t value, symbol_table *symtab )
308 {
309  const symbol *sym = symtab->lookup(name.c_str());
310  fprintf(fp," %8s ", name.c_str() );
311  if( sym == NULL ) {
312  fprintf(fp,"<unknown type> 0x%llx\n", (unsigned long long ) value.u64 );
313  return;
314  }
315  const type_info *t = sym->type();
316  if( t == NULL ) {
317  fprintf(fp,"<unknown type> 0x%llx\n", (unsigned long long ) value.u64 );
318  return;
319  }
320  type_info_key ti = t->get_key();
321 
322  switch ( ti.scalar_type() ) {
323  case S8_TYPE: fprintf(fp,".s8 %d\n", value.s8 ); break;
324  case S16_TYPE: fprintf(fp,".s16 %d\n", value.s16 ); break;
325  case S32_TYPE: fprintf(fp,".s32 %d\n", value.s32 ); break;
326  case S64_TYPE: fprintf(fp,".s64 %Ld\n", value.s64 ); break;
327  case U8_TYPE: fprintf(fp,".u8 %u [0x%02x]\n", value.u8, (unsigned) value.u8 ); break;
328  case U16_TYPE: fprintf(fp,".u16 %u [0x%04x]\n", value.u16, (unsigned) value.u16 ); break;
329  case U32_TYPE: fprintf(fp,".u32 %u [0x%08x]\n", value.u32, (unsigned) value.u32 ); break;
330  case U64_TYPE: fprintf(fp,".u64 %llu [0x%llx]\n", value.u64, value.u64 ); break;
331  case F16_TYPE: fprintf(fp,".f16 %f [0x%04x]\n", value.f16, (unsigned) value.u16 ); break;
332  case F32_TYPE: fprintf(fp,".f32 %.15lf [0x%08x]\n", value.f32, value.u32 ); break;
333  case F64_TYPE: fprintf(fp,".f64 %.15le [0x%016llx]\n", value.f64, value.u64 ); break;
334  case B8_TYPE: fprintf(fp,".b8 0x%02x\n", (unsigned) value.u8 ); break;
335  case B16_TYPE: fprintf(fp,".b16 0x%04x\n", (unsigned) value.u16 ); break;
336  case B32_TYPE: fprintf(fp,".b32 0x%08x\n", (unsigned) value.u32 ); break;
337  case B64_TYPE: fprintf(fp,".b64 0x%llx\n", (unsigned long long ) value.u64 ); break;
338  case PRED_TYPE: fprintf(fp,".pred %u\n", (unsigned) value.pred ); break;
339  default:
340  fprintf( fp, "non-scalar type\n" );
341  break;
342  }
343  fflush(fp);
344 }

379 bool ptx_thread_info::callstack_pop()
380 {
381  const symbol *rv_src = m_callstack.back().m_return_var_src;
382  const symbol *rv_dst = m_callstack.back().m_return_var_dst;
383  assert( !((rv_src != NULL) ^ (rv_dst != NULL)) ); // ensure caller and callee agree on whether there is a return value
384 
385  // read return value from callee frame
386  arg_buffer_t buffer;
387  if( rv_src != NULL )
388  buffer = copy_arg_to_buffer(this, operand_info(rv_src), rv_dst );
389 
390  m_symbol_table = m_callstack.back().m_symbol_table;
391  m_NPC = m_callstack.back().m_PC;
392  m_RPC_updated = true;
393  m_last_was_call = false;
394  m_RPC = m_callstack.back().m_RPC;
395  m_func_info = m_callstack.back().m_func_info;
396  if( m_func_info ) {
397  assert( m_local_mem_stack_pointer >= m_func_info->local_mem_framesize() );
398  m_local_mem_stack_pointer -= m_func_info->local_mem_framesize();
399  }
400  m_callstack.pop_back();
401  m_regs.pop_back();
402  m_debug_trace_regs_modified.pop_back();
403  m_debug_trace_regs_read.pop_back();
404 
405  // write return value into caller frame
406  if( rv_dst != NULL )
407  copy_buffer_to_frame(this, buffer);
408 
409  return m_callstack.empty();
410 }

446 void ptx_thread_info::dump_callstack() const
447 {
448  std::list<stack_entry>::const_iterator c=m_callstack.begin();
449  std::list<reg_map_t>::const_iterator r=m_regs.begin();
450 
451  printf("\n\n");
452  printf("Call stack for thread uid = %u (sc=%u, hwtid=%u)\n", m_uid, m_hw_sid, m_hw_tid );
453  while( c != m_callstack.end() && r != m_regs.end() ) {
454  const stack_entry &c_e = *c;
455  const reg_map_t &regs = *r;
456  if( !c_e.m_valid ) {
457  printf(" <entry> #regs = %zu\n", regs.size() );
458  } else {
459  printf(" %20s PC=%3u RV= (callee=\'%s\',caller=\'%s\') #regs = %zu\n",
460  c_e.m_func_info->get_name().c_str(), c_e.m_PC,
461  c_e.m_return_var_src->name().c_str(),
462  c_e.m_return_var_dst->name().c_str(),
463  regs.size() );
464  }
465  c++;
466  r++;
467  }
468  if( c != m_callstack.end() || r != m_regs.end() ) {
469  printf(" *** mismatch in m_regs and m_callstack sizes ***\n" );
470  }
471  printf("\n\n");
472 }
