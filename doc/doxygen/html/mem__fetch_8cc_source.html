36 mem_fetch::mem_fetch( const mem_access_t &access,
37  const warp_inst_t *inst,
38  unsigned ctrl_size,
39  unsigned wid,
40  unsigned sid,
41  unsigned tpc,
42  const class memory_config *config )
43 {
44  m_request_uid = sm_next_mf_request_uid++;
45  m_access = access;

47  m_inst = *inst;
48  assert( wid == m_inst.warp_id() );

50  m_data_size = access.get_size();
51  m_ctrl_size = ctrl_size;
52  m_sid = sid;
53  m_tpc = tpc;
54  m_wid = wid;
55  config->m_address_mapping.addrdec_tlx(access.get_addr(),&m_raw_addr);
56  m_partition_addr = config->m_address_mapping.partition_address(access.get_addr());
57  m_type = m_access.is_write()?WRITE_REQUEST:READ_REQUEST;
58  m_timestamp = gpu_sim_cycle + gpu_tot_sim_cycle;
59  m_timestamp2 = 0;
60  m_status = MEM_FETCH_INITIALIZED;
61  m_status_change = gpu_sim_cycle + gpu_tot_sim_cycle;
62  m_mem_config = config;
63  icnt_flit_size = config->icnt_flit_size;
64 }

125 unsigned mem_fetch::get_num_flits(bool simt_to_mem){
126  unsigned sz=0;
127  // If atomic, write going to memory, or read coming back from memory, size = ctrl + data. Else, only ctrl
128  if( isatomic() || (simt_to_mem && get_is_write()) || !(simt_to_mem || get_is_write()) )
129  sz = size();
130  else
131  sz = get_ctrl_size();
132 
133  return (sz/icnt_flit_size) + ( (sz % icnt_flit_size)? 1:0);
134 }
