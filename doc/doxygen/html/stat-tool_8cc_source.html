47 void add_snap_shot_trigger (snap_shot_trigger* ss_trigger)
48 {
49  // quick optimization assuming that all snap shot intervals are perfect multiples of each other
50  if (min_snap_shot_interval == 0 || min_snap_shot_interval > ss_trigger->get_interval()) {
51  min_snap_shot_interval = ss_trigger->get_interval();
52  next_snap_shot_cycle = min_snap_shot_interval; // assume that snap shots haven't started yet
53  }
54  list_ss_trigger.push_back(ss_trigger);
55 }
56 
57 void remove_snap_shot_trigger (snap_shot_trigger* ss_trigger)
58 {
59  list_ss_trigger.remove(ss_trigger);
60 }

62 void try_snap_shot (unsigned long long current_cycle)
63 {
64  if (min_snap_shot_interval == 0) return;
65  if (current_cycle != next_snap_shot_cycle) return;
66 
67  std::list<snap_shot_trigger*>::iterator ss_trigger_iter = list_ss_trigger.begin();
68  for(; ss_trigger_iter != list_ss_trigger.end(); ++ss_trigger_iter) {
69  (*ss_trigger_iter)->snap_shot(current_cycle); // WF: should be try_snap_shot
70  }
71  next_snap_shot_cycle = current_cycle + min_snap_shot_interval; // WF: stateful testing, maybe bad
72 }

96 void spill_log_to_file (FILE *fout, int final, unsigned long long current_cycle)
97 {
98  if (!final && spill_interval == 0) return;
99  if (!final && current_cycle <= next_spill_cycle) return;
100 
101  fprintf(fout, "\n"); // ensure that the spill occurs at a new line
102  std::list<spill_log_interface*>::iterator i_spill_log = list_spill_log.begin();
103  for(; i_spill_log != list_spill_log.end(); ++i_spill_log) {
104  (*i_spill_log)->spill(fout, final);
105  }
106  fflush(fout);
107 
108  next_spill_cycle = current_cycle + spill_interval; // WF: stateful testing, maybe bad
109 }

311 void shader_mem_lat_create( int n_loggers, unsigned long long logging_interval)
312 {
313  s_mem_lat_logger.assign(n_loggers,
314  linear_histogram_logger(s_mem_lat_logger_nbins, logging_interval, "ShdrMemLat"));

318  add_snap_shot_trigger(&(s_mem_lat_logger[i]));

320  }

325 void shader_mem_lat_log( int logger_id, int latency)
326 {
327  if (s_mem_lat_logger_used == false) return;
328  if (latency > (1<<(s_mem_lat_logger_nbins/2))) assert(0); // guard for out of bound bin
329  assert(latency > 0);
330 
331  int latency_bin;
332 
333  int bin; // LOG_2(latency)
334  int v = latency;
335  register unsigned int shift;
336 
337  bin = (v > 0xFFFF) << 4; v >>= bin;
338  shift = (v > 0xFF ) << 3; v >>= shift; bin |= shift;
339  shift = (v > 0xF ) << 2; v >>= shift; bin |= shift;
340  shift = (v > 0x3 ) << 1; v >>= shift; bin |= shift;
341  bin |= (v >> 1);
342  latency_bin = 2 * bin;
343  if (bin > 0) {
344  latency_bin += ((latency & (1 << (bin - 1))) != 0)? 1 : 0; // approx. for LOG_sqrt2(latency)
345  }
346 
347  s_mem_lat_logger[logger_id].log(latency_bin);
348 }
