40 TreeArbiter::TreeArbiter( Module *parent, const string &name,
41  int size, int groups, const string & arb_type )
42  : Arbiter( parent, name, size ) {
43  assert(size % groups == 0);
44  _group_arbiters.resize(groups);
45  _group_reqs.resize(groups, 0);
46  _group_size = size / groups;

53 }

88 int TreeArbiter::Arbitrate( int* id, int* pri ) {
89  if(!_num_reqs) {
90  return -1;
91  }
92  for(int i = 0; i < (int)_group_arbiters.size(); ++i) {
93  if(_group_reqs[i]) {
94  int group_id, group_pri;
95  _group_arbiters[i]->Arbitrate(&group_id, &group_pri);
96  _global_arbiter->AddRequest(i, group_id, group_pri);
97  }
98  }
99  int group = _global_arbiter->Arbitrate(NULL, NULL);
100  assert(group >= 0 && group < (int)_group_arbiters.size());
101  int group_sel = _group_arbiters[group]->LastWinner();
102  assert(group_sel >= 0 && group_sel < _group_size);
103  _selected = group * _group_size + group_sel;
104  assert(_selected >= 0 && _selected < _size);
105  return Arbiter::Arbitrate(id, pri);
106 }
