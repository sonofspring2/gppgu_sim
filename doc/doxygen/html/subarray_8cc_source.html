42 Subarray::Subarray(const DynamicParameter & dp_, bool is_fa_):
43  dp(dp_), num_rows(dp.num_r_subarray), num_cols(dp.num_c_subarray),
44  num_cols_fa_cam(dp.tag_num_c_subarray), num_cols_fa_ram(dp.data_num_c_subarray),
45  cell(dp.cell), cam_cell(dp.cam_cell), is_fa(is_fa_)
46 {
47  //num_cols=7;
48  //cout<<"num_cols ="<< num_cols <<endl;
49  if (!(is_fa || dp.pure_cam))
50  {
51  num_cols +=(g_ip->add_ecc_b_ ? (int)ceil(num_cols / num_bits_per_ecc_b_) : 0); // ECC overhead
52  uint32_t ram_num_cells_wl_stitching =
53  (dp.ram_cell_tech_type == lp_dram) ? dram_num_cells_wl_stitching_ :
54  (dp.ram_cell_tech_type == comm_dram) ? comm_dram_num_cells_wl_stitching_ : sram_num_cells_wl_stitching_;
55 
56  area.h = cell.h * num_rows;
57 
58  area.w = cell.w * num_cols +
59  ceil(num_cols / ram_num_cells_wl_stitching) * g_tp.ram_wl_stitching_overhead_; // stitching overhead
60  }
61  else //cam fa
62  {
63 
64  //should not add dummy row here since the dummy row do not need decoder
65  if (is_fa)// fully associative cache
66  {
67  num_cols_fa_cam += g_ip->add_ecc_b_ ? (int)ceil(num_cols_fa_cam / num_bits_per_ecc_b_) : 0;
68  num_cols_fa_ram += (g_ip->add_ecc_b_ ? (int)ceil(num_cols_fa_ram / num_bits_per_ecc_b_) : 0);
69  num_cols = num_cols_fa_cam + num_cols_fa_ram;
70  }
71  else
72  {
73  num_cols_fa_cam += g_ip->add_ecc_b_ ? (int)ceil(num_cols_fa_cam / num_bits_per_ecc_b_) : 0;
74  num_cols_fa_ram = 0;
75  num_cols = num_cols_fa_cam;
76  }
77 
78  area.h = cam_cell.h * (num_rows + 1);//height of subarray is decided by CAM array. blank space in sram array are filled with dummy cells
79  area.w = cam_cell.w * num_cols_fa_cam + cell.w * num_cols_fa_ram
80  + ceil((num_cols_fa_cam + num_cols_fa_ram) / sram_num_cells_wl_stitching_)*g_tp.ram_wl_stitching_overhead_
81  + 16*g_tp.wire_local.pitch //the overhead for the NAND gate to connect the two halves
82  + 128*g_tp.wire_local.pitch;//the overhead for the drivers from matchline to wordline of RAM
83  }
84 
85  assert(area.h>0);
86  assert(area.w>0);
87  compute_C();
88 }

121 void Subarray::compute_C()
122 {
123  double c_w_metal = cell.w * g_tp.wire_local.C_per_um;
124  double r_w_metal = cell.w * g_tp.wire_local.R_per_um;
125  double C_b_metal = cell.h * g_tp.wire_local.C_per_um;
126  double C_b_row_drain_C;
127 
128  if (dp.is_dram)
129  {
130  C_wl = (gate_C_pass(g_tp.dram.cell_a_w, g_tp.dram.b_w, true, true) + c_w_metal) * num_cols;
131 
132  if (dp.ram_cell_tech_type == comm_dram)
133  {
134  C_bl = num_rows * C_b_metal;
135  }
136  else
137  {
138  C_b_row_drain_C = drain_C_(g_tp.dram.cell_a_w, NCH, 1, 0, cell.w, true, true) / 2.0; // due to shared contact
139  C_bl = num_rows * (C_b_row_drain_C + C_b_metal);
140  }
141  }
142  else
143  {
144  if (!(is_fa ||dp.pure_cam))
145  {
146  C_wl = (gate_C_pass(g_tp.sram.cell_a_w, (g_tp.sram.b_w-2*g_tp.sram.cell_a_w)/2.0, false, true)*2 +
147  c_w_metal) * num_cols;
148  C_b_row_drain_C = drain_C_(g_tp.sram.cell_a_w, NCH, 1, 0, cell.w, false, true) / 2.0; // due to shared contact
149  C_bl = num_rows * (C_b_row_drain_C + C_b_metal);
150  }
151  else
152  {
153  //Following is wordline not matchline
154  //CAM portion
155  c_w_metal = cam_cell.w * g_tp.wire_local.C_per_um;
156  r_w_metal = cam_cell.w * g_tp.wire_local.R_per_um;
157  C_wl_cam = (gate_C_pass(g_tp.cam.cell_a_w, (g_tp.cam.b_w-2*g_tp.cam.cell_a_w)/2.0, false, true)*2 +
158  c_w_metal) * num_cols_fa_cam;
159  R_wl_cam = (r_w_metal) * num_cols_fa_cam;
160 
161  if (!dp.pure_cam)
162  {
163  //RAM portion
164  c_w_metal = cell.w * g_tp.wire_local.C_per_um;
165  r_w_metal = cell.w * g_tp.wire_local.R_per_um;
166  C_wl_ram = (gate_C_pass(g_tp.sram.cell_a_w, (g_tp.sram.b_w-2*g_tp.sram.cell_a_w)/2.0, false, true)*2 +
167  c_w_metal) * num_cols_fa_ram;
168  R_wl_ram = (r_w_metal) * num_cols_fa_ram;
169  }
170  else
171  {
172  C_wl_ram = R_wl_ram =0;
173  }
174  C_wl = C_wl_cam + C_wl_ram;
175  C_wl += (16+128)*g_tp.wire_local.pitch*g_tp.wire_local.C_per_um;
176 
177  R_wl = R_wl_cam + R_wl_ram;
178  R_wl += (16+128)*g_tp.wire_local.pitch*g_tp.wire_local.R_per_um;
179 
180  //there are two ways to write to a FA,
181  //1) Write to CAM array then force a match on match line to active the corresponding wordline in RAM;
182  //2) using separate wordline for read/write and search in RAM.
183  //We are using the second approach.
184 
185  //Bitline CAM portion This is bitline not searchline. We assume no sharing between bitline and searchline according to SUN's implementations.
186  C_b_metal = cam_cell.h * g_tp.wire_local.C_per_um;
187  C_b_row_drain_C = drain_C_(g_tp.cam.cell_a_w, NCH, 1, 0, cam_cell.w, false, true) / 2.0; // due to shared contact
188  C_bl_cam = (num_rows+1) * (C_b_row_drain_C + C_b_metal);
189  //height of subarray is decided by CAM array. blank space in sram array are filled with dummy cells
190  C_b_row_drain_C = drain_C_(g_tp.sram.cell_a_w, NCH, 1, 0, cell.w, false, true) / 2.0; // due to shared contact
191  C_bl = (num_rows +1) * (C_b_row_drain_C + C_b_metal);
192 
193  }
194  }
195 }
