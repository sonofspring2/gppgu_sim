58 {

317 void shader_core_ctx::init_warps( unsigned cta_id, unsigned start_thread, unsigned end_thread )
318 {
319  address_type start_pc = next_pc(start_thread);
320  if (m_config->model == POST_DOMINATOR) {
321  unsigned start_warp = start_thread / m_config->warp_size;
322  unsigned end_warp = end_thread / m_config->warp_size + ((end_thread % m_config->warp_size)? 1 : 0);
323  for (unsigned i = start_warp; i < end_warp; ++i) {
324  unsigned n_active=0;
325  simt_mask_t active_threads;
326  for (unsigned t = 0; t < m_config->warp_size; t++) {
327  unsigned hwtid = i * m_config->warp_size + t;
328  if ( hwtid < end_thread ) {
329  n_active++;
330  assert( !m_active_threads.test(hwtid) );
331  m_active_threads.set( hwtid );
332  active_threads.set(t);
333  }
334  }
335  m_simt_stack[i]->launch(start_pc,active_threads);
336  m_warp[i].init(start_pc,cta_id,i,active_threads, m_dynamic_warp_id);
337  ++m_dynamic_warp_id;
338  m_not_completed += n_active;
339  }
340  }

596 void shader_core_ctx::fetch()
597 {
598  if( !m_inst_fetch_buffer.m_valid ) {
599  // find an active warp with space in instruction buffer that is not already waiting on a cache miss
600  // and get next 1-2 instructions from i-cache...
601  for( unsigned i=0; i < m_config->max_warps_per_shader; i++ ) {
602  unsigned warp_id = (m_last_warp_fetched+1+i) % m_config->max_warps_per_shader;
603 
604  // this code checks if this warp has finished executing and can be reclaimed
605  if( m_warp[warp_id].hardware_done() && !m_scoreboard->pendingWrites(warp_id) && !m_warp[warp_id].done_exit() ) {
606  bool did_exit=false;
607  for( unsigned t=0; t<m_config->warp_size;t++) {
608  unsigned tid=warp_id*m_config->warp_size+t;
609  if( m_threadState[tid].m_active == true ) {
610  m_threadState[tid].m_active = false;
611  unsigned cta_id = m_warp[warp_id].get_cta_id();
612  register_cta_thread_exit(cta_id);
613  m_not_completed -= 1;
614  m_active_threads.reset(tid);
615  assert( m_thread[tid]!= NULL );
616  did_exit=true;
617  }
618  }
619  if( did_exit )
620  m_warp[warp_id].set_done_exit();
621  }
622 
623  // this code fetches instructions from the i-cache or generates memory requests
624  if( !m_warp[warp_id].functional_done() && !m_warp[warp_id].imiss_pending() && m_warp[warp_id].ibuffer_empty() ) {
625  address_type pc = m_warp[warp_id].get_pc();
626  address_type ppc = pc + PROGRAM_MEM_START;
627  unsigned nbytes=16;
628  unsigned offset_in_block = pc & (m_config->m_L1I_config.get_line_sz()-1);
629  if( (offset_in_block+nbytes) > m_config->m_L1I_config.get_line_sz() )
630  nbytes = (m_config->m_L1I_config.get_line_sz()-offset_in_block);
631 
632  // TODO: replace with use of allocator
633  // mem_fetch *mf = m_mem_fetch_allocator->alloc()
634  mem_access_t acc(INST_ACC_R,ppc,nbytes,false);
635  mem_fetch *mf = new mem_fetch(acc,
636  NULL/*we don't have an instruction yet*/,
637  READ_PACKET_SIZE,
638  warp_id,
639  m_sid,
640  m_tpc,
641  m_memory_config );
642  std::list<cache_event> events;
643  enum cache_request_status status = m_L1I->access( (new_addr_type)ppc, mf, gpu_sim_cycle+gpu_tot_sim_cycle,events);
644  if( status == MISS ) {
645  m_last_warp_fetched=warp_id;
646  m_warp[warp_id].set_imiss_pending();
647  m_warp[warp_id].set_last_fetch(gpu_sim_cycle);
648  } else if( status == HIT ) {
649  m_last_warp_fetched=warp_id;
650  m_inst_fetch_buffer = ifetch_buffer_t(pc,nbytes,warp_id);
651  m_warp[warp_id].set_last_fetch(gpu_sim_cycle);
652  delete mf;
653  } else {
654  m_last_warp_fetched=warp_id;
655  assert( status == RESERVATION_FAIL );
656  delete mf;
657  }
658  break;
659  }
660  }
661  }
662 
663  m_L1I->cycle();
664 
665  if( m_L1I->access_ready() ) {
666  mem_fetch *mf = m_L1I->next_access();
667  m_warp[mf->get_wid()].clear_imiss_pending();
668  delete mf;
669  }
670 }

767 template < class T >
768 void scheduler_unit::order_by_priority( std::vector< T >& result_list,
769  const typename std::vector< T >& input_list,
770  const typename std::vector< T >::const_iterator& last_issued_from_input,
771  unsigned num_warps_to_add,
772  OrderingType ordering,
773  bool (*priority_func)(T lhs, T rhs) )
774 {
775  assert( num_warps_to_add <= input_list.size() );
776  result_list.clear();
777  typename std::vector< T > temp = input_list;
778 
779  if ( ORDERING_GREEDY_THEN_PRIORITY_FUNC == ordering ) {
780  T greedy_value = *last_issued_from_input;
781  result_list.push_back( greedy_value );
782 
783  std::sort( temp.begin(), temp.end(), priority_func );
784  typename std::vector< T >::iterator iter = temp.begin();
785  for ( unsigned count = 0; count < num_warps_to_add; ++count, ++iter ) {
786  if ( *iter != greedy_value ) {
787  result_list.push_back( *iter );
788  }
789  }
790  } else if ( ORDERED_PRIORITY_FUNC_ONLY == ordering ) {
791  std::sort( temp.begin(), temp.end(), priority_func );
792  typename std::vector< T >::iterator iter = temp.begin();
793  for ( unsigned count = 0; count < num_warps_to_add; ++count, ++iter ) {
794  result_list.push_back( *iter );
795  }
796  } else {
797  fprintf( stderr, "Unknown ordering - %d\n", ordering );
798  abort();
799  }

802 void scheduler_unit::cycle()
803 {
804  SCHED_DPRINTF( "scheduler_unit::cycle()\n" );
805  bool valid_inst = false; // there was one warp with a valid instruction to issue (didn't require flush due to control hazard)
806  bool ready_inst = false; // of the valid instructions, there was one not waiting for pending register writes
807  bool issued_inst = false; // of these we issued one
808 
809  order_warps();
810  for ( std::vector< shd_warp_t* >::const_iterator iter = m_next_cycle_prioritized_warps.begin();
811  iter != m_next_cycle_prioritized_warps.end();
812  iter++ ) {
813  // Don't consider warps that are not yet valid
814  if ( (*iter) == NULL || (*iter)->done_exit() ) {
815  continue;
816  }
817  SCHED_DPRINTF( "Testing (warp_id %u, dynamic_warp_id %u)\n",
818  (*iter)->get_warp_id(), (*iter)->get_dynamic_warp_id() );
819  unsigned warp_id = (*iter)->get_warp_id();
820  unsigned checked=0;
821  unsigned issued=0;
822  unsigned max_issue = m_shader->m_config->gpgpu_max_insn_issue_per_warp;
823  while( !warp(warp_id).waiting() && !warp(warp_id).ibuffer_empty() && (checked < max_issue) && (checked <= issued) && (issued < max_issue) ) {
824  const warp_inst_t *pI = warp(warp_id).ibuffer_next_inst();
825  bool valid = warp(warp_id).ibuffer_next_valid();
826  bool warp_inst_issued = false;
827  unsigned pc,rpc;
828  m_simt_stack[warp_id]->get_pdom_stack_top_info(&pc,&rpc);
829  SCHED_DPRINTF( "Warp (warp_id %u, dynamic_warp_id %u) has valid instruction (%s)\n",
830  (*iter)->get_warp_id(), (*iter)->get_dynamic_warp_id(),
831  ptx_get_insn_str( pc).c_str() );
832  if( pI ) {
833  assert(valid);
834  if( pc != pI->pc ) {
835  SCHED_DPRINTF( "Warp (warp_id %u, dynamic_warp_id %u) control hazard instruction flush\n",
836  (*iter)->get_warp_id(), (*iter)->get_dynamic_warp_id() );
837  // control hazard
838  warp(warp_id).set_next_pc(pc);
839  warp(warp_id).ibuffer_flush();
840  } else {
841  valid_inst = true;
842  if ( !m_scoreboard->checkCollision(warp_id, pI) ) {
843  SCHED_DPRINTF( "Warp (warp_id %u, dynamic_warp_id %u) passes scoreboard\n",
844  (*iter)->get_warp_id(), (*iter)->get_dynamic_warp_id() );
845  ready_inst = true;
846  const active_mask_t &active_mask = m_simt_stack[warp_id]->get_active_mask();
847  assert( warp(warp_id).inst_in_pipeline() );
848  if ( (pI->op == LOAD_OP) || (pI->op == STORE_OP) || (pI->op == MEMORY_BARRIER_OP) ) {
849  if( m_mem_out->has_free() ) {
850  m_shader->issue_warp(*m_mem_out,pI,active_mask,warp_id);
851  issued++;
852  issued_inst=true;
853  warp_inst_issued = true;
854  }
855  } else {
856  bool sp_pipe_avail = m_sp_out->has_free();
857  bool sfu_pipe_avail = m_sfu_out->has_free();
858  if( sp_pipe_avail && (pI->op != SFU_OP) ) {
859  // always prefer SP pipe for operations that can use both SP and SFU pipelines
860  m_shader->issue_warp(*m_sp_out,pI,active_mask,warp_id);
861  issued++;
862  issued_inst=true;
863  warp_inst_issued = true;
864  } else if ( (pI->op == SFU_OP) || (pI->op == ALU_SFU_OP) ) {
865  if( sfu_pipe_avail ) {
866  m_shader->issue_warp(*m_sfu_out,pI,active_mask,warp_id);
867  issued++;
868  issued_inst=true;
869  warp_inst_issued = true;
870  }
871  } }
872  } else {
873  SCHED_DPRINTF( "Warp (warp_id %u, dynamic_warp_id %u) fails scoreboard\n",
874  (*iter)->get_warp_id(), (*iter)->get_dynamic_warp_id() );
875  }
876  }
877  } else if( valid ) {
878  // this case can happen after a return instruction in diverged warp

907  }

1078 // Returns numbers of addresses in translated_addrs, each addr points to a 4B (32-bit) word
1079 unsigned shader_core_ctx::translate_local_memaddr( address_type localaddr, unsigned tid, unsigned num_shader, unsigned datasize, new_addr_type* translated_addrs )
1080 {
1081  // During functional execution, each thread sees its own memory space for local memory, but these
1082  // need to be mapped to a shared address space for timing simulation. We do that mapping here.
1083 
1084  address_type thread_base = 0;
1085  unsigned max_concurrent_threads=0;
1086  if (m_config->gpgpu_local_mem_map) {
1087  // Dnew = D*N + T%nTpC + nTpC*C
1088  // N = nTpC*nCpS*nS (max concurent threads)
1089  // C = nS*K + S (hw cta number per gpu)
1090  // K = T/nTpC (hw cta number per core)
1091  // D = data index
1092  // T = thread
1093  // nTpC = number of threads per CTA
1094  // nCpS = number of CTA per shader
1095  //
1096  // for a given local memory address threads in a CTA map to contiguous addresses,
1097  // then distribute across memory space by CTAs from successive shader cores first,
1098  // then by successive CTA in same shader core
1099  thread_base = 4*(kernel_padded_threads_per_cta * (m_sid + num_shader * (tid / kernel_padded_threads_per_cta))
1100  + tid % kernel_padded_threads_per_cta);
1101  max_concurrent_threads = kernel_padded_threads_per_cta * kernel_max_cta_per_shader * num_shader;
1102  } else {
1103  // legacy mapping that maps the same address in the local memory space of all threads
1104  // to a single contiguous address region
1105  thread_base = 4*(m_config->n_thread_per_shader * m_sid + tid);
1106  max_concurrent_threads = num_shader * m_config->n_thread_per_shader;
1107  }
1108  assert( thread_base < 4/*word size*/*max_concurrent_threads );
1109 
1110  // If requested datasize > 4B, split into multiple 4B accesses
1111  // otherwise do one sub-4 byte memory access
1112  unsigned num_accesses = 0;
1113 
1114  if(datasize >= 4) {
1115  // >4B access, split into 4B chunks
1116  assert(datasize%4 == 0); // Must be a multiple of 4B
1117  num_accesses = datasize/4;
1118  assert(num_accesses <= MAX_ACCESSES_PER_INSN_PER_THREAD); // max 32B
1119  assert(localaddr%4 == 0); // Address must be 4B aligned - required if accessing 4B per request, otherwise access will overflow into next thread's space
1120  for(unsigned i=0; i<num_accesses; i++) {
1121  address_type local_word = localaddr/4 + i;
1122  address_type linear_address = local_word*max_concurrent_threads*4 + thread_base + LOCAL_GENERIC_START;
1123  translated_addrs[i] = linear_address;
1124  }
1125  } else {
1126  // Sub-4B access, do only one access
1127  assert(datasize > 0);
1128  num_accesses = 1;
1129  address_type local_word = localaddr/4;
1130  address_type local_word_offset = localaddr%4;
1131  assert( (localaddr+datasize-1)/4 == local_word ); // Make sure access doesn't overflow into next 4B chunk
1132  address_type linear_address = local_word*max_concurrent_threads*4 + local_word_offset + thread_base + LOCAL_GENERIC_START;
1133  translated_addrs[0] = linear_address;
1134  }
1135  return num_accesses;
1136 }

1255  */
