56 InjectionProcess * InjectionProcess::New(string const & inject, int nodes,
57  double load,
58  Configuration const * const config)
59 {
60  string process_name;
61  string param_str;
62  size_t left = inject.find_first_of('(');
63  if(left == string::npos) {
64  process_name = inject;
65  } else {
66  process_name = inject.substr(0, left);
67  size_t right = inject.find_last_of(')');
68  if(right == string::npos) {
69  param_str = inject.substr(left+1);
70  } else {
71  param_str = inject.substr(left+1, right-left-1);
72  }
73  }
74  vector<string> params = tokenize_str(param_str);
75 
76  InjectionProcess * result = NULL;
77  if(process_name == "bernoulli") {
78  result = new BernoulliInjectionProcess(nodes, load);
79  } else if(process_name == "on_off") {
80  bool missing_params = false;
81  double alpha = numeric_limits<double>::quiet_NaN();
82  if(params.size() < 1) {
83  if(config) {
84  alpha = config->GetFloat("burst_alpha");
85  } else {
86  missing_params = true;
87  }
88  } else {
89  alpha = atof(params[0].c_str());
90  }
91  double beta = numeric_limits<double>::quiet_NaN();
92  if(params.size() < 2) {
93  if(config) {
94  beta = config->GetFloat("burst_beta");
95  } else {
96  missing_params = true;
97  }
98  } else {
99  beta = atof(params[1].c_str());
100  }
101  double r1 = numeric_limits<double>::quiet_NaN();
102  if(params.size() < 3) {
103  r1 = config ? config->GetFloat("burst_r1") : -1.0;
104  } else {
105  r1 = atof(params[2].c_str());
106  }
107  if(missing_params) {
108  cout << "Missing parameters for injection process: " << inject << endl;
109  exit(-1);
110  }
111  if((alpha < 0.0 && beta < 0.0) ||
112  (alpha < 0.0 && r1 < 0.0) ||
113  (beta < 0.0 && r1 < 0.0) ||
114  (alpha >= 0.0 && beta >= 0.0 && r1 >= 0.0)) {
115  cout << "Invalid parameters for injection process: " << inject << endl;
116  exit(-1);
117  }
118  vector<int> initial(nodes);
119  if(params.size() > 3) {
120  initial = tokenize_int(params[2]);
121  initial.resize(nodes, initial.back());
122  } else {
123  for(int n = 0; n < nodes; ++n) {
124  initial[n] = RandomInt(1);
125  }
126  }
127  result = new OnOffInjectionProcess(nodes, load, alpha, beta, r1, initial);
128  } else {
129  cout << "Invalid injection process: " << inject << endl;
130  exit(-1);
131  }
132  return result;
133 }

151 OnOffInjectionProcess::OnOffInjectionProcess(int nodes, double rate,
152  double alpha, double beta,
153  double r1, vector<int> initial)
154  : InjectionProcess(nodes, rate),
155  _alpha(alpha), _beta(beta), _r1(r1), _initial(initial)
156 {
157  assert(alpha <= 1.0);
158  assert(beta <= 1.0);
159  assert(r1 <= 1.0);
160  if(alpha < 0.0) {
161  assert(beta >= 0.0);
162  assert(r1 >= 0.0);
163  _alpha = beta * rate / (r1 - rate);
164  } else if(beta < 0.0) {
165  assert(alpha >= 0.0);
166  assert(r1 >= 0.0);
167  _beta = alpha * (r1 - rate) / rate;
168  } else {
169  assert(r1 < 0.0);
170  _r1 = rate * (alpha + beta) / alpha;
171  }
172  reset();
173 }
