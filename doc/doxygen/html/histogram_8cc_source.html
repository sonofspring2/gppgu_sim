89 void pow2_histogram::add2bin (int sample) {
90  assert(sample >= 0);
91 
92  int bin;
93  int v = sample;
94  register unsigned int shift;
95 
96  bin = (v > 0xFFFF) << 4; v >>= bin;
97  shift = (v > 0xFF ) << 3; v >>= shift; bin |= shift;
98  shift = (v > 0xF ) << 2; v >>= shift; bin |= shift;
99  shift = (v > 0x3 ) << 1; v >>= shift; bin |= shift;
100  bin |= (v >> 1);
101  bin += (sample > 0)? 1:0;
102 
103  m_bin_cnts[bin] += 1;
104 
105  m_maximum = (sample > m_maximum)? sample : m_maximum;
106  m_sum += sample;
107 }
