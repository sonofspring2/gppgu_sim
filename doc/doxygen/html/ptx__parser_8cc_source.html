326 // Returns padding that needs to be inserted ahead of address to make it aligned to min(size, maxalign)
327 /*
328  * @param address the address in bytes
329  * @param size the size of the memory to be allocated in bytes
330  * @param maximum alignment in bytes. i.e. if size is too big then align to this instead
331  */
332 int pad_address (new_addr_type address, unsigned size, unsigned maxalign) {
333  assert(size >= 0);
334  assert(maxalign > 0);
335  int alignto = maxalign;
336  if (size < maxalign &&
337  (size & (size-1)) == 0) { //size is a power of 2
338  alignto = size;
339  }
340  return alignto ? ((alignto - (address % alignto)) % alignto) : 0;
341 }

343 void add_identifier( const char *identifier, int array_dim, unsigned array_ident )
344 {
345  if( g_func_decl && (g_func_info == NULL) ) {
346  // return variable decl...
347  assert( g_add_identifier_cached__identifier == NULL );
348  g_add_identifier_cached__identifier = strdup(identifier);
349  g_add_identifier_cached__array_dim = array_dim;
350  g_add_identifier_cached__array_ident = array_ident;
351  return;
352  }
353  PTX_PARSE_DPRINTF("add_identifier \"%s\" (%u)", identifier, g_ident_add_uid);
354  g_ident_add_uid++;
355  type_info *type = g_var_type;
356  type_info_key ti = type->get_key();
357  int basic_type;
358  int regnum;
359  size_t num_bits;
360  unsigned addr_pad;
361  new_addr_type addr;
362  ti.type_decode(num_bits,basic_type);
363 
364  bool duplicates = check_for_duplicates( identifier );
365  if( duplicates ) {
366  symbol *s = g_current_symbol_table->lookup(identifier);
367  g_last_symbol = s;
368  if( g_func_decl )
369  return;
370  std::string msg = std::string(identifier) + " was declared previous at " + s->decl_location() + " skipping new declaration";
371  printf("GPGPU-Sim PTX: Warning %s\n", msg.c_str());
372  return;
373  }
374 
375  assert( g_var_type != NULL );
376  switch ( array_ident ) {
377  case ARRAY_IDENTIFIER:
378  type = g_current_symbol_table->get_array_type(type,array_dim);
379  num_bits = array_dim * num_bits;
380  break;
381  case ARRAY_IDENTIFIER_NO_DIM:
382  type = g_current_symbol_table->get_array_type(type,(unsigned)-1);
383  num_bits = 0;
384  break;
385  default:
386  break;
387  }
388  g_last_symbol = g_current_symbol_table->add_variable(identifier,type,num_bits/8,g_filename,ptx_lineno);
389  switch ( ti.get_memory_space().get_type() ) {
390  case reg_space: {
391  regnum = g_current_symbol_table->next_reg_num();
392  int arch_regnum = -1;
393  for (int d = 0; d < strlen(identifier); d++) {
394  if (isdigit(identifier[d])) {
395  sscanf(identifier + d, "%d", &arch_regnum);
396  break;
397  }
398  }
399  if (strcmp(identifier, "%sp") == 0) {
400  arch_regnum = 0;
401  }
402  g_last_symbol->set_regno(regnum, arch_regnum);
403  } break;
404  case shared_space:
405  printf("GPGPU-Sim PTX: allocating shared region for \"%s\" ",
406  identifier);
407  fflush(stdout);
408  assert( (num_bits%8) == 0 );
409  addr = g_current_symbol_table->get_shared_next();
410  addr_pad = pad_address(addr, num_bits/8, 128);
411  printf("from 0x%x to 0x%lx (shared memory space)\n",
412  addr+addr_pad,
413  addr+addr_pad + num_bits/8);
414  fflush(stdout);
415  g_last_symbol->set_address( addr+addr_pad );
416  g_current_symbol_table->alloc_shared( num_bits/8 + addr_pad );
417  break;
418  case const_space:
419  if( array_ident == ARRAY_IDENTIFIER_NO_DIM ) {
420  printf("GPGPU-Sim PTX: deferring allocation of constant region for \"%s\" (need size information)\n", identifier );
421  } else {
422  printf("GPGPU-Sim PTX: allocating constant region for \"%s\" ",
423  identifier);
424  fflush(stdout);
425  assert( (num_bits%8) == 0 );
426  addr = g_current_symbol_table->get_global_next();
427  addr_pad = pad_address(addr, num_bits/8, 128);
428  printf("from 0x%x to 0x%lx (global memory space) %u\n",
429  addr+addr_pad,
430  addr+addr_pad + num_bits/8,
431  g_const_alloc++);
432  fflush(stdout);
433  g_last_symbol->set_address( addr + addr_pad );
434  g_current_symbol_table->alloc_global( num_bits/8 + addr_pad );
435  }
436  if( g_current_symbol_table == g_global_symbol_table ) {
437  g_constants.insert( identifier );
438  }
439  assert( g_current_symbol_table != NULL );
440  g_sym_name_to_symbol_table[ identifier ] = g_current_symbol_table;
441  break;
442  case global_space:
443  printf("GPGPU-Sim PTX: allocating global region for \"%s\" ",
444  identifier);
445  fflush(stdout);
446  assert( (num_bits%8) == 0 );
447  addr = g_current_symbol_table->get_global_next();
448  addr_pad = pad_address(addr, num_bits/8, 128);
449  printf("from 0x%x to 0x%lx (global memory space)\n",
450  addr+addr_pad,
451  addr+addr_pad + num_bits/8);
452  fflush(stdout);
453  g_last_symbol->set_address( addr+addr_pad );
454  g_current_symbol_table->alloc_global( num_bits/8 + addr_pad );
455  g_globals.insert( identifier );
456  assert( g_current_symbol_table != NULL );
457  g_sym_name_to_symbol_table[ identifier ] = g_current_symbol_table;
458  break;
459  case local_space:
460  if( g_func_info == NULL ) {
461  printf("GPGPU-Sim PTX: allocating local region for \"%s\" ", identifier);
462  fflush(stdout);
463  assert( (num_bits%8) == 0 );
464  addr = g_current_symbol_table->get_local_next();
465  addr_pad = pad_address(addr, num_bits/8, 128);
466  printf("from 0x%x to 0x%lx (local memory space)\n",
467  addr+addr_pad,
468  addr+addr_pad + num_bits/8);
469  fflush(stdout);
470  g_last_symbol->set_address( addr+addr_pad);
471  g_current_symbol_table->alloc_local( num_bits/8 + addr_pad);
472  } else {
473  printf("GPGPU-Sim PTX: allocating stack frame region for .local \"%s\" ",
474  identifier);
475  fflush(stdout);
476  assert( (num_bits%8) == 0 );
477  addr = g_current_symbol_table->get_local_next();
478  addr_pad = pad_address(addr, num_bits/8, 128);
479  printf("from 0x%x to 0x%lx\n",
480  addr+addr_pad,
481  addr+addr_pad + num_bits/8);
482  fflush(stdout);
483  g_last_symbol->set_address( addr+addr_pad );
484  g_current_symbol_table->alloc_local( num_bits/8 + addr_pad);
485  g_func_info->set_framesize( g_current_symbol_table->get_local_next() );
486  }
487  break;
488  case tex_space:
489  printf("GPGPU-Sim PTX: encountered texture directive %s.\n", identifier);
490  break;
491  case param_space_local:
492  printf("GPGPU-Sim PTX: allocating stack frame region for .param \"%s\" from 0x%x to 0x%lx\n",
493  identifier,
494  g_current_symbol_table->get_local_next(),
495  g_current_symbol_table->get_local_next() + num_bits/8 );
496  fflush(stdout);
497  assert( (num_bits%8) == 0 );
498  g_last_symbol->set_address( g_current_symbol_table->get_local_next() );
499  g_current_symbol_table->alloc_local( num_bits/8 );
500  g_func_info->set_framesize( g_current_symbol_table->get_local_next() );
501  break;
502  case param_space_kernel:
503  break;
504  default:
505  abort();
506  break;
507  }
508 
509  assert( !ti.is_param_unclassified() );
510  if ( ti.is_param_kernel() ) {
511  bool is_ptr = (g_ptr_spec != undefined_space);
512  g_func_info->add_param_name_type_size(g_entry_func_param_index,identifier, ti.scalar_type(), num_bits, is_ptr, g_ptr_spec);
513  g_entry_func_param_index++;
514  }
515 }
