48 HighRadix::HighRadix(
49  double SUB_SWITCH_SZ_,
50  double ROWS_,
51  double FREQUENCY_, // GHz
52  double RADIX_,
53  double VC_COUNT_,
54  double FLIT_SZ_,
55  double AF_,// activity factor
56  double DIE_LEN_,//u
57  double DIE_HT_,//u
58  double INP_BUFF_ENT_,
59  double ROW_BUFF_ENT_,
60  double COL_BUFF_ENT_,
61  TechnologyParameter::DeviceType *dt
62  ):SUB_SWITCH_SZ(SUB_SWITCH_SZ_), ROWS(ROWS_), FREQUENCY(FREQUENCY_),
63  RADIX(RADIX_), VC_COUNT(VC_COUNT_), FLIT_SZ(FLIT_SZ_), AF(AF_),
64  DIE_LEN(DIE_LEN_), DIE_HT(DIE_HT_), INP_BUFF_ENT(INP_BUFF_ENT_),
65  ROW_BUFF_ENT(ROW_BUFF_ENT_), COL_BUFF_ENT(COL_BUFF_ENT_), deviceType(dt)
66 {
67  double area_scale=1;
68  double tech_init = 90;
69  if (g_ip->F_sz_nm == 65) {
70  area_scale*=1;
71  }
72  else if(g_ip->F_sz_nm == 45) {
73  area_scale*=1;
74  }
75  else if(g_ip->F_sz_nm == 32) {
76  area_scale*=2;
77  }
78 
79  DIE_LEN = sqrt(DIE_LEN_*DIE_HT_/area_scale);
80  DIE_HT = DIE_LEN;
81 
82  COLUMNS = pow(RADIX/SUB_SWITCH_SZ, 2)/ROWS;
83  INP_BUFF_SZ = FLIT_SZ * INP_BUFF_ENT;
84  ROW_BUFF_SZ = ROW_BUFF_ENT * FLIT_SZ;
85  COL_BUFF_SZ = COL_BUFF_ENT * FLIT_SZ;
86  area.set_area(0);
87 }

89 void
90 HighRadix::compute_power()
91 {
92  num_sub = ROWS * COLUMNS;
93  //FIXME change cb power to per input
94 
95  double scale = 1;
96  while (true) {
97  Wire winit(scale, scale);
98  cb = new Crossbar(SUB_SWITCH_SZ, SUB_SWITCH_SZ, FLIT_SZ);
99  cb->compute_power();
100  if (cb->delay*1e12 < (1/FREQUENCY)*(1e3))
101  break;
102  else {
103  scale+=0.2;
104  if (scale > MAX_WIRE_SCALE) break;
105  cout << "scale = " << scale << endl;
106  }
107  }
108  cb->power.readOp.dynamic /= SUB_SWITCH_SZ; // crossbar power per message
109  scale = 1;
110 
111  while (true) {
112  Wire winit(scale, scale);
113  out_cb = new Crossbar(1, SUB_SWITCH_SZ, FLIT_SZ);
114  out_cb->compute_power();
115  if (out_cb->delay*1e12 < (1/FREQUENCY)*(1e3))
116  break;
117  else {
118  scale+=0.2;
119  if (scale > MAX_WIRE_SCALE) break;
120  cout << "scale = " << scale << endl;
121  }
122  }
123  Wire winit;
124  out_cb->power.readOp.dynamic /= SUB_SWITCH_SZ; // power per message
125 
126  //arbiter initialization
127  vc_arb = new MCPAT_Arbiter(VC_COUNT, FLIT_SZ, cb->area.w);
128  vc_arb->compute_power();
129  c_arb = new MCPAT_Arbiter(COLUMNS, FLIT_SZ, cb->area.w);
130  c_arb->compute_power();
131  cb_arb = new MCPAT_Arbiter(RADIX/ROWS, FLIT_SZ, cb->area.w);
132  cb_arb->compute_power();
133 
134  // input buffer, row/column buffer initialization
135  inp_buff = buffer_(FLIT_SZ, INP_BUFF_SZ);
136  c_buff = buffer_(FLIT_SZ, COL_BUFF_SZ*2);
137  r_buff = buffer_(FLIT_SZ, ROW_BUFF_SZ*2);
138 
139 
140  // repeated wire initialization
141  hor_bus = new Wire(g_ip->wt, DIE_LEN);
142  // effective ht of vertical bus (connecting cb to column buffer) in each sub-switch
143  double eff_ht = (ROWS * (ROWS +1)/2) * (DIE_HT/ROWS);
144  ver_bus = new Wire(g_ip->wt, eff_ht);
145 
146  // sub switch includes row buffers, column buffers, vc/crossbar/column arbitration and a 2 stage crossbar traversal
147  sub_switch_power();
148  power.readOp.dynamic += sub_sw.power.readOp.dynamic * num_sub;
149  power.readOp.leakage += sub_sw.power.readOp.leakage * num_sub;
150 
151  // input buffer power
152  power.readOp.dynamic += 2 /*r&w*/ * inp_buff->power.readOp.dynamic * RADIX;
153  power.readOp.leakage += inp_buff->power.readOp.leakage * RADIX;
154 
155  // buses
156  power.readOp.dynamic += hor_bus->power.readOp.dynamic * FLIT_SZ * SUB_SWITCH_SZ * ROWS;
157  power.readOp.leakage += hor_bus->power.readOp.leakage * FLIT_SZ * SUB_SWITCH_SZ * ROWS;
158  power.readOp.dynamic += ver_bus->power.readOp.dynamic * FLIT_SZ * COLUMNS * SUB_SWITCH_SZ;
159  power.readOp.leakage += ver_bus->power.readOp.leakage * FLIT_SZ * ROWS * COLUMNS * SUB_SWITCH_SZ;
160 
161  // To calculate contribution of each component to the total power
162  compute_crossbar_power();
163  compute_bus_power();
164  compute_arb_power();
165  compute_buff_power();
166 
167  //area
168  sub_sw.area.set_area(sub_sw.area.get_area() + cb->area.get_area());
169  sub_sw.area.set_area(sub_sw.area.get_area() + out_cb->area.get_area());
170  sub_sw.area.set_area(sub_sw.area.get_area() + r_buff->area.get_area() * VC_COUNT * SUB_SWITCH_SZ);
171  sub_sw.area.set_area(sub_sw.area.get_area() + c_buff->area.get_area() * VC_COUNT * SUB_SWITCH_SZ);
172 
173  buff_tot.area.set_area(buff_tot.area.get_area() + inp_buff->area.get_area() * RADIX);
174  buff_tot.area.set_area(buff_tot.area.get_area() + VC_COUNT * r_buff->area.get_area() * SUB_SWITCH_SZ * num_sub);
175  buff_tot.area.set_area(buff_tot.area.get_area() + VC_COUNT * c_buff->area.get_area() * SUB_SWITCH_SZ * num_sub);
176 
177  crossbar_tot.area.set_area(crossbar_tot.area.get_area() + cb->area.get_area() * num_sub);
178  crossbar_tot.area.set_area(crossbar_tot.area.get_area() + out_cb->area.get_area() * num_sub);
179 
180  wire_tot.area.set_area(hor_bus->area.get_area() * FLIT_SZ * SUB_SWITCH_SZ * ROWS);
181  wire_tot.area.set_area(ver_bus->area.get_area() * FLIT_SZ * ROWS * COLUMNS);
182 }

184 void HighRadix::compute_crossbar_power()
185 {
186  crossbar_tot.power = cb->power;
187  crossbar_tot.power = crossbar_tot.power + out_cb->power;
188  crossbar_tot.power.readOp.dynamic *= num_sub;
189  crossbar_tot.power.readOp.leakage *= num_sub;
190 }

192 void HighRadix::compute_bus_power()
193 {
194  wire_tot.power.readOp.dynamic = hor_bus->power.readOp.dynamic * FLIT_SZ * SUB_SWITCH_SZ * ROWS;
195  wire_tot.power.readOp.leakage = hor_bus->power.readOp.leakage * FLIT_SZ * SUB_SWITCH_SZ * ROWS;
196  wire_tot.power.readOp.dynamic += ver_bus->power.readOp.dynamic * FLIT_SZ * COLUMNS * SUB_SWITCH_SZ;
197  wire_tot.power.readOp.leakage += ver_bus->power.readOp.leakage * FLIT_SZ * ROWS * COLUMNS * SUB_SWITCH_SZ;
198 }

200 void HighRadix::compute_arb_power()
201 {
202  arb_tot.power = cb_arb->power;
203  arb_tot.power = arb_tot.power + vc_arb->power; // for CB traversal
204  arb_tot.power = arb_tot.power + c_arb->power;
205  arb_tot.power = arb_tot.power + vc_arb->power; // to the o/p port
206 
207  arb_tot.power.readOp.dynamic *= num_sub;
208  arb_tot.power.readOp.leakage *= num_sub;
209 }

211 void HighRadix::compute_buff_power()
212 {
213  //input buffer read/write
214  buff_tot.power.readOp.dynamic = 2 * inp_buff->power.readOp.dynamic * RADIX;
215  buff_tot.power.readOp.leakage = inp_buff->power.readOp.leakage * RADIX;
216 
217  //row buffer read/write
218  buff_tot.power.readOp.dynamic += r_buff->power.readOp.dynamic * 2 * num_sub;
219  buff_tot.power.readOp.leakage += r_buff->power.readOp.leakage * num_sub;
220 
221  //column buffer read/write
222  buff_tot.power.readOp.dynamic += c_buff->power.readOp.dynamic * 2 * num_sub;
223  buff_tot.power.readOp.leakage += c_buff->power.readOp.leakage * num_sub;
224 }

226 void
227 HighRadix::sub_switch_power()
228 {
229  // each sub-switch power
230  sub_sw.power.readOp.dynamic = sub_sw.power.readOp.dynamic +
231  r_buff->power.readOp.dynamic * 2 /* one read and one write */ * VC_COUNT;
232  sub_sw.power.readOp.leakage = sub_sw.power.readOp.leakage +
233  r_buff->power.readOp.leakage * VC_COUNT;
234  sub_sw.power = sub_sw.power + cb->power;
235 
236  sub_sw.power.readOp.dynamic = sub_sw.power.readOp.dynamic +
237  2 * c_buff->power.readOp.dynamic /* one read and one write */ * VC_COUNT;
238  sub_sw.power.readOp.leakage = sub_sw.power.readOp.leakage +
239  c_buff->power.readOp.leakage * VC_COUNT;
240  sub_sw.power = sub_sw.power + out_cb->power;
241 
242  // arbiter power
243  sub_sw.power = sub_sw.power + cb_arb->power;
244  sub_sw.power = sub_sw.power + vc_arb->power; // for CB traversal
245  sub_sw.power = sub_sw.power + c_arb->power;
246  sub_sw.power = sub_sw.power + vc_arb->power; // to the o/p port
247 }

262 {

275  dyn_p.number_way_select_signals_mat = 1;
276  dyn_p.num_act_mats_hor_dir = 1;
277  dyn_p.is_dram = false;
278  dyn_p.V_b_sense = deviceType->Vdd; // FIXME check power calc.
279  dyn_p.ram_cell_tech_type =
280  dyn_p.num_r_subarray = (int) (sz/block_sz);
281  dyn_p.num_c_subarray = (int) block_sz;
282  dyn_p.num_mats_h_dir = 1;
283  dyn_p.num_mats_v_dir = 1;
284  dyn_p.num_do_b_subbank = (int)block_sz;
285  dyn_p.num_do_b_mat = (int) block_sz;
286  dyn_p.num_di_b_mat = (int) block_sz;
287 
288  dyn_p.use_inp_params = 1;
289  dyn_p.num_wr_ports = 1;
290  dyn_p.num_rd_ports = 1;
291  dyn_p.num_rw_ports = 0;
292  dyn_p.num_se_rd_ports =0;
293  dyn_p.out_w = (int) block_sz;
294 
295 
296  dyn_p.cell.h = g_tp.sram.b_h + 2 * g_tp.wire_outside_mat.pitch * (dyn_p.num_wr_ports +
297  dyn_p.num_rw_ports - 1 + dyn_p.num_rd_ports);
298  dyn_p.cell.w = g_tp.sram.b_w + 2 * g_tp.wire_outside_mat.pitch * (dyn_p.num_rw_ports - 1 +
299  (dyn_p.num_rd_ports - dyn_p.num_se_rd_ports) +
300  dyn_p.num_wr_ports) + g_tp.wire_outside_mat.pitch * dyn_p.num_se_rd_ports;
301 
302  Mat *buff = new Mat(dyn_p);
303  buff->compute_delays(0);
304  buff->compute_power_energy();
305  return buff;
306 }
