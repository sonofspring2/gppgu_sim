75 void selection_logic::selection_power()
76 {//based on cost effective superscalar processor TR pp27-31
77  double Ctotal, Cor, Cpencode;
78  int num_arbiter;
79  double WSelORn, WSelORprequ, WSelPn, WSelPp, WSelEnn, WSelEnp;
80 
81  //TODO: the 0.8um process data is used.
82  WSelORn = 12.5 * l_ip.F_sz_um;//this was 10 micron for the 0.8 micron process
83  WSelORprequ = 50 * l_ip.F_sz_um;//this was 40 micron for the 0.8 micron process
84  WSelPn = 12.5 * l_ip.F_sz_um;//this was 10mcron for the 0.8 micron process
85  WSelPp = 18.75 * l_ip.F_sz_um;//this was 15 micron for the 0.8 micron process
86  WSelEnn = 6.25 * l_ip.F_sz_um;//this was 5 micron for the 0.8 micron process
87  WSelEnp = 12.5 * l_ip.F_sz_um;//this was 10 micron for the 0.8 micron process
88 
89 
90  Ctotal=0;
91  num_arbiter=1;
92  while(win_entries > 4)
93  {
94  win_entries = (int)ceil((double)win_entries / 4.0);
95  num_arbiter += win_entries;
96  }
97  //the 4-input OR logic to generate anyreq
98  Cor = 4 * drain_C_(WSelORn,NCH,1,1, g_tp.cell_h_def) + drain_C_(WSelORprequ,PCH,1,1, g_tp.cell_h_def);
99  power.readOp.gate_leakage = cmos_Ig_leakage(WSelORn, WSelORprequ, 4, nor)*g_tp.peri_global.Vdd;
100 
101  //The total capacity of the 4-bit priority encoder
102  Cpencode = drain_C_(WSelPn,NCH,1, 1, g_tp.cell_h_def) + drain_C_(WSelPp,PCH,1, 1, g_tp.cell_h_def) +
103  2*drain_C_(WSelPn,NCH,1, 1, g_tp.cell_h_def) + drain_C_(WSelPp,PCH,2, 1, g_tp.cell_h_def) +
104  3*drain_C_(WSelPn,NCH,1, 1, g_tp.cell_h_def) + drain_C_(WSelPp,PCH,3, 1, g_tp.cell_h_def) +
105  4*drain_C_(WSelPn,NCH,1, 1, g_tp.cell_h_def) + drain_C_(WSelPp,PCH,4, 1, g_tp.cell_h_def) +//precompute priority logic
106  2*4*gate_C(WSelEnn+WSelEnp,20.0)+
107  4*drain_C_(WSelEnn,NCH,1, 1, g_tp.cell_h_def) + 2*4*drain_C_(WSelEnp,PCH,1, 1, g_tp.cell_h_def)+//enable logic
108  (2*4+2*3+2*2+2)*gate_C(WSelPn+WSelPp,10.0);//requests signal
109 
110  Ctotal += issue_width * num_arbiter*(Cor+Cpencode);
111 
112  power.readOp.dynamic = Ctotal*g_tp.peri_global.Vdd*g_tp.peri_global.Vdd*2;//2 means the abitration signal need to travel round trip
113  power.readOp.leakage = issue_width * num_arbiter *
114  (cmos_Isub_leakage(WSelPn, WSelPp, 2, nor)/*approximate precompute with a nor gate*///grant1p
115  + cmos_Isub_leakage(WSelPn, WSelPp, 3, nor)//grant2p
116  + cmos_Isub_leakage(WSelPn, WSelPp, 4, nor)//grant3p
117  + cmos_Isub_leakage(WSelEnn, WSelEnp, 2, nor)*4//enable logic
118  + cmos_Isub_leakage(WSelEnn, WSelEnp, 1, inv)*2*3//for each grant there are two inverters, there are 3 grant sIsubnals
119  )*g_tp.peri_global.Vdd;
120  power.readOp.gate_leakage = issue_width * num_arbiter *
121  (cmos_Ig_leakage(WSelPn, WSelPp, 2, nor)/*approximate precompute with a nor gate*///grant1p
122  + cmos_Ig_leakage(WSelPn, WSelPp, 3, nor)//grant2p
123  + cmos_Ig_leakage(WSelPn, WSelPp, 4, nor)//grant3p
124  + cmos_Ig_leakage(WSelEnn, WSelEnp, 2, nor)*4//enable logic
125  + cmos_Ig_leakage(WSelEnn, WSelEnp, 1, inv)*2*3//for each grant there are two inverters, there are 3 grant signals
126  )*g_tp.peri_global.Vdd;
127 }

183 double dep_resource_conflict_check::compare_cap()
184 {
185  double c1, c2;
186 
187  WNORp = WNORp * compare_bits/2.0;//resize the big NOR gate at the DCL according to fan in.
188  /* bottom part of comparator */
189  c2 = (compare_bits)*(drain_C_(Wcompn,NCH,1,1, g_tp.cell_h_def)+drain_C_(Wcompn,NCH,2,1, g_tp.cell_h_def))+
190  drain_C_(Wevalinvp,PCH,1,1, g_tp.cell_h_def) + drain_C_(Wevalinvn,NCH,1,1, g_tp.cell_h_def);
191 
192  /* top part of comparator */
193  c1 = (compare_bits)*(drain_C_(Wcompn,NCH,1,1, g_tp.cell_h_def)+drain_C_(Wcompn,NCH,2,1, g_tp.cell_h_def)+
194  drain_C_(Wcomppreequ,NCH,1,1, g_tp.cell_h_def)) + gate_C(WNORn + WNORp,10.0) +
195  drain_C_(WNORp,NCH,2,1, g_tp.cell_h_def) + compare_bits*drain_C_(WNORn,NCH,2,1, g_tp.cell_h_def);
196  return(c1 + c2);

251 void DFFCell::compute_DFF_cell()
252 {
253  double c1, c2, c3, c4, c5, c6;
254  /* node 5 and node 6 are identical to node 1 in capacitance */
255  c1 = c5 = c6 = fpfp_node_cap(2, 1);
256  c2 = fpfp_node_cap(2, 3);
257  c3 = fpfp_node_cap(3, 2);
258  c4 = fpfp_node_cap(2, 2);
259 
260  //cap-load of the clock signal in each Dff, actually the clock signal only connected to one NAND2
261  clock_cap= 2 * gate_C(WdecNANDn + WdecNANDp, 0, is_dram);
262  e_switch.readOp.dynamic += (c4 + c1 + c2 + c3 + c5 + c6 + 2*cell_load)*0.5*g_tp.peri_global.Vdd * g_tp.peri_global.Vdd;;
263 
264  /* no 1/2 for e_keep and e_clock because clock signal switches twice in one cycle */
265  e_keep_1.readOp.dynamic += c3 * g_tp.peri_global.Vdd * g_tp.peri_global.Vdd ;
266  e_keep_0.readOp.dynamic += c2 * g_tp.peri_global.Vdd * g_tp.peri_global.Vdd ;
267  e_clock.readOp.dynamic += clock_cap* g_tp.peri_global.Vdd * g_tp.peri_global.Vdd;;
268 
269  /* static power */
270  e_switch.readOp.leakage += (cmos_Isub_leakage(WdecNANDn, WdecNANDp, 2, nand)*5//5 NAND2 and 1 NAND3 in a DFF
271  + cmos_Isub_leakage(WdecNANDn, WdecNANDn, 3, nand))*g_tp.peri_global.Vdd;
272  e_switch.readOp.gate_leakage += (cmos_Ig_leakage(WdecNANDn, WdecNANDp, 2, nand)*5//5 NAND2 and 1 NAND3 in a DFF
273  + cmos_Ig_leakage(WdecNANDn, WdecNANDn, 3, nand))*g_tp.peri_global.Vdd;
274  //printf("leakage =%E\n",cmos_Ileak(1, is_dram) );
275 }

333 void Pipeline::compute_stage_vector()
334 {
335  double num_stages, tot_stage_vector, per_stage_vector;
336  int opcode_length = coredynp.x86? coredynp.micro_opcode_length:coredynp.opcode_length;
337  //Hthread = thread_clock_gated? 1:num_thread;
338 
339  if (!is_core_pipeline)
340  {
341  num_piperegs=l_ip.pipeline_stages*l_ip.per_stage_vector;//The number of pipeline stages are calculated based on the achievable throughput and required throughput
342  }
343  else
344  {
345  if (coredynp.core_ty==Inorder)
346  {
347  /* assume 6 pipe stages and try to estimate bits per pipe stage */
348  /* pipe stage 0/IF */
349  num_piperegs += coredynp.pc_width*2*coredynp.num_hthreads;
350  /* pipe stage IF/ID */
351  num_piperegs += coredynp.fetchW*(coredynp.instruction_length + coredynp.pc_width)*coredynp.num_hthreads;
352  /* pipe stage IF/ThreadSEL */
353  if (coredynp.multithreaded) num_piperegs += coredynp.num_hthreads*coredynp.perThreadState; //8 bit thread states
354  /* pipe stage ID/EXE */
355  num_piperegs += coredynp.decodeW*(coredynp.instruction_length + coredynp.pc_width + pow(2.0,opcode_length)+ 2*coredynp.int_data_width)*coredynp.num_hthreads;
356  /* pipe stage EXE/MEM */
357  num_piperegs += coredynp.issueW*(3 * coredynp.arch_ireg_width + pow(2.0,opcode_length) + 8*2*coredynp.int_data_width/*+2*powers (2,reg_length)*/);
358  /* pipe stage MEM/WB the 2^opcode_length means the total decoded signal for the opcode*/
359  num_piperegs += coredynp.issueW*(2*coredynp.int_data_width + pow(2.0,opcode_length) + 8*2*coredynp.int_data_width/*+2*powers (2,reg_length)*/);
360 // /* pipe stage 5/6 */
361 // num_piperegs += issueWidth*(data_width + powers (2,opcode_length)/*+2*powers (2,reg_length)*/);
362 // /* pipe stage 6/7 */
363 // num_piperegs += issueWidth*(data_width + powers (2,opcode_length)/*+2*powers (2,reg_length)*/);
364 // /* pipe stage 7/8 */
365 // num_piperegs += issueWidth*(data_width + powers (2,opcode_length)/**2*powers (2,reg_length)*/);
366 // /* assume 50% extra in control signals (rule of thumb) */
367  num_stages=6;
368 
369  }
370  else
371  {
372  /* assume 12 stage pipe stages and try to estimate bits per pipe stage */
373  /*OOO: Fetch, decode, rename, IssueQ, dispatch, regread, EXE, MEM, WB, CM */
374 
375  /* pipe stage 0/1F*/
376  num_piperegs += coredynp.pc_width*2*coredynp.num_hthreads ;//PC and Next PC
377  /* pipe stage IF/ID */
378  num_piperegs += coredynp.fetchW*(coredynp.instruction_length + coredynp.pc_width)*coredynp.num_hthreads;//PC is used to feed branch predictor in ID
379  /* pipe stage 1D/Renaming*/
380  num_piperegs += coredynp.decodeW*(coredynp.instruction_length + coredynp.pc_width)*coredynp.num_hthreads;//PC is for branch exe in later stage.
381  /* pipe stage Renaming/wire_drive */
382  num_piperegs += coredynp.decodeW*(coredynp.instruction_length + coredynp.pc_width);
383  /* pipe stage Renaming/IssueQ */
384  num_piperegs += coredynp.issueW*(coredynp.instruction_length + coredynp.pc_width + 3*coredynp.phy_ireg_width)*coredynp.num_hthreads;//3*coredynp.phy_ireg_width means 2 sources and 1 dest
385  /* pipe stage IssueQ/Dispatch */
386  num_piperegs += coredynp.issueW*(coredynp.instruction_length + 3 * coredynp.phy_ireg_width);
387  /* pipe stage Dispatch/EXE */
388 
389  num_piperegs += coredynp.issueW*(3 * coredynp.phy_ireg_width + coredynp.pc_width + pow(2.0,opcode_length)/*+2*powers (2,reg_length)*/);
390  /* 2^opcode_length means the total decoded signal for the opcode*/
391  num_piperegs += coredynp.issueW*(2*coredynp.int_data_width + pow(2.0,opcode_length)/*+2*powers (2,reg_length)*/);
392  /*2 source operands in EXE; Assume 2EXE stages* since we do not really distinguish OP*/
393  num_piperegs += coredynp.issueW*(2*coredynp.int_data_width + pow(2.0,opcode_length)/*+2*powers (2,reg_length)*/);
394  /* pipe stage EXE/MEM, data need to be read/write, address*/
395  num_piperegs += coredynp.issueW*(coredynp.int_data_width + coredynp.v_address_width + pow(2.0,opcode_length)/*+2*powers (2,reg_length)*/);//memory Opcode still need to be passed

955  /*
956  * Instruction decoder is different from n to 2^n decoders
957  * that are commonly used in row decoders in memory arrays.
958  * The RISC instruction decoder is typically a very simple device.
959  * We can decode an instruction by simply
960  * separating the machine word into small parts using wire slices
961  * The RISC instruction decoder can be approximate by the n to 2^n decoders,
962  * although this approximation usually underestimate power since each decoded
963  * instruction normally has more than 1 active signal.
964  *
965  * However, decoding a CISC instruction word is much more difficult
966  * than the RISC case. A CISC decoder is typically set up as a state machine.
967  * The machine reads the opcode field to determine
968  * what type of instruction it is,
969  * and where the other data values are.
970  * The instruction word is read in piece by piece,
971  * and decisions are made at each stage as to
972  * how the remainder of the instruction word will be read.
973  * (sequencer and ROM are usually needed)
974  * An x86 decoder can be even more complex since
975  * it involve both decoding instructions into u-ops and
976  * merge u-ops when doing micro-ops fusion.
977  */
