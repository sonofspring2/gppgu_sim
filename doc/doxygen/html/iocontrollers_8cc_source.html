46 /*
47 SUN Niagara 2 I/O power analysis:
48 total signal bits: 711
49 Total FBDIMM bits: (14+10)*2*8= 384
50 PCIe bits: (8 + 8)*2 = 32
51 10Gb NIC: (4*2+4*2)*2 = 32
52 Debug I/Os: 168
53 Other I/Os: 711- 32-32 - 384 - 168 = 95
54 
55 According to "Implementation of an 8-Core, 64-Thread, Power-Efficient SPARC Server on a Chip"
56 90% of I/Os are SerDers (the calucaltion is 384+64/(711-168)=83% about the same as the 90% reported in the paper)
57 --> around 80Pins are common I/Os.
58 Common I/Os consumes 71mW/Gb/s according to Cadence ChipEstimate @65nm
59 Niagara 2 I/O clock is 1/4 of core clock. --> 87pin (<--((711-168)*17%)) * 71mW/Gb/s *0.25*1.4Ghz = 2.17W
60 
61 Total dynamic power of FBDIMM, NIC, PCIe = 84*0.132 + 84*0.049*0.132 = 11.14 - 2.17 = 8.98
62 Further, if assuming I/O logic power is about 50% of I/Os then Total energy of FBDIMM, NIC, PCIe = 11.14 - 2.17*1.5 = 7.89
63  */

104  //SerDer_dyn is power not energy, scaling from 10mw/Gb/s @90nm
105  SerDer_dyn = 0.01*10*sqrt(interface_ip.F_sz_um/0.09)*g_tp.peri_global.Vdd/1.2*g_tp.peri_global.Vdd/1.2;

112  PMOS_sizing = 5*g_tp.min_w_nmos_*pmos_to_nmos_sizing_r;

114

118  // ---When there are multiple IP (same kind or not) selected, Cadence ChipEstimator results are not
119  // a simple summation of all IPs. Ignore this effect

121  frontend_area = 0.1 * (interface_ip.F_sz_um/0.065)* (interface_ip.F_sz_um/0.065);//Frontend is the PCS layer
122  SerDer_area = 0.35 * (interface_ip.F_sz_um/0.065)* (interface_ip.F_sz_um/0.065);
123  //Compare 130um implementation in "A 1.2-V-Only 900-mW 10 Gb Ethernet Transceiver and XAUI Interface With Robust VCO Tuning Technique"

125  //total area
126  area.set_area((mac_area + frontend_area + SerDer_area)*1e6);
127  //Power
128  //Cadence ChipEstimate using 65nm (mac, front_end are all energy. E=P*T = P/F = 1.37/1Ghz = 1.37e-9);
129  mac_dyn = 1.257e-9*g_tp.peri_global.Vdd/1.1*g_tp.peri_global.Vdd/1.1*(interface_ip.F_sz_nm/65.0);//niup.clockRate; //2.19W@1GHz fully active according to Cadence ChipEstimate @65nm
130  //Cadence ChipEstimate using 65nm soft IP;

132  //SerDer_dyn is power not energy, scaling from 216mw/10Gb/s @130nm
133  SerDer_dyn = 0.0216*10*(interface_ip.F_sz_um/0.13)*g_tp.peri_global.Vdd/1.2*g_tp.peri_global.Vdd/1.2;
134  SerDer_dyn /= niup.clockRate;//covert to energy per clock cycle of whole NIU
135 
136  mac_gates = 111700;
137  frontend_gates = 52000;
138 
139  NMOS_sizing = g_tp.min_w_nmos_;
140  PMOS_sizing = g_tp.min_w_nmos_*pmos_to_nmos_sizing_r;
141 
142  }
143 
144  power_t.readOp.dynamic = mac_dyn + frontend_dyn + SerDer_dyn;
145  power_t.readOp.leakage = (mac_gates + frontend_gates + frontend_gates)*cmos_Isub_leakage(NMOS_sizing, PMOS_sizing, 2, nand)*g_tp.peri_global.Vdd;//unit W
146  double long_channel_device_reduction = longer_channel_device_reduction(Uncore_device);
147  power_t.readOp.longer_channel_leakage = power_t.readOp.leakage * long_channel_device_reduction;
