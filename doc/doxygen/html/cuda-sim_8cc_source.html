117 unsigned int intLOGB2( unsigned int v ) {
118  unsigned int shift;
119  unsigned int r;
120 
121  r = 0;
122 
123  shift = (( v & 0xFFFF0000) != 0 ) << 4; v >>= shift; r |= shift;
124  shift = (( v & 0xFF00 ) != 0 ) << 3; v >>= shift; r |= shift;
125  shift = (( v & 0xF0 ) != 0 ) << 2; v >>= shift; r |= shift;
126  shift = (( v & 0xC ) != 0 ) << 1; v >>= shift; r |= shift;
127  shift = (( v & 0x2 ) != 0 ) << 0; v >>= shift; r |= shift;
128 
129  return r;
130 }

133 {
134  m_TextureRefToCudaArray[texref] = array;
135  unsigned int texel_size_bits = array->desc.w + array->desc.x + array->desc.y + array->desc.z;
136  unsigned int texel_size = texel_size_bits/8;
137  unsigned int Tx, Ty;
138  int r;
139 
140  printf("GPGPU-Sim PTX: texel size = %d\n", texel_size);

142  //first determine base Tx size for given linesize
143  switch (m_function_model_config.get_texcache_linesize()) {
144  case 16: Tx = 4; break;
145  case 32: Tx = 8; break;
146  case 64: Tx = 8; break;
147  case 128: Tx = 16; break;
148  case 256: Tx = 16; break;

161  Ty = m_function_model_config.get_texcache_linesize()/(Tx*texel_size);

183 void function_info::ptx_assemble()
184 {
185  if( m_assembled ) {
186  return;
187  }
188 
189  // get the instructions into instruction memory...
190  unsigned num_inst = m_instructions.size();
191  m_instr_mem_size = MAX_INST_SIZE*(num_inst+1);
192  m_instr_mem = new ptx_instruction*[ m_instr_mem_size ];
193 
194  printf("GPGPU-Sim PTX: instruction assembly for function \'%s\'... ", m_name.c_str() );
195  fflush(stdout);
196  std::list<ptx_instruction*>::iterator i;
197 
198  addr_t PC = g_assemble_code_next_pc; // globally unique address (across functions)
199  // start function on an aligned address
200  for( unsigned i=0; i < (PC%MAX_INST_SIZE); i++ )
201  s_g_pc_to_insn.push_back((ptx_instruction*)NULL);
202  PC += PC%MAX_INST_SIZE;
203  m_start_PC = PC;
204 
205  addr_t n=0; // offset in m_instr_mem
206  s_g_pc_to_insn.reserve(s_g_pc_to_insn.size() + MAX_INST_SIZE*m_instructions.size());
207  for ( i=m_instructions.begin(); i != m_instructions.end(); i++ ) {
208  ptx_instruction *pI = *i;
209  if ( pI->is_label() ) {
210  const symbol *l = pI->get_label();
211  labels[l->name()] = n;
212  } else {
213  g_pc_to_finfo[PC] = this;
214  m_instr_mem[n] = pI;
215  s_g_pc_to_insn.push_back(pI);
216  assert(pI == s_g_pc_to_insn[PC]);
217  pI->set_m_instr_mem_index(n);
218  pI->set_PC(PC);
219  assert( pI->inst_size() <= MAX_INST_SIZE );
220  for( unsigned i=1; i < pI->inst_size(); i++ ) {
221  s_g_pc_to_insn.push_back((ptx_instruction*)NULL);
222  m_instr_mem[n+i]=NULL;
223  }
224  n += pI->inst_size();
225  PC += pI->inst_size();
226  }
227  }
228  g_assemble_code_next_pc=PC;
229  for ( unsigned ii=0; ii < n; ii += m_instr_mem[ii]->inst_size() ) { // handle branch instructions
230  ptx_instruction *pI = m_instr_mem[ii];
231  if ( pI->get_opcode() == BRA_OP || pI->get_opcode() == BREAKADDR_OP || pI->get_opcode() == CALLP_OP) {
232  operand_info &target = pI->dst(); //get operand, e.g. target name
233  if ( labels.find(target.name()) == labels.end() ) {
234  printf("GPGPU-Sim PTX: Loader error (%s:%u): Branch label \"%s\" does not appear in assembly code.",
235  pI->source_file(),pI->source_line(), target.name().c_str() );
236  abort();
237  }
238  unsigned index = labels[ target.name() ]; //determine address from name
239  unsigned PC = m_instr_mem[index]->get_PC();
240  m_symtab->set_label_address( target.get_symbol(), PC );
241  target.set_type(label_t);
242  }
243  }
244 
245  printf(" done.\n");
246  fflush(stdout);
247  printf("GPGPU-Sim PTX: finding reconvergence points for \'%s\'...\n", m_name.c_str() );
248 
249  create_basic_blocks();
250  connect_basic_blocks();
251  bool modified = false;
252  do {
253  find_dominators();
254  find_idominators();
255  modified = connect_break_targets();
256  } while (modified == true);
257 
258  if ( g_debug_execution>=50 ) {
259  print_basic_blocks();
260  print_basic_block_links();
261  print_basic_block_dot();
262  }
263  if ( g_debug_execution>=2 ) {
264  print_dominators();
265  }
266  find_postdominators();
267  find_ipostdominators();
268  if ( g_debug_execution>=50 ) {
269  print_postdominators();
270  print_ipostdominators();
271  }
272 
273  printf("GPGPU-Sim PTX: pre-decoding instructions for \'%s\'...\n", m_name.c_str() );
274  for ( unsigned ii=0; ii < n; ii += m_instr_mem[ii]->inst_size() ) { // handle branch instructions
275  ptx_instruction *pI = m_instr_mem[ii];
276  pI->pre_decode();
277  }
278  printf("GPGPU-Sim PTX: ... done pre-decoding instructions for \'%s\'.\n", m_name.c_str() );
279  fflush(stdout);
280 
281  m_assembled = true;
282 }

820 void ptx_instruction::pre_decode()
821 {
822  pc = m_PC;
823  isize = m_inst_size;
824  for( unsigned i=0; i<4; i++) {
825  out[i] = 0;
826  in[i] = 0;
827  }
828  is_vectorin = 0;
829  is_vectorout = 0;
830  std::fill_n(arch_reg.src, MAX_REG_OPERANDS, -1);
831  std::fill_n(arch_reg.dst, MAX_REG_OPERANDS, -1);
832  pred = 0;
833  ar1 = 0;
834  ar2 = 0;
835  space = m_space_spec;
836  memory_op = no_memory_op;
837  data_size = 0;
838  if ( has_memory_read() || has_memory_write() ) {
839  unsigned to_type = get_type();
840  data_size = datatype2size(to_type);
841  memory_op = has_memory_read() ? memory_load : memory_store;
842  }
843 
844  bool has_dst = false ;
845 
846  switch ( get_opcode() ) {
847 #define OP_DEF(OP,FUNC,STR,DST,CLASSIFICATION) case OP: has_dst = (DST!=0); break;
848 #include "opcodes.def"
849 #undef OP_DEF
850  default:
851  printf( "Execution error: Invalid opcode (0x%x)\n", get_opcode() );
852  break;
853  }
854 
855  switch( m_cache_option ) {
856  case CA_OPTION: cache_op = CACHE_ALL; break;
857  case CG_OPTION: cache_op = CACHE_GLOBAL; break;
858  case CS_OPTION: cache_op = CACHE_STREAMING; break;
859  case LU_OPTION: cache_op = CACHE_LAST_USE; break;
860  case CV_OPTION: cache_op = CACHE_VOLATILE; break;
861  case WB_OPTION: cache_op = CACHE_WRITE_BACK; break;
862  case WT_OPTION: cache_op = CACHE_WRITE_THROUGH; break;
863  default:
864  if( m_opcode == LD_OP || m_opcode == LDU_OP )
865  cache_op = CACHE_ALL;
866  else if( m_opcode == ST_OP )
867  cache_op = CACHE_WRITE_BACK;
868  else if( m_opcode == ATOM_OP )
869  cache_op = CACHE_GLOBAL;
870  break;
871  }
872 
873  set_opcode_and_latency();
874  set_bar_type();
875  // Get register operands
876  int n=0,m=0;
877  ptx_instruction::const_iterator opr=op_iter_begin();
878  for ( ; opr != op_iter_end(); opr++, n++ ) { //process operands
879  const operand_info &o = *opr;
880  if ( has_dst && n==0 ) {
881  // Do not set the null register "_" as an architectural register
882  if ( o.is_reg() && !o.is_non_arch_reg() ) {
883  out[0] = o.reg_num();
884  arch_reg.dst[0] = o.arch_reg_num();
885  } else if ( o.is_vector() ) {
886  is_vectorin = 1;
887  unsigned num_elem = o.get_vect_nelem();
888  if( num_elem >= 1 ) out[0] = o.reg1_num();
889  if( num_elem >= 2 ) out[1] = o.reg2_num();
890  if( num_elem >= 3 ) out[2] = o.reg3_num();
891  if( num_elem >= 4 ) out[3] = o.reg4_num();
892  for (int i = 0; i < num_elem; i++)
893  arch_reg.dst[i] = o.arch_reg_num(i);
894  }
895  } else {
896  if ( o.is_reg() && !o.is_non_arch_reg() ) {
897  int reg_num = o.reg_num();
898  arch_reg.src[m] = o.arch_reg_num();
899  switch ( m ) {
900  case 0: in[0] = reg_num; break;
901  case 1: in[1] = reg_num; break;
902  case 2: in[2] = reg_num; break;
903  default: break;
904  }
905  m++;
906  } else if ( o.is_vector() ) {
907  //assert(m == 0); //only support 1 vector operand (for textures) right now
908  is_vectorout = 1;
909  unsigned num_elem = o.get_vect_nelem();
910  if( num_elem >= 1 ) in[0] = o.reg1_num();
911  if( num_elem >= 2 ) in[1] = o.reg2_num();
912  if( num_elem >= 3 ) in[2] = o.reg3_num();
913  if( num_elem >= 4 ) in[3] = o.reg4_num();
914  for (int i = 0; i < num_elem; i++)
915  arch_reg.src[i] = o.arch_reg_num(i);
916  m+=4;
917  }
918  }
919  }
920 
921  // Get predicate
922  if(has_pred()) {
923  const operand_info &p = get_pred();
924  pred = p.reg_num();
925  }
926 
927  // Get address registers inside memory operands.
928  // Assuming only one memory operand per instruction,
929  // and maximum of two address registers for one memory operand.
930  if( has_memory_read() || has_memory_write() ) {
931  ptx_instruction::const_iterator op=op_iter_begin();
932  for ( ; op != op_iter_end(); op++, n++ ) { //process operands
933  const operand_info &o = *op;
934 
935  if(o.is_memory_operand()) {
936  // We do not support the null register as a memory operand
937  assert( !o.is_non_arch_reg() );
938 
939  // Check PTXPlus-type operand
940  // memory operand with addressing (ex. s[0x4] or g[$r1])
941  if(o.is_memory_operand2()) {
942 
943  // memory operand with one address register (ex. g[$r1+0x4] or s[$r2+=0x4])
944  if(o.get_double_operand_type() == 0 || o.get_double_operand_type() == 3){
945  ar1 = o.reg_num();
946  arch_reg.src[4] = o.arch_reg_num();
947  // TODO: address register in $r2+=0x4 should be an output register as well
948  }
949  // memory operand with two address register (ex. s[$r1+$r1] or g[$r1+=$r2])
950  else if(o.get_double_operand_type() == 1 || o.get_double_operand_type() == 2) {
951  ar1 = o.reg1_num();
952  arch_reg.src[4] = o.arch_reg_num();
953  ar2 = o.reg2_num();
954  arch_reg.src[5] = o.arch_reg_num();
955  // TODO: first address register in $r1+=$r2 should be an output register as well
956  }
957  }

959 
960  }

965  }

969  }
970
