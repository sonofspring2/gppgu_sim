71 void CMesh::_ComputeSize( const Configuration &config ) {
72 
73  int k = config.GetInt( "k" );
74  int n = config.GetInt( "n" );
75  assert(n <= 2); // broken for n > 2
76  int c = config.GetInt( "c" );
77  assert(c == 4); // broken for c != 4
78 
79  ostringstream router_name;
80  //how many routers in the x or y direction
81  _xcount = config.GetInt("x");
82  _ycount = config.GetInt("y");
83  assert(_xcount == _ycount); // broken for asymmetric topologies
84  //configuration of hohw many clients in X and Y per router
85  _xrouter = config.GetInt("xr");
86  _yrouter = config.GetInt("yr");
87  assert(_xrouter == _yrouter); // broken for asymmetric concentration
88 
89  gK = _k = k ;
90  gN = _n = n ;
91  gC = _c = c ;
92 
93  assert(c == _xrouter*_yrouter);
94 
95  _nodes = _c * powi( _k, _n); // Number of nodes in network
96  _size = powi( _k, _n); // Number of routers in network
97  _channels = 2 * _n * _size; // Number of channels in network
98 
99  _cX = _c / _n ; // Concentration in X Dimension
100  _cY = _c / _cX ; // Concentration in Y Dimension
101 
102  //
103  _memo_NodeShiftX = _cX >> 1 ;
104  _memo_NodeShiftY = log_two(gK * _cX) + ( _cY >> 1 ) ;
105  _memo_PortShiftY = log_two(gK * _cX) ;
106 
107 }

109 void CMesh::_BuildNet( const Configuration& config ) {
110 
111  int x_index ;
112  int y_index ;
113 
114  //standard trace configuration
115  if(gTrace){
116  cout<<"Setup Finished Router"<<endl;
117  }
118 
119  //latency type, noc or conventional network
120  bool use_noc_latency;
121  use_noc_latency = (config.GetInt("use_noc_latency")==1);
122 
123  ostringstream name;
124  // The following vector is used to check that every
125  // processor in the system is connected to the network
126  vector<bool> channel_vector(_nodes, false) ;
127 
128  //
129  // Routers and Channel
130  //
131  for (int node = 0; node < _size; ++node) {
132 
133  // Router index derived from mesh index
134  y_index = node / _k ;
135  x_index = node % _k ;
136 
137  const int degree_in = 2 *_n + _c ;
138  const int degree_out = 2 *_n + _c ;
139 
140  name << "router_" << y_index << '_' << x_index;
141  _routers[node] = Router::NewRouter( config,
142  this,
143  name.str(),
144  node,
145  degree_in,
146  degree_out);
147  _timed_modules.push_back(_routers[node]);
148  name.str("");
149 
150  //
151  // Port Numbering: as best as I can determine, the order in
152  // which the input and output channels are added to the
153  // router determines the associated port number that must be
154  // used by the router. Output port number increases with
155  // each new channel
156  //
157 
158  //
159  // Processing node channels
160  //
161  for (int y = 0; y < _cY ; y++) {
162  for (int x = 0; x < _cX ; x++) {
163  int link = (_k * _cX) * (_cY * y_index + y) + (_cX * x_index + x) ;
164  assert( link >= 0 ) ;
165  assert( link < _nodes ) ;
166  assert( channel_vector[ link ] == false ) ;
167  channel_vector[link] = true ;
168  // Ingress Ports
169  _routers[node]->AddInputChannel(_inject[link], _inject_cred[link]);
170  // Egress Ports
171  _routers[node]->AddOutputChannel(_eject[link], _eject_cred[link]);
172  //injeciton ejection latency is 1
173  _inject[link]->SetLatency( 1 );
174  _eject[link]->SetLatency( 1 );
175  }
176  }
177 
178  //
179  // router to router channels
180  //
181  const int x = node % _k ;
182  const int y = node / _k ;
183  const int offset = powi( _k, _n ) ;
184 
185  //the channel number of the input output channels.
186  int px_out = _k * y + x + 0 * offset ;
187  int nx_out = _k * y + x + 1 * offset ;
188  int py_out = _k * y + x + 2 * offset ;
189  int ny_out = _k * y + x + 3 * offset ;
190  int px_in = _k * y + ((x+1)) + 1 * offset ;
191  int nx_in = _k * y + ((x-1)) + 0 * offset ;
192  int py_in = _k * ((y+1)) + x + 3 * offset ;
193  int ny_in = _k * ((y-1)) + x + 2 * offset ;
194 
195  // Express Channels
196  if (x == 0){
197  // Router on left edge of mesh. Connect to -x output of
198  // another router on the left edge of the mesh.
199  if (y < _k / 2 )
200  nx_in = _k * (y + _k/2) + x + offset ;
201  else
202  nx_in = _k * (y - _k/2) + x + offset ;
203  }
204 
205  if (x == (_k-1)){
206  // Router on right edge of mesh. Connect to +x output of
207  // another router on the right edge of the mesh.
208  if (y < _k / 2)
209  px_in = _k * (y + _k/2) + x ;
210  else
211  px_in = _k * (y - _k/2) + x ;
212  }
213 
214  if (y == 0) {
215  // Router on bottom edge of mesh. Connect to -y output of
216  // another router on the bottom edge of the mesh.
217  if (x < _k / 2)
218  ny_in = _k * y + (x + _k/2) + 3 * offset ;
219  else
220  ny_in = _k * y + (x - _k/2) + 3 * offset ;
221  }
222 
223  if (y == (_k-1)) {
224  // Router on top edge of mesh. Connect to +y output of
225  // another router on the top edge of the mesh
226  if (x < _k / 2)
227  py_in = _k * y + (x + _k/2) + 2 * offset ;
228  else
229  py_in = _k * y + (x - _k/2) + 2 * offset ;
230  }
231 
232  /*set latency and add the channels*/
233 
234  // Port 0: +x channel
235  if(use_noc_latency) {
236  int const px_latency = (x == _k-1) ? (_cY*_k/2) : _cX;
237  _chan[px_out]->SetLatency( px_latency );
238  _chan_cred[px_out]->SetLatency( px_latency );
239  } else {
240  _chan[px_out]->SetLatency( 1 );
241  _chan_cred[px_out]->SetLatency( 1 );
242  }
243  _routers[node]->AddOutputChannel( _chan[px_out], _chan_cred[px_out] );
244  _routers[node]->AddInputChannel( _chan[px_in], _chan_cred[px_in] );
245 
246  if(gTrace) {
247  cout<<"Link "<<" "<<px_out<<" "<<px_in<<" "<<node<<" "<<_chan[px_out]->GetLatency()<<endl;
248  }
249 
250  // Port 1: -x channel
251  if(use_noc_latency) {
252  int const nx_latency = (x == 0) ? (_cY*_k/2) : _cX;
253  _chan[nx_out]->SetLatency( nx_latency );
254  _chan_cred[nx_out]->SetLatency( nx_latency );
255  } else {
256  _chan[nx_out]->SetLatency( 1 );
257  _chan_cred[nx_out]->SetLatency( 1 );
258  }
259  _routers[node]->AddOutputChannel( _chan[nx_out], _chan_cred[nx_out] );
260  _routers[node]->AddInputChannel( _chan[nx_in], _chan_cred[nx_in] );
261 
262  if(gTrace){
263  cout<<"Link "<<" "<<nx_out<<" "<<nx_in<<" "<<node<<" "<<_chan[nx_out]->GetLatency()<<endl;
264  }
265 
266  // Port 2: +y channel
267  if(use_noc_latency) {
268  int const py_latency = (y == _k-1) ? (_cX*_k/2) : _cY;
269  _chan[py_out]->SetLatency( py_latency );
270  _chan_cred[py_out]->SetLatency( py_latency );
271  } else {
272  _chan[py_out]->SetLatency( 1 );
273  _chan_cred[py_out]->SetLatency( 1 );
274  }
275  _routers[node]->AddOutputChannel( _chan[py_out], _chan_cred[py_out] );
276  _routers[node]->AddInputChannel( _chan[py_in], _chan_cred[py_in] );
277 
278  if(gTrace){
279  cout<<"Link "<<" "<<py_out<<" "<<py_in<<" "<<node<<" "<<_chan[py_out]->GetLatency()<<endl;
280  }
281 
282  // Port 3: -y channel
283  if(use_noc_latency){
284  int const ny_latency = (y == 0) ? (_cX*_k/2) : _cY;
285  _chan[ny_out]->SetLatency( ny_latency );
286  _chan_cred[ny_out]->SetLatency( ny_latency );
287  } else {
288  _chan[ny_out]->SetLatency( 1 );
289  _chan_cred[ny_out]->SetLatency( 1 );
290  }
291  _routers[node]->AddOutputChannel( _chan[ny_out], _chan_cred[ny_out] );
292  _routers[node]->AddInputChannel( _chan[ny_in], _chan_cred[ny_in] );
293 
294  if(gTrace){
295  cout<<"Link "<<" "<<ny_out<<" "<<ny_in<<" "<<node<<" "<<_chan[ny_out]->GetLatency()<<endl;
296  }
297 
298  }
299 
300  // Check that all processors were connected to the network
301  for ( int i = 0 ; i < _nodes ; i++ )
302  assert( channel_vector[i] == true ) ;
303 
304  if(gTrace){
305  cout<<"Setup Finished Link"<<endl;
306  }
307 }

316 int CMesh::NodeToRouter( int address ) {
317 
318  int y = (address / (_cX*gK))/_cY ;
319  int x = (address % (_cX*gK))/_cY ;
320  int router = y*gK + x ;
321 
322  return router ;

325 int CMesh::NodeToPort( int address ) {
326 
327  const int maskX = _cX - 1 ;
328  const int maskY = _cY - 1 ;
329 
330  int x = address & maskX ;
331  int y = (int)(address/(2*gK)) & maskY ;
332 
333  return (gC / 2) * y + x;
334 }

343 int cmesh_xy( int cur, int dest ) {
344 
345  const int POSITIVE_X = 0 ;
346  const int NEGATIVE_X = 1 ;
347  const int POSITIVE_Y = 2 ;
348  const int NEGATIVE_Y = 3 ;
349 
350  int cur_y = cur / gK;
351  int cur_x = cur % gK;
352  int dest_y = dest / gK;
353  int dest_x = dest % gK;
354 
355  // Dimension-order Routing: x , y
356  if (cur_x < dest_x) {
357  // Express?
358  if ((dest_x - cur_x) > 1){
359  if (cur_y == 0)
360  return gC + NEGATIVE_Y ;
361  if (cur_y == (gK-1))
362  return gC + POSITIVE_Y ;
363  }
364  return gC + POSITIVE_X ;
365  }
366  if (cur_x > dest_x) {
367  // Express ?
368  if ((cur_x - dest_x) > 1){
369  if (cur_y == 0)
370  return gC + NEGATIVE_Y ;
371  if (cur_y == (gK-1))
372  return gC + POSITIVE_Y ;
373  }
374  return gC + NEGATIVE_X ;
375  }
376  if (cur_y < dest_y) {
377  // Express?
378  if ((dest_y - cur_y) > 1) {
379  if (cur_x == 0)
380  return gC + NEGATIVE_X ;
381  if (cur_x == (gK-1))
382  return gC + POSITIVE_X ;
383  }
384  return gC + POSITIVE_Y ;
385  }
386  if (cur_y > dest_y) {
387  // Express ?
388  if ((cur_y - dest_y) > 1 ){
389  if (cur_x == 0)
390  return gC + NEGATIVE_X ;
391  if (cur_x == (gK-1))
392  return gC + POSITIVE_X ;
393  }
394  return gC + NEGATIVE_Y ;
395  }
396  return 0;

455 void xy_yx_cmesh( const Router *r, const Flit *f, int in_channel,
456  OutputSet *outputs, bool inject )
457 {
458 
459  // ( Traffic Class , Routing Order ) -> Virtual Channel Range
460  int vcBegin = 0, vcEnd = gNumVCs-1;
461  if ( f->type == Flit::READ_REQUEST ) {
462  vcBegin = gReadReqBeginVC;
463  vcEnd = gReadReqEndVC;
464  } else if ( f->type == Flit::WRITE_REQUEST ) {
465  vcBegin = gWriteReqBeginVC;
466  vcEnd = gWriteReqEndVC;
467  } else if ( f->type == Flit::READ_REPLY ) {
468  vcBegin = gReadReplyBeginVC;
469  vcEnd = gReadReplyEndVC;
470  } else if ( f->type == Flit::WRITE_REPLY ) {
471  vcBegin = gWriteReplyBeginVC;
472  vcEnd = gWriteReplyEndVC;
473  }
474  assert(((f->vc >= vcBegin) && (f->vc <= vcEnd)) || (inject && (f->vc < 0)));
475 
476  int out_port;
477 
478  if(inject) {
479 
480  out_port = -1;
481 
482  } else {
483 
484  // Current Router
485  int cur_router = r->GetID();
486 
487  // Destination Router
488  int dest_router = CMesh::NodeToRouter( f->dest ) ;
489 
490  if (dest_router == cur_router) {
491 
492  // Forward to processing element
493  out_port = CMesh::NodeToPort( f->dest );
494 
495  } else {
496 
497  // Forward to neighbouring router
498 
499  //each class must have at least 2 vcs assigned or else xy_yx will deadlock
500  int const available_vcs = (vcEnd - vcBegin + 1) / 2;
501  assert(available_vcs > 0);
502 
503  // randomly select dimension order at first hop
504  bool x_then_y = ((in_channel < gC) ?
505  (RandomInt(1) > 0) :
506  (f->vc < (vcBegin + available_vcs)));
507 
508  if(x_then_y) {
509  out_port = cmesh_xy( cur_router, dest_router );
510  vcEnd -= available_vcs;
511  } else {
512  out_port = cmesh_yx( cur_router, dest_router );
513  vcBegin += available_vcs;
514  }
515  }
516 
517  }
518 
519  outputs->Clear();
520 
521  outputs->AddRange( out_port , vcBegin, vcEnd );
522 }
