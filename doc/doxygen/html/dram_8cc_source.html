44 dram_t::dram_t( unsigned int partition_id, const struct memory_config *config, memory_stats_t *stats,
45  memory_partition_unit *mp )
46 {
47  id = partition_id;
48  m_memory_partition_unit = mp;
49  m_stats = stats;
50  m_config = config;
51 
52  CCDc = 0;
53  RRDc = 0;
54  RTWc = 0;
55  WTRc = 0;
56 
57  rw = READ; //read mode is default
58 
59  bkgrp = (bankgrp_t**) calloc(sizeof(bankgrp_t*), m_config->nbkgrp);
60  bkgrp[0] = (bankgrp_t*) calloc(sizeof(bank_t), m_config->nbkgrp);
61  for (unsigned i=1; i<m_config->nbkgrp; i++) {
62  bkgrp[i] = bkgrp[0] + i;
63  }
64  for (unsigned i=0; i<m_config->nbkgrp; i++) {
65  bkgrp[i]->CCDLc = 0;
66  bkgrp[i]->RTPLc = 0;
67  }
68 
69  bk = (bank_t**) calloc(sizeof(bank_t*),m_config->nbk);
70  bk[0] = (bank_t*) calloc(sizeof(bank_t),m_config->nbk);
71  for (unsigned i=1;i<m_config->nbk;i++)
72  bk[i] = bk[0] + i;
73  for (unsigned i=0;i<m_config->nbk;i++) {
74  bk[i]->state = BANK_IDLE;
75  bk[i]->bkgrpindex = i/(m_config->nbk/m_config->nbkgrp);
76  }
77  prio = 0;
78  rwq = new fifo_pipeline<dram_req_t>("rwq",m_config->CL,m_config->CL+1);
79  mrqq = new fifo_pipeline<dram_req_t>("mrqq",0,2);
80  returnq = new fifo_pipeline<mem_fetch>("dramreturnq",0,m_config->gpgpu_dram_return_queue_size==0?1024:m_config->gpgpu_dram_return_queue_size);
81  m_frfcfs_scheduler = NULL;
82  if ( m_config->scheduler_type == DRAM_FRFCFS )
83  m_frfcfs_scheduler = new frfcfs_scheduler(m_config,this,stats);
84  n_cmd = 0;
85  n_activity = 0;
86  n_nop = 0;
87  n_act = 0;
88  n_pre = 0;
89  n_rd = 0;
90  n_wr = 0;
91  n_req = 0;
92  max_mrqs_temp = 0;
93  bwutil = 0;
94  max_mrqs = 0;
95  ave_mrqs = 0;
96 
97  for (unsigned i=0;i<10;i++) {
98  dram_util_bins[i]=0;
99  dram_eff_bins[i]=0;
100  }
101  last_n_cmd = last_n_activity = last_bwutil = 0;
102 
103  n_cmd_partial = 0;
104  n_activity_partial = 0;
105  n_nop_partial = 0;
106  n_act_partial = 0;
107  n_pre_partial = 0;
108  n_req_partial = 0;
109  ave_mrqs_partial = 0;
110  bwutil_partial = 0;
111 
112  if ( queue_limit() )
113  mrqq_Dist = StatCreate("mrqq_length",1, queue_limit());
114  else //queue length is unlimited;
115  mrqq_Dist = StatCreate("mrqq_length",1,64); //track up to 64 entries
116 }

168 void dram_t::push( class mem_fetch *data )
169 {
170  assert(id == data->get_tlx_addr().chip); // Ensure request is in correct memory partition
171 
172  dram_req_t *mrq = new dram_req_t(data);
173  data->set_status(IN_PARTITION_MC_INTERFACE_QUEUE,gpu_sim_cycle+gpu_tot_sim_cycle);
174  mrqq->push(mrq);
175 
176  // stats...
177  n_req += 1;
178  n_req_partial += 1;
179  if ( m_config->scheduler_type == DRAM_FRFCFS ) {
180  unsigned nreqs = m_frfcfs_scheduler->num_pending();
181  if ( nreqs > max_mrqs_temp)
182  max_mrqs_temp = nreqs;
183  } else {
184  max_mrqs_temp = (max_mrqs_temp > mrqq->get_length())? max_mrqs_temp : mrqq->get_length();
185  }
186  m_stats->memlatstat_dram_access(data);
187 }

189 void dram_t::scheduler_fifo()
190 {
191  if (!mrqq->empty()) {
192  unsigned int bkn;
193  dram_req_t *head_mrqq = mrqq->top();
194  head_mrqq->data->set_status(IN_PARTITION_MC_BANK_ARB_QUEUE,gpu_sim_cycle+gpu_tot_sim_cycle);
195  bkn = head_mrqq->bk;
196  if (!bk[bkn]->mrq)
197  bk[bkn]->mrq = mrqq->pop();
198  }
199 }

205 void dram_t::cycle()
206 {
207 
208  if( !returnq->full() ) {
209  dram_req_t *cmd = rwq->pop();
210  if( cmd ) {
211 #ifdef DRAM_VIEWCMD
212  printf("\tDQ: BK%d Row:%03x Col:%03x", cmd->bk, cmd->row, cmd->col + cmd->dqbytes);
213 #endif
214  cmd->dqbytes += m_config->dram_atom_size;
215  if (cmd->dqbytes >= cmd->nbytes) {
216  mem_fetch *data = cmd->data;
217  data->set_status(IN_PARTITION_MC_RETURNQ,gpu_sim_cycle+gpu_tot_sim_cycle);
218  if( data->get_access_type() != L1_WRBK_ACC && data->get_access_type() != L2_WRBK_ACC ) {
219  data->set_reply();
220  returnq->push(data);
221  } else {
222  m_memory_partition_unit->set_done(data);
223  delete data;
224  }
225  delete cmd;
226  }
227 #ifdef DRAM_VIEWCMD
228  printf("\n");
229 #endif
230  }
231  }
232 
233  /* check if the upcoming request is on an idle bank */
234  /* Should we modify this so that multiple requests are checked? */
235 
236  switch (m_config->scheduler_type) {
237  case DRAM_FIFO: scheduler_fifo(); break;
238  case DRAM_FRFCFS: scheduler_frfcfs(); break;
239  default:
240  printf("Error: Unknown DRAM scheduler type\n");
241  assert(0);
242  }
243  if ( m_config->scheduler_type == DRAM_FRFCFS ) {
244  unsigned nreqs = m_frfcfs_scheduler->num_pending();
245  if ( nreqs > max_mrqs) {
246  max_mrqs = nreqs;
247  }
248  ave_mrqs += nreqs;
249  ave_mrqs_partial += nreqs;
250  } else {
251  if (mrqq->get_length() > max_mrqs) {
252  max_mrqs = mrqq->get_length();
253  }
254  ave_mrqs += mrqq->get_length();
255  ave_mrqs_partial += mrqq->get_length();
256  }
257 
258  unsigned k=m_config->nbk;
259  bool issued = false;
260 
261  // check if any bank is ready to issue a new read
262  for (unsigned i=0;i<m_config->nbk;i++) {
263  unsigned j = (i + prio) % m_config->nbk;
264  unsigned grp = j>>m_config->bk_tag_length;
265  if (bk[j]->mrq) { //if currently servicing a memory request
266  bk[j]->mrq->data->set_status(IN_PARTITION_DRAM,gpu_sim_cycle+gpu_tot_sim_cycle);
267  // correct row activated for a READ
268  if ( !issued && !CCDc && !bk[j]->RCDc &&
269  !(bkgrp[grp]->CCDLc) &&
270  (bk[j]->curr_row == bk[j]->mrq->row) &&
271  (bk[j]->mrq->rw == READ) && (WTRc == 0 ) &&
272  (bk[j]->state == BANK_ACTIVE) &&
273  !rwq->full() ) {
274  if (rw==WRITE) {
275  rw=READ;
276  rwq->set_min_length(m_config->CL);
277  }
278  rwq->push(bk[j]->mrq);
279  bk[j]->mrq->txbytes += m_config->dram_atom_size;
280  CCDc = m_config->tCCD;
281  bkgrp[grp]->CCDLc = m_config->tCCDL;
282  RTWc = m_config->tRTW;
283  bk[j]->RTPc = m_config->BL/m_config->data_command_freq_ratio;
284  bkgrp[grp]->RTPLc = m_config->tRTPL;
285  issued = true;
286  n_rd++;
287  bwutil += m_config->BL/m_config->data_command_freq_ratio;
288  bwutil_partial += m_config->BL/m_config->data_command_freq_ratio;
289  bk[j]->n_access++;
290 #ifdef DRAM_VERIFY
291  PRINT_CYCLE=1;
292  printf("\tRD Bk:%d Row:%03x Col:%03x \n",
293  j, bk[j]->curr_row,
294  bk[j]->mrq->col + bk[j]->mrq->txbytes - m_config->dram_atom_size);
295 #endif
296  // transfer done
297  if ( !(bk[j]->mrq->txbytes < bk[j]->mrq->nbytes) ) {
298  bk[j]->mrq = NULL;
299  }
300  } else
301  // correct row activated for a WRITE
302  if ( !issued && !CCDc && !bk[j]->RCDWRc &&
303  !(bkgrp[grp]->CCDLc) &&
304  (bk[j]->curr_row == bk[j]->mrq->row) &&
305  (bk[j]->mrq->rw == WRITE) && (RTWc == 0 ) &&
306  (bk[j]->state == BANK_ACTIVE) &&
307  !rwq->full() ) {
308  if (rw==READ) {
309  rw=WRITE;
310  rwq->set_min_length(m_config->WL);
311  }
312  rwq->push(bk[j]->mrq);
313 
314  bk[j]->mrq->txbytes += m_config->dram_atom_size;
315  CCDc = m_config->tCCD;
316  bkgrp[grp]->CCDLc = m_config->tCCDL;
317  WTRc = m_config->tWTR;
318  bk[j]->WTPc = m_config->tWTP;
319  issued = true;
320  n_wr++;
321  bwutil += m_config->BL/m_config->data_command_freq_ratio;
322  bwutil_partial += m_config->BL/m_config->data_command_freq_ratio;
323 #ifdef DRAM_VERIFY
324  PRINT_CYCLE=1;
325  printf("\tWR Bk:%d Row:%03x Col:%03x \n",
326  j, bk[j]->curr_row,
327  bk[j]->mrq->col + bk[j]->mrq->txbytes - m_config->dram_atom_size);
328 #endif
329  // transfer done
330  if ( !(bk[j]->mrq->txbytes < bk[j]->mrq->nbytes) ) {
331  bk[j]->mrq = NULL;
332  }
333  }
334 
335  else
336  // bank is idle
337  if ( !issued && !RRDc &&
338  (bk[j]->state == BANK_IDLE) &&
339  !bk[j]->RPc && !bk[j]->RCc ) {
340 #ifdef DRAM_VERIFY
341  PRINT_CYCLE=1;
342  printf("\tACT BK:%d NewRow:%03x From:%03x \n",
343  j,bk[j]->mrq->row,bk[j]->curr_row);
344 #endif
345  // activate the row with current memory request
346  bk[j]->curr_row = bk[j]->mrq->row;
347  bk[j]->state = BANK_ACTIVE;
348  RRDc = m_config->tRRD;
349  bk[j]->RCDc = m_config->tRCD;
350  bk[j]->RCDWRc = m_config->tRCDWR;
351  bk[j]->RASc = m_config->tRAS;
352  bk[j]->RCc = m_config->tRC;
353  prio = (j + 1) % m_config->nbk;
354  issued = true;
355  n_act_partial++;
356  n_act++;
357  }
358 
359  else
360  // different row activated
361  if ( (!issued) &&
362  (bk[j]->curr_row != bk[j]->mrq->row) &&
363  (bk[j]->state == BANK_ACTIVE) &&
364  (!bk[j]->RASc && !bk[j]->WTPc &&
365  !bk[j]->RTPc &&
366  !bkgrp[grp]->RTPLc) ) {
367  // make the bank idle again
368  bk[j]->state = BANK_IDLE;
369  bk[j]->RPc = m_config->tRP;
370  prio = (j + 1) % m_config->nbk;
371  issued = true;
372  n_pre++;
373  n_pre_partial++;
374 #ifdef DRAM_VERIFY
375  PRINT_CYCLE=1;
376  printf("\tPRE BK:%d Row:%03x \n", j,bk[j]->curr_row);
377 #endif
378  }
379  } else {
380  if (!CCDc && !RRDc && !RTWc && !WTRc && !bk[j]->RCDc && !bk[j]->RASc
381  && !bk[j]->RCc && !bk[j]->RPc && !bk[j]->RCDWRc) k--;
382  bk[j]->n_idle++;
383  }
384  }
385  if (!issued) {
386  n_nop++;
387  n_nop_partial++;
388 #ifdef DRAM_VIEWCMD
389  printf("\tNOP ");
390 #endif
391  }
392  if (k) {
393  n_activity++;
394  n_activity_partial++;
395  }
396  n_cmd++;
397  n_cmd_partial++;
398 
399  // decrements counters once for each time dram_issueCMD is called
400  DEC2ZERO(RRDc);
401  DEC2ZERO(CCDc);
402  DEC2ZERO(RTWc);
403  DEC2ZERO(WTRc);
404  for (unsigned j=0;j<m_config->nbk;j++) {
405  DEC2ZERO(bk[j]->RCDc);
406  DEC2ZERO(bk[j]->RASc);
407  DEC2ZERO(bk[j]->RCc);
408  DEC2ZERO(bk[j]->RPc);
409  DEC2ZERO(bk[j]->RCDWRc);
410  DEC2ZERO(bk[j]->WTPc);
411  DEC2ZERO(bk[j]->RTPc);
412  }
413  for (unsigned j=0; j<m_config->nbkgrp; j++) {
414  DEC2ZERO(bkgrp[j]->CCDLc);
415  DEC2ZERO(bkgrp[j]->RTPLc);
416  }
417 
418 #ifdef DRAM_VISUALIZE
419  visualize();
420 #endif
421 }
