750 int TrafficManager::_IssuePacket( int source, int cl )
751 {
752  int result = 0;
753  if(_use_read_write[cl]){ //use read and write
754  //check queue for waiting replies.
755  //check to make sure it is on time yet
756  if (!_repliesPending[source].empty()) {

758  result = -1;
759  }
760  } else {
761 
762  //produce a packet

764

766  result = (RandomFloat() < _write_fraction[cl]) ? 2 : 1;

768  _requestsOutstanding[source]++;

832  }
833

835  ( ( _sim_state == draining ) && ( time < _drain_time ) ) ) {
836  record = _measure_stats[cl];
837  }
838

840  RandomInt(_subnets-1) :
841  _subnet[packet_type]);

844  *gWatchOut << GetSimTime() << " | "
845  << "node" << source << " | "
846  << "Enqueuing packet " << pid

898  f->tail = true;
899  } else {
900  f->tail = false;
901  }
902 
903  f->vc = -1;
904 
905  if ( f->watch ) {
906  *gWatchOut << GetSimTime() << " | "
907  << "node" << source << " | "
908  << "Enqueuing flit " << f->id
909  << " (packet " << f->pid
910  << ") at time " << time
911  << "." << endl;
912  }
913 
914  _partial_packets[source][cl].push_back( f );
915  }

951 {

953  for(int c = 0; c < _classes; ++c) {
954  flits_in_flight |= !_total_in_flight_flits[c].empty();
955  }

958  cout << "WARNING: Possible network deadlock.\n";
959  }
960

962 
963  for ( int subnet = 0; subnet < _subnets; ++subnet ) {
964  for ( int n = 0; n < _nodes; ++n ) {

977  ++_accepted_flits[f->cl][n];

979  ++_accepted_packets[f->cl][n];

981  }
982  }
983

985  if ( c ) {

988  int const vc = *iter;
989  assert(!_outstanding_classes[n][subnet][vc].empty());

991  _outstanding_classes[n][subnet][vc].pop();

1003  if ( !_empty_network ) {
1004  _Inject();
1005  }

1007  for(int subnet = 0; subnet < _subnets; ++subnet) {
1008 
1009  for(int n = 0; n < _nodes; ++n) {
1010

1012

1014

1016

1024  assert(f->vc == _last_vc[n][subnet][last_class]);

1029  }
1030  }
1031 
1032  for(int i = 1; i <= class_limit; ++i) {
1033 
1034  int const c = (last_class + i) % _classes;
1035

1037 
1038  if(pp.empty()) {
1039  continue;

1043  assert(cf);
1044  assert(cf->cl == c);

1061  assert(se.output_port == -1);

1104  vc_start :

1118  } else {

1120  *gWatchOut << GetSimTime() << " | " << FullName() << " | "
1121  << " Selected output VC " << vc << "." << endl;

1124  break;

1126  }
1127  }
1128  }

1131  if(cf->watch) {
1132  *gWatchOut << GetSimTime() << " | " << FullName() << " | "
1133  << "No output VC found for flit " << cf->id
1134  << "." << endl;
1135  }
1136  } else {
1137  if(dest_buf->IsFullFor(cf->vc)) {
1138  if(cf->watch) {
1139  *gWatchOut << GetSimTime() << " | " << FullName() << " | "
1140  << "Selected output VC " << cf->vc
1141  << " is full for flit " << cf->id
1142  << "." << endl;
1143  }
1144  } else {
1145  f = cf;
1146  }
1147  }
1148  }

1151 
1152  assert(f->subnetwork == subnet);
1153

1155 
1156  if(f->head) {
1157

1159  if(!_noq) {
1160  const FlitChannel * inject = _net[subnet]->GetInject(n);
