28 /*anynet
29  *
30  *Network setup file format
31  *example 1:
32  *router 0 router 1 15 router 2
33  *
34  *Router 0 is connect to router 1 with a 15-cycle channel, and router 0 is connected to
35  * router 2 with a 1-cycle channel, the channels latency are unidirectional, so channel
36  * from router 1 back to router 0 is only single-cycle because it was not specified
37  *
38  *example 2:
39  *router 0 node 0 node 1 5 node 2 5
40  *
41  *Router 0 is directly connected to node 0-2. Channel latency is 5cycles for 1 and 2. In
42  * this case the latency specification is bidirectional, the injeciton and ejection lat
43  * for node 1 and 2 are 5-cycle
44  *
45  *other notes:
46  *
47  *Router and node numbers must be sequential starting with 0
48  *Credit channel latency follows the channel latency, even though it travels in revse
49  * direction this might not be desired
50  *
51  */

255 void AnyNet::route(int r_start){
256  int* dist = new int[_size];
257  int* prev = new int[_size];
258  set<int> rlist;
259  for(int i = 0; i<_size; i++){
260  dist[i] = numeric_limits<int>::max();
261  prev[i] = -1;
262  rlist.insert(i);
263  }
264  dist[r_start] = 0;
265  while(!rlist.empty()){
266  //find min
267  int min_dist = numeric_limits<int>::max();
268  int min_cand = -1;
269  for(set<int>::iterator i = rlist.begin();
270  i!=rlist.end();
271  i++){
272  if(dist[*i]<min_dist){
273  min_dist = dist[*i];
274  min_cand = *i;
275  }
276  }
277  rlist.erase(min_cand);
278 
279  //neighbor
280  for(map<int,pair<int,int> >::iterator i = router_list[1][min_cand].begin();
281  i!=router_list[1][min_cand].end();
282  i++){
283  int new_dist = dist[min_cand] + i->second.second;//distance is hops not cycles
284  if(new_dist < dist[i->first]){
285  dist[i->first] = new_dist;
286  prev[i->first] = min_cand;
287  }
288  }
289  }
290 
291  //post process from the prev list
292  for(int i = 0; i<_size; i++){
293  if(prev[i] ==-1){ //self
294  assert(i == r_start);
295  for(map<int, pair<int, int> >::iterator iter = router_list[0][i].begin();
296  iter!=router_list[0][i].end();
297  iter++){
298  routing_table[r_start][iter->first]=iter->second.first;
299  //cout<<"node "<<iter->first<<" port "<< iter->second.first<<endl;
300  }
301  } else {
302  int distance=0;
303  int neighbor=i;
304  while(prev[neighbor]!=r_start){
305  assert(router_list[1][neighbor].count(prev[neighbor])>0);
306  distance+=router_list[1][prev[neighbor]][neighbor].second;//REVERSE lat
307  neighbor= prev[neighbor];
308  }
309  distance+=router_list[1][prev[neighbor]][neighbor].second;//lat
310 
311  assert( router_list[1][r_start].count(neighbor)!=0);
312  int port = router_list[1][r_start][neighbor].first;
313  for(map<int, pair<int,int> >::iterator iter = router_list[0][i].begin();
314  iter!=router_list[0][i].end();
315  iter++){
316  routing_table[r_start][iter->first]=port;
317  //cout<<"node "<<iter->first<<" port "<< port<<" dist "<<distance<<endl;
318  }
319  }
320  }

324 void AnyNet::readFile(){
325 
326  ifstream network_list;
327  string line;
328  enum ParseState{HEAD_TYPE=0,
329  HEAD_ID,
330  BODY_TYPE,
331  BODY_ID,
332  LINK_WEIGHT};
333  enum ParseType{NODE=0,
334  ROUTER,
335  UNKNOWN};
336 
337  network_list.open(file_name.c_str());
338  if(!network_list.is_open()){
339  cout<<"Anynet:can't open network file "<<file_name<<endl;
340  exit(-1);
341  }
342 
343  //loop through the entire file
344  while(!network_list.eof()){
345  getline(network_list,line);
346  if(line==""){
347  continue;
348  }
349 
350  ParseState state=HEAD_TYPE;
351  //position to parse out white sspace
352  int pos = 0;
353  int next_pos=-1;
354  string temp;
355  //the first node and its type
356  int head_id = -1;
357  ParseType head_type = UNKNOWN;
358  //stuff that head are linked to
359  ParseType body_type = UNKNOWN;
360  int body_id = -1;
361  int link_weight = 1;
362 
363  do{
364 
365  //skip empty spaces
366  next_pos = line.find(" ",pos);
367  temp = line.substr(pos,next_pos-pos);
368  pos = next_pos+1;
369  if(temp=="" || temp==" "){
370  continue;
371  }
372 
373  switch(state){
374  case HEAD_TYPE:
375  if(temp=="router"){
376  head_type = ROUTER;
377  } else if (temp == "node"){
378  head_type = NODE;
379  } else {
380  cout<<"Anynet:Unknow head of line type "<<temp<<"\n";
381  assert(false);
382  }
383  state=HEAD_ID;
384  break;
385  case HEAD_ID:
386  //need better error check
387  head_id = atoi(temp.c_str());
388 
389  //intialize router structures
390  if(router_list[NODE].count(head_id) == 0){
391  router_list[NODE][head_id] = map<int, pair<int,int> >();
392  }
393  if(router_list[ROUTER].count(head_id) == 0){
394  router_list[ROUTER][head_id] = map<int, pair<int,int> >();
395  }
396 
397  state=BODY_TYPE;
398  break;
399  case LINK_WEIGHT:
400  if(temp=="router"||
401  temp == "node"){
402  //ignore
403  } else {
404  link_weight= atoi(temp.c_str());
405  router_list[head_type][head_id][body_id].second=link_weight;
406  break;
407  }
408  //intentionally letting it flow through
409  case BODY_TYPE:
410  if(temp=="router"){
411  body_type = ROUTER;
412  } else if (temp == "node"){
413  body_type = NODE;
414  } else {
415  cout<<"Anynet:Unknow body type "<<temp<<"\n";
416  assert(false);
417  }
418  state=BODY_ID;
419  break;
420  case BODY_ID:
421  body_id = atoi(temp.c_str());
422  //intialize router structures if necessary
423  if(body_type==ROUTER){
424  if(router_list[NODE].count(body_id) ==0){
425  router_list[NODE][body_id] = map<int, pair<int,int> >();
426  }
427  if(router_list[ROUTER].count(body_id) == 0){
428  router_list[ROUTER][body_id] = map<int, pair<int,int> >();
429  }
430  }
431 
432  if(head_type==NODE && body_type==NODE){
433 
434  cout<<"Anynet:Cannot connect node to node "<<temp<<"\n";
435  assert(false);
436 
437  } else if(head_type==NODE && body_type==ROUTER){
438 
439  if(node_list.count(head_id)!=0 &&
440  node_list[head_id]!=body_id){
441  cout<<"Anynet:Node "<<body_id<<" trying to connect to multiple router "
442  <<body_id<<" and "<<node_list[head_id]<<endl;
443  assert(false);
444  }
445  node_list[head_id]=body_id;
446  router_list[NODE][body_id][head_id]=pair<int, int>(-1,1);
447 
448  } else if(head_type==ROUTER && body_type==NODE){
449  //insert and check node
450  if(node_list.count(body_id) != 0 &&
451  node_list[body_id]!=head_id){
452  cout<<"Anynet:Node "<<body_id<<" trying to connect to multiple router "
453  <<body_id<<" and "<<node_list[head_id]<<endl;
454  assert(false);
455  }
456  node_list[body_id] = head_id;
457  router_list[NODE][head_id][body_id]=pair<int, int>(-1,1);
458 
459  } else if(head_type==ROUTER && body_type==ROUTER){
460  router_list[ROUTER][head_id][body_id]=pair<int, int>(-1,1);
461  if(router_list[ROUTER][body_id].count(head_id)==0){
462  router_list[ROUTER][body_id][head_id]=pair<int, int>(-1,1);
463  }
464  }
465  state=LINK_WEIGHT;
466  break ;
467  default:
468  cout<<"Anynet:Unknow parse state\n";
469  assert(false);
470  break;
471  }
472 
473  } while(pos!=0);
474  if(state!=LINK_WEIGHT &&
475  state!=BODY_TYPE){
476  cout<<"Anynet:Incomplete parse of the line: "<<line<<endl;
477  }
478 
479  }
480 
481  //map verification, make sure the information contained in bother maps
482  //are the same
483  assert(router_list[0].size() == router_list[1].size());
484 
485  //traffic generator assumes node list is sequenctial and starts at 0
486  vector<int> node_check;
487  for(map<int,int>::iterator i = node_list.begin();
488  i!=node_list.end();
489  i++){
490  node_check.push_back(i->first);
491  }
492  sort(node_check.begin(), node_check.end());
493  for(size_t i = 0; i<node_check.size(); i++){
494  if(node_check[i] != i){
495  cout<<"Anynet:booksim trafficmanager assumes sequential node numbering starting at 0\n";
496  assert(false);
497  }
498  }
499 
500 }
