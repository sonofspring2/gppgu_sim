47 void SelAlloc::Allocate( )
48 {
49  int input;
50  int output;
51 
52  int input_offset;
53  int output_offset;
54 
55  map<int, sRequest>::iterator p;
56  set<int>::iterator outer_iter;
57  bool wrapped;
58 
59  int max_index;
60  int max_pri;
61 
62  vector<int> grants(_outputs, -1);
63 
64  for ( int iter = 0; iter < _iter; ++iter ) {
65  // Grant phase
66 
67  for( outer_iter = _out_occ.begin( );
68  outer_iter != _out_occ.end( ); ++outer_iter ) {
69  output = *outer_iter;
70 
71  // Skip loop if there are no requests
72  // or the output is already matched or
73  // the output is masked
74  if ( ( _out_req[output].empty( ) ) ||
75  ( _outmatch[output] != -1 ) ||
76  ( _outmask[output] != 0 ) ) {
77  continue;
78  }
79 
80  // A round-robin arbiter between input requests
81  input_offset = _gptrs[output];
82 
83  p = _out_req[output].begin( );
84  while( ( p != _out_req[output].end( ) ) &&
85  ( p->second.port < input_offset ) ) {
86  p++;
87  }
88 
89  max_index = -1;
90  max_pri = 0;
91 
92  wrapped = false;
93  while( (!wrapped) ||
94  ( ( p != _out_req[output].end() ) &&
95  ( p->second.port < input_offset ) ) ) {
96  if ( p == _out_req[output].end( ) ) {
97  if ( wrapped ) { break; }
98  // p is valid here because empty lists
99  // are skipped (above)
100  p = _out_req[output].begin( );
101  wrapped = true;
102  }
103 
104  input = p->second.port;
105 
106  // we know the output is free (above) and
107  // if the input is free, check if request is the
108  // highest priority so far
109  if ( ( _inmatch[input] == -1 ) &&
110  ( ( p->second.out_pri > max_pri ) || ( max_index == -1 ) ) ) {
111  max_pri = p->second.out_pri;
112  max_index = input;
113  }
114 
115  p++;
116  }
117 
118  if ( max_index != -1 ) { // grant
119  grants[output] = max_index;
120  }
121  }
122 
123 #ifdef DEBUG_SELALLOC
124  cout << "grants: ";
125  for ( int i = 0; i < _outputs; ++i ) {
126  cout << grants[i] << " ";
127  }
128  cout << endl;
129 
130  cout << "aptrs: ";
131  for ( int i = 0; i < _inputs; ++i ) {
132  cout << _aptrs[i] << " ";
133  }
134  cout << endl;
135 #endif
136 
137  // Accept phase
138 
139  for ( outer_iter = _in_occ.begin( );
140  outer_iter != _in_occ.end( ); ++outer_iter ) {
141  input = *outer_iter;
142 
143  if ( _in_req[input].empty( ) ) {
144  continue;
145  }
146 
147  // A round-robin arbiter between output grants
148  output_offset = _aptrs[input];
149 
150  p = _in_req[input].begin( );
151  while( ( p != _in_req[input].end( ) ) &&
152  ( p->second.port < output_offset ) ) {
153  p++;
154  }
155 
156  max_index = -1;
157  max_pri = 0;
158 
159  wrapped = false;
160  while( (!wrapped) ||
161  ( ( p != _in_req[input].end() ) &&
162  ( p->second.port < output_offset ) ) ) {
163  if ( p == _in_req[input].end( ) ) {
164  if ( wrapped ) { break; }
165  // p is valid here because empty lists
166  // are skipped (above)
167  p = _in_req[input].begin( );
168  wrapped = true;
169  }
170 
171  output = p->second.port;
172 
173  // we know the output is free (above) and
174  // if the input is free, check if the highest
175  // priroity
176  if ( ( grants[output] == input ) &&
177  ( !_out_req[output].empty( ) ) &&
178  ( ( p->second.in_pri > max_pri ) || ( max_index == -1 ) ) ) {
179  max_pri = p->second.in_pri;
180  max_index = output;
181  }
182 
183  p++;
184  }
185 
186  if ( max_index != -1 ) {
187  // Accept
188  output = max_index;
189 
190  _inmatch[input] = output;
191  _outmatch[output] = input;
192 
193  // Only update pointers if accepted during the 1st iteration
194  if ( iter == 0 ) {
195  _gptrs[output] = ( input + 1 ) % _inputs;
196  _aptrs[input] = ( output + 1 ) % _outputs;
197  }
198  }
199  }
200  }
201 
202 #ifdef DEBUG_SELALLOC
203  cout << "input match: ";
204  for ( int i = 0; i < _inputs; ++i ) {
205  cout << _inmatch[i] << " ";
206  }
207  cout << endl;
208 
209  cout << "output match: ";
210  for ( int j = 0; j < _outputs; ++j ) {
211  cout << _outmatch[j] << " ";
212  }
213  cout << endl;
214 #endif
215 }
