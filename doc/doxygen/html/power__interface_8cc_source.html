41 void mcpat_cycle(const gpgpu_sim_config &config, const struct shader_core_config *shdr_config, class gpgpu_sim_wrapper *wrapper, class power_stat_t *power_stats, unsigned stat_sample_freq, unsigned tot_cycle, unsigned cycle, unsigned tot_inst, unsigned inst){
42 
43  static bool mcpat_init=true;
44 
45  if(mcpat_init){ // If first cycle, don't have any power numbers yet
46  mcpat_init=false;
47  return;
48  }
49 
50  if ((tot_cycle+cycle) % stat_sample_freq == 0) {
51 
52  wrapper->set_inst_power(shdr_config->gpgpu_clock_gated_lanes,
53  stat_sample_freq, stat_sample_freq,
54  power_stats->get_total_inst(), power_stats->get_total_int_inst(),
55  power_stats->get_total_fp_inst(), power_stats->get_l1d_read_accesses(),
56  power_stats->get_l1d_write_accesses(), power_stats->get_committed_inst());
57 
58  // Single RF for both int and fp ops
59  wrapper->set_regfile_power(power_stats->get_regfile_reads(), power_stats->get_regfile_writes(), power_stats->get_non_regfile_operands());
60 
61  //Instruction cache stats
62  wrapper->set_icache_power(power_stats->get_inst_c_hits(), power_stats->get_inst_c_misses());
63 
64  //Constant Cache, shared memory, texture cache
65  wrapper->set_ccache_power(power_stats->get_constant_c_hits(), power_stats->get_constant_c_misses());
66  wrapper->set_tcache_power(power_stats->get_texture_c_hits(), power_stats->get_texture_c_misses());
67  wrapper->set_shrd_mem_power(power_stats->get_shmem_read_access());
68 
69  wrapper->set_l1cache_power(power_stats->get_l1d_read_hits(), power_stats->get_l1d_read_misses(),
70  power_stats->get_l1d_write_hits(), power_stats->get_l1d_write_misses());
71 
72 
73  wrapper->set_l2cache_power(power_stats->get_l2_read_hits(), power_stats->get_l2_read_misses(),
74  power_stats->get_l2_write_hits(), power_stats->get_l2_write_misses());
75 
76 
77  float active_sms=(*power_stats->m_active_sms)/stat_sample_freq;
78  float num_cores = shdr_config->num_shader();
79  float num_idle_core = num_cores - active_sms;
80  wrapper->set_idle_core_power(num_idle_core);
81 
82  //pipeline power - pipeline_duty_cycle *= percent_active_sms;
83  float pipeline_duty_cycle=((*power_stats->m_average_pipeline_duty_cycle/( stat_sample_freq)) < 0.8)?((*power_stats->m_average_pipeline_duty_cycle)/stat_sample_freq):0.8;
84  wrapper->set_duty_cycle_power(pipeline_duty_cycle);
85 
86  //Memory Controller
87  wrapper->set_mem_ctrl_power(power_stats->get_dram_rd(), power_stats->get_dram_wr(), power_stats->get_dram_pre());
88 
89  //Execution pipeline accesses
90  //FPU (SP) accesses, Integer ALU (not present in Tesla), Sfu accesses
91  wrapper->set_exec_unit_power(power_stats->get_tot_fpu_accessess(), power_stats->get_ialu_accessess(), power_stats->get_tot_sfu_accessess());
92 
93  //Average active lanes for sp and sfu pipelines
94  float avg_sp_active_lanes=(power_stats->get_sp_active_lanes())/stat_sample_freq;
95  float avg_sfu_active_lanes=(power_stats->get_sfu_active_lanes())/stat_sample_freq;
96  assert(avg_sp_active_lanes<=32);
97  assert(avg_sfu_active_lanes<=32);
98  wrapper->set_active_lanes_power((power_stats->get_sp_active_lanes())/stat_sample_freq,
99  (power_stats->get_sfu_active_lanes())/stat_sample_freq);
100 
101 
102  double n_icnt_simt_to_mem = (double)power_stats->get_icnt_simt_to_mem(); // # flits from SIMT clusters to memory partitions
103  double n_icnt_mem_to_simt = (double)power_stats->get_icnt_mem_to_simt(); // # flits from memory partitions to SIMT clusters
104  wrapper->set_NoC_power(n_icnt_mem_to_simt, n_icnt_simt_to_mem); // Number of flits traversing the interconnect
105 
106  wrapper->compute();
107 
108 
109  wrapper->update_components_power();
110  wrapper->print_trace_files();
111  power_stats->save_stats();
112 
113  wrapper->detect_print_steady_state(0,tot_inst+inst);
114 
115  wrapper->power_metrics_calculations();
116 
117 
118  wrapper->dump();
119  }
120  //wrapper->close_files();
121 }
