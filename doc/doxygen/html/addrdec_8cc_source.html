67 new_addr_type linear_to_raw_address_translation::partition_address( new_addr_type addr ) const
68 {
69  if (!gap) {
70  return addrdec_packbits( ~(addrdec_mask[CHIP] | sub_partition_id_mask), addr, 64, 0 );
71  } else {
72  // see addrdec_tlx for explanation
73  unsigned long long int partition_addr;
74  partition_addr = ( (addr>>ADDR_CHIP_S) / m_n_channel) << ADDR_CHIP_S;
75  partition_addr |= addr & ((1 << ADDR_CHIP_S) - 1);
76  // remove the part of address that constributes to the sub partition ID
77  partition_addr = addrdec_packbits( ~sub_partition_id_mask, partition_addr, 64, 0);
78  return partition_addr;
79  }

82 void linear_to_raw_address_translation::addrdec_tlx(new_addr_type addr, addrdec_t *tlx) const
83 {
84  unsigned long long int addr_for_chip,rest_of_addr;
85  if (!gap) {
86  tlx->chip = addrdec_packbits(addrdec_mask[CHIP], addr, addrdec_mkhigh[CHIP], addrdec_mklow[CHIP]);
87  tlx->bk = addrdec_packbits(addrdec_mask[BK], addr, addrdec_mkhigh[BK], addrdec_mklow[BK]);
88  tlx->row = addrdec_packbits(addrdec_mask[ROW], addr, addrdec_mkhigh[ROW], addrdec_mklow[ROW]);
89  tlx->col = addrdec_packbits(addrdec_mask[COL], addr, addrdec_mkhigh[COL], addrdec_mklow[COL]);
90  tlx->burst= addrdec_packbits(addrdec_mask[BURST], addr, addrdec_mkhigh[BURST], addrdec_mklow[BURST]);
91  } else {
92  // Split the given address at ADDR_CHIP_S into (MSBs,LSBs)
93  // - extract chip address using modulus of MSBs
94  // - recreate the rest of the address by stitching the quotient of MSBs and the LSBs
95  addr_for_chip = (addr>>ADDR_CHIP_S) % m_n_channel;
96  rest_of_addr = ( (addr>>ADDR_CHIP_S) / m_n_channel) << ADDR_CHIP_S;
97  rest_of_addr |= addr & ((1 << ADDR_CHIP_S) - 1);
98 
99  tlx->chip = addr_for_chip;
100  tlx->bk = addrdec_packbits(addrdec_mask[BK], rest_of_addr, addrdec_mkhigh[BK], addrdec_mklow[BK]);
101  tlx->row = addrdec_packbits(addrdec_mask[ROW], rest_of_addr, addrdec_mkhigh[ROW], addrdec_mklow[ROW]);
102  tlx->col = addrdec_packbits(addrdec_mask[COL], rest_of_addr, addrdec_mkhigh[COL], addrdec_mklow[COL]);
103  tlx->burst= addrdec_packbits(addrdec_mask[BURST], rest_of_addr, addrdec_mkhigh[BURST], addrdec_mklow[BURST]);
104  }
105 
106  // combine the chip address and the lower bits of DRAM bank address to form the subpartition ID
107  unsigned sub_partition_addr_mask = m_n_sub_partition_in_channel - 1;
108  tlx->sub_partition = tlx->chip * m_n_sub_partition_in_channel
109  + (tlx->bk & sub_partition_addr_mask);
110 }

112 void linear_to_raw_address_translation::addrdec_parseoption(const char *option)
113 {
114  unsigned int dramid_start = 0;
115  int dramid_parsed = sscanf(option, "dramid@%d", &dramid_start);
116  if (dramid_parsed == 1) {
117  ADDR_CHIP_S = dramid_start;
118  } else {
119  ADDR_CHIP_S = -1;
120  }
121 
122  const char *cmapping = strchr(option, ';');
123  if (cmapping == NULL) {
124  cmapping = option;
125  } else {
126  cmapping += 1;
127  }
128 
129  addrdec_mask[CHIP] = 0x0;
130  addrdec_mask[BK] = 0x0;
131  addrdec_mask[ROW] = 0x0;
132  addrdec_mask[COL] = 0x0;
133  addrdec_mask[BURST]= 0x0;
134 
135  int ofs = 63;
136  while ((*cmapping) != '\0') {
137  switch (*cmapping) {
138  case 'D': case 'd':
139  assert(dramid_parsed != 1); addrdec_mask[CHIP] |= (1ULL << ofs); ofs--; break;
140  case 'B': case 'b': addrdec_mask[BK] |= (1ULL << ofs); ofs--; break;
141  case 'R': case 'r': addrdec_mask[ROW] |= (1ULL << ofs); ofs--; break;
142  case 'C': case 'c': addrdec_mask[COL] |= (1ULL << ofs); ofs--; break;
143  case 'S': case 's': addrdec_mask[BURST] |= (1ULL << ofs); addrdec_mask[COL] |= (1ULL << ofs); ofs--; break;
144  // ignore bit
145  case '0': ofs--; break;
146  // ignore character
147  case '|':
148  case ' ':
149  case '.': break;
150  default:
151  fprintf(stderr, "ERROR: Invalid address mapping character '%c' in option '%s'\n", *cmapping, option);
152  }
153  cmapping += 1;
154  }
155 
156  if (ofs != -1) {
157  fprintf(stderr, "ERROR: Invalid address mapping length (%d) in option '%s'\n", 63 - ofs, option);
158  assert(ofs == -1);
159  }
160 }

355 // a simple sweep test to ensure that two linear addresses are not mapped to the same raw address
356 void linear_to_raw_address_translation::sweep_test() const
357 {
358  new_addr_type sweep_range = 16 * 1024 * 1024;
359 
360 #if tr1_hash_map_ismap == 1
361  typedef tr1_hash_map<addrdec_t, new_addr_type> history_map_t;
362 #else
363  typedef tr1_hash_map<addrdec_t, new_addr_type, hash_addrdec_t> history_map_t;
364 #endif
365  history_map_t history_map;
366 
367  for (new_addr_type raw_addr = 4; raw_addr < sweep_range; raw_addr += 4) {
368  addrdec_t tlx;
369  addrdec_tlx(raw_addr, &tlx);
370 
371  history_map_t::iterator h = history_map.find(tlx);
372 
373  if (h != history_map.end()) {
374  printf("[AddrDec] ** Error: address decoding mapping aliases two addresses to same partition with same intra-partition address: %llx %llx\n", h->second, raw_addr);
375  abort();
376  } else {
377  assert((int)tlx.chip < m_n_channel);
378  // ensure that partition_address() returns the concatenated address
379  if ((ADDR_CHIP_S != -1 and raw_addr >= (1ULL << ADDR_CHIP_S)) or
380  (ADDR_CHIP_S == -1 and raw_addr >= (1ULL << addrdec_mklow[CHIP]))) {
381  assert(raw_addr != partition_address(raw_addr));
382  }
383  history_map[tlx] = raw_addr;
384  }
385 
386  if ((raw_addr & 0xffff) == 0) printf("%llu scaned\n", raw_addr);
387  }
388 }

411 static unsigned int LOGB2_32( unsigned int v )
412 {
413  unsigned int shift;
414  unsigned int r;
415 
416  r = 0;
417 
418  shift = (( v & 0xFFFF0000) != 0 ) << 4; v >>= shift; r |= shift;
419  shift = (( v & 0xFF00 ) != 0 ) << 3; v >>= shift; r |= shift;
420  shift = (( v & 0xF0 ) != 0 ) << 2; v >>= shift; r |= shift;
421  shift = (( v & 0xC ) != 0 ) << 1; v >>= shift; r |= shift;
422  shift = (( v & 0x2 ) != 0 ) << 0; v >>= shift; r |= shift;
423 
424  return r;
425 }

427 static new_addr_type addrdec_packbits( new_addr_type mask, new_addr_type val, unsigned char high, unsigned char low)
428 {
429  unsigned pos=0;
430  new_addr_type result = 0;
431  for (unsigned i=low;i<high;i++) {
432  if ((mask & ((unsigned long long int)1<<i)) != 0) {
433  result |= ((val & ((unsigned long long int)1<<i)) >> i) << pos;
434  pos++;
435  }
436  }
437  return result;
