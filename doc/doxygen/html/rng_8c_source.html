56 void ran_start(seed) /* do this before using ran_array */
57  long seed; /* selector for different streams */
58 #endif
59 {
60  register int t,j;
61  long x[KK+KK-1]; /* the preparation buffer */
62  register long ss=(seed+2)&(MM-2);
63  for (j=0;j<KK;j++) {
64  x[j]=ss; /* bootstrap the buffer */
65  ss<<=1; if (ss>=MM) ss-=MM-2; /* cyclic shift 29 bits */
66  }
67  x[1]++; /* make x[1] (and only x[1]) odd */
68  for (ss=seed&(MM-1),t=TT-1; t; ) {
69  for (j=KK-1;j>0;j--) x[j+j]=x[j], x[j+j-1]=0; /* "square" */
70  for (j=KK+KK-2;j>=KK;j--)
71  x[j-(KK-LL)]=mod_diff(x[j-(KK-LL)],x[j]),
72  x[j-KK]=mod_diff(x[j-KK],x[j]);
73  if (is_odd(ss)) { /* "multiply by z" */
74  for (j=KK;j>0;j--) x[j]=x[j-1];
75  x[0]=x[KK]; /* shift the buffer cyclically */
76  x[LL]=mod_diff(x[LL],x[KK]);
77  }
78  if (ss) ss>>=1; else t--;
79  }
80  for (j=0;j<LL;j++) ran_x[j+KK-LL]=x[j];
81  for (;j<KK;j++) ran_x[j-LL]=x[j];
82  for (j=0;j<10;j++) ran_array(x,KK+KK-1); /* warm things up */
83  ran_arr_ptr=&ran_arr_started;
84 }
