129 void ChaosRouter::ReadInputs( )
130 {
131  Flit *f;
132  Credit *c;
133 
134  for ( int input = 0; input < _inputs; ++input ) {
135  f = _input_channels[input]->Receive();
136 
137  if ( f ) {
138  _input_frame[input].push( f );
139 
140  if ( f->watch ) {
141  *gWatchOut << GetSimTime() << " | " << FullName() << " | "
142  << "Flit arriving at " << FullName()
143  << " on channel " << input << endl
144  << *f;
145  }
146 
147  switch( _input_state[input] ) {
148  case empty:
149  if ( f->head ) {
150  if ( f->tail ) {
151  _input_state[input] = full;
152  } else {
153  _input_state[input] = filling;
154  }
155  _rf( this, f, input, _input_route[input], false );
156  } else {
157  cout << *f;
158  Error( "Empty buffer received non-head flit!" );
159  }
160  break;
161 
162  case filling:
163  if ( f->tail ) {
164  _input_state[input] = full;
165  } else if ( f->head ) {
166  Error( "Input buffer received another head before previous tail!" );
167  }
168  break;
169 
170  case full:
171  Error( "Received flit while full!" );
172  break;
173 
174  case leaving:
175  if ( f->head ) {
176  _input_state[input] = shared;
177 
178  if ( f->tail ) {
179  Error( "Received single-flit packet in leaving state!" );
180  }
181  } else {
182  cout << *f;
183  Error( "Received non-head flit while packet leaving!" );
184  }
185  break;
186 
187  case cut_through:
188  if ( f->tail ) {
189  _input_state[input] = leaving;
190  }
191  if ( f->head ) {
192  cout << *f;
193  Error( "Received head flit in cut through buffer!" );
194  }
195  break;
196 
197  case shared:
198  if ( f->head ) {
199  Error( "Shared buffer received another head!" );
200  } else if ( f->tail ) {
201  cout << "Input " << input << endl;
202  cout << *f;
203  Error( "Shared buffer received another tail!" );
204  }
205  break;
206  }
207  }
208  }
209 
210  // Process incoming credits
211 
212  for ( int output = 0; output < _outputs; ++output ) {
213  c = _output_credits[output]->Receive();
214 
215  if ( c ) {
216  _next_queue_cnt[output]--;
217 
218  if ( _next_queue_cnt[output] < 0 ) {
219  Error( "Next queue count fell below zero!" );
220  }
221 
222  c->Free();
223  }
224  }

300 int ChaosRouter::_MultiQueueForOutput( int output ) const
301 {
302  // return oldest multi queue that prefers the output,
303  // or if none prefer and the multi queue is full,
304  // return a random entry
305 
306  int mq_oldest = -1;
307  int mq_age;
308 
309  int m, r;
310 
311  bool isfull = true;
312 
313  for ( int i = 0; i < _multi_queue_size; ++i ) {
314  if ( ( _multi_match[i] == -1 ) &&
315  ( ( _multi_state[i] == full ) ||
316  ( _multi_state[i] == filling ) ) ) {
317 
318  if ( ( !_mq_route[i]->OutputEmpty( output ) ) &&
319  ( ( mq_oldest == -1 ) || ( _mq_age[i] > mq_age ) ) ) {
320  mq_oldest = i;
321  mq_age = _mq_age[i];
322  }
323  }
324 
325  // deroute only if all queues contain head flits ...
326 
327  if ( ( _multi_state[i] != full ) &&
328  ( _multi_state[i] != filling ) &&
329  ( _multi_state[i] != shared ) ) {
330  isfull = false;
331  }
332  }
333 
334  // Don't deroute MQs to the ejection channel
335  if ( ( mq_oldest == -1 ) && isfull &&
336  ( !_IsEjectionChan( output ) ) ) {
337  r = RandomInt( _multi_queue_size - 1 );
338 
339  // Find first routable multi-queue
340  for ( int i = 0; i < _multi_queue_size; ++i ) {
341  m = ( i + r ) % _multi_queue_size;
342  if ( ( _multi_state[m] == filling ) ||
343  ( _multi_state[m] == full ) ) {
344  mq_oldest = m;
345  //cout << "DEROUTING at " << FullName() << endl;
346  break;
347  }
348  }
349 
350  if ( mq_oldest == -1 ) {
351  cout << "write stall" << endl;
352  }
353  }
354 
355  return mq_oldest;
356 }

375 void ChaosRouter::_NextInterestingChannel( )
376 {
377  bool interesting;
378 
379  int mq_index;
380  int in_index;
381  int mq_avail;
382 
383  int c;
384 
385  interesting = false;
386  mq_index = -1;
387  in_index = -1;
388 
389  // A channel is interesting if
390  //
391  // ( output frame available and
392  // ( ( a multiqueue packet wants output channel ) or
393  // ( an input packet wants output channel ) or
394  // ( the multiqueue is full ) ) )
395  // or
396  // ( the packet at the input channel is stalled )
397 
398  for ( c = 0; ( c < _inputs ) && ( !interesting ); ++c ) {
399  if ( _OutputAvail( _cur_channel ) ) {
400  mq_index = _MultiQueueForOutput( _cur_channel );
401  in_index = _InputForOutput( _cur_channel );
402 
403  if ( ( mq_index != -1 ) || ( in_index != -1 ) ) {
404  interesting = true;
405  }
406  }
407 
408  if ( _input_state[_cur_channel] == full ) {
409  interesting = true;
410  }
411 
412  if ( !interesting ) {
413  _cur_channel = ( _cur_channel + 1 ) % _inputs;
414  }
415  }
416 
417  if ( interesting ) {
418  //cout << _cur_channel << " is interesting at " << FullName() << endl;
419 
420  if ( mq_index != -1 ) {
421  //cout << "Match for multi-queue " << mq_index << " at " << FullName()
422  // << ", output matched = " << _output_matched[_cur_channel] << endl;
423 
424  _output_matched[_cur_channel] = true;
425  _multi_match[mq_index] = _cur_channel;
426  } else if ( in_index != -1 ) {
427  _output_matched[_cur_channel] = true;
428  _input_output_match[in_index] = _cur_channel;
429 
430  //cout << "Match for input " << in_index << " at " << FullName() << endl;
431 
432  if ( _input_state[in_index] == full ) {
433  _input_state[in_index] = leaving;
434  } else if ( _input_state[in_index] == filling ) {
435  _input_state[in_index] = cut_through;
436  } else {
437  Error( "Tried to route input through crossbar that was not full or filling!" );
438  }
439  }
440 
441  // Any non-injection channel that is routable is
442  // directed to the multi-queue
443  if ( ( ( _input_state[_cur_channel] == filling ) ||
444  ( _input_state[_cur_channel] == full ) ) &&
445  ( !_IsInjectionChan( _cur_channel ) ) ) {
446  ++_read_stall;
447  } else {
448  // go to next channel for the next cycle
449  _cur_channel = ( _cur_channel + 1 ) % _inputs;
450  _read_stall = 0;
451  }
452  }
453 
454  if ( _read_stall > 0 ) {
455  mq_avail = _FindAvailMultiQueue( );
456 
457  if ( mq_avail != -1 ) {
458  if ( _input_state[_cur_channel] == full ) {
459  _input_state[_cur_channel] = leaving;
460  } else if ( _input_state[_cur_channel] == filling ) {
461  _input_state[_cur_channel] = cut_through;
462  } else {
463  cout << "Input " << _cur_channel << " state = "
464  << _input_state[_cur_channel] << endl;
465  Error( "Tried to route input throught multi-queue that was not full or filling!" );
466  }
467 
468  _input_mq_match[_cur_channel] = mq_avail;
469  _mq_matched[mq_avail] = true;
470 
471  // go to next channel for the next cycle
472  _cur_channel = ( _cur_channel + 1 ) % _inputs;
473  _read_stall = 0;
474  } else {
475  ++_read_stall;
476  //cout << "stalling at input " << _cur_channel << " (count = " << _read_stall << ")" << endl;
477  }
478  }
479 }
