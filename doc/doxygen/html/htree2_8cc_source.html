103 // nand gate sizing calculation
104 void Htree2::input_nand(double s1, double s2, double l_eff)
105 {
106  Wire w1(wt, l_eff);
107  double pton_size = deviceType->n_to_p_eff_curr_drv_ratio;
108  // input capacitance of a repeater = input capacitance of nand.
109  double nsize = s1*(1 + pton_size)/(2 + pton_size);
110  nsize = (nsize < 1) ? 1 : nsize;
111 
112  double tc = 2*tr_R_on(nsize*min_w_nmos, NCH, 1) *
113  (drain_C_(nsize*min_w_nmos, NCH, 1, 1, g_tp.cell_h_def)*2 +
114  2 * gate_C(s2*(min_w_nmos + min_w_pmos), 0));
115  delay+= horowitz (w1.out_rise_time, tc,
116  deviceType->Vth/deviceType->Vdd, deviceType->Vth/deviceType->Vdd, RISE);
117  power.readOp.dynamic += 0.5 *
118  (2*drain_C_(pton_size * nsize*min_w_pmos, PCH, 1, 1, g_tp.cell_h_def)
119  + drain_C_(nsize*min_w_nmos, NCH, 1, 1, g_tp.cell_h_def)
120  + 2*gate_C(s2*(min_w_nmos + min_w_pmos), 0)) *
121  deviceType->Vdd * deviceType->Vdd;
122 
123  power.searchOp.dynamic += 0.5 *
124  (2*drain_C_(pton_size * nsize*min_w_pmos, PCH, 1, 1, g_tp.cell_h_def)
125  + drain_C_(nsize*min_w_nmos, NCH, 1, 1, g_tp.cell_h_def)
126  + 2*gate_C(s2*(min_w_nmos + min_w_pmos), 0)) *
127  deviceType->Vdd * deviceType->Vdd * wire_bw ;
128  power.readOp.leakage += (wire_bw*cmos_Isub_leakage(min_w_nmos*(nsize*2), min_w_pmos * nsize * 2, 2, nand))*deviceType->Vdd;
129  power.readOp.gate_leakage += (wire_bw*cmos_Ig_leakage(min_w_nmos*(nsize*2), min_w_pmos * nsize * 2, 2, nand))*deviceType->Vdd;
130 }

134 // tristate buffer model consisting of not, nand, nor, and driver transistors
135 void Htree2::output_buffer(double s1, double s2, double l_eff)
136 {
137  Wire w1(wt, l_eff);
138  double pton_size = deviceType->n_to_p_eff_curr_drv_ratio;
139  // input capacitance of repeater = input capacitance of nand + nor.
140  double size = s1*(1 + pton_size)/(2 + pton_size + 1 + 2*pton_size);
141  double s_eff = //stage eff of a repeater in a wire
142  (gate_C(s2*(min_w_nmos + min_w_pmos), 0) + w1.wire_cap(l_eff*1e-6,true))/
143  gate_C(s2*(min_w_nmos + min_w_pmos), 0);
144  double tr_size = gate_C(s1*(min_w_nmos + min_w_pmos), 0) * 1/2/(s_eff*gate_C(min_w_pmos, 0));
145  size = (size < 1) ? 1 : size;
146 
147  double res_nor = 2*tr_R_on(size*min_w_pmos, PCH, 1);
148  double res_ptrans = tr_R_on(tr_size*min_w_nmos, NCH, 1);
149  double cap_nand_out = drain_C_(size*min_w_nmos, NCH, 1, 1, g_tp.cell_h_def) +
150  drain_C_(size*min_w_pmos, PCH, 1, 1, g_tp.cell_h_def)*2 +
151  gate_C(tr_size*min_w_pmos, 0);
152  double cap_ptrans_out = 2 *(drain_C_(tr_size*min_w_pmos, PCH, 1, 1, g_tp.cell_h_def) +
153  drain_C_(tr_size*min_w_nmos, NCH, 1, 1, g_tp.cell_h_def)) +
154  gate_C(s1*(min_w_nmos + min_w_pmos), 0);
155 
156  double tc = res_nor * cap_nand_out + (res_nor + res_ptrans) * cap_ptrans_out;
157 
158 
159  delay += horowitz (w1.out_rise_time, tc,
160  deviceType->Vth/deviceType->Vdd, deviceType->Vth/deviceType->Vdd, RISE);
161 
162  //nand
163  power.readOp.dynamic += 0.5 *
164  (2*drain_C_(size*min_w_pmos, PCH, 1, 1, g_tp.cell_h_def) +
165  drain_C_(size*min_w_nmos, NCH, 1, 1, g_tp.cell_h_def) +
166  gate_C(tr_size*(min_w_pmos), 0)) *
167  deviceType->Vdd * deviceType->Vdd;
168 
169  power.searchOp.dynamic += 0.5 *
170  (2*drain_C_(size*min_w_pmos, PCH, 1, 1, g_tp.cell_h_def) +
171  drain_C_(size*min_w_nmos, NCH, 1, 1, g_tp.cell_h_def) +
172  gate_C(tr_size*(min_w_pmos), 0)) *
173  deviceType->Vdd * deviceType->Vdd*init_wire_bw;
174 
175  //not
176  power.readOp.dynamic += 0.5 *
177  (drain_C_(size*min_w_pmos, PCH, 1, 1, g_tp.cell_h_def)
178  +drain_C_(size*min_w_nmos, NCH, 1, 1, g_tp.cell_h_def)
179  +gate_C(size*(min_w_nmos + min_w_pmos), 0)) *
180  deviceType->Vdd * deviceType->Vdd;
181 
182  power.searchOp.dynamic += 0.5 *
183  (drain_C_(size*min_w_pmos, PCH, 1, 1, g_tp.cell_h_def)
184  +drain_C_(size*min_w_nmos, NCH, 1, 1, g_tp.cell_h_def)
185  +gate_C(size*(min_w_nmos + min_w_pmos), 0)) *
186  deviceType->Vdd * deviceType->Vdd*init_wire_bw;
187 
188  //nor
189  power.readOp.dynamic += 0.5 *
190  (drain_C_(size*min_w_pmos, PCH, 1, 1, g_tp.cell_h_def)
191  + 2*drain_C_(size*min_w_nmos, NCH, 1, 1, g_tp.cell_h_def)
192  +gate_C(tr_size*(min_w_nmos + min_w_pmos), 0)) *
193  deviceType->Vdd * deviceType->Vdd;
194 
195  power.searchOp.dynamic += 0.5 *
196  (drain_C_(size*min_w_pmos, PCH, 1, 1, g_tp.cell_h_def)
197  + 2*drain_C_(size*min_w_nmos, NCH, 1, 1, g_tp.cell_h_def)
198  +gate_C(tr_size*(min_w_nmos + min_w_pmos), 0)) *
199  deviceType->Vdd * deviceType->Vdd*init_wire_bw;
200 
201  //output transistor
202  power.readOp.dynamic += 0.5 *
203  ((drain_C_(tr_size*min_w_pmos, PCH, 1, 1, g_tp.cell_h_def)
204  +drain_C_(tr_size*min_w_nmos, NCH, 1, 1, g_tp.cell_h_def))*2
205  + gate_C(s1*(min_w_nmos + min_w_pmos), 0)) *
206  deviceType->Vdd * deviceType->Vdd;
207 
208  power.searchOp.dynamic += 0.5 *
209  ((drain_C_(tr_size*min_w_pmos, PCH, 1, 1, g_tp.cell_h_def)
210  +drain_C_(tr_size*min_w_nmos, NCH, 1, 1, g_tp.cell_h_def))*2
211  + gate_C(s1*(min_w_nmos + min_w_pmos), 0)) *
212  deviceType->Vdd * deviceType->Vdd*init_wire_bw;
213 
214  if(uca_tree) {
215  power.readOp.leakage += cmos_Isub_leakage(min_w_nmos*tr_size*2, min_w_pmos*tr_size*2, 1, inv)*deviceType->Vdd*wire_bw;/*inverter + output tr*/
216  power.readOp.leakage += cmos_Isub_leakage(min_w_nmos*size*3, min_w_pmos*size*3, 2, nand)*deviceType->Vdd*wire_bw;//nand
217  power.readOp.leakage += cmos_Isub_leakage(min_w_nmos*size*3, min_w_pmos*size*3, 2, nor)*deviceType->Vdd*wire_bw;//nor
218 
219  power.readOp.gate_leakage += cmos_Ig_leakage(min_w_nmos*tr_size*2, min_w_pmos*tr_size*2, 1, inv)*deviceType->Vdd*wire_bw;/*inverter + output tr*/
220  power.readOp.gate_leakage += cmos_Ig_leakage(min_w_nmos*size*3, min_w_pmos*size*3, 2, nand)*deviceType->Vdd*wire_bw;//nand
221  power.readOp.gate_leakage += cmos_Ig_leakage(min_w_nmos*size*3, min_w_pmos*size*3, 2, nor)*deviceType->Vdd*wire_bw;//nor
222  //power.readOp.gate_leakage *=;
223  }
224  else {
225  power.readOp.leakage += cmos_Isub_leakage(min_w_nmos*tr_size*2, min_w_pmos*tr_size*2, 1, inv)*deviceType->Vdd*wire_bw;/*inverter + output tr*/
226  power.readOp.leakage += cmos_Isub_leakage(min_w_nmos*size*3, min_w_pmos*size*3, 2, nand)*deviceType->Vdd*wire_bw;//nand
227  power.readOp.leakage += cmos_Isub_leakage(min_w_nmos*size*3, min_w_pmos*size*3, 2, nor)*deviceType->Vdd*wire_bw;//nor
228 
229  power.readOp.gate_leakage += cmos_Ig_leakage(min_w_nmos*tr_size*2, min_w_pmos*tr_size*2, 1, inv)*deviceType->Vdd*wire_bw;/*inverter + output tr*/
230  power.readOp.gate_leakage += cmos_Ig_leakage(min_w_nmos*size*3, min_w_pmos*size*3, 2, nand)*deviceType->Vdd*wire_bw;//nand
231  power.readOp.gate_leakage += cmos_Ig_leakage(min_w_nmos*size*3, min_w_pmos*size*3, 2, nor)*deviceType->Vdd*wire_bw;//nor
232  //power.readOp.gate_leakage *=deviceType->Vdd*wire_bw;
233  }

238 /* calculates the input h-tree delay/power
239  * A nand gate is used at each node to
240  * limit the signal
241  * The area of an unbalanced htree (rows != columns)
242  * depends on how data is traversed.
243  * In the following function, if ( no. of rows < no. of columns),
244  * then data first traverse in excess hor. links until vertical
245  * and horizontal nodes are same.
246  * If no. of rows is bigger, then data traverse in
247  * a hor. link followed by a ver. link in a repeated
248  * fashion (similar to a balanced tree) until there are no
249  * hor. links left. After this it goes through the remaining vertical
250  * links.
251  */
252  void
253 Htree2::in_htree()
254 {
255  //temp var
256  double s1 = 0, s2 = 0, s3 = 0;
257  double l_eff = 0;
258  Wire *wtemp1 = 0, *wtemp2 = 0, *wtemp3 = 0;
259  double len = 0, ht = 0;
260  int option = 0;
261 
262  int h = (int) _log2(ndwl/2); // horizontal nodes
263  int v = (int) _log2(ndbl/2); // vertical nodes
264  double len_temp;
265  double ht_temp;
266  if (uca_tree)
267  {//Sheng: this computation do not consider the wires that route from edge to middle.
268  ht_temp = (mat_height*ndbl/2 +/* since uca_tree models interbank tree, mat_height => bank height */
269  ((add_bits + data_in_bits + data_out_bits + (search_data_in_bits + search_data_out_bits)) * g_tp.wire_outside_mat.pitch *
270  2 * (1-pow(0.5,h))))/2;
271  len_temp = (mat_width*ndwl/2 +
272  ((add_bits + data_in_bits + data_out_bits + (search_data_in_bits + search_data_out_bits)) * g_tp.wire_outside_mat.pitch *
273  2 * (1-pow(0.5,v))))/2;
274  }
275  else
276  {
277  if (ndwl == ndbl) {
278  ht_temp = ((mat_height*ndbl/2) +
279  ((add_bits + (search_data_in_bits + search_data_out_bits))* (ndbl/2-1) * g_tp.wire_outside_mat.pitch) +
280  ((data_in_bits + data_out_bits) * g_tp.wire_outside_mat.pitch * h)
281  )/2;
282  len_temp = (mat_width*ndwl/2 +
283  ((add_bits + (search_data_in_bits + search_data_out_bits)) * (ndwl/2-1) * g_tp.wire_outside_mat.pitch) +
284  ((data_in_bits + data_out_bits) * g_tp.wire_outside_mat.pitch * v))/2;
285  }
286  else if (ndwl > ndbl) {
287  double excess_part = (_log2(ndwl/2) - _log2(ndbl/2));
288  ht_temp = ((mat_height*ndbl/2) +
289  ((add_bits + + (search_data_in_bits + search_data_out_bits)) * ((ndbl/2-1) + excess_part) * g_tp.wire_outside_mat.pitch) +
290  (data_in_bits + data_out_bits) * g_tp.wire_outside_mat.pitch *
291  (2*(1 - pow(0.5, h-v)) + pow(0.5, v-h) * v))/2;
292  len_temp = (mat_width*ndwl/2 +
293  ((add_bits + (search_data_in_bits + search_data_out_bits))* (ndwl/2-1) * g_tp.wire_outside_mat.pitch) +
294  ((data_in_bits + data_out_bits) * g_tp.wire_outside_mat.pitch * v))/2;
295  }
296  else {
297  double excess_part = (_log2(ndbl/2) - _log2(ndwl/2));
298  ht_temp = ((mat_height*ndbl/2) +
299  ((add_bits + (search_data_in_bits + search_data_out_bits))* ((ndwl/2-1) + excess_part) * g_tp.wire_outside_mat.pitch) +
300  ((data_in_bits + data_out_bits) * g_tp.wire_outside_mat.pitch * h)
301  )/2;
302  len_temp = (mat_width*ndwl/2 +
303  ((add_bits + (search_data_in_bits + search_data_out_bits)) * ((ndwl/2-1) + excess_part) * g_tp.wire_outside_mat.pitch) +
304  (data_in_bits + data_out_bits) * g_tp.wire_outside_mat.pitch * (h + 2*(1-pow(0.5, v-h))))/2;
305  }
306  }
307 
308  area.h = ht_temp * 2;
309  area.w = len_temp * 2;
310  delay = 0;
311  power.readOp.dynamic = 0;
312  power.readOp.leakage = 0;
313  power.searchOp.dynamic =0;
314  len = len_temp;
315  ht = ht_temp/2;
316 
317  while (v > 0 || h > 0)
318  {
319  if (wtemp1) delete wtemp1;
320  if (wtemp2) delete wtemp2;
321  if (wtemp3) delete wtemp3;
322 
323  if (h > v)
324  {
325  //the iteration considers only one horizontal link
326  wtemp1 = new Wire(wt, len); // hor
327  wtemp2 = new Wire(wt, len/2); // ver
328  len_temp = len;
329  len /= 2;
330  wtemp3 = 0;
331  h--;
332  option = 0;
333  }
334  else if (v>0 && h>0)
335  {
336  //considers one horizontal link and one vertical link
337  wtemp1 = new Wire(wt, len); // hor
338  wtemp2 = new Wire(wt, ht); // ver
339  wtemp3 = new Wire(wt, len/2); // next hor
340  len_temp = len;
341  ht_temp = ht;
342  len /= 2;
343  ht /= 2;
344  v--;
345  h--;
346  option = 1;
347  }
348  else
349  {
350  // considers only one vertical link
351  assert(h == 0);
352  wtemp1 = new Wire(wt, ht); // ver
353  wtemp2 = new Wire(wt, ht/2); // hor
354  ht_temp = ht;
355  ht /= 2;
356  wtemp3 = 0;
357  v--;
358  option = 2;
359  }
360 
361  delay += wtemp1->delay;
362  power.readOp.dynamic += wtemp1->power.readOp.dynamic;
363  power.searchOp.dynamic += wtemp1->power.readOp.dynamic*wire_bw;
364  power.readOp.leakage += wtemp1->power.readOp.leakage*wire_bw;
365  power.readOp.gate_leakage += wtemp1->power.readOp.gate_leakage*wire_bw;
366  if ((uca_tree == false && option == 2) || search_tree==true)
367  {
368  wire_bw*=2; // wire bandwidth doubles only for vertical branches
369  }
370 
371  if (uca_tree == false)
372  {
373  if (len_temp > wtemp1->repeater_spacing)
374  {
375  s1 = wtemp1->repeater_size;
376  l_eff = wtemp1->repeater_spacing;
377  }
378  else
379  {
380  s1 = (len_temp/wtemp1->repeater_spacing) * wtemp1->repeater_size;
381  l_eff = len_temp;
382  }
383 
384  if (ht_temp > wtemp2->repeater_spacing)
385  {
386  s2 = wtemp2->repeater_size;
387  }
388  else
389  {
390  s2 = (len_temp/wtemp2->repeater_spacing) * wtemp2->repeater_size;
391  }
392  // first level
393  input_nand(s1, s2, l_eff);
394  }
395 
396 
397  if (option != 1)
398  {
399  continue;
400  }
401 
402  // second level
403  delay += wtemp2->delay;
404  power.readOp.dynamic += wtemp2->power.readOp.dynamic;
405  power.searchOp.dynamic += wtemp2->power.readOp.dynamic*wire_bw;
406  power.readOp.leakage += wtemp2->power.readOp.leakage*wire_bw;
407  power.readOp.gate_leakage += wtemp2->power.readOp.gate_leakage*wire_bw;
408 
409  if (uca_tree)
410  {
411  power.readOp.leakage += (wtemp2->power.readOp.leakage*wire_bw);
412  power.readOp.gate_leakage += wtemp2->power.readOp.gate_leakage*wire_bw;
413  }
414  else
415  {
416  power.readOp.leakage += (wtemp2->power.readOp.leakage*wire_bw);
417  power.readOp.gate_leakage += wtemp2->power.readOp.gate_leakage*wire_bw;
418  wire_bw*=2;
419 
420  if (ht_temp > wtemp3->repeater_spacing)
421  {
422  s3 = wtemp3->repeater_size;
423  l_eff = wtemp3->repeater_spacing;
424  }
425  else
426  {
427  s3 = (len_temp/wtemp3->repeater_spacing) * wtemp3->repeater_size;
428  l_eff = ht_temp;
429  }
430 
431  input_nand(s2, s3, l_eff);
432  }
433  }
434 
435  if (wtemp1) delete wtemp1;
436  if (wtemp2) delete wtemp2;
437  if (wtemp3) delete wtemp3;
438 }

454 void Htree2::out_htree()
455 {
456  //temp var
457  double s1 = 0, s2 = 0, s3 = 0;
458  double l_eff = 0;
459  Wire *wtemp1 = 0, *wtemp2 = 0, *wtemp3 = 0;
460  double len = 0, ht = 0;
461  int option = 0;
462
