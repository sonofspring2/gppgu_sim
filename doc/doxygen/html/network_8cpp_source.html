80 Network * Network::New(const Configuration & config, const string & name)
81 {
82  const string topo = config.GetStr( "topology" );
83  Network * n = NULL;
84  if ( topo == "torus" ) {
85  KNCube::RegisterRoutingFunctions() ;
86  n = new KNCube( config, name, false );
87  } else if ( topo == "mesh" ) {
88  KNCube::RegisterRoutingFunctions() ;
89  n = new KNCube( config, name, true );
90  } else if ( topo == "cmesh" ) {
91  CMesh::RegisterRoutingFunctions() ;
92  n = new CMesh( config, name );
93  } else if ( topo == "fly" ) {
94  KNFly::RegisterRoutingFunctions() ;
95  n = new KNFly( config, name );
96  } else if ( topo == "qtree" ) {
97  QTree::RegisterRoutingFunctions() ;
98  n = new QTree( config, name );
99  } else if ( topo == "tree4" ) {
100  Tree4::RegisterRoutingFunctions() ;
101  n = new Tree4( config, name );
102  } else if ( topo == "fattree" ) {
103  FatTree::RegisterRoutingFunctions() ;
104  n = new FatTree( config, name );
105  } else if ( topo == "flatfly" ) {
106  FlatFlyOnChip::RegisterRoutingFunctions() ;
107  n = new FlatFlyOnChip( config, name );
108  } else if ( topo == "anynet"){
109  AnyNet::RegisterRoutingFunctions() ;
110  n = new AnyNet(config, name);
111  } else if ( topo == "dragonflynew"){
112  DragonFlyNew::RegisterRoutingFunctions() ;
113  n = new DragonFlyNew(config, name);
114  } else {
115  cerr << "Unknown topology: " << topo << endl;
116  }
117 
118  /*legacy code that insert random faults in the networks
119  *not sure how to use this
120  */
121  if ( n && ( config.GetInt( "link_failures" ) > 0 ) ) {
122  n->InsertRandomFaults( config );
123  }
124  return n;
125 }

127 void Network::_Alloc( )
128 {
129  assert( ( _size != -1 ) &&
130  ( _nodes != -1 ) &&
131  ( _channels != -1 ) );
132 
133  _routers.resize(_size);
134  gNodes = _nodes;
135 
136  /*booksim used arrays of flits as the channels which makes have capacity of
137  *one. To simulate channel latency, flitchannel class has been added
138  *which are fifos with depth = channel latency and each cycle the channel
139  *shifts by one
140  *credit channels are the necessary counter part
141  */
142  _inject.resize(_nodes);
143  _inject_cred.resize(_nodes);
144  for ( int s = 0; s < _nodes; ++s ) {
145  ostringstream name;
146  name << Name() << "_fchan_ingress" << s;
147  _inject[s] = new FlitChannel(this, name.str(), _classes);
148  _inject[s]->SetSource(NULL, s);
149  _timed_modules.push_back(_inject[s]);
150  name.str("");
151  name << Name() << "_cchan_ingress" << s;
152  _inject_cred[s] = new CreditChannel(this, name.str());
153  _timed_modules.push_back(_inject_cred[s]);
154  }
155  _eject.resize(_nodes);
156  _eject_cred.resize(_nodes);
157  for ( int d = 0; d < _nodes; ++d ) {
158  ostringstream name;
159  name << Name() << "_fchan_egress" << d;
160  _eject[d] = new FlitChannel(this, name.str(), _classes);
161  _eject[d]->SetSink(NULL, d);
162  _timed_modules.push_back(_eject[d]);
163  name.str("");
164  name << Name() << "_cchan_egress" << d;
165  _eject_cred[d] = new CreditChannel(this, name.str());
166  _timed_modules.push_back(_eject_cred[d]);
167  }
168  _chan.resize(_channels);
169  _chan_cred.resize(_channels);
170  for ( int c = 0; c < _channels; ++c ) {
171  ostringstream name;
172  name << Name() << "_fchan_" << c;
173  _chan[c] = new FlitChannel(this, name.str(), _classes);
174  _timed_modules.push_back(_chan[c]);
175  name.str("");
176  name << Name() << "_cchan_" << c;
177  _chan_cred[c] = new CreditChannel(this, name.str());
178  _timed_modules.push_back(_chan_cred[c]);
179  }
180 }
