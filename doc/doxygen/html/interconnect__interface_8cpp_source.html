193 void* InterconnectInterface::Pop(unsigned deviceID)
194 {
195  int icntID = _node_map[deviceID];
196 #if DEBUG
197  cout<<"Call interconnect POP " << output<<endl;
198 #endif
199 
200  void* data = NULL;
201 
202  // 0-_n_shader-1 indicates reply(network 1), otherwise request(network 0)
203  int subnet = 0;
204  if (deviceID < _n_shader)
205  subnet = 1;
206 
207  int turn = _round_robin_turn[subnet][icntID];
208  for (int vc=0;(vc<_vcs) && (data==NULL);vc++) {
209  if (_boundary_buffer[subnet][icntID][turn].HasPacket()) {
210  data = _boundary_buffer[subnet][icntID][turn].PopPacket();
211  }
212  turn++;
213  if (turn == _vcs) turn = 0;
214  }
215  if (data) {
216  _round_robin_turn[subnet][icntID] = turn;
217  }
218 
219  return data;

387 void InterconnectInterface::_CreateNodeMap(unsigned n_shader, unsigned n_mem, unsigned n_node, int use_map)
388 {
389  if (use_map) {
390  // The (<SM, Memory>, Memory Location Vector) map
391  map<pair<unsigned,unsigned>, vector<unsigned> > preset_memory_map;
392 
393  // preset memory and shader map, optimized for mesh
394  // good for 8 SMs and 8 memory ports, the map is as follows:
395  // +--+--+--+--+
396  // |C0|M0|C1|M1|
397  // +--+--+--+--+
398  // |M2|C2|M3|C3|
399  // +--+--+--+--+
400  // |C4|M4|C5|M5|
401  // +--+--+--+--+
402  // |M6|C6|M7|C7|
403  // +--+--+--+--+
404  {
405  unsigned memory_node[] = {1, 3, 4, 6, 9, 11, 12, 14};
406  preset_memory_map[make_pair(8,8)] = vector<unsigned>(memory_node, memory_node+8);
407  }
408 
409  // good for 28 SMs and 8 memory ports
410  {
411  unsigned memory_node[] = {3, 7, 10, 12, 23, 25, 28, 32};
412  preset_memory_map[make_pair(28,8)] = vector<unsigned>(memory_node, memory_node+8);
413  }
414 
415  // good for 56 SMs and 8 memory cores
416  {
417  unsigned memory_node[] = {3, 15, 17, 29, 36, 47, 49, 61};
418  preset_memory_map[make_pair(56,8)] = vector<unsigned>(memory_node, memory_node+sizeof(memory_node)/sizeof(unsigned));
419  }
420 
421  // good for 110 SMs and 11 memory cores
422  {
423  unsigned memory_node[] = {12, 20, 25, 28, 57, 60, 63, 92, 95,100,108};
424  preset_memory_map[make_pair(110, 11)] = vector<unsigned>(memory_node, memory_node+sizeof(memory_node)/sizeof(unsigned));
425  }
426  const vector<int> config_memory_node(_icnt_config->GetIntArray("memory_node_map"));
427  if (!config_memory_node.empty()) {
428  if (config_memory_node.size() != _n_mem) {
429  cerr << "Number of memory nodes in memory_node_map should equal to memory ports" << endl;
430  assert( config_memory_node.size() == _n_mem);
431  }
432  vector<unsigned> t_memory_node(config_memory_node.size());
433  copy(config_memory_node.begin(), config_memory_node.end(), t_memory_node.begin());
434  preset_memory_map[make_pair(_n_shader, _n_mem)] = t_memory_node;
435  }
436 
437  const vector<unsigned> &memory_node = preset_memory_map[make_pair(_n_shader, _n_mem)];
438  if (memory_node.empty()) {
439  cerr<<"ERROR!!! NO MAPPING IMPLEMENTED YET FOR THIS CONFIG"<<endl;
440  assert(0);
441  }
442 
443  // create node map
444  unsigned next_node = 0;
445  unsigned memory_node_index = 0;
446  for (unsigned i = 0; i < n_shader; ++i) {
447  while (next_node == memory_node[memory_node_index]) {
448  next_node += 1;
449  memory_node_index += 1;
450  }
451  _node_map[i] = next_node;
452  next_node += 1;
453  }
454  for (unsigned i = n_shader; i < n_shader+n_mem; ++i) {
455  _node_map[i] = memory_node[i-n_shader];
456  }
457  } else { //not use preset map
458  for (unsigned i=0;i<n_node;i++) {
459  _node_map[i]=i;
460  }
461  }
462 
463  for (unsigned i = 0; i < n_node ; i++) {
464  for (unsigned j = 0; j< n_node ; j++) {
465  if ( _node_map[j] == i ) {
466  _reverse_node_map[i]=j;
467  break;
468  }
469  }
470  }
471 
472  //FIXME: should compatible with non-square number
473  _DisplayMap((int) sqrt(n_node), n_node);
474 
475 }
