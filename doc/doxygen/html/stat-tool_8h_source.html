39 // logger snapshot trigger:
40 // - automate the snap_shot part of loggers to avoid modifying simulation loop everytime
41 // a new time-dependent stat is added
43 
44 class snap_shot_trigger {
45 public:
46  snap_shot_trigger(unsigned long long interval) : m_snap_shot_interval(interval) {}
47  virtual ~snap_shot_trigger() {}
48 
49  void try_snap_shot(unsigned long long current_cycle) {
50  if ((current_cycle % m_snap_shot_interval == 0) && current_cycle != 0) {
51  snap_shot(current_cycle);
52  }
53  }
54 
55  virtual void snap_shot(unsigned long long current_cycle) = 0;
56 
57  const unsigned long long & get_interval() const { return m_snap_shot_interval;}
58 
59 protected:
60  unsigned long long m_snap_shot_interval;
61 };

137  m_id(s_ids++) {}
138 
139  insn_warp_occ_logger(const insn_warp_occ_logger& other)
140  : m_simd_width(other.m_simd_width),
141  m_insn_warp_occ(other.m_insn_warp_occ.size(), linear_histogram(1, "", m_simd_width)),
142  m_id(s_ids++) {}
143 
144  ~insn_warp_occ_logger() {}
145 
146  insn_warp_occ_logger& operator=(const insn_warp_occ_logger& p) {
147  printf("insn_warp_occ_logger Operator= called: %02d \n", m_id);
148  assert(0);
149  return *this;
150  }
151 
152  void set_id(int id) { m_id = id; }
153 
154  void log(address_type pc, int warp_occ) {
155  if( pc >= m_insn_warp_occ.size() )
156  m_insn_warp_occ.resize(2*pc, linear_histogram(1, "", m_simd_width));
157  m_insn_warp_occ[pc].add2bin(warp_occ - 1);
158  }
159 
160  void print(FILE *fout) const
161  {
162  for (unsigned i = 0; i < m_insn_warp_occ.size(); i++) {
163  fprintf(fout, "InsnWarpOcc%02d-%d", m_id, i);
164  m_insn_warp_occ[i].fprint(fout);
165  fprintf(fout, "\n");
166  }
167  }
168 
169 private:
170 
171  int m_simd_width;
172  std::vector<linear_histogram> m_insn_warp_occ;
173  int m_id;

182 class linear_histogram_snapshot {
183 public:
184  linear_histogram_snapshot(int n_bins, unsigned long long cycle)
185  : m_cycle(cycle),
186  m_linear_histogram(n_bins,0)
187  { }
188 
189  linear_histogram_snapshot(const linear_histogram_snapshot& other)
190  : m_cycle(other.m_cycle),
191  m_linear_histogram(other.m_linear_histogram)
192  { }
193 
194  ~linear_histogram_snapshot() { }
195 
196  void addsample(int pos) {
197  assert((size_t)pos < m_linear_histogram.size());
198  m_linear_histogram[pos] += 1;
199  }
200 
201  void subsample(int pos) {
202  assert((size_t)pos < m_linear_histogram.size());
203  m_linear_histogram[pos] -= 1;
204  }
205 
206  void reset(unsigned long long cycle) {
207  m_cycle = cycle;
208  m_linear_histogram.assign(m_linear_histogram.size(), 0);
209  }
210 
211  void set_cycle(unsigned long long cycle) { m_cycle = cycle; }
212 
213  void print(FILE *fout) const {
214  fprintf(fout, "%d = ", (int)m_cycle);
215  for (unsigned int i = 0; i < m_linear_histogram.size(); i++) {
216  fprintf(fout, "%d ", m_linear_histogram[i]);
217  }
218  }
219 
220  void print_visualizer(FILE *fout) const {
221  for (unsigned int i = 0; i < m_linear_histogram.size(); i++) {
222  fprintf(fout, "%d ", m_linear_histogram[i]);
223  }
224  }
225 
226  void print_visualizer(gzFile fout) const {
227  for (unsigned int i = 0; i < m_linear_histogram.size(); i++) {
228  gzprintf(fout, "%d ", m_linear_histogram[i]);
229  }
230  }
231 
232 private:
233  unsigned long long m_cycle;
234  std::vector<int> m_linear_histogram;
235 };
