75  htree_in_add = new Htree2(g_ip->wt, bank.area.w, bank.area.h,

80  num_addr_b_bank, num_di_b_bank, 0, num_do_b_bank, 0, num_banks_ver_dir*2, num_banks_hor_dir*2, Data_out_htree, true);

121 double UCA::compute_delays(double inrisetime)
122 {
123  double outrisetime = bank.compute_delays(inrisetime);
124 
125  double delay_array_to_mat = htree_in_add->delay + bank.htree_in_add->delay;
126  double max_delay_before_row_decoder = delay_array_to_mat + bank.mat.r_predec->delay;
127  delay_array_to_sa_mux_lev_1_decoder = delay_array_to_mat +
128  bank.mat.sa_mux_lev_1_predec->delay +
129  bank.mat.sa_mux_lev_1_dec->delay;
130  delay_array_to_sa_mux_lev_2_decoder = delay_array_to_mat +
131  bank.mat.sa_mux_lev_2_predec->delay +
132  bank.mat.sa_mux_lev_2_dec->delay;
133  double delay_inside_mat = bank.mat.row_dec->delay + bank.mat.delay_bitline + bank.mat.delay_sa;
134 
135  delay_before_subarray_output_driver =
136  MAX(MAX(max_delay_before_row_decoder + delay_inside_mat, // row_path
137  delay_array_to_mat + bank.mat.b_mux_predec->delay + bank.mat.bit_mux_dec->delay + bank.mat.delay_sa), // col_path
138  MAX(delay_array_to_sa_mux_lev_1_decoder, // sa_mux_lev_1_path
139  delay_array_to_sa_mux_lev_2_decoder)); // sa_mux_lev_2_path
140  delay_from_subarray_out_drv_to_out = bank.mat.delay_subarray_out_drv_htree +
141  bank.htree_out_data->delay + htree_out_data->delay;
142  access_time = bank.mat.delay_comparator;
143 
144  double ram_delay_inside_mat;
145  if (dp.fully_assoc)
146  {
147  //delay of FA contains both CAM tag and RAM data
148  { //delay of CAM
149  ram_delay_inside_mat = bank.mat.delay_bitline + bank.mat.delay_matchchline;
150  access_time = htree_in_add->delay + bank.htree_in_add->delay;
151  //delay of fully-associative data array
152  access_time += ram_delay_inside_mat + delay_from_subarray_out_drv_to_out;
153  }
154  }
155  else
156  {
157  access_time = delay_before_subarray_output_driver + delay_from_subarray_out_drv_to_out; //data_acc_path
158  }
159 
160  if (dp.is_main_mem)
161  {
162  double t_rcd = max_delay_before_row_decoder + delay_inside_mat;
163  double cas_latency = MAX(delay_array_to_sa_mux_lev_1_decoder, delay_array_to_sa_mux_lev_2_decoder) +
164  delay_from_subarray_out_drv_to_out;
165  access_time = t_rcd + cas_latency;
166  }
167 
168  double temp;
169 
170  if (!dp.fully_assoc)
171  {
172  temp = delay_inside_mat + bank.mat.delay_wl_reset + bank.mat.delay_bl_restore;//TODO: Sheng: revisit
173  if (dp.is_dram)
174  {
175  temp += bank.mat.delay_writeback; // temp stores random cycle time
176  }
177 
178 
179  temp = MAX(temp, bank.mat.r_predec->delay);
180  temp = MAX(temp, bank.mat.b_mux_predec->delay);
181  temp = MAX(temp, bank.mat.sa_mux_lev_1_predec->delay);
182  temp = MAX(temp, bank.mat.sa_mux_lev_2_predec->delay);
183  }
184  else
185  {
186  ram_delay_inside_mat = bank.mat.delay_bitline + bank.mat.delay_matchchline;
187  temp = ram_delay_inside_mat + bank.mat.delay_cam_sl_restore + bank.mat.delay_cam_ml_reset + bank.mat.delay_bl_restore
188  + bank.mat.delay_hit_miss_reset + bank.mat.delay_wl_reset;
189 
190  temp = MAX(temp, bank.mat.b_mux_predec->delay);//TODO: Sheng revisit whether distinguish cam and ram bitline etc.
191  temp = MAX(temp, bank.mat.sa_mux_lev_1_predec->delay);
192  temp = MAX(temp, bank.mat.sa_mux_lev_2_predec->delay);
193  }
194 
195  // The following is true only if the input parameter "repeaters_in_htree" is set to false --Nav
196  if (g_ip->rpters_in_htree == false)
197  {
198  temp = MAX(temp, bank.htree_in_add->max_unpipelined_link_delay);
199  }
200  cycle_time = temp;
201 
202  double delay_req_network = max_delay_before_row_decoder;
203  double delay_rep_network = delay_from_subarray_out_drv_to_out;
204  multisubbank_interleave_cycle_time = MAX(delay_req_network, delay_rep_network);
205 
206  if (dp.is_main_mem)
207  {
208  multisubbank_interleave_cycle_time = htree_in_add->delay;
209  precharge_delay = htree_in_add->delay +
210  bank.htree_in_add->delay + bank.mat.delay_writeback +
211  bank.mat.delay_wl_reset + bank.mat.delay_bl_restore;
212  cycle_time = access_time + precharge_delay;
213  }
214  else
215  {
216  precharge_delay = 0;
217  }
218  return outrisetime;
219 }

223 // note: currently, power numbers are for a bank of an array
224 void UCA::compute_power_energy()
225 {
226  bank.compute_power_energy();
227  power = bank.power;
228 
229  power_routing_to_bank.readOp.dynamic = htree_in_add->power.readOp.dynamic + htree_out_data->power.readOp.dynamic;
230  power_routing_to_bank.writeOp.dynamic = htree_in_add->power.readOp.dynamic + htree_in_data->power.readOp.dynamic;
231  if (dp.fully_assoc || dp.pure_cam)
232  power_routing_to_bank.searchOp.dynamic= htree_in_search->power.searchOp.dynamic + htree_out_search->power.searchOp.dynamic;
233 
234  power_routing_to_bank.readOp.leakage += htree_in_add->power.readOp.leakage +
235  htree_in_data->power.readOp.leakage +
236  htree_out_data->power.readOp.leakage;
237 
238  power_routing_to_bank.readOp.gate_leakage += htree_in_add->power.readOp.gate_leakage +
239  htree_in_data->power.readOp.gate_leakage +
240  htree_out_data->power.readOp.gate_leakage;
241  if (dp.fully_assoc || dp.pure_cam)
242  {
243  power_routing_to_bank.readOp.leakage += htree_in_search->power.readOp.leakage + htree_out_search->power.readOp.leakage;
244  power_routing_to_bank.readOp.gate_leakage += htree_in_search->power.readOp.gate_leakage + htree_out_search->power.readOp.gate_leakage;
245  }
246 
247  power.searchOp.dynamic += power_routing_to_bank.searchOp.dynamic;
248  power.readOp.dynamic += power_routing_to_bank.readOp.dynamic;
249  power.readOp.leakage += power_routing_to_bank.readOp.leakage;
250  power.readOp.gate_leakage += power_routing_to_bank.readOp.gate_leakage;
251 
252  // calculate total write energy per access
253  power.writeOp.dynamic = power.readOp.dynamic
254  - bank.mat.power_bitline.readOp.dynamic * dp.num_act_mats_hor_dir
255  + bank.mat.power_bitline.writeOp.dynamic * dp.num_act_mats_hor_dir
256  - power_routing_to_bank.readOp.dynamic
257  + power_routing_to_bank.writeOp.dynamic
258  + bank.htree_in_data->power.readOp.dynamic
259  - bank.htree_out_data->power.readOp.dynamic;
260 
261  if (dp.is_dram == false)
262  {
263  power.writeOp.dynamic -= bank.mat.power_sa.readOp.dynamic * dp.num_act_mats_hor_dir;
264  }
265 
266  dyn_read_energy_from_closed_page = power.readOp.dynamic;
267  dyn_read_energy_from_open_page = power.readOp.dynamic -
268  (bank.mat.r_predec->power.readOp.dynamic +
269  bank.mat.power_row_decoders.readOp.dynamic +
270  bank.mat.power_bl_precharge_eq_drv.readOp.dynamic +
271  bank.mat.power_sa.readOp.dynamic +
272  bank.mat.power_bitline.readOp.dynamic) * dp.num_act_mats_hor_dir;
273 
274  dyn_read_energy_remaining_words_in_burst =
275  (MAX((g_ip->burst_len / g_ip->int_prefetch_w), 1) - 1) *
276  ((bank.mat.sa_mux_lev_1_predec->power.readOp.dynamic +
277  bank.mat.sa_mux_lev_2_predec->power.readOp.dynamic +
278  bank.mat.power_sa_mux_lev_1_decoders.readOp.dynamic +
279  bank.mat.power_sa_mux_lev_2_decoders.readOp.dynamic +
280  bank.mat.power_subarray_out_drv.readOp.dynamic) * dp.num_act_mats_hor_dir +
281  bank.htree_out_data->power.readOp.dynamic +
282  power_routing_to_bank.readOp.dynamic);
283  dyn_read_energy_from_closed_page += dyn_read_energy_remaining_words_in_burst;
284  dyn_read_energy_from_open_page += dyn_read_energy_remaining_words_in_burst;
285 
286  activate_energy = htree_in_add->power.readOp.dynamic +
287  bank.htree_in_add->power_bit.readOp.dynamic * bank.num_addr_b_routed_to_mat_for_act +
288  (bank.mat.r_predec->power.readOp.dynamic +
289  bank.mat.power_row_decoders.readOp.dynamic +
290  bank.mat.power_sa.readOp.dynamic) * dp.num_act_mats_hor_dir;
291  read_energy = (htree_in_add->power.readOp.dynamic +
292  bank.htree_in_add->power_bit.readOp.dynamic * bank.num_addr_b_routed_to_mat_for_rd_or_wr +
293  (bank.mat.sa_mux_lev_1_predec->power.readOp.dynamic +
294  bank.mat.sa_mux_lev_2_predec->power.readOp.dynamic +
295  bank.mat.power_sa_mux_lev_1_decoders.readOp.dynamic +
296  bank.mat.power_sa_mux_lev_2_decoders.readOp.dynamic +
297  bank.mat.power_subarray_out_drv.readOp.dynamic) * dp.num_act_mats_hor_dir +
298  bank.htree_out_data->power.readOp.dynamic +
299  htree_in_data->power.readOp.dynamic) * g_ip->burst_len;
300  write_energy = (htree_in_add->power.readOp.dynamic +
301  bank.htree_in_add->power_bit.readOp.dynamic * bank.num_addr_b_routed_to_mat_for_rd_or_wr +
302  htree_in_data->power.readOp.dynamic +
303  bank.htree_in_data->power.readOp.dynamic +
304  (bank.mat.sa_mux_lev_1_predec->power.readOp.dynamic +
305  bank.mat.sa_mux_lev_2_predec->power.readOp.dynamic +
306  bank.mat.power_sa_mux_lev_1_decoders.readOp.dynamic +
307  bank.mat.power_sa_mux_lev_2_decoders.readOp.dynamic) * dp.num_act_mats_hor_dir) * g_ip->burst_len;
308  precharge_energy = (bank.mat.power_bitline.readOp.dynamic +
309  bank.mat.power_bl_precharge_eq_drv.readOp.dynamic) * dp.num_act_mats_hor_dir;
310 
311  leak_power_subbank_closed_page =
312  (bank.mat.r_predec->power.readOp.leakage +
313  bank.mat.b_mux_predec->power.readOp.leakage +
314  bank.mat.sa_mux_lev_1_predec->power.readOp.leakage +
315  bank.mat.sa_mux_lev_2_predec->power.readOp.leakage +
316  bank.mat.power_row_decoders.readOp.leakage +
317  bank.mat.power_bit_mux_decoders.readOp.leakage +
318  bank.mat.power_sa_mux_lev_1_decoders.readOp.leakage +
319  bank.mat.power_sa_mux_lev_2_decoders.readOp.leakage +
320  bank.mat.leak_power_sense_amps_closed_page_state) * dp.num_act_mats_hor_dir;
321 
322  leak_power_subbank_closed_page +=
323  (bank.mat.r_predec->power.readOp.gate_leakage +
324  bank.mat.b_mux_predec->power.readOp.gate_leakage +
325  bank.mat.sa_mux_lev_1_predec->power.readOp.gate_leakage +
326  bank.mat.sa_mux_lev_2_predec->power.readOp.gate_leakage +
327  bank.mat.power_row_decoders.readOp.gate_leakage +
328  bank.mat.power_bit_mux_decoders.readOp.gate_leakage +
329  bank.mat.power_sa_mux_lev_1_decoders.readOp.gate_leakage +
330  bank.mat.power_sa_mux_lev_2_decoders.readOp.gate_leakage) * dp.num_act_mats_hor_dir; //+
331  //bank.mat.leak_power_sense_amps_closed_page_state) * dp.num_act_mats_hor_dir;
332 
333  leak_power_subbank_open_page =
334  (bank.mat.r_predec->power.readOp.leakage +
335  bank.mat.b_mux_predec->power.readOp.leakage +
336  bank.mat.sa_mux_lev_1_predec->power.readOp.leakage +
337  bank.mat.sa_mux_lev_2_predec->power.readOp.leakage +
338  bank.mat.power_row_decoders.readOp.leakage +
339  bank.mat.power_bit_mux_decoders.readOp.leakage +
340  bank.mat.power_sa_mux_lev_1_decoders.readOp.leakage +
341  bank.mat.power_sa_mux_lev_2_decoders.readOp.leakage +
342  bank.mat.leak_power_sense_amps_open_page_state) * dp.num_act_mats_hor_dir;
343 
344  leak_power_subbank_open_page +=
345  (bank.mat.r_predec->power.readOp.gate_leakage +
346  bank.mat.b_mux_predec->power.readOp.gate_leakage +
347  bank.mat.sa_mux_lev_1_predec->power.readOp.gate_leakage +
348  bank.mat.sa_mux_lev_2_predec->power.readOp.gate_leakage +
349  bank.mat.power_row_decoders.readOp.gate_leakage +
350  bank.mat.power_bit_mux_decoders.readOp.gate_leakage +
351  bank.mat.power_sa_mux_lev_1_decoders.readOp.gate_leakage +
352  bank.mat.power_sa_mux_lev_2_decoders.readOp.gate_leakage ) * dp.num_act_mats_hor_dir;
353  //bank.mat.leak_power_sense_amps_open_page_state) * dp.num_act_mats_hor_dir;
354 
355  leak_power_request_and_reply_networks =
356  power_routing_to_bank.readOp.leakage +
357  bank.htree_in_add->power.readOp.leakage +
358  bank.htree_in_data->power.readOp.leakage +
359  bank.htree_out_data->power.readOp.leakage;
360 
361  leak_power_request_and_reply_networks +=
362  power_routing_to_bank.readOp.gate_leakage +
363  bank.htree_in_add->power.readOp.gate_leakage +
364  bank.htree_in_data->power.readOp.gate_leakage +
365  bank.htree_out_data->power.readOp.gate_leakage;
366 
367  if (dp.fully_assoc || dp.pure_cam)
368  {
369  leak_power_request_and_reply_networks += htree_in_search->power.readOp.leakage + htree_out_search->power.readOp.leakage;
370  leak_power_request_and_reply_networks += htree_in_search->power.readOp.gate_leakage + htree_out_search->power.readOp.gate_leakage;
371  }
372 
373 
374  if (dp.is_dram)
375  { // if DRAM, add contribution of power spent in row predecoder drivers, blocks and decoders to refresh power
376  refresh_power = (bank.mat.r_predec->power.readOp.dynamic * dp.num_act_mats_hor_dir +
377  bank.mat.row_dec->power.readOp.dynamic) * dp.num_r_subarray * dp.num_subarrays;
378  refresh_power += bank.mat.per_bitline_read_energy * dp.num_c_subarray * dp.num_r_subarray * dp.num_subarrays;
379  refresh_power += bank.mat.power_bl_precharge_eq_drv.readOp.dynamic * dp.num_act_mats_hor_dir;
380  refresh_power += bank.mat.power_sa.readOp.dynamic * dp.num_act_mats_hor_dir;
381  refresh_power /= dp.dram_refresh_period;
382  }
383 
384 
385  if (dp.is_tag == false)
386  {
387  power.readOp.dynamic = dyn_read_energy_from_closed_page;
388  power.writeOp.dynamic = dyn_read_energy_from_closed_page
389  - dyn_read_energy_remaining_words_in_burst
390  - bank.mat.power_bitline.readOp.dynamic * dp.num_act_mats_hor_dir
391  + bank.mat.power_bitline.writeOp.dynamic * dp.num_act_mats_hor_dir
392  + (power_routing_to_bank.writeOp.dynamic -
393  power_routing_to_bank.readOp.dynamic -
394  bank.htree_out_data->power.readOp.dynamic +
395  bank.htree_in_data->power.readOp.dynamic) *
396  (MAX((g_ip->burst_len / g_ip->int_prefetch_w), 1) - 1); //FIXME
397 
398  if (dp.is_dram == false)
399  {
400  power.writeOp.dynamic -= bank.mat.power_sa.readOp.dynamic * dp.num_act_mats_hor_dir;
401  }
402  }
403 
404  // if DRAM, add refresh power to total leakage
405  if (dp.is_dram)
406  {
407  power.readOp.leakage += refresh_power;
408  }
409 
410  // TODO: below should be avoided.
411  /*if (dp.is_main_mem)
412  {
413  power.readOp.leakage += MAIN_MEM_PER_CHIP_STANDBY_CURRENT_mA * 1e-3 * g_tp.peri_global.Vdd / g_ip->nbanks;
414  }*/
