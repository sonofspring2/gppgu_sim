183 void warp_inst_t::generate_mem_accesses()
184 {
185  if( empty() || op == MEMORY_BARRIER_OP || m_mem_accesses_created )
186  return;
187  if ( !((op == LOAD_OP) || (op == STORE_OP)) )
188  return;
189  if( m_warp_active_mask.count() == 0 )
190  return; // predicated off
191 
192  const size_t starting_queue_size = m_accessq.size();
193 
194  assert( is_load() || is_store() );
195  assert( m_per_scalar_thread_valid ); // need address information per thread
196 
197  bool is_write = is_store();
198 
199  mem_access_type access_type;
200  switch (space.get_type()) {
201  case const_space:
202  case param_space_kernel:
203  access_type = CONST_ACC_R;
204  break;
205  case tex_space:
206  access_type = TEXTURE_ACC_R;
207  break;
208  case global_space:
209  access_type = is_write? GLOBAL_ACC_W: GLOBAL_ACC_R;
210  break;
211  case local_space:
212  case param_space_local:
213  access_type = is_write? LOCAL_ACC_W: LOCAL_ACC_R;
214  break;
215  case shared_space: break;
216  default: assert(0); break;
217  }
218 
219  // Calculate memory accesses generated by this warp
220  new_addr_type cache_block_size = 0; // in bytes
221 
222  switch( space.get_type() ) {
223  case shared_space: {
224  unsigned subwarp_size = m_config->warp_size / m_config->mem_warp_parts;
225  unsigned total_accesses=0;
226  for( unsigned subwarp=0; subwarp < m_config->mem_warp_parts; subwarp++ ) {
227 
228  // data structures used per part warp
229  std::map<unsigned,std::map<new_addr_type,unsigned> > bank_accs; // bank -> word address -> access count
230 
231  // step 1: compute accesses to words in banks
232  for( unsigned thread=subwarp*subwarp_size; thread < (subwarp+1)*subwarp_size; thread++ ) {
233  if( !active(thread) )
234  continue;
235  new_addr_type addr = m_per_scalar_thread[thread].memreqaddr[0];
236  //FIXME: deferred allocation of shared memory should not accumulate across kernel launches
237  //assert( addr < m_config->gpgpu_shmem_size );
238  unsigned bank = m_config->shmem_bank_func(addr);
239  new_addr_type word = line_size_based_tag_func(addr,m_config->WORD_SIZE);
240  bank_accs[bank][word]++;
241  }
242 
243  if (m_config->shmem_limited_broadcast) {
244  // step 2: look for and select a broadcast bank/word if one occurs
245  bool broadcast_detected = false;
246  new_addr_type broadcast_word=(new_addr_type)-1;
247  unsigned broadcast_bank=(unsigned)-1;
248  std::map<unsigned,std::map<new_addr_type,unsigned> >::iterator b;
249  for( b=bank_accs.begin(); b != bank_accs.end(); b++ ) {
250  unsigned bank = b->first;
251  std::map<new_addr_type,unsigned> &access_set = b->second;
252  std::map<new_addr_type,unsigned>::iterator w;
253  for( w=access_set.begin(); w != access_set.end(); ++w ) {
254  if( w->second > 1 ) {
255  // found a broadcast
256  broadcast_detected=true;
257  broadcast_bank=bank;
258  broadcast_word=w->first;
259  break;
260  }
261  }
262  if( broadcast_detected )
263  break;
264  }
265 
266  // step 3: figure out max bank accesses performed, taking account of broadcast case
267  unsigned max_bank_accesses=0;
268  for( b=bank_accs.begin(); b != bank_accs.end(); b++ ) {
269  unsigned bank_accesses=0;
270  std::map<new_addr_type,unsigned> &access_set = b->second;
271  std::map<new_addr_type,unsigned>::iterator w;
272  for( w=access_set.begin(); w != access_set.end(); ++w )
273  bank_accesses += w->second;
274  if( broadcast_detected && broadcast_bank == b->first ) {
275  for( w=access_set.begin(); w != access_set.end(); ++w ) {
276  if( w->first == broadcast_word ) {
277  unsigned n = w->second;
278  assert(n > 1); // or this wasn't a broadcast
279  assert(bank_accesses >= (n-1));
280  bank_accesses -= (n-1);
281  break;
282  }
283  }
284  }
285  if( bank_accesses > max_bank_accesses )
286  max_bank_accesses = bank_accesses;
287  }
288 
289  // step 4: accumulate
290  total_accesses+= max_bank_accesses;
291  } else {
292  // step 2: look for the bank with the maximum number of access to different words
293  unsigned max_bank_accesses=0;
294  std::map<unsigned,std::map<new_addr_type,unsigned> >::iterator b;
295  for( b=bank_accs.begin(); b != bank_accs.end(); b++ ) {
296  max_bank_accesses = std::max(max_bank_accesses, (unsigned)b->second.size());
297  }
298 
299  // step 3: accumulate
300  total_accesses+= max_bank_accesses;
301  }
302  }
303  assert( total_accesses > 0 && total_accesses <= m_config->warp_size );
304  cycles = total_accesses; // shared memory conflicts modeled as larger initiation interval
305  ptx_file_line_stats_add_smem_bank_conflict( pc, total_accesses );
306  break;
307  }
308 
309  case tex_space:
310  cache_block_size = m_config->gpgpu_cache_texl1_linesize;
311  break;
312  case const_space: case param_space_kernel:
313  cache_block_size = m_config->gpgpu_cache_constl1_linesize;
314  break;
315 
316  case global_space: case local_space: case param_space_local:
317  if( m_config->gpgpu_coalesce_arch == 13 ) {
318  if(isatomic())
319  memory_coalescing_arch_13_atomic(is_write, access_type);
320  else
321  memory_coalescing_arch_13(is_write, access_type);
322  } else abort();
323 
324  break;
325 
326  default:
327  abort();
328  }
329 
330  if( cache_block_size ) {
331  assert( m_accessq.empty() );
332  mem_access_byte_mask_t byte_mask;
333  std::map<new_addr_type,active_mask_t> accesses; // block address -> set of thread offsets in warp
334  std::map<new_addr_type,active_mask_t>::iterator a;
335  for( unsigned thread=0; thread < m_config->warp_size; thread++ ) {
336  if( !active(thread) )
337  continue;
338  new_addr_type addr = m_per_scalar_thread[thread].memreqaddr[0];
339  unsigned block_address = line_size_based_tag_func(addr,cache_block_size);
340  accesses[block_address].set(thread);
341  unsigned idx = addr-block_address;
342  for( unsigned i=0; i < data_size; i++ )
343  byte_mask.set(idx+i);
344  }
345  for( a=accesses.begin(); a != accesses.end(); ++a )
346  m_accessq.push_back( mem_access_t(access_type,a->first,cache_block_size,is_write,a->second,byte_mask) );
347  }
348 
349  if ( space.get_type() == global_space ) {
350  ptx_file_line_stats_add_uncoalesced_gmem( pc, m_accessq.size() - starting_queue_size );
351  }
352  m_mem_accesses_created=true;
353 }

355 void warp_inst_t::memory_coalescing_arch_13( bool is_write, mem_access_type access_type )
356 {
357  // see the CUDA manual where it discusses coalescing rules before reading this
358  unsigned segment_size = 0;
359  unsigned warp_parts = m_config->mem_warp_parts;
360  switch( data_size ) {
361  case 1: segment_size = 32; break;
362  case 2: segment_size = 64; break;
363  case 4: case 8: case 16: segment_size = 128; break;
364  }
365  unsigned subwarp_size = m_config->warp_size / warp_parts;
366 
367  for( unsigned subwarp=0; subwarp < warp_parts; subwarp++ ) {
368  std::map<new_addr_type,transaction_info> subwarp_transactions;
369 
370  // step 1: find all transactions generated by this subwarp
371  for( unsigned thread=subwarp*subwarp_size; thread<subwarp_size*(subwarp+1); thread++ ) {
372  if( !active(thread) )
373  continue;
374 
375  unsigned data_size_coales = data_size;
376  unsigned num_accesses = 1;
377 
378  if( space.get_type() == local_space || space.get_type() == param_space_local ) {
379  // Local memory accesses >4B were split into 4B chunks
380  if(data_size >= 4) {
381  data_size_coales = 4;
382  num_accesses = data_size/4;
383  }
384  // Otherwise keep the same data_size for sub-4B access to local memory
385  }
386 
387 
388  assert(num_accesses <= MAX_ACCESSES_PER_INSN_PER_THREAD);
389 
390  for(unsigned access=0; access<num_accesses; access++) {
391  new_addr_type addr = m_per_scalar_thread[thread].memreqaddr[access];
392  unsigned block_address = line_size_based_tag_func(addr,segment_size);
393  unsigned chunk = (addr&127)/32; // which 32-byte chunk within in a 128-byte chunk does this thread access?
394  transaction_info &info = subwarp_transactions[block_address];
395 
396  // can only write to one segment
397  assert(block_address == line_size_based_tag_func(addr+data_size_coales-1,segment_size));
398 
399  info.chunks.set(chunk);
400  info.active.set(thread);
401  unsigned idx = (addr&127);
402  for( unsigned i=0; i < data_size_coales; i++ )
403  info.bytes.set(idx+i);
404  }
405  }
406 
407  // step 2: reduce each transaction size, if possible
408  std::map< new_addr_type, transaction_info >::iterator t;
409  for( t=subwarp_transactions.begin(); t !=subwarp_transactions.end(); t++ ) {
410  new_addr_type addr = t->first;
411  const transaction_info &info = t->second;
412 
413  memory_coalescing_arch_13_reduce_and_send(is_write, access_type, info, addr, segment_size);
414 
415  }
416  }
417 }

419 void warp_inst_t::memory_coalescing_arch_13_atomic( bool is_write, mem_access_type access_type )
420 {
421 
422  assert(space.get_type() == global_space); // Atomics allowed only for global memory
423 
424  // see the CUDA manual where it discusses coalescing rules before reading this
425  unsigned segment_size = 0;
426  unsigned warp_parts = 2;
427  switch( data_size ) {
428  case 1: segment_size = 32; break;
429  case 2: segment_size = 64; break;
430  case 4: case 8: case 16: segment_size = 128; break;
431  }
432  unsigned subwarp_size = m_config->warp_size / warp_parts;
433 
434  for( unsigned subwarp=0; subwarp < warp_parts; subwarp++ ) {
435  std::map<new_addr_type,std::list<transaction_info> > subwarp_transactions; // each block addr maps to a list of transactions
436 
437  // step 1: find all transactions generated by this subwarp
438  for( unsigned thread=subwarp*subwarp_size; thread<subwarp_size*(subwarp+1); thread++ ) {
439  if( !active(thread) )
440  continue;
441 
442  new_addr_type addr = m_per_scalar_thread[thread].memreqaddr[0];
443  unsigned block_address = line_size_based_tag_func(addr,segment_size);
444  unsigned chunk = (addr&127)/32; // which 32-byte chunk within in a 128-byte chunk does this thread access?
445 
446  // can only write to one segment
447  assert(block_address == line_size_based_tag_func(addr+data_size-1,segment_size));
448 
449  // Find a transaction that does not conflict with this thread's accesses
450  bool new_transaction = true;
451  std::list<transaction_info>::iterator it;
452  transaction_info* info;
453  for(it=subwarp_transactions[block_address].begin(); it!=subwarp_transactions[block_address].end(); it++) {
454  unsigned idx = (addr&127);
455  if(not it->test_bytes(idx,idx+data_size-1)) {
456  new_transaction = false;
457  info = &(*it);
458  break;
459  }
460  }
461  if(new_transaction) {
462  // Need a new transaction
463  subwarp_transactions[block_address].push_back(transaction_info());
464  info = &subwarp_transactions[block_address].back();
465  }
466  assert(info);
467 
468  info->chunks.set(chunk);
469  info->active.set(thread);
470  unsigned idx = (addr&127);
471  for( unsigned i=0; i < data_size; i++ ) {
472  assert(!info->bytes.test(idx+i));
473  info->bytes.set(idx+i);
474  }
475  }
476 
477  // step 2: reduce each transaction size, if possible
478  std::map< new_addr_type, std::list<transaction_info> >::iterator t_list;
479  for( t_list=subwarp_transactions.begin(); t_list !=subwarp_transactions.end(); t_list++ ) {
480  // For each block addr
481  new_addr_type addr = t_list->first;
482  const std::list<transaction_info>& transaction_list = t_list->second;
483 
484  std::list<transaction_info>::const_iterator t;
485  for(t=transaction_list.begin(); t!=transaction_list.end(); t++) {
486  // For each transaction
487  const transaction_info &info = *t;
488  memory_coalescing_arch_13_reduce_and_send(is_write, access_type, info, addr, segment_size);
489  }
490  }
491  }
492 }

494 void warp_inst_t::memory_coalescing_arch_13_reduce_and_send( bool is_write, mem_access_type access_type, const transaction_info &info, new_addr_type addr, unsigned segment_size )
495 {
496  assert( (addr & (segment_size-1)) == 0 );
497 
498  const std::bitset<4> &q = info.chunks;
499  assert( q.count() >= 1 );
500  std::bitset<2> h; // halves (used to check if 64 byte segment can be compressed into a single 32 byte segment)
501 
502  unsigned size=segment_size;
503  if( segment_size == 128 ) {
504  bool lower_half_used = q[0] || q[1];
505  bool upper_half_used = q[2] || q[3];
506  if( lower_half_used && !upper_half_used ) {
507  // only lower 64 bytes used
508  size = 64;
509  if(q[0]) h.set(0);
510  if(q[1]) h.set(1);
511  } else if ( (!lower_half_used) && upper_half_used ) {
512  // only upper 64 bytes used
513  addr = addr+64;
514  size = 64;
515  if(q[2]) h.set(0);
516  if(q[3]) h.set(1);
517  } else {
518  assert(lower_half_used && upper_half_used);
519  }
520  } else if( segment_size == 64 ) {
521  // need to set halves
522  if( (addr % 128) == 0 ) {
523  if(q[0]) h.set(0);
524  if(q[1]) h.set(1);
525  } else {
526  assert( (addr % 128) == 64 );
527  if(q[2]) h.set(0);
528  if(q[3]) h.set(1);
529  }
530  }
531  if( size == 64 ) {
532  bool lower_half_used = h[0];
533  bool upper_half_used = h[1];
534  if( lower_half_used && !upper_half_used ) {
535  size = 32;
536  } else if ( (!lower_half_used) && upper_half_used ) {
537  addr = addr+32;
538  size = 32;
539  } else {
540  assert(lower_half_used && upper_half_used);
541  }
542  }
543  m_accessq.push_back( mem_access_t(access_type,addr,size,is_write,info.active,info.bytes) );
544 }

650 void simt_stack::update( simt_mask_t &thread_done, addr_vector_t &next_pc, address_type recvg_pc, op_type next_inst_op,unsigned next_inst_size, address_type next_inst_pc )
651 {
652  assert(m_stack.size() > 0);
653 
654  assert( next_pc.size() == m_warp_size );
655 
656  simt_mask_t top_active_mask = m_stack.back().m_active_mask;
657  address_type top_recvg_pc = m_stack.back().m_recvg_pc;
658  address_type top_pc = m_stack.back().m_pc; // the pc of the instruction just executed
659  stack_entry_type top_type = m_stack.back().m_type;
660  assert(top_pc==next_inst_pc);
661  assert(top_active_mask.any());
662 
663  const address_type null_pc = -1;
664  bool warp_diverged = false;
665  address_type new_recvg_pc = null_pc;
666  unsigned num_divergent_paths=0;
667 
668  std::map<address_type,simt_mask_t> divergent_paths;
669  while (top_active_mask.any()) {
670 
671  // extract a group of threads with the same next PC among the active threads in the warp
672  address_type tmp_next_pc = null_pc;
673  simt_mask_t tmp_active_mask;
674  for (int i = m_warp_size - 1; i >= 0; i--) {
675  if ( top_active_mask.test(i) ) { // is this thread active?
676  if (thread_done.test(i)) {
677  top_active_mask.reset(i); // remove completed thread from active mask
678  } else if (tmp_next_pc == null_pc) {
679  tmp_next_pc = next_pc[i];
680  tmp_active_mask.set(i);
681  top_active_mask.reset(i);
682  } else if (tmp_next_pc == next_pc[i]) {
683  tmp_active_mask.set(i);
684  top_active_mask.reset(i);
685  }
686  }
687  }
688 
689  if(tmp_next_pc == null_pc) {
690  assert(!top_active_mask.any()); // all threads done
691  continue;
692  }
693 
694  divergent_paths[tmp_next_pc]=tmp_active_mask;
695  num_divergent_paths++;
696  }
697 
698 
699  address_type not_taken_pc = next_inst_pc+next_inst_size;
700  assert(num_divergent_paths<=2);
701  for(unsigned i=0; i<num_divergent_paths; i++){
702  address_type tmp_next_pc = null_pc;
703  simt_mask_t tmp_active_mask;
704  tmp_active_mask.reset();
705  if(divergent_paths.find(not_taken_pc)!=divergent_paths.end()){
706  assert(i==0);
707  tmp_next_pc=not_taken_pc;
708  tmp_active_mask=divergent_paths[tmp_next_pc];
709  divergent_paths.erase(tmp_next_pc);
710  }else{
711  std::map<address_type,simt_mask_t>:: iterator it=divergent_paths.begin();
712  tmp_next_pc=it->first;
713  tmp_active_mask=divergent_paths[tmp_next_pc];
714  divergent_paths.erase(tmp_next_pc);
715  }
716 
717  // HANDLE THE SPECIAL CASES FIRST
718  if (next_inst_op== CALL_OPS){
719  // Since call is not a divergent instruction, all threads should have executed a call instruction
720  assert(num_divergent_paths == 1);
721 
722  simt_stack_entry new_stack_entry;
723  new_stack_entry.m_pc = tmp_next_pc;
724  new_stack_entry.m_active_mask = tmp_active_mask;
725  new_stack_entry.m_branch_div_cycle = gpu_sim_cycle+gpu_tot_sim_cycle;
726  new_stack_entry.m_type = STACK_ENTRY_TYPE_CALL;
727  m_stack.push_back(new_stack_entry);
728  return;
729  }else if(next_inst_op == RET_OPS && top_type==STACK_ENTRY_TYPE_CALL){
730  // pop the CALL Entry
731  assert(num_divergent_paths == 1);
732  m_stack.pop_back();
733 
734  assert(m_stack.size() > 0);
735  m_stack.back().m_pc=tmp_next_pc;// set the PC of the stack top entry to return PC from the call stack;
736  // Check if the New top of the stack is reconverging
737  if (tmp_next_pc == m_stack.back().m_recvg_pc && m_stack.back().m_type!=STACK_ENTRY_TYPE_CALL){
738  assert(m_stack.back().m_type==STACK_ENTRY_TYPE_NORMAL);
739  m_stack.pop_back();
740  }
741  return;
742  }
743 
744  // discard the new entry if its PC matches with reconvergence PC
745  // that automatically reconverges the entry
746  // If the top stack entry is CALL, dont reconverge.
747  if (tmp_next_pc == top_recvg_pc && (top_type != STACK_ENTRY_TYPE_CALL)) continue;
748 
749  // this new entry is not converging
750  // if this entry does not include thread from the warp, divergence occurs
751  if ((num_divergent_paths>1) && !warp_diverged ) {
752  warp_diverged = true;
753  // modify the existing top entry into a reconvergence entry in the pdom stack
754  new_recvg_pc = recvg_pc;
755  if (new_recvg_pc != top_recvg_pc) {
756  m_stack.back().m_pc = new_recvg_pc;
757  m_stack.back().m_branch_div_cycle = gpu_sim_cycle+gpu_tot_sim_cycle;
758 
759  m_stack.push_back(simt_stack_entry());
760  }
761  }
762 
763  // discard the new entry if its PC matches with reconvergence PC
764  if (warp_diverged && tmp_next_pc == new_recvg_pc) continue;
765 
766  // update the current top of pdom stack
767  m_stack.back().m_pc = tmp_next_pc;
768  m_stack.back().m_active_mask = tmp_active_mask;
769  if (warp_diverged) {
770  m_stack.back().m_calldepth = 0;
771  m_stack.back().m_recvg_pc = new_recvg_pc;
772  } else {
773  m_stack.back().m_recvg_pc = top_recvg_pc;
774  }
775 
776  m_stack.push_back(simt_stack_entry());
777  }
778  assert(m_stack.size() > 0);
779  m_stack.pop_back();
780 
781 
782  if (warp_diverged) {
783  ptx_file_line_stats_add_warp_divergence(top_pc, 1);
784  }
785 }
