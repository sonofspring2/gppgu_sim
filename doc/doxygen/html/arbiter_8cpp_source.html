91 Arbiter *Arbiter::NewArbiter( Module *parent, const string& name,

100  size_t left = 4;

103  size_t right = arb_type.find_last_of(')');
104  assert(right != string::npos);
105  string groups_str = arb_type.substr(left+1, middle-left-1);
106  int groups = atoi(groups_str.c_str());
107  string sub_arb_type = arb_type.substr(middle+1, right-middle-1);
108  a = new TreeArbiter( parent, name, size, groups, sub_arb_type );
109  } else assert(false);
