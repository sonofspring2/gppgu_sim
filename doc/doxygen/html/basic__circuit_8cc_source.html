163 double drain_C_(
164  double width,
165  int nchannel,
166  int stack,
167  int next_arg_thresh_folding_width_or_height_cell,
168  double fold_dimension,
169  bool _is_dram,
170  bool _is_cell,
171  bool _is_wl_tr)
172 {
173  double w_folded_tr=0;
174  const TechnologyParameter::DeviceType * dt;
175 
176  if ((_is_dram) && (_is_cell))
177  {
178  dt = &g_tp.dram_acc; // DRAM cell access transistor
179  }
180  else if ((_is_dram) && (_is_wl_tr))
181  {
182  dt = &g_tp.dram_wl; // DRAM wordline transistor
183  }
184  else if ((!_is_dram) && _is_cell)
185  {
186  dt = &g_tp.sram_cell; // SRAM cell access transistor
187  }
188  else
189  {
190  dt = &g_tp.peri_global;
191  }
192 
193  double c_junc_area = dt->C_junc;
194  double c_junc_sidewall = dt->C_junc_sidewall;
195  double c_fringe = 2*dt->C_fringe;
196  double c_overlap = 2*dt->C_overlap;
197  double drain_C_metal_connecting_folded_tr = 0;
198 
199  // determine the width of the transistor after folding (if it is getting folded)
200  if (next_arg_thresh_folding_width_or_height_cell == 0)
201  { // interpret fold_dimension as the the folding width threshold
202  // i.e. the value of transistor width above which the transistor gets folded
203  w_folded_tr = fold_dimension;
204  }
205  else
206  { // interpret fold_dimension as the height of the cell that this transistor is part of.
207  double h_tr_region = fold_dimension - 2 * g_tp.HPOWERRAIL;
208  // TODO : w_folded_tr must come from Component::compute_gate_area()
209  double ratio_p_to_n = 2.0 / (2.0 + 1.0);
210  if (nchannel)
211  {
212  w_folded_tr = (1 - ratio_p_to_n) * (h_tr_region - g_tp.MIN_GAP_BET_P_AND_N_DIFFS);
213  }
214  else
215  {
216  w_folded_tr = ratio_p_to_n * (h_tr_region - g_tp.MIN_GAP_BET_P_AND_N_DIFFS);
217  }
218  }
219  int num_folded_tr = (int) (ceil(width / w_folded_tr));
220 
221  if (num_folded_tr < 2)
222  {
223  w_folded_tr = width;
224  }
225 
226  double total_drain_w = (g_tp.w_poly_contact + 2 * g_tp.spacing_poly_to_contact) + // only for drain
227  (stack - 1) * g_tp.spacing_poly_to_poly;
228  double drain_h_for_sidewall = w_folded_tr;
229  double total_drain_height_for_cap_wrt_gate = w_folded_tr + 2 * w_folded_tr * (stack - 1);
230  if (num_folded_tr > 1)
231  {
232  total_drain_w += (num_folded_tr - 2) * (g_tp.w_poly_contact + 2 * g_tp.spacing_poly_to_contact) +
233  (num_folded_tr - 1) * ((stack - 1) * g_tp.spacing_poly_to_poly);
234 
235  if (num_folded_tr%2 == 0)
236  {
237  drain_h_for_sidewall = 0;
238  }
239  total_drain_height_for_cap_wrt_gate *= num_folded_tr;
240  drain_C_metal_connecting_folded_tr = g_tp.wire_local.C_per_um * total_drain_w;
241  }
242 
243  double drain_C_area = c_junc_area * total_drain_w * w_folded_tr;
244  double drain_C_sidewall = c_junc_sidewall * (drain_h_for_sidewall + 2 * total_drain_w);
245  double drain_C_wrt_gate = (c_fringe + c_overlap) * total_drain_height_for_cap_wrt_gate;
246 
247  return (drain_C_area + drain_C_sidewall + drain_C_wrt_gate + drain_C_metal_connecting_folded_tr);
248 }

336 // "Timing Models for MOS Circuits" by Mark Horowitz, 1984
337 double horowitz(
338  double inputramptime, // input rise time
339  double tf, // time constant of gate
340  double vs1, // threshold voltage
341  double vs2, // threshold voltage
342  int rise) // whether input rises or fall
343 {
344  if (inputramptime == 0 && vs1 == vs2)
345  {
346  return tf * (vs1 < 1 ? -log(vs1) : log(vs1));
347  }
348  double a, b, td;
349 
350  a = inputramptime / tf;
351  if (rise == RISE)
352  {
353  b = 0.5;
354  td = tf * sqrt(log(vs1)*log(vs1) + 2*a*b*(1.0 - vs1)) + tf*(log(vs1) - log(vs2));
355  }
356  else
357  {
358  b = 0.4;
359  td = tf * sqrt(log(1.0 - vs1)*log(1.0 - vs1) + 2*a*b*(vs1)) + tf*(log(1.0 - vs1) - log(1.0 - vs2));
360  }
361  return (td);
362 }

496 double cmos_Isub_leakage(
497  double nWidth,
498  double pWidth,
499  int fanin,
500  enum Gate_type g_type,
501  bool _is_dram,
502  bool _is_cell,
503  bool _is_wl_tr,
504  enum Half_net_topology topo)
505 {
506  assert (fanin>=1);
507  double nmos_leak = simplified_nmos_leakage(nWidth, _is_dram, _is_cell, _is_wl_tr);
508  double pmos_leak = simplified_pmos_leakage(pWidth, _is_dram, _is_cell, _is_wl_tr);
509  double Isub=0;
510  int num_states;
511  int num_off_tx;
512 
513  num_states = int(pow(2.0, fanin));
514 
515  switch (g_type)
516  {
517  case nmos:
518  if (fanin==1)
519  {
520  Isub = nmos_leak/num_states;
521  }
522  else
523  {
524  if (topo==parallel)
525  {
526  Isub=nmos_leak*fanin/num_states; //only when all tx are off, leakage power is non-zero. The possibility of this state is 1/num_states
527  }
528  else
529  {
530  for (num_off_tx=1; num_off_tx<=fanin; num_off_tx++) //when num_off_tx ==0 there is no leakage power
531  {
532  //Isub += nmos_leak*pow(UNI_LEAK_STACK_FACTOR,(num_off_tx-1))*(factorial(fanin)/(factorial(fanin, num_off_tx)*factorial(num_off_tx)));
533  Isub += nmos_leak*pow(UNI_LEAK_STACK_FACTOR,(num_off_tx-1))*combination(fanin, num_off_tx);
534  }
535  Isub /=num_states;
536  }
537 
538  }
539  break;
540  case pmos:
541  if (fanin==1)
542  {
543  Isub = pmos_leak/num_states;
544  }
545  else
546  {
547  if (topo==parallel)
548  {
549  Isub=pmos_leak*fanin/num_states; //only when all tx are off, leakage power is non-zero. The possibility of this state is 1/num_states
550  }
551  else
552  {
553  for (num_off_tx=1; num_off_tx<=fanin; num_off_tx++) //when num_off_tx ==0 there is no leakage power
554  {
555  //Isub += pmos_leak*pow(UNI_LEAK_STACK_FACTOR,(num_off_tx-1))*(factorial(fanin)/(factorial(fanin, num_off_tx)*factorial(num_off_tx)));
556  Isub += pmos_leak*pow(UNI_LEAK_STACK_FACTOR,(num_off_tx-1))*combination(fanin, num_off_tx);
557  }
558  Isub /=num_states;
559  }
560 
561  }
562  break;
563  case inv:
564  Isub = (nmos_leak + pmos_leak)/2;
565  break;
566  case nand:
567  Isub += fanin*pmos_leak;//the pullup network
568  for (num_off_tx=1; num_off_tx<=fanin; num_off_tx++) // the pulldown network
569  {
570  //Isub += nmos_leak*pow(UNI_LEAK_STACK_FACTOR,(num_off_tx-1))*(factorial(fanin)/(factorial(fanin, num_off_tx)*factorial(num_off_tx)));
571  Isub += nmos_leak*pow(UNI_LEAK_STACK_FACTOR,(num_off_tx-1))*combination(fanin, num_off_tx);
572  }
573  Isub /=num_states;
574  break;
575  case nor:
576  for (num_off_tx=1; num_off_tx<=fanin; num_off_tx++) // the pullup network
577  {
578  //Isub += pmos_leak*pow(UNI_LEAK_STACK_FACTOR,(num_off_tx-1))*(factorial(fanin)/(factorial(fanin, num_off_tx)*factorial(num_off_tx)));
579  Isub += pmos_leak*pow(UNI_LEAK_STACK_FACTOR,(num_off_tx-1))*combination(fanin, num_off_tx);
580  }
581  Isub += fanin*nmos_leak;//the pulldown network
582  Isub /=num_states;
583  break;
584  case tri:
585  Isub += (nmos_leak + pmos_leak)/2;//enabled
586  Isub += nmos_leak*UNI_LEAK_STACK_FACTOR; //disabled upper bound of leakage power
587  Isub /=2;
588  break;
589  case tg:
590  Isub = (nmos_leak + pmos_leak)/2;
591  break;
592  default:
593  assert(0);
594  break;
595  }
596 
597  return Isub;
598 }
