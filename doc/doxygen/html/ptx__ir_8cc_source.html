175 bool symbol_table::add_function_decl( const char *name, int entry_point, function_info **func_info, symbol_table **sym_table )
176 {
177  std::string key = std::string(name);
178  bool prior_decl = false;
179  if( m_function_info_lookup.find(key) != m_function_info_lookup.end() ) {
180  *func_info = m_function_info_lookup[key];
181  prior_decl = true;
182  } else {
183  *func_info = new function_info(entry_point);
184  (*func_info)->set_name(name);
185  m_function_info_lookup[key] = *func_info;
186  }
187 
188  if( m_function_symtab_lookup.find(key) != m_function_symtab_lookup.end() ) {
189  assert( prior_decl );
190  *sym_table = m_function_symtab_lookup[key];
191  } else {
192  assert( !prior_decl );
193  *sym_table = new symbol_table( "", entry_point, this );
194 
195  // Initial setup code to support a register represented as "_".
196  // This register is used when an instruction operand is
197  // not read or written. However, the parser must recognize it
198  // as a legitimate register but we do not want to pass
199  // it to the micro-architectural register to the performance simulator.
200  // For this purpose we add a symbol to the symbol table but
201  // mark it as a non_arch_reg so it does not effect the performance sim.
202  type_info_key null_key( reg_space, 0, 0, 0, 0, 0 );
203  null_key.set_is_non_arch_reg();
204  // First param is null - which is bad.
205  // However, the first parameter is actually unread in the constructor...
206  // TODO - remove the symbol_table* from type_info
207  type_info* null_type_info = new type_info( NULL, null_key );
208  symbol *null_reg = (*sym_table)->add_variable( "_", null_type_info, 0, "", 0 );
209  null_reg->set_regno(0, 0);
210 
211  (*sym_table)->set_name(name);
212  (*func_info)->set_symtab(*sym_table);
213  m_function_symtab_lookup[key] = *sym_table;
214  assert( (*func_info)->get_symtab() == *sym_table );
215  register_ptx_function(name,*func_info);
216  }
217  return prior_decl;
218 }

287 void function_info::create_basic_blocks()
288 {
289  std::list<ptx_instruction*> leaders;
290  std::list<ptx_instruction*>::iterator i, l;
291 
292  // first instruction is a leader
293  i=m_instructions.begin();
294  leaders.push_back(*i);
295  i++;
296  while( i!=m_instructions.end() ) {
297  ptx_instruction *pI = *i;
298  if( pI->is_label() ) {
299  leaders.push_back(pI);
300  i = find_next_real_instruction(++i);
301  } else {
302  switch( pI->get_opcode() ) {
303  case BRA_OP: case RET_OP: case EXIT_OP: case RETP_OP: case BREAK_OP:
304  i++;
305  if( i != m_instructions.end() )
306  leaders.push_back(*i);
307  i = find_next_real_instruction(i);
308  break;
309  case CALL_OP: case CALLP_OP:
310  if( pI->has_pred() ) {
311  printf("GPGPU-Sim PTX: Warning found predicated call\n");
312  i++;
313  if( i != m_instructions.end() )
314  leaders.push_back(*i);
315  i = find_next_real_instruction(i);
316  } else i++;
317  break;
318  default:
319  i++;
320  }
321  }
322  }
323 
324  if( leaders.empty() ) {
325  printf("GPGPU-Sim PTX: Function \'%s\' has no basic blocks\n", m_name.c_str());
326  return;
327  }
328 
329  unsigned bb_id = 0;
330  l=leaders.begin();
331  i=m_instructions.begin();
332  m_basic_blocks.push_back( new basic_block_t(bb_id++,*find_next_real_instruction(i),NULL,1,0) );
333  ptx_instruction *last_real_inst=*(l++);
334 
335  for( ; i!=m_instructions.end(); i++ ) {
336  ptx_instruction *pI = *i;
337  if( l != leaders.end() && *i == *l ) {
338  // found start of next basic block
339  m_basic_blocks.back()->ptx_end = last_real_inst;
340  if( find_next_real_instruction(i) != m_instructions.end() ) { // if not bogus trailing label
341  m_basic_blocks.push_back( new basic_block_t(bb_id++,*find_next_real_instruction(i),NULL,0,0) );
342  last_real_inst = *find_next_real_instruction(i);
343  }
344  // start search for next leader
345  l++;
346  }
347  pI->assign_bb( m_basic_blocks.back() );
348  if( !pI->is_label() ) last_real_inst = pI;
349  }
350  m_basic_blocks.back()->ptx_end = last_real_inst;
351  m_basic_blocks.push_back( /*exit basic block*/ new basic_block_t(bb_id,NULL,NULL,0,1) );
352 }

409 operand_info* function_info::find_break_target( ptx_instruction * p_break_insn ) //find the target of a break instruction
410 {
411  const basic_block_t *break_bb = p_break_insn->get_bb();
412  // go through the dominator tree
413  for(const basic_block_t *p_bb = break_bb;
414  p_bb->immediatedominator_id != -1;
415  p_bb = m_basic_blocks[p_bb->immediatedominator_id])
416  {
417  // reverse search through instructions in basic block for breakaddr instruction
418  unsigned insn_addr = p_bb->ptx_end->get_m_instr_mem_index();
419  while (insn_addr >= p_bb->ptx_begin->get_m_instr_mem_index()) {
420  ptx_instruction *pI = m_instr_mem[insn_addr];
421  insn_addr -= 1;
422  if (pI == NULL) continue; // temporary solution for variable size instructions
423  if (pI->get_opcode() == BREAKADDR_OP) {
424  return &(pI->dst());
425  }
426  }
427  }
428 
429  assert(0);
430 
431  // lazy fallback: just traverse backwards?
432  for (int insn_addr = p_break_insn->get_m_instr_mem_index();
433  insn_addr >= 0; insn_addr--)
434  {
435  ptx_instruction *pI = m_instr_mem[insn_addr];
436  if (pI->get_opcode() == BREAKADDR_OP) {
437  return &(pI->dst());
438  }
439  }
440 
441  return NULL;
442 }
443 void function_info::connect_basic_blocks( ) //iterate across m_basic_blocks of function, connecting basic blocks together
444 {
445  std::vector<basic_block_t*>::iterator bb_itr;
446  std::vector<basic_block_t*>::iterator bb_target_itr;
447  basic_block_t* exit_bb = m_basic_blocks.back();
448 
449  //start from first basic block, which we know is the entry point
450  bb_itr = m_basic_blocks.begin();
451  for (bb_itr = m_basic_blocks.begin();bb_itr != m_basic_blocks.end(); bb_itr++) {
452  ptx_instruction *pI = (*bb_itr)->ptx_end;
453  if ((*bb_itr)->is_exit) //reached last basic block, no successors to link
454  continue;
455  if (pI->get_opcode() == RETP_OP || pI->get_opcode() == RET_OP || pI->get_opcode() == EXIT_OP ) {
456  (*bb_itr)->successor_ids.insert(exit_bb->bb_id);
457  exit_bb->predecessor_ids.insert((*bb_itr)->bb_id);
458  if( pI->has_pred() ) {
459  printf("GPGPU-Sim PTX: Warning detected predicated return/exit.\n");
460  // if predicated, add link to next block
461  unsigned next_addr = pI->get_m_instr_mem_index() + 1;
462  if( next_addr < m_instr_mem_size && m_instr_mem[next_addr] ) {
463  basic_block_t *next_bb = m_instr_mem[next_addr]->get_bb();
464  (*bb_itr)->successor_ids.insert(next_bb->bb_id);
465  next_bb->predecessor_ids.insert((*bb_itr)->bb_id);
466  }
467  }
468  continue;
469  } else if (pI->get_opcode() == BRA_OP) {
470  //find successor and link that basic_block to this one
471  operand_info &target = pI->dst(); //get operand, e.g. target name
472  unsigned addr = labels[ target.name() ];
473  ptx_instruction *target_pI = m_instr_mem[addr];
474  basic_block_t *target_bb = target_pI->get_bb();
475  (*bb_itr)->successor_ids.insert(target_bb->bb_id);
476  target_bb->predecessor_ids.insert((*bb_itr)->bb_id);
477  }
478 
479  if ( !(pI->get_opcode()==BRA_OP && (!pI->has_pred())) ) {
480  // if basic block does not end in an unpredicated branch,
481  // then next basic block is also successor
482  // (this is better than testing for .uni)
483  unsigned next_addr = pI->get_m_instr_mem_index() + pI->inst_size();
484  basic_block_t *next_bb = m_instr_mem[next_addr]->get_bb();
485  (*bb_itr)->successor_ids.insert(next_bb->bb_id);
486  next_bb->predecessor_ids.insert((*bb_itr)->bb_id);
487  } else
488  assert(pI->get_opcode() == BRA_OP);
489  }
490 }
491 bool function_info::connect_break_targets() //connecting break instructions with proper targets
492 {
493  std::vector<basic_block_t*>::iterator bb_itr;
494  std::vector<basic_block_t*>::iterator bb_target_itr;
495  bool modified = false;
496 
497  //start from first basic block, which we know is the entry point
498  bb_itr = m_basic_blocks.begin();
499  for (bb_itr = m_basic_blocks.begin();bb_itr != m_basic_blocks.end(); bb_itr++) {
500  basic_block_t *p_bb = *bb_itr;
501  ptx_instruction *pI = p_bb->ptx_end;
502  if (p_bb->is_exit) //reached last basic block, no successors to link
503  continue;
504  if (pI->get_opcode() == BREAK_OP) {
505  // backup existing successor_ids for stability check
506  std::set<int> orig_successor_ids = p_bb->successor_ids;
507 
508  // erase the previous linkage with old successors
509  for(std::set<int>::iterator succ_ids = p_bb->successor_ids.begin(); succ_ids != p_bb->successor_ids.end(); ++succ_ids) {
510  basic_block_t *successor_bb = m_basic_blocks[*succ_ids];
511  successor_bb->predecessor_ids.erase(p_bb->bb_id);
512  }
513  p_bb->successor_ids.clear();
514 
515  //find successor and link that basic_block to this one
516  //successor of a break is set by an preceeding breakaddr instruction
517  operand_info *target = find_break_target(pI);
518  unsigned addr = labels[ target->name() ];
519  ptx_instruction *target_pI = m_instr_mem[addr];
520  basic_block_t *target_bb = target_pI->get_bb();
521  p_bb->successor_ids.insert(target_bb->bb_id);
522  target_bb->predecessor_ids.insert(p_bb->bb_id);
523 
524  if (pI->has_pred()) {
525  // predicated break - add link to next basic block
526  unsigned next_addr = pI->get_m_instr_mem_index() + pI->inst_size();
527  basic_block_t *next_bb = m_instr_mem[next_addr]->get_bb();
528  p_bb->successor_ids.insert(next_bb->bb_id);
529  next_bb->predecessor_ids.insert(p_bb->bb_id);
530  }
531 
532  modified = modified || (orig_successor_ids != p_bb->successor_ids);
533  }
534  }
535 
536  return modified;
537 }

573  // find dominators using algorithm of Muchnick's Adv. Compiler Design & Implemmntation Fig 7.14
574  printf("GPGPU-Sim PTX: Finding dominators for \'%s\'...\n", m_name.c_str() );
575  fflush(stdout);
576  assert( m_basic_blocks.size() >= 2 ); // must have a distinquished entry block
577  std::vector<basic_block_t*>::iterator bb_itr = m_basic_blocks.begin();
578  (*bb_itr)->dominator_ids.insert((*bb_itr)->bb_id); // the only dominator of the entry block is the entry
579  //copy all basic blocks to all dominator lists EXCEPT for the entry block
580  for (++bb_itr;bb_itr != m_basic_blocks.end(); bb_itr++) {
581  for (unsigned i = 0; i < m_basic_blocks.size(); i++)
582  (*bb_itr)->dominator_ids.insert(i);
583  }
584  bool change = true;
585  while (change) {
586  change = false;
587  for ( int h = 1/*skip entry*/; h < m_basic_blocks.size(); ++h ) {
588  assert( m_basic_blocks[h]->bb_id == (unsigned)h );
589  std::set<int> T;
590  for (unsigned i=0;i< m_basic_blocks.size();i++)
591  T.insert(i);
592  for ( std::set<int>::iterator s = m_basic_blocks[h]->predecessor_ids.begin();s != m_basic_blocks[h]->predecessor_ids.end();s++)
593  intersect(T, m_basic_blocks[*s]->dominator_ids);
594  T.insert(h);
595  if (!is_equal(T, m_basic_blocks[h]->dominator_ids)) {
596  change = true;
597  m_basic_blocks[h]->dominator_ids = T;
598  }
599  }
600  }
601  //clean the basic block of dominators of it has no predecessors -- except for entry block
602  bb_itr = m_basic_blocks.begin();
603  for (++bb_itr;bb_itr != m_basic_blocks.end(); bb_itr++) {
604  if ((*bb_itr)->predecessor_ids.empty())
605  (*bb_itr)->dominator_ids.clear();
606  }

640 void function_info::find_ipostdominators( )
641 {
642  // find immediate postdominator blocks, using algorithm of
643  // Muchnick's Adv. Compiler Design & Implemmntation Fig 7.15
644  printf("GPGPU-Sim PTX: Finding immediate postdominators for \'%s\'...\n", m_name.c_str() );
645  fflush(stdout);
646  assert( m_basic_blocks.size() >= 2 ); // must have a distinquished exit block
647  for (unsigned i=0; i<m_basic_blocks.size(); i++) { //initialize Tmp(n) to all pdoms of n except for n
648  m_basic_blocks[i]->Tmp_ids = m_basic_blocks[i]->postdominator_ids;
649  assert( m_basic_blocks[i]->bb_id == i );
650  m_basic_blocks[i]->Tmp_ids.erase(i);
651  }
652  for ( int n = m_basic_blocks.size()-2; n >=0;--n) {
653  // point iterator to basic block before the exit
654  for( std::set<int>::iterator s=m_basic_blocks[n]->Tmp_ids.begin(); s != m_basic_blocks[n]->Tmp_ids.end(); s++ ) {
655  int bb_s = *s;
656  for( std::set<int>::iterator t=m_basic_blocks[n]->Tmp_ids.begin(); t != m_basic_blocks[n]->Tmp_ids.end(); ) {
657  std::set<int>::iterator t_next = t; t_next++; // might erase thing pointed to be t, invalidating iterator t
658  if( *s == *t ) {
659  t = t_next;
660  continue;
661  }
662  int bb_t = *t;
663  if( m_basic_blocks[bb_s]->postdominator_ids.find(bb_t) != m_basic_blocks[bb_s]->postdominator_ids.end() )
664  m_basic_blocks[n]->Tmp_ids.erase(bb_t);
665  t = t_next;
666  }
667  }
668  }
669  unsigned num_ipdoms=0;
670  for ( int n = m_basic_blocks.size()-1; n >=0;--n) {
671  assert( m_basic_blocks[n]->Tmp_ids.size() <= 1 );
672  // if the above assert fails we have an error in either postdominator
673  // computation, the flow graph does not have a unique exit, or some other error
674  if( !m_basic_blocks[n]->Tmp_ids.empty() ) {
675  m_basic_blocks[n]->immediatepostdominator_id = *m_basic_blocks[n]->Tmp_ids.begin();
676  num_ipdoms++;
677  }
678  }
679  assert( num_ipdoms == m_basic_blocks.size()-1 );
680  // the exit node does not have an immediate post dominator, but everyone else should
