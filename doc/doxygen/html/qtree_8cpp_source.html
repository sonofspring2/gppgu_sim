88 
89  for (h = 0; h < _n; h++) {
90  for (pos = 0 ; pos < powi( _k, h ) ; ++pos ) {
91 
92  int id = h * 256 + pos;

94 
95  routerName << "router_" << h << "_" << pos;

99  routerName.str( ),

102  routerName.str("");
103  }
104  }
105 
106  // Injection & Ejection Channels
107  for ( pos = 0 ; pos < powi( _k, _n-1 ) ; ++pos ) {
108  r = _RouterIndex( _n-1, pos );
109  for ( port = 0 ; port < _k ; port++ ) {
110 
111  _routers[r]->AddInputChannel( _inject[_k*pos+port],
112  _inject_cred[_k*pos+port]);

115  _eject_cred[_k*pos+port]);
116  }
117  }
118 
119  int c;
120  for ( h = 0 ; h < _n ; ++h ) {
121  for ( pos = 0 ; pos < powi( _k, h ) ; ++pos ) {
122  for ( port = 0 ; port < _k ; port++ ) {
123 
124  r = _RouterIndex( h, pos );
125
