55 void Wavefront::Allocate( )
56 {
57 
58  int first_diag = -1;
59 
60  if(_num_requests == 0)
61 
62  // bypass allocator completely if there were no requests
63  return;
64 
65  if(_num_requests == 1) {
66 
67  // if we only had a single request, we can immediately grant it
68  _inmatch[_last_in] = _last_out;
69  _outmatch[_last_out] = _last_in;
70  first_diag = _last_in + _last_out;
71 
72  } else {
73 
74  // otherwise we have to loop through the diagonals of request matrix
75 
76  for(set<pair<int, int> >::const_reverse_iterator iter =
77  _priorities.rbegin();
78  iter != _priorities.rend(); ++iter) {
79 
80  for ( int p = 0; p < _square; ++p ) {
81  for ( int output = 0; output < _square; ++output ) {
82  int input = ( ( _pri + p ) + ( _square - output ) ) % _square;
83  if ( ( input < _inputs ) && ( output < _outputs ) &&
84  ( _inmatch[input] == -1 ) && ( _outmatch[output] == -1 ) &&
85  ( _request[input][output].label != -1 ) &&
86  ( _request[input][output].in_pri == iter->second ) &&
87  ( _request[input][output].out_pri == iter->first ) ) {
88  // Grant!
89  _inmatch[input] = output;
90  _outmatch[output] = input;
91  if(first_diag < 0) {
92  first_diag = input + output;
93  }
94  }
95  }
96  }
97  }
98  }
99 
100  _num_requests = 0;
101  _last_in = -1;
102  _last_out = -1;
103  _priorities.clear();
104 
105  assert(first_diag >= 0);
106 
107  // Round-robin the priority diagonal
108  _pri = ( ( _skip_diags ? first_diag : _pri ) + 1 ) % _square;
109 }
