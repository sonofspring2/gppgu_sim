86

100 
101  for ( int dim = 0; dim < _n; ++dim ) {
102 
103  //find the neighbor
104  left_node = _LeftNode( node, dim );
105  right_node = _RightNode( node, dim );
106 
107  //
108  // Current (N)ode
109  // (L)eft node
110  // (R)ight node
111  //
112  // L--->N<---R

121  left_input = _RightChannel( left_node, dim );
122

188 int KNCube::_LeftNode( int node, int dim )
189 {
190  int k_to_dim = powi( _k, dim );
191  int loc_in_dim = ( node / k_to_dim ) % _k;
192  int left_node;
193  // if at the left edge of the dimension, wraparound
194  if ( loc_in_dim == 0 ) {
195  left_node = node + (_k-1)*k_to_dim;
196  } else {
197  left_node = node - k_to_dim;
198  }
199 
200  return left_node;
201 }

203 int KNCube::_RightNode( int node, int dim )
204 {
205  int k_to_dim = powi( _k, dim );
206  int loc_in_dim = ( node / k_to_dim ) % _k;
207  int right_node;
208  // if at the right edge of the dimension, wraparound
209  if ( loc_in_dim == ( _k-1 ) ) {
210  right_node = node - (_k-1)*k_to_dim;
211  } else {
212  right_node = node + k_to_dim;
213  }
214 
215  return right_node;
216 }

247 
248  for ( i = 0; i < _size; ++i ) {

250 
251  // edge test
252  edge = false;
253  for ( n = 0; n < _n; ++n ) {
254  if ( ( ( node % _k ) == 0 ) ||
255  ( ( node % _k ) == _k - 1 ) ) {
256  edge = true;
257  }
258  node /= _k;
259  }
260 
261  if ( edge ) {
262  fail_nodes[i] = true;
263  } else {
264  fail_nodes[i] = false;
265  }
266  }
267 
268  for ( i = 0; i < num_fails; ++i ) {
269  j = RandomInt( _size - 1 );
270  available = false;
271 
272  for ( t = 0; ( t < _size ) && (!available); ++t ) {
273  node = ( j + t ) % _size;
274 
275  if ( !fail_nodes[node] ) {
276  // check neighbors
277  c = RandomInt( 2*_n - 1 );
278 
279  for ( n = 0; ( n < 2*_n ) && (!available); ++n ) {
280  chan = ( n + c ) % 2*_n;
281 
282  if ( chan % 1 ) {
283  available = fail_nodes[_LeftNode( node, chan/2 )];
284  } else {
285  available = fail_nodes[_RightNode( node, chan/2 )];
286  }
287  }
288  }
289 
290  if ( !available ) {
291  cout << "skipping " << node << endl;
292  }
293  }
294 
295  if ( t == _size ) {
296  Error( "Could not find another possible fault channel" );
297  }
298 
299 
300  OutChannelFault( node, chan );
301  fail_nodes[node] = true;
302 
303  for ( n = 0; ( n < _n ) && available ; ++n ) {
304  fail_nodes[_LeftNode( node, n )] = true;
305  fail_nodes[_RightNode( node, n )] = true;
306  }
307 
308  cout << "failure at node " << node << ", channel "
309  << chan << endl;
310  }
311 
312  RandomSeed( prev_seed );
313  }
