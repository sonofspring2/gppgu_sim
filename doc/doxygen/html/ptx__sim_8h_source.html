77  {
78  bits.ms = 0;
79  bits.ls = 0;

88  u8=0;

93  f64=0;
94  pred=0;
95  u32 = x;
96  }
97  operator unsigned int() { return u32;}
98  operator unsigned short() { return u16;}
99  operator unsigned char() { return u8;}
100  operator unsigned long long() { return u64;}

102  void mask_and( unsigned ms, unsigned ls )
103  {
104  bits.ms &= ms;
105  bits.ls &= ls;
106  }
107 
108  void mask_or( unsigned ms, unsigned ls )
109  {
110  bits.ms |= ms;
111  bits.ls |= ls;
112  }
113  int get_bit( unsigned bit )
114  {
115  if ( bit < 32 )
116  return(bits.ls >> bit) & 1;
117  else
118  return(bits.ms >> (bit-32)) & 1;
119  }

123  signed int s32;
124  signed long long s64;

224  void set_target( const char *sm_ver, const char *ext, const char *ext2 )
225  {
226  assert( m_valid );
227  m_sm_version_str = sm_ver;
228  check_target_extension(ext);
229  check_target_extension(ext2);
230  sscanf(sm_ver,"%u",&m_sm_version);
231  m_sm_version_valid=true;
232  }

235  unsigned extensions() const { assert(m_valid); return m_ptx_extensions; }
236 private:
237  void check_target_extension( const char *ext )
238  {
239  if( ext ) {
240  if( !strcmp(ext,"texmode_independent") )
241  m_texmode_unified=false;
242  else if( !strcmp(ext,"texmode_unified") )
243  m_texmode_unified=true;
244  else if( !strcmp(ext,"map_f64_to_f32") )
245  m_map_f64_to_f32 = true;
246  else abort();
247  }
248  }
