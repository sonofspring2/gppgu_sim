167 void EventRouter::_InternalStep( )
168 {
169  // Receive incoming flits
170  _IncomingFlits( );
171 
172  // The input pipe simulates routing delay
173  _arrival_pipe->Advance( );
174 
175  // Clear output requests
176  for ( int output = 0; output < _outputs; ++output ) {
177  _arrival_arbiter[output]->Clear( );
178  }
179 
180  // Check input arrival queues and generate
181  // requests for the outputs
182  for ( int input = 0; input < _inputs; ++input ) {
183  _ArrivalRequests( input );
184  }
185 
186  // Arbitrate between requests at outputs
187  for ( int output = 0; output < _outputs; ++output ) {
188  _ArrivalArb( output );
189  }
190 
191  for ( int input = 0; input < _inputs; ++input ) {
192  _transport_arbiter[input]->Clear( );
193  }
194 
195  _crossbar_pipe->WriteAll( 0 );
196  _credit_pipe->WriteAll( 0 );
197 
198  // Generate transport events and their
199  // requests for the inputs
200  for ( int output = 0; output < _outputs; ++output ) {
201  _TransportRequests( output );
202  }
203 
204  // Arbitrate between requests at inputs
205  for ( int input = 0; input < _inputs; ++input ) {
206  _TransportArb( input );
207  }
208 
209  _crossbar_pipe->Advance( );
210  _credit_pipe->Advance( );
211 
212  _OutputQueuing( );
213 }

247 void EventRouter::_ProcessWaiting( int output, int out_vc )
248 {
249  // out_vc just sent the transport event for out_vc,
250  // check if any events are queued on that vc. if so,
251  // generate another transport event and set the
252  // owner of the vc, otherwise set the vc to idle.
253 
254  int credits;
255 
256  tTransportEvent *tevt;
257 
258  EventNextVCState::tWaiting *w;
259 
260  if ( _output_state[output]->IsWaiting( out_vc ) ) {
261 
262  // State remains as busy, but the waiting VC takes over
263  w = _output_state[output]->PopWaiting( out_vc );
264 
265  _output_state[output]->SetState( out_vc, EventNextVCState::busy );
266  _output_state[output]->SetInput( out_vc, w->input );
267  _output_state[output]->SetInputVC( out_vc, w->vc );
268 
269  if ( w->watch ) {
270  cout << "Dequeuing waiting arrival event at " << FullName()
271  << " for flit " << w->id << endl;
272  }
273 
274  credits = _output_state[output]->GetCredits( out_vc );
275 
276  // Try to queue a transmit event for a waiting packet
277  if ( credits > 0 ) {
278  tevt = new tTransportEvent;
279  tevt->src_vc = w->vc;
280  tevt->dst_vc = out_vc;
281  tevt->input = w->input;
282  tevt->watch = w->watch; // just to have something here
283  tevt->id = w->id;
284 
285  _transport_queue[output].push( tevt );
286 
287  if ( tevt->watch ) {
288  cout << "Injecting transport event at " << FullName()
289  << " for flit " << tevt->id << endl;
290  }
291 
292  credits--;
293  _output_state[output]->SetCredits( out_vc, credits );
294  _output_state[output]->SetPresence( out_vc, w->pres - 1 );
295 
296  } else {
297  // No credits available, just store presence
298  _output_state[output]->SetPresence( out_vc, w->pres );
299  }
300 
301  delete w;
302 
303  } else {
304  // Tail sent, none waiting => VC is idle
305  _output_state[output]->SetState( out_vc, EventNextVCState::idle );
306  }

309 void EventRouter::_IncomingFlits( )
310 {
311  Flit *f;
312  Buffer *cur_buf;
313 
314  tArrivalEvent *aevt;
315 
316  _arrival_pipe->WriteAll( 0 );
317 
318  for ( int input = 0; input < _inputs; ++input ) {
319  if ( !_input_buffer[input].empty( ) ) {
320  f = _input_buffer[input].front( );
321  _input_buffer[input].pop( );
322 
323  cur_buf = _buf[input];
324  int vc = f->vc;
325 
326  cur_buf->AddFlit( vc, f );
327 
328  // Head flit arriving at idle VC
329  if ( !_active[input][vc] ) {
330 
331  if ( !f->head ) {
332  cout << "Non-head flit:" << endl;
333  cout << *f;
334  Error( "Received non-head flit at idle VC" );
335  }
336 
337  const OutputSet *route_set;
338  int out_vc, out_port;
339 
340  cur_buf->Route( vc, _rf, this, f, input );
341  route_set = cur_buf->GetRouteSet( vc );
342 
343  if ( !route_set->GetPortVC( &out_port, &out_vc ) ) {
344  Error( "The event-driven router requires routing functions with a single (port,vc) output" );
345  }
346 
347  cur_buf->SetOutput( vc, out_port, out_vc );
348  _active[input][vc] = true;
349  } else {
350  if ( f->head ) {
351  cout << *f;
352  Error( "Received head flit at non-idle VC." );
353  }
354  }
355 
356  if ( f->watch ) {
357  *gWatchOut << GetSimTime() << " | " << FullName() << " | "
358  << "Received flit at " << FullName() << ". Output port = "
359  << cur_buf->GetOutputPort( vc ) << ", output VC = "
360  << cur_buf->GetOutputVC( vc ) << endl
361  << *f;
362  }
363 
364  // In cut-through mode, only head flits generate arrivals,
365  // otherwise all flits generate
366 
367  if ( ( !_vct ) || ( _vct && f->head ) ) {
368  // Add the arrival event to a delay pipeline to
369  // account for routing/decoding time
370 
371  aevt = new tArrivalEvent;
372 
373  aevt->input = input;
374  aevt->output = cur_buf->GetOutputPort( vc );
375  aevt->src_vc = f->vc;
376  aevt->dst_vc = cur_buf->GetOutputVC( vc );
377  aevt->head = f->head;
378  aevt->tail = f->tail;
379 
380  //if ( f->head && f->tail ) {
381  // Error( "Head/tail packets not supported." );
382  //}
383 
384  aevt->watch = f->watch;
385  aevt->id = f->id;
386 
387  _arrival_pipe->Write( aevt, input );
388 
389  if ( aevt->watch ) {
390  cout << "Injected arrival event at " << FullName()
391  << " for flit " << aevt->id << endl;
392  }
393  }
394  }
395  }

413 void EventRouter::_SendTransport( int input, int output, tArrivalEvent *aevt )
414 {
415  // Try to send a transport event
416 
417  tTransportEvent *tevt;
418 
419  int credits;
420  int pres;
421 
422  credits = _output_state[output]->GetCredits( aevt->dst_vc );
423 
424  if ( credits > 0 ) {
425  // Take a credit and queue a transport event
426  credits--;
427  _output_state[output]->SetCredits( aevt->dst_vc, credits );
428 
429  tevt = new tTransportEvent;
430  tevt->src_vc = aevt->src_vc;
431  tevt->dst_vc = aevt->dst_vc;
432  tevt->input = input;
433  tevt->watch = aevt->watch;
434  tevt->id = aevt->id;
435 
436  _transport_queue[output].push( tevt );
437 
438  if ( tevt->watch ) {
439  cout << "Injecting transport event at " << FullName()
440  << " for flit " << tevt->id << endl;
441  }
442  } else {
443  if ( aevt->watch ) {
444  cout << "No credits available at " << FullName()
445  << " for flit " << aevt->id << " storing presence." << endl;
446  }
447 
448  // No credits available, just store presence
449  pres = _output_state[output]->GetPresence( aevt->dst_vc );
450  _output_state[output]->SetPresence( aevt->dst_vc, pres + 1 );
451  }

454 void EventRouter::_ArrivalArb( int output )
455 {
456  tArrivalEvent *aevt;
457  tTransportEvent *tevt;
458  Credit *c;
459 
460  EventNextVCState::tWaiting *w;
461 
462  int input;
463  int credits;
464  int pres;
465 
466  // Incoming credits can produce or enable
467  // transport events --- process them first
468 
469  if ( !_out_cred_buffer[output].empty( ) ) {
470  c = _out_cred_buffer[output].front( );
471  _out_cred_buffer[output].pop( );
472 
473  assert( c->vc.size() == 1 );
474  int vc = *c->vc.begin();
475 
476  EventNextVCState::eNextVCState state =
477  _output_state[output]->GetState( vc );
478 
479  credits = _output_state[output]->GetCredits( vc );
480  pres = _output_state[output]->GetPresence( vc );
481 
482  if ( _vct ) {
483  // In cut-through mode, only head credits indicate a change in
484  // channel state.
485 
486  if ( c->head ) {
487  credits++;
488  _output_state[output]->SetCredits( vc, credits );
489  _ProcessWaiting( output, vc );
490  }
491  } else {
492  credits++;
493  _output_state[output]->SetCredits( vc, credits );
494 
495  if ( c->tail ) { // tail flit -- recycle VC
496  if ( state != EventNextVCState::busy ) {
497  Error( "Received tail credit at non-busy output VC" );
498  }
499 
500  _ProcessWaiting( output, vc );
501  } else if ( ( state == EventNextVCState::busy ) && ( pres > 0 ) ) {
502  // Flit is present => generate transport event
503 
504  tevt = new tTransportEvent;
505  tevt->input = _output_state[output]->GetInput( vc );
506  tevt->src_vc = _output_state[output]->GetInputVC( vc );
507  tevt->dst_vc = vc;
508  tevt->watch = false;
509  tevt->id = -1;
510 
511  _transport_queue[output].push( tevt );
512 
513  pres--;
514  credits--;
515  _output_state[output]->SetPresence( vc, pres );
516  _output_state[output]->SetCredits( vc, credits );
517  }
518  }
519 
520  c->Free();
521  }
522 
523  // Now process arrival events
524 
525  _arrival_arbiter[output]->Arbitrate( );
526  input = _arrival_arbiter[output]->Match( );
527 
528  if ( input != -1 ) {
529  // Winning arrival event gets access to output
530 
531  aevt = _arrival_queue[input].front( );
532  _arrival_queue[input].pop( );
533 
534  if ( aevt->watch ) {
535  cout << "Processing arrival event at " << FullName()
536  << " for flit " << aevt->id << endl;
537  }
538 
539  EventNextVCState::eNextVCState state =
540  _output_state[output]->GetState( aevt->dst_vc );
541 
542  if ( aevt->head ) { // Head flits
543  if ( state == EventNextVCState::idle ) {
544  // Allocate the output VC and queue a transport event
545  _output_state[output]->SetState( aevt->dst_vc, EventNextVCState::busy );
546  _output_state[output]->SetInput( aevt->dst_vc, input );
547  _output_state[output]->SetInputVC( aevt->dst_vc, aevt->src_vc );
548 
549  _SendTransport( input, output, aevt );
550  } else {
551  // VC busy => queue a waiting event
552 
553  w = new EventNextVCState::tWaiting;
554 
555  w->input = input;
556  w->vc = aevt->src_vc;
557  w->id = aevt->id;
558  w->watch = aevt->watch;
559  w->pres = 1;
560 
561  _output_state[output]->PushWaiting( aevt->dst_vc, w );
562  }
563  } else {
564  if ( _vct ) {
565  Error( "Received arrival event for non-head flit in cut-through mode" );
566  }
567 
568  if ( state != EventNextVCState::busy ) {
569  cout << "flit id = " << aevt->id << endl;
570  Error( "Received a body flit at a non-busy output VC" );
571  }
572 
573  if ( ( !_output_state[output]->IsInputWaiting( aevt->dst_vc, input, aevt->src_vc ) ) &&
574  ( input == _output_state[output]->GetInput( aevt->dst_vc ) ) &&
575  ( aevt->src_vc == _output_state[output]->GetInputVC( aevt->dst_vc ) ) ) {
576  // Body flit part of the current active VC => queue transport event
577  // (the weird IsInputWaiting call handles a body flit waiting in addition
578  // to a head flit)
579 
580  _SendTransport( input, output, aevt );
581  } else {
582 
583  // VC busy with a differnet transaction => update waiting event
584  _output_state[output]->IncrWaiting( aevt->dst_vc, input, aevt->src_vc );
585  }
586  }
587 
588  delete aevt;
589  }
590 }

602 void EventRouter::_TransportArb( int input )
603 {
604  tTransportEvent *tevt;
605 
606  int output;
607  Buffer *cur_buf;
608  Flit *f;
609  Credit *c;
610 
611  if ( _transport_free[input] ) {
612  _transport_arbiter[input]->Arbitrate( );
613  output = _transport_arbiter[input]->Match( );
614  } else {
615  output = _transport_match[input];
616  }
617 
618  if ( output != -1 ) {
619  // This completes the match from input to output =>
620  // one flit can be transferred
621 
622  tevt = _transport_queue[output].front( );
623 
624  if ( tevt->watch ) {
625  cout << "Processing transport event at " << FullName()
626  << " for flit " << tevt->id << endl;
627  }
628 
629  cur_buf = _buf[input];
630  int vc = tevt->src_vc;
631 
632  // Some sanity checking first
633 
634  if ( !_active[input][vc] ) {
635  Error( "Non-active VC received grant." );
636  }
637 
638  if ( cur_buf->Empty( vc ) ) {
639  return; //Error( "Empty VC received grant." );
640  }
641 
642  if ( tevt->dst_vc != cur_buf->GetOutputVC( vc ) ) {
643  Error( "Transport event's VC does not match input's destination VC." );
644  }
645 
646  f = cur_buf->RemoveFlit( vc );
647 
648  if ( _vct ) {
649  if ( f->tail ) {
650  _transport_free[input] = true;
651  _transport_match[input] = -1;
652 
653  _transport_queue[output].pop( );
654  delete tevt;
655 
656  _active[input][vc] = false;
657  } else {
658  _transport_free[input] = false;
659  _transport_match[input] = output;
660  }
661  } else {
662  _transport_free[input] = true;
663  _transport_match[input] = -1;
664 
665  _transport_queue[output].pop( );
666  delete tevt;
667 
668  if ( f->tail ) {
669  _active[input][vc] = false;
670  }
671  }
672 
673  c = Credit::New( );
674  c->vc.insert(f->vc);
675  c->head = f->head;
676  c->tail = f->tail;
677  c->id = f->id;
678  _credit_pipe->Write( c, input );
679 
680  if ( f->watch && c->tail ) {
681  *gWatchOut << GetSimTime() << " | " << FullName() << " | "
682  << FullName() << " sending tail credit back for flit " << f->id << endl;
683  }
684 
685  // Update and forward the flit to the crossbar
686 
687  f->hops++;
688  f->vc = cur_buf->GetOutputVC( vc );
689  _crossbar_pipe->Write( f, output );
690 
691  if ( f->watch ) {
692  *gWatchOut << GetSimTime() << " | " << FullName() << " | "
693  << "Forwarding flit through crossbar at " << FullName() << ":" << endl
694  << *f;
695  }
696  }
697 }
