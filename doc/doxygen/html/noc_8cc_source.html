150 void NoC::computeEnergy(bool is_tdp)
151 {
152  //power_point_product_masks
153  double pppm_t[4] = {1,1,1,1};
154  double M=nocdynp.duty_cycle;
155  // nocdynp.executionTime=XML->sys.total_cycles/(XML->sys.target_core_clockrate*1e6);//Syed
156  //cout<<"NOC Total Cycles: "<<XML->sys.total_cycles<<endl;
157  //cout<<"NOC Clock Rate: "<<XML->sys.target_core_clockrate<<endl;
158  if (is_tdp)
159  {
160  //init stats for TDP
161  stats_t.readAc.access = M;
162  tdp_stats = stats_t;
163  if (router_exist)
164  {
165  set_pppm(pppm_t, 1*M, 1, 1, 1);//reset traffic pattern
166  router->power = router->power*pppm_t;
167  set_pppm(pppm_t, nocdynp.total_nodes, nocdynp.total_nodes, nocdynp.total_nodes, nocdynp.total_nodes);
168  power = power + router->power*pppm_t;
169  }
170  if (link_bus_exist)
171  {
172  if (nocdynp.type)
173  set_pppm(pppm_t, 1*M_traffic_pattern*M*(nocdynp.min_ports -1), nocdynp.global_linked_ports,
174  nocdynp.global_linked_ports, nocdynp.global_linked_ports);
175  //reset traffic pattern; local port do not have router links
176  else
177  set_pppm(pppm_t, 1*M_traffic_pattern*M*(nocdynp.min_ports), nocdynp.global_linked_ports,
178  nocdynp.global_linked_ports, nocdynp.global_linked_ports);//reset traffic pattern
179 
180  link_bus_tot_per_Router.power = link_bus->power*pppm_t;
181 
182  set_pppm(pppm_t, nocdynp.total_nodes,
183  nocdynp.total_nodes,
184  nocdynp.total_nodes,
185  nocdynp.total_nodes);
186  power = power + link_bus_tot_per_Router.power*pppm_t;
187 
188  }
189  }
190  else
191  {
192  rt_power.reset();
193  router->buffer.rt_power.reset();
194  router->crossbar.rt_power.reset();
195  router->arbiter.rt_power.reset();
196  router->rt_power.reset();
197  //link_bus->rt_power.reset();
198 
199  //init stats for runtime power (RTP)
200  stats_t.readAc.access = XML->sys.NoC[ithNoC].total_accesses;
201  //cout<<"NOC(computeEnergy) read accesses: "<< stats_t.readAc.access<<endl;
202  rtp_stats = stats_t;
203  set_pppm(pppm_t, 1, 0 , 0, 0);
204  if (router_exist)
205  {
206  router->buffer.rt_power.readOp.dynamic = (router->buffer.power.readOp.dynamic + router->buffer.power.writeOp.dynamic)*rtp_stats.readAc.access ;
207  router->crossbar.rt_power.readOp.dynamic = router->crossbar.power.readOp.dynamic*rtp_stats.readAc.access ;
208  router->arbiter.rt_power.readOp.dynamic = router->arbiter.power.readOp.dynamic*rtp_stats.readAc.access ;
209 
210  router->rt_power = router->rt_power + (router->buffer.rt_power + router->crossbar.rt_power + router->arbiter.rt_power)*pppm_t +
211  router->power*pppm_lkg;//TDP power must be calculated first!
212  rt_power = rt_power + router->rt_power;
213  }
214  if (link_bus_exist)
215  {
216  link_bus->rt_power.reset();
217  set_pppm(pppm_t, rtp_stats.readAc.access, 1 , 1, rtp_stats.readAc.access);
218  link_bus->rt_power = link_bus->power * pppm_t;
219  rt_power = rt_power + link_bus->rt_power;
220  }
221 
222  }

319 void NoC::set_noc_param()
320 {
321 
322  nocdynp.type = XML->sys.NoC[ithNoC].type;
323  nocdynp.clockRate =XML->sys.NoC[ithNoC].clockrate;
324  nocdynp.clockRate *= 1e6;
325  nocdynp.executionTime = XML->sys.total_cycles/(XML->sys.target_core_clockrate*1e6);
326 
327  nocdynp.flit_size = XML->sys.NoC[ithNoC].flit_bits;
328  if (nocdynp.type)
329  {
330  nocdynp.input_ports = XML->sys.NoC[ithNoC].input_ports;
331  nocdynp.output_ports = XML->sys.NoC[ithNoC].output_ports;//later minus 1
332  nocdynp.min_ports = min(nocdynp.input_ports,nocdynp.output_ports);
333  nocdynp.global_linked_ports = (nocdynp.input_ports-1) + (nocdynp.output_ports-1);
334  /*
335  * Except local i/o ports, all ports needs links( global_linked_ports);
336  * However only min_ports can be fully active simultaneously
337  * since the fewer number of ports (input or output ) is the bottleneck.
338  */
339  }
340  else
341  {
342  nocdynp.input_ports = 1;
343  nocdynp.output_ports = 1;
344  nocdynp.min_ports = min(nocdynp.input_ports,nocdynp.output_ports);
345  nocdynp.global_linked_ports = 1;
346  }
347 
348  nocdynp.virtual_channel_per_port = XML->sys.NoC[ithNoC].virtual_channel_per_port;
349  nocdynp.input_buffer_entries_per_vc = XML->sys.NoC[ithNoC].input_buffer_entries_per_vc;
350 
351  nocdynp.horizontal_nodes = XML->sys.NoC[ithNoC].horizontal_nodes;
352  nocdynp.vertical_nodes = XML->sys.NoC[ithNoC].vertical_nodes;
353  nocdynp.total_nodes = nocdynp.horizontal_nodes*nocdynp.vertical_nodes;
354  nocdynp.duty_cycle = XML->sys.NoC[ithNoC].duty_cycle;
355  nocdynp.has_global_link = XML->sys.NoC[ithNoC].has_global_link;
356  nocdynp.link_throughput = XML->sys.NoC[ithNoC].link_throughput;
357  nocdynp.link_latency = XML->sys.NoC[ithNoC].link_latency;
358  nocdynp.chip_coverage = XML->sys.NoC[ithNoC].chip_coverage;
359  nocdynp.route_over_perc = XML->sys.NoC[ithNoC].route_over_perc;
360 
361  assert (nocdynp.chip_coverage <=1);
362  assert (nocdynp.route_over_perc <=1);
363 
364  if (nocdynp.type)
365  name = "NOC";
366  else
367  name = "BUSES";
368 
369 }
