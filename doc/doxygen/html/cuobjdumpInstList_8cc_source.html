445 // increment register list and parse register
446 std::string cuobjdumpInstList::parseCuobjdumpRegister(std::string reg, bool lo, int vectorFlag)
447 {
448  std::string origReg = reg;
449  // Make sure entry list is not empty
450  if(m_entryList.size() == 0) {
451  output("ERROR: Adding a register before adding an entry.\n");
452  assert(0);
453  }
454 
455  // remove minus sign if exists
456  if(reg.substr(0,1) == "-")
457  reg = reg.substr(1, reg.size()-1);
458 
459  // if lo or hi register, get register name only (remove 'H' or 'L')
460  if(lo)
461  reg = reg.substr(0, reg.size()-1);
462 
463  // Increase register number if needed
464  // Two types of registers, R# or A#
465  if(reg.substr(0, 1) == "R") {
466  reg = reg.substr(1, reg.size()-1);
467  int regNum = atoi(reg.c_str());
468 
469  // Remove register overlap at 64
470  // TODO: is this still needed?
471  /*if(regNum > 63 && regNum < 124) {
472  regNum -= 64;
473  // Fix the origReg string
474  std::stringstream out;
475  out << ((origReg.substr(0,1)=="-") ? "-" : "")
476  << "$r" << regNum
477  << (lo ? origReg.substr(origReg.size()-3, 3) : "");
478  origReg = out.str();
479  }*/
480 
481  if(vectorFlag==64)
482  regNum += 1;
483  if(vectorFlag==128)
484  regNum += 3;
485 
486  if( m_entryList.back().m_largestRegIndex < regNum && regNum < 124 )
487  m_entryList.back().m_largestRegIndex = regNum;
488  else if( regNum == 124 )
489  m_entryList.back().m_reg124 = true;
490  } else if(reg.substr(0, 1) == "A") {
491  reg = reg.substr(1, reg.size()-1);
492  int regNum = atoi(reg.c_str());
493 
494  if( m_entryList.back().m_largestOfsRegIndex < regNum && regNum < 124 )
495  m_entryList.back().m_largestOfsRegIndex = regNum;
496  } else if(reg == "o [0x7f]") {
497  m_entryList.back().m_oreg127 = true;
498  } else if (reg.substr(0,3) == "SR_") {
499  if(reg.substr(3,3)=="Tid") {
500  origReg = "%%tid";
501  if(reg.substr(7,1)=="X") {
502  origReg += ".x";
503  }
504  }
505  } else {
506  output("ERROR: unknown register type.\n");
507  printf("\nERROR: unknown register type: ");
508  printf(reg.c_str());
509  printf("\n");
510  assert(0);
511  }
512  return origReg;
513 }
