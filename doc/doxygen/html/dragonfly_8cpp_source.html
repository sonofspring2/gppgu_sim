42 //calculate the hop count between src and estination
43 int dragonflynew_hopcnt(int src, int dest)
44 {
45  int hopcnt;
46  int dest_grp_ID, src_grp_ID;
47  int src_hopcnt, dest_hopcnt;
48  int src_intm, dest_intm;
49  int grp_output, dest_grp_output;
50  int grp_output_RID;
51 
52  int _grp_num_routers= gA;
53  int _grp_num_nodes =_grp_num_routers*gP;
54 
55  dest_grp_ID = int(dest/_grp_num_nodes);
56  src_grp_ID = int(src / _grp_num_nodes);
57 
58  //source and dest are in the same group, either 0-1 hop
59  if (dest_grp_ID == src_grp_ID) {
60  if ((int)(dest / gP) == (int)(src /gP))
61  hopcnt = 0;
62  else
63  hopcnt = 1;
64 
65  } else {
66  //source and dest are in the same group
67  //find the number of hops in the source group
68  //find the number of hops in the dest group
69  if (src_grp_ID > dest_grp_ID) {
70  grp_output = dest_grp_ID;
71  dest_grp_output = src_grp_ID - 1;
72  }
73  else {
74  grp_output = dest_grp_ID - 1;
75  dest_grp_output = src_grp_ID;
76  }
77  grp_output_RID = ((int) (grp_output / (gP))) + src_grp_ID * _grp_num_routers;
78  src_intm = grp_output_RID * gP;
79 
80  grp_output_RID = ((int) (dest_grp_output / (gP))) + dest_grp_ID * _grp_num_routers;
81  dest_intm = grp_output_RID * gP;
82 
83  //hop count in source group
84  if ((int)( src_intm / gP) == (int)( src / gP ) )
85  src_hopcnt = 0;
86  else
87  src_hopcnt = 1;
88 
89  //hop count in destination group
90  if ((int)( dest_intm / gP) == (int)( dest / gP ) ){
91  dest_hopcnt = 0;
92  }else{
93  dest_hopcnt = 1;
94  }
95 
96  //tally
97  hopcnt = src_hopcnt + 1 + dest_hopcnt;
98  }
99 
100  return hopcnt;
101 }

105 int dragonfly_port(int rID, int source, int dest){
106  int _grp_num_routers= gA;
107  int _grp_num_nodes =_grp_num_routers*gP;
108 
109  int out_port = -1;
110  int grp_ID = int(rID / _grp_num_routers);
111  int dest_grp_ID = int(dest/_grp_num_nodes);
112  int grp_output=-1;
113  int grp_RID=-1;
114  int group_dest=-1;
115 
116  //which router within this group the packet needs to go to
117  if (dest_grp_ID == grp_ID) {
118  grp_RID = int(dest / gP);
119  } else {
120  if (grp_ID > dest_grp_ID) {
121  grp_output = dest_grp_ID;
122  } else {
123  grp_output = dest_grp_ID - 1;
124  }
125  grp_RID = int(grp_output /gP) + grp_ID * _grp_num_routers;
126  group_dest = grp_RID * gP;
127  }
128 
129  //At the last hop
130  if (dest >= rID*gP && dest < (rID+1)*gP) {
131  out_port = dest%gP;
132  } else if (grp_RID == rID) {
133  //At the optical link
134  out_port = gP + (gA-1) + grp_output %(gP);
135  } else {
136  //need to route within a group
137  assert(grp_RID!=-1);
138 
139  if (rID < grp_RID){
140  out_port = (grp_RID % _grp_num_routers) - 1 + gP;
141  }else{
142  out_port = (grp_RID % _grp_num_routers) + gP;
143  }
144  }
145 
146  assert(out_port!=-1);
147  return out_port;
148 }

160 void DragonFlyNew::_ComputeSize( const Configuration &config )
161 {
162 
163  // LIMITATION
164  // -- only one dimension between the group
165  // _n == # of dimensions within a group
166  // _p == # of processors within a router
167  // inter-group ports : _p
168  // terminal ports : _p
169  // intra-group ports : 2*_p - 1
170  _p = config.GetInt( "k" ); // # of ports in each switch
171  _n = config.GetInt( "n" );
172 
173 
174  assert(_n==1);
175  // dimension
176 
177  if (_n == 1)
178  _k = _p + _p + 2*_p - 1;
179  else
180  _k = _p + _p + 2*_p;
181 
182 
183  // FIX...
184  gK = _p; gN = _n;
185 
186  // with 1 dimension, total of 2p routers per group
187  // N = 2p * p * (2p^2 + 1)
188  // a = # of routers per group
189  // = 2p (if n = 1)
190  // = p^(n) (if n > 2)
191  // g = # of groups
192  // = a * p + 1
193  // N = a * p * g;
194 
195  if (_n == 1)
196  _a = 2 * _p;
197  else
198  _a = powi(_p, _n);
199 
200  _g = _a * _p + 1;
201  _nodes = _a * _p * _g;
202 
203  _num_of_switch = _nodes / _p;
204  _channels = _num_of_switch * (_k - _p);
205  _size = _num_of_switch;
206 
207 
208 
209  gG = _g;
210  gP = _p;
211  gA = _a;
212  _grp_num_routers = gA;
213  _grp_num_nodes =_grp_num_routers*gP;
214 
215 }

468 void ugal_dragonflynew( const Router *r, const Flit *f, int in_channel,
469  OutputSet *outputs, bool inject )
470 {
471  //need 3 VCs for deadlock freedom
472 
473  assert(gNumVCs==3);
474  outputs->Clear( );
475  if(inject) {
476  int inject_vc= RandomInt(gNumVCs-1);
477  outputs->AddRange(-1, inject_vc, inject_vc);
478  return;
479  }
480 
481  //this constant biases the adaptive decision toward minimum routing
482  //negative value woudl biases it towards nonminimum routing
483  int adaptive_threshold = 30;
484 
485  int _grp_num_routers= gA;
486  int _grp_num_nodes =_grp_num_routers*gP;
487  int _network_size = gA * gP * gG;
488 
489 
490  int dest = f->dest;
491  int rID = r->GetID();
492  int grp_ID = (int) (rID / _grp_num_routers);
493  int dest_grp_ID = int(dest/_grp_num_nodes);
494 
495  int debug = f->watch;
496  int out_port = -1;
497  int out_vc = 0;
498  int min_queue_size, min_hopcnt;
499  int nonmin_queue_size, nonmin_hopcnt;
500  int intm_grp_ID;
501  int intm_rID;
502 
503  if(debug){
504  cout<<"At router "<<rID<<endl;
505  }
506  int min_router_output, nonmin_router_output;
507 
508  //at the source router, make the adaptive routing decision
509  if ( in_channel < gP ) {
510  //dest are in the same group, only use minimum routing
511  if (dest_grp_ID == grp_ID) {
512  f->ph = 2;
513  } else {
514  //select a random node
515  f->intm =RandomInt(_network_size - 1);
516  intm_grp_ID = (int)(f->intm/_grp_num_nodes);
517  if (debug){
518  cout<<"Intermediate node "<<f->intm<<" grp id "<<intm_grp_ID<<endl;
519  }
520 
521  //random intermediate are in the same group, use minimum routing
522  if(grp_ID == intm_grp_ID){
523  f->ph = 1;
524  } else {
525  //congestion metrics using queue length, obtained by GetUsedCredit()
526  min_hopcnt = dragonflynew_hopcnt(f->src, f->dest);
527  min_router_output = dragonfly_port(rID, f->src, f->dest);
528  min_queue_size = max(r->GetUsedCredit(min_router_output), 0) ;
529 
530 
531  nonmin_hopcnt = dragonflynew_hopcnt(f->src, f->intm) +
532  dragonflynew_hopcnt(f->intm,f->dest);
533  nonmin_router_output = dragonfly_port(rID, f->src, f->intm);
534  nonmin_queue_size = max(r->GetUsedCredit(nonmin_router_output), 0);
535 
536  //congestion comparison, could use hopcnt instead of 1 and 2
537  if ((1 * min_queue_size ) <= (2 * nonmin_queue_size)+adaptive_threshold ) {
538  if (debug) cout << " MINIMAL routing " << endl;
539  f->ph = 1;
540  } else {
541  f->ph = 0;
542  }
543  }
544  }
545  }
546 
547  //transition from nonminimal phase to minimal
548  if(f->ph==0){
549  intm_rID= (int)(f->intm/gP);
550  if( rID == intm_rID){
551  f->ph = 1;
552  }
553  }
554 
555  //port assignement based on the phase
556  if(f->ph == 0){
557  out_port = dragonfly_port(rID, f->src, f->intm);
558  } else if(f->ph == 1){
559  out_port = dragonfly_port(rID, f->src, f->dest);

564  }
565 
566  //optical dateline
567  if (f->ph == 1 && out_port >=gP + (gA-1)) {
568  f->ph = 2;
569  }
570 
571  //vc assignemnt based on phase
572  out_vc = f->ph;
573 
574  outputs->AddRange( out_port, out_vc, out_vc );
575 }
