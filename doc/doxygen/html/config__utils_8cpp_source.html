227 bool ParseArgs(Configuration * cf, int argc, char * * argv)
228 {
229  bool rc = false;
230 
231  //all dashed variables are ignored by the arg parser
232  for(int i = 1; i < argc; ++i) {
233  string arg(argv[i]);
234  size_t pos = arg.find('=');
235  bool dash = (argv[i][0] =='-');
236  if(pos == string::npos && !dash) {
237  // parse config file
238  cf->ParseFile( argv[i] );
239  ifstream in(argv[i]);
240  cout << "BEGIN Configuration File: " << argv[i] << endl;
241  while (!in.eof()) {
242  char c;
243  in.get(c);
244  cout << c ;
245  }
246  cout << "END Configuration File: " << argv[i] << endl;
247  rc = true;
248  } else if(pos != string::npos) {
249  // override individual parameter
250  cout << "OVERRIDE Parameter: " << arg << endl;
251  cf->ParseString(argv[i]);
252  }
253  }
254 
255  return rc;
256 }

326 vector<string> tokenize_str(string const & data)
327 {
328  vector<string> values;
329 
330  // no elements, no braces --> empty list
331  if(data.empty()) {
332  return values;
333  }
334 
335  // doesn't start with an opening brace --> treat as single element
336  // note that this element can potentially contain nested lists
337  if(data[0] != '{') {
338  values.push_back(data);
339  return values;
340  }
341 
342  size_t start = 1;
343  int nested = 0;
344 
345  size_t curr = start;
346 
347  while(string::npos != (curr = data.find_first_of("{,}", curr))) {
348 
349  if(data[curr] == '{') {
350  ++nested;
351  } else if((data[curr] == '}') && nested) {
352  --nested;
353  } else if(!nested) {
354  if(curr > start) {
355  string token = data.substr(start, curr - start);
356  values.push_back(token);
357  }
358  start = curr + 1;
359  }
360  ++curr;
361  }
362  assert(!nested);
363 
364  return values;
365 }
