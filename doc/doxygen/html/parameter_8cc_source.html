225  ram_cell_tech_type = (is_tag) ? g_ip->tag_arr_ram_cell_tech_type : g_ip->data_arr_ram_cell_tech_type;
226  is_dram = ((ram_cell_tech_type == lp_dram) || (ram_cell_tech_type == comm_dram));
227 
228  unsigned int capacity_per_die = g_ip->cache_sz / NUMBER_STACKED_DIE_LAYERS; // capacity per stacked die layer
229  const TechnologyParameter::InterconnectType & wire_local = g_tp.wire_local;
230  fully_assoc = (g_ip->fully_assoc) ? true : false;
231

233 
234 
235 
236  if (fully_assoc || pure_cam)
237  { // fully-assocative cache -- ref: CACTi 2.0 report
238  if (Ndwl != 1 || //Ndwl is fixed to 1 for FA
239  Ndcm != 1 || //Ndcm is fixed to 1 for FA
240  Nspd < 1 || Nspd > 1 || //Nspd is fixed to 1 for FA
241  Ndsam_lev_1 != 1 || //Ndsam_lev_1 is fixed to one
242  Ndsam_lev_2 != 1 || //Ndsam_lev_2 is fixed to one
243  Ndbl < 2)
244  {
245  return;
246  }
247  }
248 
249  if ((is_dram) && (!is_tag) && (Ndcm > 1))
250  {
251  return; // For a DRAM array, each bitline has its own sense-amp
252  }
253 
254  // If it's not an FA tag/data array, Ndwl should be at least two and Ndbl should be
255  // at least two because an array is assumed to have at least one mat. And a mat
256  // is formed out of two horizontal subarrays and two vertical subarrays
257  if (fully_assoc == false && (Ndwl < 1 || Ndbl < 1))
258  {
259  return;
260  }
261 
262  //***********compute row, col of an subarray
263  if (!(fully_assoc || pure_cam))//Not fully_asso nor cam
264  {
265  // if data array, let tagbits = 0
266  if (is_tag)
267  {
268  if (g_ip->specific_tag)
269  {
270  tagbits = g_ip->tag_w;
271  }
272  else
273  {
274  tagbits = ADDRESS_BITS + EXTRA_TAG_BITS - _log2(capacity_per_die) +
275  _log2(g_ip->tag_assoc*2 - 1) - _log2(g_ip->nbanks);
276 
277  }
278  tagbits = (((tagbits + 3) >> 2) << 2);
279 
280  num_r_subarray = (int)ceil(capacity_per_die / (g_ip->nbanks *
281  g_ip->block_sz * g_ip->tag_assoc * Ndbl * Nspd));// + EPSILON);
282  num_c_subarray = (int)ceil((tagbits * g_ip->tag_assoc * Nspd / Ndwl));// + EPSILON);
283  //burst_length = 1;
284  }
285  else
286  {
287  num_r_subarray = (int)ceil(capacity_per_die / (g_ip->nbanks *
288  g_ip->block_sz * g_ip->data_assoc * Ndbl * Nspd));// + EPSILON);
289  num_c_subarray = (int)ceil((8 * g_ip->block_sz * g_ip->data_assoc * Nspd / Ndwl));// + EPSILON); + EPSILON);
290  // burst_length = g_ip->block_sz * 8 / g_ip->out_w;
291  }
292 
293  if (num_r_subarray < MINSUBARRAYROWS) return;
294  if (num_r_subarray == 0) return;
295  if (num_r_subarray > MAXSUBARRAYROWS) return;
296  if (num_c_subarray < MINSUBARRAYCOLS) return;
297  if (num_c_subarray > MAXSUBARRAYCOLS) return;
298 
299  }
300 
301  else
302  {//either fully-asso or cam
303  if (pure_cam)
304  {
305  if (g_ip->specific_tag)
306  {
307  tagbits = int(ceil(g_ip->tag_w/8.0)*8);
308  }
309  else
310  {
311  tagbits = int(ceil((ADDRESS_BITS + EXTRA_TAG_BITS)/8.0)*8);
312 // cout<<"Pure CAM needs tag width to be specified"<<endl;

314  }
315  //tagbits = (((tagbits + 3) >> 2) << 2);

318  //tag_num_c_subarray = (int)(tagbits + EPSILON);
319  tag_num_c_subarray = tagbits;
320  if (tag_num_r_subarray == 0) return;
321  if (tag_num_r_subarray > MAXSUBARRAYROWS) return;
322  if (tag_num_c_subarray < MINSUBARRAYCOLS) return;
323  if (tag_num_c_subarray > MAXSUBARRAYCOLS) return;
324  num_r_subarray = tag_num_r_subarray;
325  }
326  else //fully associative
327  {
328  if (g_ip->specific_tag)
329  {
330  tagbits = g_ip->tag_w;
331  }
332  else
333  {
334  tagbits = ADDRESS_BITS + EXTRA_TAG_BITS - _log2(g_ip->block_sz);//TODO: should be the page_offset=log2(page size), but this info is not avail with CACTI, for McPAT this is no problem.
335  }
336  tagbits = (((tagbits + 3) >> 2) << 2);
337 
338  tag_num_r_subarray = (int)(capacity_per_die / (g_ip->nbanks*g_ip->block_sz * Ndbl));
339  tag_num_c_subarray = (int)ceil((tagbits * Nspd / Ndwl));// + EPSILON);
340  if (tag_num_r_subarray == 0) return;
341  if (tag_num_r_subarray > MAXSUBARRAYROWS) return;
342  if (tag_num_c_subarray < MINSUBARRAYCOLS) return;
343  if (tag_num_c_subarray > MAXSUBARRAYCOLS) return;
344 
345  data_num_r_subarray = tag_num_r_subarray;
346  data_num_c_subarray = 8 * g_ip->block_sz;
347  if (data_num_r_subarray == 0) return;
348  if (data_num_r_subarray > MAXSUBARRAYROWS) return;
349  if (data_num_c_subarray < MINSUBARRAYCOLS) return;
350  if (data_num_c_subarray > MAXSUBARRAYCOLS) return;
351  num_r_subarray = tag_num_r_subarray;
352  }
353  }
354 
355  num_subarrays = Ndwl * Ndbl;
356  //****************end of computation of row, col of an subarray
357 
358  // calculate wire parameters
359  if (fully_assoc || pure_cam)
360  {

376  g_ip->num_wr_ports);

380  }

382  {

384  {
385  cell.h = g_tp.dram.b_h;
386  cell.w = g_tp.dram.b_w;
387  }
388  else
389  {
390  cell.h = g_tp.sram.b_h + 2 * wire_local.pitch * (g_ip->num_wr_ports +
391  g_ip->num_rw_ports - 1 + g_ip->num_rd_ports);
392  cell.w = g_tp.sram.b_w + 2 * wire_local.pitch * (g_ip->num_rw_ports - 1 +
393  (g_ip->num_rd_ports - g_ip->num_se_rd_ports) +
394  g_ip->num_wr_ports) + g_tp.wire_local.pitch * g_ip->num_se_rd_ports;
395  }
396  }
397  }

399  double c_b_metal = cell.h * wire_local.C_per_um;
400  double C_bl;
401 
402  if (!(fully_assoc || pure_cam))
403  {
404  if (is_dram)
405  {
406  deg_bl_muxing = 1;
407  if (ram_cell_tech_type == comm_dram)
408  {
409  C_bl = num_r_subarray * c_b_metal;
410  V_b_sense = (g_tp.dram_cell_Vdd/2) * g_tp.dram_cell_C / (g_tp.dram_cell_C + C_bl);
411  if (V_b_sense < VBITSENSEMIN)
412  {
413  return;
414  }
415  V_b_sense = VBITSENSEMIN; // in any case, we fix sense amp input signal to a constant value
416  dram_refresh_period = 64e-3;
417  }
418  else
419  {

421  C_bl = num_r_subarray * (Cbitrow_drain_cap + c_b_metal);
422  V_b_sense = (g_tp.dram_cell_Vdd/2) * g_tp.dram_cell_C /(g_tp.dram_cell_C + C_bl);
423 
424  if (V_b_sense < VBITSENSEMIN)
425  {
426  return; //Sense amp input signal is smaller that minimum allowable sense amp input signal
427  }
428  V_b_sense = VBITSENSEMIN; // in any case, we fix sense amp input signal to a constant value
429  //v_storage_worst = g_tp.dram_cell_Vdd / 2 - VBITSENSEMIN * (g_tp.dram_cell_C + C_bl) / g_tp.dram_cell_C;
430  //dram_refresh_period = 1.1 * g_tp.dram_cell_C * v_storage_worst / g_tp.dram_cell_I_off_worst_case_len_temp;
431  dram_refresh_period = 0.9 * g_tp.dram_cell_C * VDD_STORAGE_LOSS_FRACTION_WORST * g_tp.dram_cell_Vdd / g_tp.dram_cell_I_off_worst_case_len_temp;

433  }
434  else
