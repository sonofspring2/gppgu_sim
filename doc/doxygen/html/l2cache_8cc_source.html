86 memory_partition_unit::arbitration_metadata::arbitration_metadata(const struct memory_config *config)
87 : m_last_borrower(config->m_n_sub_partition_per_memory_channel - 1),
88  m_private_credit(config->m_n_sub_partition_per_memory_channel, 0),
89  m_shared_credit(0)
90 {
91  // each sub partition get at least 1 credit for forward progress
92  // the rest is shared among with other partitions
93  m_private_credit_limit = 1;
94  m_shared_credit_limit = config->gpgpu_frfcfs_dram_sched_queue_size
95  + config->gpgpu_dram_return_queue_size
96  - (config->m_n_sub_partition_per_memory_channel - 1);
97  if (config->gpgpu_frfcfs_dram_sched_queue_size == 0
98  or config->gpgpu_dram_return_queue_size == 0)
99  {
100  m_shared_credit_limit = 0; // no limit if either of the queue has no limit in size
101  }
102  assert(m_shared_credit_limit >= 0);
103 }

105 bool memory_partition_unit::arbitration_metadata::has_credits(int inner_sub_partition_id) const
106 {
107  int spid = inner_sub_partition_id;
108  if (m_private_credit[spid] < m_private_credit_limit) {
109  return true;
110  } else if (m_shared_credit_limit == 0 || m_shared_credit < m_shared_credit_limit) {
111  return true;
112  } else {
113  return false;
114  }

117 void memory_partition_unit::arbitration_metadata::borrow_credit(int inner_sub_partition_id)
118 {
119  int spid = inner_sub_partition_id;
120  if (m_private_credit[spid] < m_private_credit_limit) {
121  m_private_credit[spid] += 1;
122  } else if (m_shared_credit_limit == 0 || m_shared_credit < m_shared_credit_limit) {
123  m_shared_credit += 1;
124  } else {
125  assert(0 && "DRAM arbitration error: Borrowing from depleted credit!");
126  }
127  m_last_borrower = spid;
128 }

130 void memory_partition_unit::arbitration_metadata::return_credit(int inner_sub_partition_id)
131 {
132  int spid = inner_sub_partition_id;
133  if (m_private_credit[spid] > 0) {
134  m_private_credit[spid] -= 1;
135  } else {
136  m_shared_credit -= 1;
137  }
138  assert((m_shared_credit >= 0) && "DRAM arbitration error: Returning more than available credits!");
139 }

195 void memory_partition_unit::dram_cycle()
196 {
197  // pop completed memory request from dram and push it to dram-to-L2 queue
198  // of the original sub partition
199  mem_fetch* mf_return = m_dram->return_queue_top();
200  if (mf_return) {
201  unsigned dest_global_spid = mf_return->get_sub_partition_id();
202  int dest_spid = global_sub_partition_id_to_local_id(dest_global_spid);
203  assert(m_sub_partition[dest_spid]->get_id() == dest_global_spid);
204  if (!m_sub_partition[dest_spid]->dram_L2_queue_full()) {
205  if( mf_return->get_access_type() == L1_WRBK_ACC ) {
206  m_sub_partition[dest_spid]->set_done(mf_return);
207  delete mf_return;
208  } else {
209  m_sub_partition[dest_spid]->dram_L2_queue_push(mf_return);
210  mf_return->set_status(IN_PARTITION_DRAM_TO_L2_QUEUE,gpu_sim_cycle+gpu_tot_sim_cycle);
211  m_arbitration_metadata.return_credit(dest_spid);
212  MEMPART_DPRINTF("mem_fetch request %p return from dram to sub partition %d\n", mf_return, dest_spid);
213  }
214  m_dram->return_queue_pop();
215  }
216  } else {
217  m_dram->return_queue_pop();
218  }
219 
220  m_dram->cycle();
221  m_dram->dram_log(SAMPLELOG);
222 
223  if( !m_dram->full() ) {
224  // L2->DRAM queue to DRAM latency queue
225  // Arbitrate among multiple L2 subpartitions
226  int last_issued_partition = m_arbitration_metadata.last_borrower();
227  for (unsigned p = 0; p < m_config->m_n_sub_partition_per_memory_channel; p++) {
228  int spid = (p + last_issued_partition + 1) % m_config->m_n_sub_partition_per_memory_channel;
229  if (!m_sub_partition[spid]->L2_dram_queue_empty() && can_issue_to_dram(spid)) {
230  mem_fetch *mf = m_sub_partition[spid]->L2_dram_queue_top();
231  m_sub_partition[spid]->L2_dram_queue_pop();
232  MEMPART_DPRINTF("Issue mem_fetch request %p from sub partition %d to dram\n", mf, spid);
233  dram_delay_t d;
234  d.req = mf;
235  d.ready_cycle = gpu_sim_cycle+gpu_tot_sim_cycle + m_config->dram_latency;
236  m_dram_latency_queue.push_back(d);
237  mf->set_status(IN_PARTITION_DRAM_LATENCY_QUEUE,gpu_sim_cycle+gpu_tot_sim_cycle);
238  m_arbitration_metadata.borrow_credit(spid);
239  break; // the DRAM should only accept one request per cycle
240  }
241  }
242  }
243 
244  // DRAM latency queue
245  if( !m_dram_latency_queue.empty() && ( (gpu_sim_cycle+gpu_tot_sim_cycle) >= m_dram_latency_queue.front().ready_cycle ) && !m_dram->full() ) {
246  mem_fetch* mf = m_dram_latency_queue.front().req;
247  m_dram_latency_queue.pop_front();
248  m_dram->push(mf);
249  }
250 }

252 void memory_partition_unit::set_done( mem_fetch *mf )
253 {
254  unsigned global_spid = mf->get_sub_partition_id();
255  int spid = global_sub_partition_id_to_local_id(global_spid);
256  assert(m_sub_partition[spid]->get_id() == global_spid);
257  if (mf->get_access_type() == L1_WRBK_ACC || mf->get_access_type() == L2_WRBK_ACC) {
258  m_arbitration_metadata.return_credit(spid);
259  MEMPART_DPRINTF("mem_fetch request %p return from dram to sub partition %d\n", mf, spid);
260  }
261  m_sub_partition[spid]->set_done(mf);
262 }

335 void memory_sub_partition::cache_cycle( unsigned cycle )
336 {
337  // L2 fill responses
338  if( !m_config->m_L2_config.disabled()) {
339  if ( m_L2cache->access_ready() && !m_L2_icnt_queue->full() ) {
340  mem_fetch *mf = m_L2cache->next_access();
341  if(mf->get_access_type() != L2_WR_ALLOC_R){ // Don't pass write allocate read request back to upper level cache
342  mf->set_reply();
343  mf->set_status(IN_PARTITION_L2_TO_ICNT_QUEUE,gpu_sim_cycle+gpu_tot_sim_cycle);
344  m_L2_icnt_queue->push(mf);
345  }else{
346  m_request_tracker.erase(mf);
347  delete mf;
348  }
349  }
350  }
351 
352  // DRAM to L2 (texture) and icnt (not texture)
353  if ( !m_dram_L2_queue->empty() ) {
354  mem_fetch *mf = m_dram_L2_queue->top();
355  if ( !m_config->m_L2_config.disabled() && m_L2cache->waiting_for_fill(mf) ) {
356  if (m_L2cache->fill_port_free()) {
357  mf->set_status(IN_PARTITION_L2_FILL_QUEUE,gpu_sim_cycle+gpu_tot_sim_cycle);
358  m_L2cache->fill(mf,gpu_sim_cycle+gpu_tot_sim_cycle);
359  m_dram_L2_queue->pop();
360  }
361  } else if ( !m_L2_icnt_queue->full() ) {
362  mf->set_status(IN_PARTITION_L2_TO_ICNT_QUEUE,gpu_sim_cycle+gpu_tot_sim_cycle);
363  m_L2_icnt_queue->push(mf);
364  m_dram_L2_queue->pop();
365  }
366  }
367 
368  // prior L2 misses inserted into m_L2_dram_queue here
369  if( !m_config->m_L2_config.disabled() )
370  m_L2cache->cycle();
371 
372  // new L2 texture accesses and/or non-texture accesses
373  if ( !m_L2_dram_queue->full() && !m_icnt_L2_queue->empty() ) {
374  mem_fetch *mf = m_icnt_L2_queue->top();
375  if ( !m_config->m_L2_config.disabled() &&
376  ( (m_config->m_L2_texure_only && mf->istexture()) || (!m_config->m_L2_texure_only) )
377  ) {
378  // L2 is enabled and access is for L2
379  bool output_full = m_L2_icnt_queue->full();
380  bool port_free = m_L2cache->data_port_free();
381  if ( !output_full && port_free ) {
382  std::list<cache_event> events;
383  enum cache_request_status status = m_L2cache->access(mf->get_addr(),mf,gpu_sim_cycle+gpu_tot_sim_cycle,events);
384  bool write_sent = was_write_sent(events);
385  bool read_sent = was_read_sent(events);
386 
387  if ( status == HIT ) {
388  if( !write_sent ) {
389  // L2 cache replies
390  assert(!read_sent);
391  if( mf->get_access_type() == L1_WRBK_ACC ) {
392  m_request_tracker.erase(mf);
393  delete mf;
394  } else {
395  mf->set_reply();
396  mf->set_status(IN_PARTITION_L2_TO_ICNT_QUEUE,gpu_sim_cycle+gpu_tot_sim_cycle);
397  m_L2_icnt_queue->push(mf);
398  }
399  m_icnt_L2_queue->pop();
400  } else {
401  assert(write_sent);
402  m_icnt_L2_queue->pop();
403  }
404  } else if ( status != RESERVATION_FAIL ) {
405  // L2 cache accepted request
406  m_icnt_L2_queue->pop();
407  } else {
408  assert(!write_sent);
409  assert(!read_sent);
410  // L2 cache lock-up: will try again next cycle
411  }
412  }
413  } else {
414  // L2 is disabled or non-texture access to texture-only L2
415  mf->set_status(IN_PARTITION_L2_TO_DRAM_QUEUE,gpu_sim_cycle+gpu_tot_sim_cycle);
416  m_L2_dram_queue->push(mf);
417  m_icnt_L2_queue->pop();
418  }
419  }
420 
421  // ROP delay queue
422  if( !m_rop.empty() && (cycle >= m_rop.front().ready_cycle) && !m_icnt_L2_queue->full() ) {
423  mem_fetch* mf = m_rop.front().req;
424  m_rop.pop();
425  m_icnt_L2_queue->push(mf);
426  mf->set_status(IN_PARTITION_ICNT_TO_L2_QUEUE,gpu_sim_cycle+gpu_tot_sim_cycle);
427  }
428 }

493 void gpgpu_sim::print_dram_stats(FILE *fout) const
494 {
495  unsigned cmd=0;
496  unsigned activity=0;
497  unsigned nop=0;
498  unsigned act=0;
499  unsigned pre=0;
500  unsigned rd=0;
501  unsigned wr=0;
502  unsigned req=0;
503  unsigned tot_cmd=0;
504  unsigned tot_nop=0;
505  unsigned tot_act=0;
506  unsigned tot_pre=0;
507  unsigned tot_rd=0;
508  unsigned tot_wr=0;
509  unsigned tot_req=0;
510 
511  for (unsigned i=0;i<m_memory_config->m_n_mem;i++){
512  m_memory_partition_unit[i]->set_dram_power_stats(cmd,activity,nop,act,pre,rd,wr,req);
513  tot_cmd+=cmd;
514  tot_nop+=nop;
515  tot_act+=act;
516  tot_pre+=pre;
517  tot_rd+=rd;
518  tot_wr+=wr;
519  tot_req+=req;
520  }
521  fprintf(fout,"gpgpu_n_dram_reads = %d\n",tot_rd );
522  fprintf(fout,"gpgpu_n_dram_writes = %d\n",tot_wr );
523  fprintf(fout,"gpgpu_n_dram_activate = %d\n",tot_act );
524  fprintf(fout,"gpgpu_n_dram_commands = %d\n",tot_cmd);
