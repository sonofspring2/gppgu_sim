54 #else
55 void ranf_start(seed) /* do this before using ranf_array */
56  long seed; /* selector for different streams */
57 #endif
58 {
59  register int t,s,j;
60  double u[KK+KK-1];
61  double ulp=(1.0/(1L<<30))/(1L<<22); /* 2 to the -52 */
62  double ss=2.0*ulp*((seed&0x3fffffff)+2);
63 
64  for (j=0;j<KK;j++) {
65  u[j]=ss; /* bootstrap the buffer */
66  ss+=ss; if (ss>=1.0) ss-=1.0-2*ulp; /* cyclic shift of 51 bits */
67  }
68  u[1]+=ulp; /* make u[1] (and only u[1]) "odd" */
69  for (s=seed&0x3fffffff,t=TT-1; t; ) {
70  for (j=KK-1;j>0;j--)
71  u[j+j]=u[j],u[j+j-1]=0.0; /* "square" */
72  for (j=KK+KK-2;j>=KK;j--) {
73  u[j-(KK-LL)]=mod_sum(u[j-(KK-LL)],u[j]);
74  u[j-KK]=mod_sum(u[j-KK],u[j]);
75  }
76  if (is_odd(s)) { /* "multiply by z" */
77  for (j=KK;j>0;j--) u[j]=u[j-1];
78  u[0]=u[KK]; /* shift the buffer cyclically */
79  u[LL]=mod_sum(u[LL],u[KK]);
80  }
81  if (s) s>>=1; else t--;
82  }
83  for (j=0;j<LL;j++) ran_u[j+KK-LL]=u[j];
84  for (;j<KK;j++) ran_u[j-LL]=u[j];
85  for (j=0;j<10;j++) ranf_array(u,KK+KK-1); /* warm things up */
86  ranf_arr_ptr=&ranf_arr_started;
87 }
