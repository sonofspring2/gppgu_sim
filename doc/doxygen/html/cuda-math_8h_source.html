271 // float to integer conversion
272 int float2int(float a, enum cudaRoundMode mode)
273 {
274  int tmp;
275  switch (mode) {
276  case cuda_math::cudaRoundZero: tmp = truncf(a); break;
277  case cuda_math::cudaRoundNearest: tmp = nearbyintf(a); break;
278  case cuda_math::cudaRoundMinInf: tmp = floorf(a); break;
279  case cuda_math::cudaRoundPosInf: tmp = ceilf(a); break;
280  default: abort();
281  }
282  return tmp;
283 }

320 // intrinsic for saturate (clamp values beyond 0 and 1)
321 float __saturatef(float a)
322 {
323  float b;
324  if (isnan(a)) b = 0.0f;
325  else if (a >= 1.0f) b = 1.0f;
326  else if (a <= 0.0f) b = 0.0f;
327  else b = a;
328  return b;
