67 void *gpgpu_sim_thread_sequential(void*)
68 {
69  // at most one kernel running at a time
70  bool done;
71  do {
72  sem_wait(&g_sim_signal_start);
73  done = true;
74  if( g_the_gpu->get_more_cta_left() ) {
75  done = false;
76  g_the_gpu->init();
77  while( g_the_gpu->active() ) {
78  g_the_gpu->cycle();
79  g_the_gpu->deadlock_check();
80  }
81  g_the_gpu->print_stats();
82  g_the_gpu->update_stats();
83  print_simulation_time();
84  }
85  sem_post(&g_sim_signal_finish);
86  } while(!done);
87  sem_post(&g_sim_signal_exit);
88  return NULL;
89 }

95 void *gpgpu_sim_thread_concurrent(void*)
96 {
97  // concurrent kernel execution simulation thread
98  do {
99  if(g_debug_execution >= 3) {
100  printf("GPGPU-Sim: *** simulation thread starting and spinning waiting for work ***\n");
101  fflush(stdout);
102  }
103  while( g_stream_manager->empty_protected() && !g_sim_done )
104  ;
105  if(g_debug_execution >= 3) {
106  printf("GPGPU-Sim: ** START simulation thread (detected work) **\n");
107  g_stream_manager->print(stdout);
108  fflush(stdout);
109  }
110  pthread_mutex_lock(&g_sim_lock);
111  g_sim_active = true;
112  pthread_mutex_unlock(&g_sim_lock);
113  bool active = false;
114  bool sim_cycles = false;
115  g_the_gpu->init();
116  do {
117  // check if a kernel has completed
118  // launch operation on device if one is pending and can be run
119 
120  // Need to break this loop when a kernel completes. This was a
121  // source of non-deterministic behaviour in GPGPU-Sim (bug 147).
122  // If another stream operation is available, g_the_gpu remains active,
123  // causing this loop to not break. If the next operation happens to be
124  // another kernel, the gpu is not re-initialized and the inter-kernel
125  // behaviour may be incorrect. Check that a kernel has finished and
126  // no other kernel is currently running.
127  if(g_stream_manager->operation(&sim_cycles) && !g_the_gpu->active())
128  break;
129 
130  if( g_the_gpu->active() ) {
131  g_the_gpu->cycle();
132  sim_cycles = true;
133  g_the_gpu->deadlock_check();
134  }
135  active=g_the_gpu->active() || !g_stream_manager->empty_protected();
136  } while( active );
137  if(g_debug_execution >= 3) {
138  printf("GPGPU-Sim: ** STOP simulation thread (no work) **\n");
139  fflush(stdout);
140  }
141  if(sim_cycles) {
142  g_the_gpu->update_stats();
143  print_simulation_time();
144  }
145  pthread_mutex_lock(&g_sim_lock);
146  g_sim_active = false;
147  pthread_mutex_unlock(&g_sim_lock);
148  } while( !g_sim_done );
149  if(g_debug_execution >= 3) {
150  printf("GPGPU-Sim: *** simulation thread exiting ***\n");
151  fflush(stdout);
152  }
153  sem_post(&g_sim_signal_exit);
154  return NULL;
155 }

230 void print_simulation_time()
231 {
232  time_t current_time, difference, d, h, m, s;
233  current_time = time((time_t *)NULL);
234  difference = MAX(current_time - g_simulation_starttime, 1);
235 
236  d = difference/(3600*24);
237  h = difference/3600 - 24*d;
238  m = difference/60 - 60*(h + 24*d);
239  s = difference - 60*(m + 60*(h + 24*d));
240 
241  fflush(stderr);
242  printf("\n\ngpgpu_simulation_time = %u days, %u hrs, %u min, %u sec (%u sec)\n",
243  (unsigned)d, (unsigned)h, (unsigned)m, (unsigned)s, (unsigned)difference );
244  printf("gpgpu_simulation_rate = %u (inst/sec)\n", (unsigned)(g_the_gpu->gpu_tot_sim_insn / difference) );
245  printf("gpgpu_simulation_rate = %u (cycle/sec)\n", (unsigned)(gpu_tot_sim_cycle / difference) );
246  fflush(stdout);
247 }
