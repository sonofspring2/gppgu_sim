101 void Stats::AddSample( double val )
102 {
103  ++_num_samples;
104  _sample_sum += val;
105 
106  // NOTE: the negation ensures that NaN values are handled correctly!
107  _max = !(val <= _max) ? val : _max;
108  _min = !(val >= _min) ? val : _min;
109 
110  //double clamp between 0 and num_bins-1
111  int b = (int)fmax(floor( val / _bin_size ), 0.0);
112  b = (b >= _num_bins) ? (_num_bins - 1) : b;
113 
114  _hist[b]++;
115 }
