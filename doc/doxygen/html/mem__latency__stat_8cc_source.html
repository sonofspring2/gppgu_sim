136 // record the total latency
137 unsigned memory_stats_t::memlatstat_done(mem_fetch *mf )
138 {
139  unsigned mf_latency;
140  mf_latency = (gpu_sim_cycle+gpu_tot_sim_cycle) - mf->get_timestamp();
141  mf_num_lat_pw++;
142  mf_tot_lat_pw += mf_latency;
143  unsigned idx = LOGB2(mf_latency);
144  assert(idx<32);
145  mf_lat_table[idx]++;
146  shader_mem_lat_log(mf->get_sid(), mf_latency);
147  mf_total_lat_table[mf->get_tlx_addr().chip][mf->get_tlx_addr().bk] += mf_latency;
148  if (mf_latency > max_mf_latency)
149  max_mf_latency = mf_latency;
150  return mf_latency;
151 }

153 void memory_stats_t::memlatstat_read_done(mem_fetch *mf)
154 {
155  if (m_memory_config->gpgpu_memlatency_stat) {
156  unsigned mf_latency = memlatstat_done(mf);
157  if (mf_latency > mf_max_lat_table[mf->get_tlx_addr().chip][mf->get_tlx_addr().bk])
158  mf_max_lat_table[mf->get_tlx_addr().chip][mf->get_tlx_addr().bk] = mf_latency;
159  unsigned icnt2sh_latency;
160  icnt2sh_latency = (gpu_tot_sim_cycle+gpu_sim_cycle) - mf->get_return_timestamp();
161  icnt2sh_lat_table[LOGB2(icnt2sh_latency)]++;
162  if (icnt2sh_latency > max_icnt2sh_latency)
163  max_icnt2sh_latency = icnt2sh_latency;
164  }

167 void memory_stats_t::memlatstat_dram_access(mem_fetch *mf)
168 {
169  unsigned dram_id = mf->get_tlx_addr().chip;
170  unsigned bank = mf->get_tlx_addr().bk;
171  if (m_memory_config->gpgpu_memlatency_stat) {
172  if (mf->get_is_write()) {
173  if ( mf->get_sid() < m_n_shader ) { //do not count L2_writebacks here
174  bankwrites[mf->get_sid()][dram_id][bank]++;
175  shader_mem_acc_log( mf->get_sid(), dram_id, bank, 'w');
176  }
177  totalbankwrites[dram_id][bank]++;
178  } else {
179  bankreads[mf->get_sid()][dram_id][bank]++;
180  shader_mem_acc_log( mf->get_sid(), dram_id, bank, 'r');
181  totalbankreads[dram_id][bank]++;
182  }
183  mem_access_type_stats[mf->get_access_type()][dram_id][bank]++;
184  }
185  if (mf->get_pc() != (unsigned)-1)
186  ptx_file_line_stats_add_dram_traffic(mf->get_pc(), mf->get_data_size());
187 }
