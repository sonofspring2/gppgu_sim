49 
50  int input_offset;

53  int lonely;
54  int lonely_cnt;
55 
56  // Count phase --- the number of requests
57  // per output is counted
58 
59  for ( int j = 0; j < _outputs; ++j ) {
60  _counts[j] = 0;
61  for ( int i = 0; i < _inputs; ++i ) {
62  _counts[j] += ( _request[i][j].label != -1 ) ? 1 : 0;
63  }
64  }
65 
66  // Request phase
67  for ( input = 0; input < _inputs; ++input ) {
68 
69  // Find the lonely output
70  output_offset = _rptr[input];
71  lonely = -1;
72  lonely_cnt = _inputs + 1;
73 
74  for ( int o = 0; o < _outputs; ++o ) {
75  output = ( o + output_offset ) % _outputs;
76 
77  if ( ( _request[input][output].label != -1 ) &&
78  ( _counts[output] < lonely_cnt ) ) {
79  lonely = output;
80  lonely_cnt = _counts[output];
81  }
82  }
83 
84  // Request the lonely output (-1 for no request)
85  _req[input] = lonely;
86  }
87 
88  // Grant phase
89  for ( output = 0; output < _outputs; ++output ) {
90  input_offset = _gptr[output];
91 
92  for ( int i = 0; i < _inputs; ++i ) {
93  input = ( i + input_offset ) % _inputs;
94 
95  if ( _req[input] == output ) {
96  // Grant!
97 
98  _inmatch[input] = output;
99  _outmatch[output] = input;
100 
101  _rptr[input] = ( _rptr[input] + 1 ) % _outputs;
102  _gptr[output] = ( _gptr[output] + 1 ) % _inputs;
103 
104  break;
105  }
106  }
