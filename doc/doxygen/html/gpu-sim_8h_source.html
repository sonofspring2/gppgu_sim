147 struct memory_config {
148  memory_config()
149  {
150  m_valid = false;
151  gpgpu_dram_timing_opt=NULL;
152  gpgpu_L2_queue_config=NULL;
153  }
154  void init()
155  {
156  assert(gpgpu_dram_timing_opt);
157  if (strchr(gpgpu_dram_timing_opt, '=') == NULL) {
158  // dram timing option in ordered variables (legacy)
159  // Disabling bank groups if their values are not specified
160  nbkgrp = 1;
161  tCCDL = 0;
162  tRTPL = 0;
163  sscanf(gpgpu_dram_timing_opt,"%d:%d:%d:%d:%d:%d:%d:%d:%d:%d:%d:%d:%d:%d",
164  &nbk,&tCCD,&tRRD,&tRCD,&tRAS,&tRP,&tRC,&CL,&WL,&tCDLR,&tWR,&nbkgrp,&tCCDL,&tRTPL);
165  } else {
166  // named dram timing options (unordered)
167  option_parser_t dram_opp = option_parser_create();
168 
169  option_parser_register(dram_opp, "nbk", OPT_UINT32, &nbk, "number of banks", "");
170  option_parser_register(dram_opp, "CCD", OPT_UINT32, &tCCD, "column to column delay", "");
171  option_parser_register(dram_opp, "RRD", OPT_UINT32, &tRRD, "minimal delay between activation of rows in different banks", "");
172  option_parser_register(dram_opp, "RCD", OPT_UINT32, &tRCD, "row to column delay", "");
173  option_parser_register(dram_opp, "RAS", OPT_UINT32, &tRAS, "time needed to activate row", "");
174  option_parser_register(dram_opp, "RP", OPT_UINT32, &tRP, "time needed to precharge (deactivate) row", "");
175  option_parser_register(dram_opp, "RC", OPT_UINT32, &tRC, "row cycle time", "");
176  option_parser_register(dram_opp, "CDLR", OPT_UINT32, &tCDLR, "switching from write to read (changes tWTR)", "");
177  option_parser_register(dram_opp, "WR", OPT_UINT32, &tWR, "last data-in to row precharge", "");
178 
179  option_parser_register(dram_opp, "CL", OPT_UINT32, &CL, "CAS latency", "");
180  option_parser_register(dram_opp, "WL", OPT_UINT32, &WL, "Write latency", "");
181 
182  //Disabling bank groups if their values are not specified
183  option_parser_register(dram_opp, "nbkgrp", OPT_UINT32, &nbkgrp, "number of bank groups", "1");
184  option_parser_register(dram_opp, "CCDL", OPT_UINT32, &tCCDL, "column to column delay between accesses to different bank groups", "0");
185  option_parser_register(dram_opp, "RTPL", OPT_UINT32, &tRTPL, "read to precharge delay between accesses to different bank groups", "0");
186 
187  option_parser_delimited_string(dram_opp, gpgpu_dram_timing_opt, "=:;");
188  fprintf(stdout, "DRAM Timing Options:\n");
189  option_parser_print(dram_opp, stdout);
190  option_parser_destroy(dram_opp);
191  }
192 
193  int nbkt = nbk/nbkgrp;
194  unsigned i;
195  for (i=0; nbkt>0; i++) {
196  nbkt = nbkt>>1;
197  }
198  bk_tag_length = i;
199  assert(nbkgrp>0 && "Number of bank groups cannot be zero");
200  tRCDWR = tRCD-(WL+1);
201  tRTW = (CL+(BL/data_command_freq_ratio)+2-WL);
202  tWTR = (WL+(BL/data_command_freq_ratio)+tCDLR);
203  tWTP = (WL+(BL/data_command_freq_ratio)+tWR);
204  dram_atom_size = BL * busW * gpu_n_mem_per_ctrlr; // burst length x bus width x # chips per partition
205 
206  assert( m_n_sub_partition_per_memory_channel > 0 );
207  assert( (nbk % m_n_sub_partition_per_memory_channel == 0)
208  && "Number of DRAM banks must be a perfect multiple of memory sub partition");
209  m_n_mem_sub_partition = m_n_mem * m_n_sub_partition_per_memory_channel;
210  fprintf(stdout, "Total number of memory sub partition = %u\n", m_n_mem_sub_partition);
211 
212  m_address_mapping.init(m_n_mem, m_n_sub_partition_per_memory_channel);
213  m_L2_config.init(&m_address_mapping);
214 
215  m_valid = true;
216  icnt_flit_size = 32; // Default 32
217  }

247  unsigned tRCDWR; //row to column delay for a write command

259  unsigned tWTP; //time to switch from write to precharge in the same bank
260  unsigned busW;

263  unsigned bk_tag_length; //number of bits that define a bank inside a bank group
264

268  unsigned dram_atom_size; // number of bytes transferred per read or write command
