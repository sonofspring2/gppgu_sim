124 BufferState::SharedBufferPolicy::SharedBufferPolicy(Configuration const & config, BufferState * parent, const string & name)
125  : BufferPolicy(config, parent, name), _shared_buf_occupancy(0)
126 {
127  int const vcs = config.GetInt( "num_vcs" );
128  int num_private_bufs = config.GetInt("private_bufs");
129  if(num_private_bufs < 0) {
130  num_private_bufs = vcs;
131  } else if(num_private_bufs == 0) {
132  num_private_bufs = 1;
133  }
134 
135  _private_buf_occupancy.resize(num_private_bufs, 0);
136 
137  _buf_size = config.GetInt("buf_size");
138  if(_buf_size < 0) {
139  _buf_size = vcs * config.GetInt("vc_buf_size");
140  }
141 
142  _private_buf_size = config.GetIntArray("private_buf_size");
143  if(_private_buf_size.empty()) {
144  int const bs = config.GetInt("private_buf_size");
145  if(bs < 0) {
146  _private_buf_size.push_back(_buf_size / num_private_bufs);
147  } else {
148  _private_buf_size.push_back(bs);
149  }
150  }
151  _private_buf_size.resize(num_private_bufs, _private_buf_size.back());
152 
153  vector<int> start_vc = config.GetIntArray("private_buf_start_vc");
154  if(start_vc.empty()) {
155  int const sv = config.GetInt("private_buf_start_vc");
156  if(sv < 0) {
157  start_vc.resize(num_private_bufs);
158  for(int i = 0; i < num_private_bufs; ++i) {
159  start_vc[i] = i * vcs / num_private_bufs;
160  }
161  } else {
162  start_vc.push_back(sv);
163  }
164  }
165 
166  vector<int> end_vc = config.GetIntArray("private_buf_end_vc");
167  if(end_vc.empty()) {
168  int const ev = config.GetInt("private_buf_end_vc");
169  if(ev < 0) {
170  end_vc.resize(num_private_bufs);
171  for(int i = 0; i < num_private_bufs; ++i) {
172  end_vc[i] = (i + 1) * vcs / num_private_bufs - 1;
173  }
174  } else {
175  end_vc.push_back(ev);
176  }
177  }
178 
179  _private_buf_vc_map.resize(vcs, -1);
180  _shared_buf_size = _buf_size;
181  for(int i = 0; i < num_private_bufs; ++i) {
182  _shared_buf_size -= _private_buf_size[i];
183  assert(start_vc[i] <= end_vc[i]);
184  for(int v = start_vc[i]; v <= end_vc[i]; ++v) {
185  assert(_private_buf_vc_map[v] < 0);
186  _private_buf_vc_map[v] = i;
187  }
188  }
189  assert(_shared_buf_size >= 0);
190 
191  _reserved_slots.resize(vcs, 0);
192 }

210 void BufferState::SharedBufferPolicy::SendingFlit(Flit const * const f)
211 {
212  int const vc = f->vc;
213  if(_reserved_slots[vc] > 0) {
214  --_reserved_slots[vc];
215  } else {

217  ++_private_buf_occupancy[i];

221  Error("Shared buffer overflow.");

395 int BufferState::FeedbackSharedBufferPolicy::_ComputeRTT(int vc, int last_rtt) const
396 {
397  // compute moving average of round-trip time
398  int rtt = _round_trip_time[vc];
399  if(rtt < 0) {
400  return last_rtt;
401  }
402  return ((rtt << _aging_scale) + last_rtt - rtt) >> _aging_scale;
403 }

405 int BufferState::FeedbackSharedBufferPolicy::_ComputeLimit(int rtt) const
406 {
407  // for every cycle that the measured average round trip time exceeded the
408  // observed minimum round trip time, reduce buffer occupancy limit by one
409  assert(_min_latency >= 0);
410  return max((_min_latency << 1) - rtt + _offset, 1);
411 }

572 void BufferState::ProcessCredit( Credit const * const c )
573 {
574  assert( c );
575 
576  set<int>::iterator iter = c->vc.begin();
577  while(iter != c->vc.end()) {
578 
579  int const vc = *iter;
580 
581  assert( ( vc >= 0 ) && ( vc < _vcs ) );
582 
583  if ( ( _wait_for_tail_credit ) &&
584  ( _in_use_by[vc] < 0 ) ) {
585  ostringstream err;
586  err << "Received credit for idle VC " << vc;
587  Error( err.str() );
588  }
589  --_occupancy;
590  if(_occupancy < 0) {
591  Error("Buffer occupancy fell below zero.");
592  }
593  --_vc_occupancy[vc];
594  if(_vc_occupancy[vc] < 0) {
595  ostringstream err;
596  err << "Buffer occupancy fell below zero for VC " << vc;
597  Error(err.str());
598  }
599  if(_wait_for_tail_credit && !_vc_occupancy[vc] && _tail_sent[vc]) {
600  assert(_in_use_by[vc] >= 0);
601  _in_use_by[vc] = -1;
602  }
603 
604 #ifdef TRACK_BUFFERS
605  assert(!_outstanding_classes[vc].empty());
606  int cl = _outstanding_classes[vc].front();
607  _outstanding_classes[vc].pop();
608  assert((cl >= 0) && (cl < _classes));
609  assert(_class_occupancy[cl] > 0);
610  --_class_occupancy[cl];
611 #endif
612 
613  _buffer_policy->FreeSlotFor(vc);
614 
615  ++iter;
616  }
617 }
