45 void KNFly::_ComputeSize( const Configuration &config )
46 {
47  _k = config.GetInt( "k" );
48  _n = config.GetInt( "n" );
49 
50  gK = _k; gN = _n;
51 
52  _nodes = powi( _k, _n );
53 
54  // n stages of k^(n-1) k x k switches
55  _size = _n*powi( _k, _n-1 );
56 
57  // n-1 sets of wiring between the stages
58  _channels = (_n-1)*_nodes;
59 }

61 void KNFly::_BuildNet( const Configuration &config )
62 {
63  ostringstream router_name;
64 
65  int per_stage = powi( _k, _n-1 );
66 
67  int node = 0;
68  int c;
69 
70  for ( int stage = 0; stage < _n; ++stage ) {
71  for ( int addr = 0; addr < per_stage; ++addr ) {
72 
73  router_name << "router_" << stage << "_" << addr;
74  _routers[node] = Router::NewRouter( config, this, router_name.str( ),
75  node, _k, _k );
76  _timed_modules.push_back(_routers[node]);
77  router_name.str("");
78 
79 #ifdef DEBUG_FLY
80  cout << "connecting node " << node << " to:" << endl;
81 #endif
82 
83  for ( int port = 0; port < _k; ++port ) {
84  // Input connections
85  if ( stage == 0 ) {
86  c = addr*_k + port;
87  _routers[node]->AddInputChannel( _inject[c], _inject_cred[c] );
88 #ifdef DEBUG_FLY
89  cout << " injection channel " << c << endl;
90 #endif
91  } else {
92  c = _InChannel( stage, addr, port );
93  _routers[node]->AddInputChannel( _chan[c], _chan_cred[c] );
94  _chan[c]->SetLatency( 1 );
95 
96 #ifdef DEBUG_FLY
97  cout << " input channel " << c << endl;
98 #endif
99  }
100 
101  // Output connections
102  if ( stage == _n - 1 ) {
103  c = addr*_k + port;
104  _routers[node]->AddOutputChannel( _eject[c], _eject_cred[c] );
105 #ifdef DEBUG_FLY
106  cout << " ejection channel " << c << endl;
107 #endif
108  } else {
109  c = _OutChannel( stage, addr, port );
110  _routers[node]->AddOutputChannel( _chan[c], _chan_cred[c] );
111 #ifdef DEBUG_FLY
112  cout << " output channel " << c << endl;
113 #endif
114  }
115  }

117  ++node;
118  }

127 int KNFly::_InChannel( int stage, int addr, int port ) const
128 {
129  int in_addr;
130  int in_port;
131 
132  // Channels are between {node,port}
133  // { d_{n-1} ... d_{n-stage} ... d_0 } and
134  // { d_{n-1} ... d_0 ... d_{n-stage} }
135 
136  int shift = powi( _k, _n-stage-1 );
137 
138  int last_digit = port;
139  int zero_digit = ( addr / shift ) % _k;
140 
141  // swap zero and last digit to get first node's address
142  in_addr = addr - zero_digit*shift + last_digit*shift;
143  in_port = zero_digit;
144 
145  return (stage-1)*_nodes + in_addr*_k + in_port;
146 }
