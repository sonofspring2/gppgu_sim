388 void gpgpu_sim_wrapper::power_metrics_calculations()
389 {
390  total_sample_count++;
391  kernel_sample_count++;
392 
393  // Current sample power
394  double sample_power = proc->rt_power.readOp.dynamic + sample_cmp_pwr[CONST_DYNAMICP];
395 
396  // Average power
397  // Previous + new + constant dynamic power (e.g., dynamic clocking power)
398  kernel_tot_power += sample_power;
399  kernel_power.avg = kernel_tot_power / kernel_sample_count;
400  for(unsigned ind=0; ind<num_pwr_cmps; ++ind){
401  kernel_cmp_pwr[ind].avg += (double)sample_cmp_pwr[ind];
402  }
403 
404  for(unsigned ind=0; ind<num_perf_counters; ++ind){
405  kernel_cmp_perf_counters[ind].avg += (double)sample_perf_counters[ind];
406  }
407 
408  // Max Power
409  if(sample_power > kernel_power.max){
410  kernel_power.max = sample_power;
411  for(unsigned ind=0; ind<num_pwr_cmps; ++ind){
412  kernel_cmp_pwr[ind].max = (double)sample_cmp_pwr[ind];
413  }
414  for(unsigned ind=0; ind<num_perf_counters; ++ind){
415  kernel_cmp_perf_counters[ind].max = sample_perf_counters[ind];
416  }
417 
418  }
419 
420  // Min Power
421  if(sample_power < kernel_power.min ||(kernel_power.min==0) ){
422  kernel_power.min = sample_power;
423  for(unsigned ind=0; ind<num_pwr_cmps; ++ind){
424  kernel_cmp_pwr[ind].min = (double)sample_cmp_pwr[ind];
425  }
426  for(unsigned ind=0; ind<num_perf_counters; ++ind){
427  kernel_cmp_perf_counters[ind].min = sample_perf_counters[ind];
428  }
429  }
430 
431  gpu_tot_power.avg = (gpu_tot_power.avg + sample_power);
432  gpu_tot_power.max = (sample_power > gpu_tot_power.max) ? sample_power : gpu_tot_power.max;
433  gpu_tot_power.min = ((sample_power < gpu_tot_power.min) || (gpu_tot_power.min == 0)) ? sample_power : gpu_tot_power.min;
434 
435 }

457 void gpgpu_sim_wrapper::update_coefficients()
458 {
459 
460  initpower_coeff[FP_INT]=proc->cores[0]->get_coefficient_fpint_insts();
461  effpower_coeff[FP_INT]=initpower_coeff[FP_INT] * p->sys.scaling_coefficients[FP_INT];
462 
463  initpower_coeff[TOT_INST]=proc->cores[0]->get_coefficient_tot_insts();
464  effpower_coeff[TOT_INST]=initpower_coeff[TOT_INST] * p->sys.scaling_coefficients[TOT_INST];
465 
466  initpower_coeff[REG_RD]=proc->cores[0]->get_coefficient_regreads_accesses()*(proc->cores[0]->exu->rf_fu_clockRate/proc->cores[0]->exu->clockRate);
467  initpower_coeff[REG_WR]=proc->cores[0]->get_coefficient_regwrites_accesses()*(proc->cores[0]->exu->rf_fu_clockRate/proc->cores[0]->exu->clockRate);
468  initpower_coeff[NON_REG_OPs]=proc->cores[0]->get_coefficient_noregfileops_accesses()*(proc->cores[0]->exu->rf_fu_clockRate/proc->cores[0]->exu->clockRate);
469  effpower_coeff[REG_RD]=initpower_coeff[REG_RD]*p->sys.scaling_coefficients[REG_RD];
470  effpower_coeff[REG_WR]=initpower_coeff[REG_WR]*p->sys.scaling_coefficients[REG_WR];
471  effpower_coeff[NON_REG_OPs]=initpower_coeff[NON_REG_OPs]*p->sys.scaling_coefficients[NON_REG_OPs];
472 
473  initpower_coeff[IC_H]=proc->cores[0]->get_coefficient_icache_hits();
474  initpower_coeff[IC_M]=proc->cores[0]->get_coefficient_icache_misses();
475  effpower_coeff[IC_H]=initpower_coeff[IC_H]*p->sys.scaling_coefficients[IC_H];
476  effpower_coeff[IC_M]=initpower_coeff[IC_M]*p->sys.scaling_coefficients[IC_M];
477 
478  initpower_coeff[CC_H]=(proc->cores[0]->get_coefficient_ccache_readhits()+proc->get_coefficient_readcoalescing());
479  initpower_coeff[CC_M]=(proc->cores[0]->get_coefficient_ccache_readmisses()+proc->get_coefficient_readcoalescing());
480  effpower_coeff[CC_H]=initpower_coeff[CC_H]*p->sys.scaling_coefficients[CC_H];
481  effpower_coeff[CC_M]=initpower_coeff[CC_M]*p->sys.scaling_coefficients[CC_M];

693 void gpgpu_sim_wrapper::detect_print_steady_state(int position, double init_val)
694 {
695  // Calculating Average
696  if(g_power_simulation_enabled && g_steady_power_levels_enabled){
697  steady_state_tacking_file = gzopen(g_steady_state_tracking_filename,"a");
698  if(position==0){
699  if(samples.size() == 0){
700  // First sample
701  sample_start = total_sample_count;
702  sample_val = proc->rt_power.readOp.dynamic;
703  init_inst_val=init_val;
704  samples.push_back(proc->rt_power.readOp.dynamic);
705  assert(samples_counter.size() == 0);
706  assert(pwr_counter.size() == 0);
707 
708  for(unsigned i=0; i<(num_perf_counters); ++i){
709  samples_counter.push_back(sample_perf_counters[i]);
710  }
711 
712  for(unsigned i=0; i<(num_pwr_cmps); ++i){
713  pwr_counter.push_back(sample_cmp_pwr[i]);
714  }
715  assert(pwr_counter.size() == (double)num_pwr_cmps);
716  assert(samples_counter.size() == (double)num_perf_counters);
717  }else{
718  // Get current average
719  double temp_avg = sample_val / (double)samples.size() ;
720 
721  if( abs(proc->rt_power.readOp.dynamic-temp_avg) < gpu_steady_power_deviation){ // Value is within threshold
722  sample_val += proc->rt_power.readOp.dynamic;
723  samples.push_back(proc->rt_power.readOp.dynamic);
724  for(unsigned i=0; i<(num_perf_counters); ++i){
725  samples_counter.at(i) += sample_perf_counters[i];
726  }
727 
728  for(unsigned i=0; i<(num_pwr_cmps); ++i){
729  pwr_counter.at(i) += sample_cmp_pwr[i];
730  }
731 
732  }else{ // Value exceeds threshold, not considered steady state
733  print_steady_state(position, init_val);
734  }
735  }
736  }else{
737  print_steady_state(position, init_val);
738  }
739  gzclose(steady_state_tacking_file);
740  }
