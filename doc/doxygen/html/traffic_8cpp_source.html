47 TrafficPattern * TrafficPattern::New(string const & pattern, int nodes,
48  Configuration const * const config)
49 {
50  string pattern_name;
51  string param_str;
52  size_t left = pattern.find_first_of('(');
53  if(left == string::npos) {
54  pattern_name = pattern;
55  } else {
56  pattern_name = pattern.substr(0, left);
57  size_t right = pattern.find_last_of(')');
58  if(right == string::npos) {
59  param_str = pattern.substr(left+1);
60  } else {
61  param_str = pattern.substr(left+1, right-left-1);
62  }
63  }
64  vector<string> params = tokenize_str(param_str);
65 
66  TrafficPattern * result = NULL;
67  if(pattern_name == "bitcomp") {
68  result = new BitCompTrafficPattern(nodes);
69  } else if(pattern_name == "transpose") {
70  result = new TransposeTrafficPattern(nodes);
71  } else if(pattern_name == "bitrev") {
72  result = new BitRevTrafficPattern(nodes);
73  } else if(pattern_name == "shuffle") {
74  result = new ShuffleTrafficPattern(nodes);
75  } else if(pattern_name == "randperm") {
76  int perm_seed = -1;
77  if(params.empty()) {
78  if(config) {
79  perm_seed = config->GetInt("perm_seed");
80  } else {
81  cout << "Error: Missing parameter for random permutation traffic pattern: " << pattern << endl;
82  exit(-1);
83  }
84  } else {
85  perm_seed = atoi(params[0].c_str());
86  }
87  result = new RandomPermutationTrafficPattern(nodes, perm_seed);
88  } else if(pattern_name == "uniform") {
89  result = new UniformRandomTrafficPattern(nodes);
90  } else if(pattern_name == "background") {
91  vector<int> excludes = tokenize_int(params[0]);
92  result = new UniformBackgroundTrafficPattern(nodes, excludes);
93  } else if(pattern_name == "diagonal") {
94  result = new DiagonalTrafficPattern(nodes);
95  } else if(pattern_name == "asymmetric") {
96  result = new AsymmetricTrafficPattern(nodes);
97  } else if(pattern_name == "taper64") {
98  result = new Taper64TrafficPattern(nodes);
99  } else if(pattern_name == "bad_dragon") {
100  bool missing_params = false;
101  int k = -1;
102  if(params.size() < 1) {
103  if(config) {
104  k = config->GetInt("k");
105  } else {
106  missing_params = true;
107  }
108  } else {
109  k = atoi(params[0].c_str());
110  }
111  int n = -1;
112  if(params.size() < 2) {
113  if(config) {
114  n = config->GetInt("n");
115  } else {
116  missing_params = true;
117  }
118  } else {
119  n = atoi(params[1].c_str());
120  }
121  if(missing_params) {
122  cout << "Error: Missing parameters for dragonfly bad permutation traffic pattern: " << pattern << endl;
123  exit(-1);
124  }
125  result = new BadPermDFlyTrafficPattern(nodes, k, n);
126  } else if((pattern_name == "tornado") || (pattern_name == "neighbor") ||
127  (pattern_name == "badperm_yarc")) {
128  bool missing_params = false;
129  int k = -1;
130  if(params.size() < 1) {
131  if(config) {
132  k = config->GetInt("k");
133  } else {
134  missing_params = true;
135  }
136  } else {
137  k = atoi(params[0].c_str());
138  }
139  int n = -1;
140  if(params.size() < 2) {
141  if(config) {
142  n = config->GetInt("n");
143  } else {
144  missing_params = true;
145  }
146  } else {
147  n = atoi(params[1].c_str());
148  }
149  int xr = -1;
150  if(params.size() < 3) {
151  if(config) {
152  xr = config->GetInt("xr");
153  } else {
154  missing_params = true;
155  }
156  } else {
157  xr = atoi(params[2].c_str());
158  }
159  if(missing_params) {

161  exit(-1);
162  }
163  if(pattern_name == "tornado") {
164  result = new TornadoTrafficPattern(nodes, k, n, xr);
165  } else if(pattern_name == "neighbor") {
166  result = new NeighborTrafficPattern(nodes, k, n, xr);
167  } else if(pattern_name == "badperm_yarc") {
168  result = new BadPermYarcTrafficPattern(nodes, k, n, xr);
169  }
170  } else if(pattern_name == "hotspot") {
171  if(params.empty()) {
172  params.push_back("-1");
173  }
174  vector<int> hotspots = tokenize_int(params[0]);
175  for(size_t i = 0; i < hotspots.size(); ++i) {
176  if(hotspots[i] < 0) {
177  hotspots[i] = RandomInt(nodes - 1);
178  }
179  }
180  vector<int> rates;
181  if(params.size() >= 2) {
182  rates = tokenize_int(params[1]);
183  rates.resize(hotspots.size(), rates.back());
184  } else {
185  rates.resize(hotspots.size(), 1);
186  }
187  result = new HotSpotTrafficPattern(nodes, hotspots, rates);
188  } else {
189  cout << "Error: Unknown traffic pattern: " << pattern << endl;
190  exit(-1);
191  }
192  return result;
193 }

239 {

242  int const mask_hi = mask_lo << _shift;

333 void RandomPermutationTrafficPattern::randomize(int seed)
334 {
335  unsigned long prev_seed = RandomIntLong( );
336  RandomSeed(seed);
337 
338  _dest.assign(_nodes, -1);
339 
340  for(int i = 0; i < _nodes; ++i) {
341  int ind = RandomInt(_nodes - 1 - i);
342 
343  int j = 0;
344  int cnt = 0;
345  while((cnt < ind) || (_dest[j] != -1)) {
346  if(_dest[j] == -1) {
347  ++cnt;
348  }
349  ++j;
350  assert(j < _nodes);
351  }
352 
353  _dest[j] = i;
354  }
355 
356  RandomSeed(prev_seed);
357 }
