107 
108  int max_index, max_pri;
109  bool wrapped;
110 
111  //MERGENOTE
112  //booksim does not have this if statement

114  if ( _requests.begin( ) != _requests.end( ) ) {
115  // A round-robin arbiter between input requests
116  p = _requests.begin( );
117  while( ( p != _requests.end( ) ) &&
118  ( p->in < _rr_ptr ) ) {
119  p++;
120  }
121 
122  max_index = -1;
123  max_pri = 0;
124 
125  wrapped = false;
126  while( (!wrapped) || ( p->in < _rr_ptr ) ) {
127  if ( p == _requests.end( ) ) {
128  if ( wrapped ) { break; }
129  // p is valid here because empty lists
130  // are skipped (above)
131  p = _requests.begin( );
132  wrapped = true;
133  }
134 
135  // check if request is the highest priority so far
136  if ( ( p->pri > max_pri ) || ( max_index == -1 ) ) {
137  max_pri = p->pri;
138  max_index = p->in;
139  }
140 
141  p++;
142  }
143 
144  _match = max_index; // -1 for no match
145  if ( _match != -1 ) {
146  _rr_ptr = ( _match + 1 ) % _inputs;
147  }
148 
149  } else {
150  _match = -1;
151  }
