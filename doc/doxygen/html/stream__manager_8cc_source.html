114 void stream_operation::do_operation( gpgpu_sim *gpu )
115 {
116  if( is_noop() )
117  return;
118 
119  assert(!m_done && m_stream);
120  if(g_debug_execution >= 3)
121  printf("GPGPU-Sim API: stream %u performing ", m_stream->get_uid() );
122  switch( m_type ) {
123  case stream_memcpy_host_to_device:
124  if(g_debug_execution >= 3)
125  printf("memcpy host-to-device\n");
126  gpu->memcpy_to_gpu(m_device_address_dst,m_host_address_src,m_cnt);
127  m_stream->record_next_done();
128  break;
129  case stream_memcpy_device_to_host:
130  if(g_debug_execution >= 3)
131  printf("memcpy device-to-host\n");
132  gpu->memcpy_from_gpu(m_host_address_dst,m_device_address_src,m_cnt);
133  m_stream->record_next_done();
134  break;
135  case stream_memcpy_device_to_device:
136  if(g_debug_execution >= 3)
137  printf("memcpy device-to-device\n");
138  gpu->memcpy_gpu_to_gpu(m_device_address_dst,m_device_address_src,m_cnt);
139  m_stream->record_next_done();
140  break;
141  case stream_memcpy_to_symbol:
142  if(g_debug_execution >= 3)
143  printf("memcpy to symbol\n");
144  gpgpu_ptx_sim_memcpy_symbol(m_symbol,m_host_address_src,m_cnt,m_offset,1,gpu);
145  m_stream->record_next_done();
146  break;
147  case stream_memcpy_from_symbol:
148  if(g_debug_execution >= 3)
149  printf("memcpy from symbol\n");
150  gpgpu_ptx_sim_memcpy_symbol(m_symbol,m_host_address_dst,m_cnt,m_offset,0,gpu);
151  m_stream->record_next_done();
152  break;
153  case stream_kernel_launch:
154  if( gpu->can_start_kernel() ) {
155  gpu->set_cache_config(m_kernel->name());
156  printf("kernel \'%s\' transfer to GPU hardware scheduler\n", m_kernel->name().c_str() );
157  if( m_sim_mode )
158  gpgpu_cuda_ptx_sim_main_func( *m_kernel );
159  else
160  gpu->launch( m_kernel );
161  }
162  break;
163  case stream_event: {
164  printf("event update\n");
165  time_t wallclock = time((time_t *)NULL);
166  m_event->update( gpu_tot_sim_cycle, wallclock );
167  m_stream->record_next_done();
168  }
169  break;
170  default:
171  abort();
172  }
173  m_done=true;
174  fflush(stdout);
175 }

223 {

226  CUstream_st *stream = m_grid_id_to_stream[grid_uid];
227  kernel_info_t *kernel = stream->front().get_kernel();
228  assert( grid_uid == kernel->get_uid() );
229  stream->record_next_done();
230  m_grid_id_to_stream.erase(grid_uid);

233  }else{
234  return false;
235  }

239 stream_operation stream_manager::front()
240 {
241  // called by gpu simulation thread
242  stream_operation result;
243  if( concurrent_streams_empty() )
244  m_service_stream_zero = true;
245  if( m_service_stream_zero ) {
246  if( !m_stream_zero.empty() ) {
247  if( !m_stream_zero.busy() ) {
248  result = m_stream_zero.next();
249  if( result.is_kernel() ) {
250  unsigned grid_id = result.get_kernel()->get_uid();
251  m_grid_id_to_stream[grid_id] = &m_stream_zero;
252  }
253  }
254  } else {
255  m_service_stream_zero = false;
256  }
257  } else {
258  std::list<struct CUstream_st*>::iterator s;
259  for( s=m_streams.begin(); s != m_streams.end(); s++) {
260  CUstream_st *stream = *s;
261  if( !stream->busy() && !stream->empty() ) {
262  result = stream->next();
263  if( result.is_kernel() ) {
264  unsigned grid_id = result.get_kernel()->get_uid();
265  m_grid_id_to_stream[grid_id] = stream;
266  }
267  break;
268  }
269  }
270  }
271  return result;
272 }

356 void stream_manager::push( stream_operation op )
357 {
358  struct CUstream_st *stream = op.get_stream();
359 
360  // block if stream 0 (or concurrency disabled) and pending concurrent operations exist
361  bool block= !stream || m_cuda_launch_blocking;
362  while(block) {
363  pthread_mutex_lock(&m_lock);
364  block = !concurrent_streams_empty();
365  pthread_mutex_unlock(&m_lock);
366  };
367 
368  pthread_mutex_lock(&m_lock);
369  if( stream && !m_cuda_launch_blocking ) {
370  stream->push(op);
371  } else {
372  op.set_stream(&m_stream_zero);
373  m_stream_zero.push(op);
374  }
375  if(g_debug_execution >= 3)
376  print_impl(stdout);
377  pthread_mutex_unlock(&m_lock);
378  if( m_cuda_launch_blocking || stream == NULL ) {
379  unsigned int wait_amount = 100;
380  unsigned int wait_cap = 100000; // 100ms
381  while( !empty() ) {
382  // sleep to prevent CPU hog by empty spin
383  // sleep time increased exponentially ensure fast response when needed
384  usleep(wait_amount);
385  wait_amount *= 2;
386  if (wait_amount > wait_cap)
387  wait_amount = wait_cap;
388  }
389  }
390 }
