216 /*
217  * The fall time of an input signal to the first stage of a circuit is
218  * assumed to be same as the fall time of the output signal of two
219  * inverters connected in series (refer: CACTI 1 Technical report,
220  * section 6.1.3)
221  */
222  double
223 Wire::signal_fall_time ()
224 {
225 
226  /* rise time of inverter 1's output */
227  double rt;
228  /* fall time of inverter 2's output */
229  double ft;
230  double timeconst;
231 
232  timeconst = (drain_C_(g_tp.min_w_nmos_, NCH, 1, 1, g_tp.cell_h_def) +
233  drain_C_(min_w_pmos, PCH, 1, 1, g_tp.cell_h_def) +
234  gate_C(min_w_pmos + g_tp.min_w_nmos_, 0)) *
235  tr_R_on(min_w_pmos, PCH, 1);
236  rt = horowitz (0, timeconst, deviceType->Vth/deviceType->Vdd, deviceType->Vth/deviceType->Vdd, FALL) / (deviceType->Vdd - deviceType->Vth);
237  timeconst = (drain_C_(g_tp.min_w_nmos_, NCH, 1, 1, g_tp.cell_h_def) +
238  drain_C_(min_w_pmos, PCH, 1, 1, g_tp.cell_h_def) +
239  gate_C(min_w_pmos + g_tp.min_w_nmos_, 0)) *
240  tr_R_on(g_tp.min_w_nmos_, NCH, 1);
241  ft = horowitz (rt, timeconst, deviceType->Vth/deviceType->Vdd, deviceType->Vth/deviceType->Vdd, RISE) / deviceType->Vth;
242  return ft;
243 }

285 double Wire::wire_cap (double len /* in m */, bool call_from_outside)
286 {
287  //TODO: this should be consistent with the wire_res in technology file
288  double sidewall, adj, tot_cap;
289  double wire_height;
290  double epsilon0 = 8.8542e-12;
291  double aspect_ratio, horiz_dielectric_constant, vert_dielectric_constant, miller_value,ild_thickness;
292 
293  switch (wire_placement)
294  {
295  case outside_mat:
296  {
297  aspect_ratio = g_tp.wire_outside_mat.aspect_ratio;
298  horiz_dielectric_constant = g_tp.wire_outside_mat.horiz_dielectric_constant;
299  vert_dielectric_constant = g_tp.wire_outside_mat.vert_dielectric_constant;
300  miller_value = g_tp.wire_outside_mat.miller_value;
301  ild_thickness = g_tp.wire_outside_mat.ild_thickness;
302  break;
303  }
304  case inside_mat :
305  {
306  aspect_ratio = g_tp.wire_inside_mat.aspect_ratio;
307  horiz_dielectric_constant = g_tp.wire_inside_mat.horiz_dielectric_constant;
308  vert_dielectric_constant = g_tp.wire_inside_mat.vert_dielectric_constant;
309  miller_value = g_tp.wire_inside_mat.miller_value;
310  ild_thickness = g_tp.wire_inside_mat.ild_thickness;
311  break;
312  }
313  default:
314  {
315  aspect_ratio = g_tp.wire_local.aspect_ratio;
316  horiz_dielectric_constant = g_tp.wire_local.horiz_dielectric_constant;
317  vert_dielectric_constant = g_tp.wire_local.vert_dielectric_constant;
318  miller_value = g_tp.wire_local.miller_value;
319  ild_thickness = g_tp.wire_local.ild_thickness;
320  break;
321  }
322  }
323 
324  if (call_from_outside)
325  {
326  wire_width *= 1e-6;
327  wire_spacing *= 1e-6;
328  }
329  wire_height = wire_width/w_scale*aspect_ratio;
330  /*
331  * assuming height does not change. wire_width = width_original*w_scale
332  * So wire_height does not change as wire width increases
333  */
334 
335 // capacitance between wires in the same level
336 // sidewall = 2*miller_value * horiz_dielectric_constant * (wire_height/wire_spacing)
337 // * epsilon0;
338 
339  sidewall = miller_value * horiz_dielectric_constant * (wire_height/wire_spacing)
340  * epsilon0;
341 
342 
343  // capacitance between wires in adjacent levels
344  //adj = miller_value * vert_dielectric_constant *w_scale * epsilon0;
345  //adj = 2*vert_dielectric_constant *wire_width/(ild_thickness*1e-6) * epsilon0;
346 
347  adj = miller_value *vert_dielectric_constant *wire_width/(ild_thickness*1e-6) * epsilon0;
348  //Change ild_thickness from micron to M
349 
350  //tot_cap = (sidewall + adj + (deviceType->C_fringe * 1e6)); //F/m
351  tot_cap = (sidewall + adj + (g_tp.fringe_cap * 1e6)); //F/m
352 
353  if (call_from_outside)
354  {
355  wire_width *= 1e6;
356  wire_spacing *= 1e6;
357  }
358  return (tot_cap*len); // (F)
359 }

392 /*
393  * Calculates the delay, power and area of the transmitter circuit.
394  *
395  * The transmitter delay is the sum of nand gate delay, inverter delay
396  * low swing nmos delay, and the wire delay
397  * (ref: Technical report 6)
398  */
399  void
400 Wire::low_swing_model()
401 {
402  double len = wire_length;
403  double beta = pmos_to_nmos_sz_ratio();
404 
405 
406  double inputrise = (in_rise_time == 0) ? signal_rise_time() : in_rise_time;
407 
408  /* Final nmos low swing driver size calculation:
409  * Try to size the driver such that the delay
410  * is less than 8FO4.
411  * If the driver size is greater than
412  * the max allowable size, assume max size for the driver.
413  * In either case, recalculate the delay using
414  * the final driver size assuming slow input with
415  * finite rise time instead of ideal step input
416  *
417  * (ref: Technical report 6)
418  */
419  double cwire = wire_cap(len); /* load capacitance */
420  double rwire = wire_res(len);
421 
422 #define RES_ADJ (8.6) // Increase in resistance due to low driving vol.
423 
424  double driver_res = (-8*g_tp.FO4/(log(0.5) * cwire))/RES_ADJ;
425  double nsize = R_to_w(driver_res, NCH);
426 
427  nsize = MIN(nsize, g_tp.max_w_nmos_);
428  nsize = MAX(nsize, g_tp.min_w_nmos_);
429 
430  if(rwire*cwire > 8*g_tp.FO4)
431  {
432  nsize = g_tp.max_w_nmos_;
433  }
434 
435  // size the inverter appropriately to minimize the transmitter delay
436  // Note - In order to minimize leakage, we are not adding a set of inverters to
437  // bring down delay. Instead, we are sizing the single gate
438  // based on the logical effort.
439  double st_eff = sqrt((2+beta/1+beta)*gate_C(nsize, 0)/(gate_C(2*g_tp.min_w_nmos_, 0)
440  + gate_C(2*min_w_pmos, 0)));
441  double req_cin = ((2+beta/1+beta)*gate_C(nsize, 0))/st_eff;
442  double inv_size = req_cin/(gate_C(min_w_pmos, 0) + gate_C(g_tp.min_w_nmos_, 0));
443  inv_size = MAX(inv_size, 1);
444 
445  /* nand gate delay */
446  double res_eq = (2 * tr_R_on(g_tp.min_w_nmos_, NCH, 1));
447  double cap_eq = 2 * drain_C_(min_w_pmos, PCH, 1, 1, g_tp.cell_h_def) +
448  drain_C_(2*g_tp.min_w_nmos_, NCH, 1, 1, g_tp.cell_h_def) +
449  gate_C(inv_size*g_tp.min_w_nmos_, 0) +
450  gate_C(inv_size*min_w_pmos, 0);
451 
452  double timeconst = res_eq * cap_eq;
453 
454  delay = horowitz(inputrise, timeconst, deviceType->Vth/deviceType->Vdd,
455  deviceType->Vth/deviceType->Vdd, RISE);
456  double temp_power = cap_eq*deviceType->Vdd*deviceType->Vdd;
457 
458  inputrise = delay / (deviceType->Vdd - deviceType->Vth); /* for the next stage */
459 
460  /* Inverter delay:
461  * The load capacitance of this inv depends on
462  * the gate capacitance of the final stage nmos
463  * transistor which in turn depends on nsize
464  */
465  res_eq = tr_R_on(inv_size*min_w_pmos, PCH, 1);
466  cap_eq = drain_C_(inv_size*min_w_pmos, PCH, 1, 1, g_tp.cell_h_def) +
467  drain_C_(inv_size*g_tp.min_w_nmos_, NCH, 1, 1, g_tp.cell_h_def) +
468  gate_C(nsize, 0);
469  timeconst = res_eq * cap_eq;
470 
471  delay += horowitz(inputrise, timeconst, deviceType->Vth/deviceType->Vdd,
472  deviceType->Vth/deviceType->Vdd, FALL);
473  temp_power += cap_eq*deviceType->Vdd*deviceType->Vdd;
474 
475 
476  transmitter.delay = delay;
477  transmitter.power.readOp.dynamic = temp_power*2; /* since it is a diff. model*/
478  transmitter.power.readOp.leakage = deviceType->Vdd *
479  (4 * cmos_Isub_leakage(g_tp.min_w_nmos_, min_w_pmos, 2, nand) +
480  4 * cmos_Isub_leakage(g_tp.min_w_nmos_, min_w_pmos, 1, inv));
481 
482  transmitter.power.readOp.gate_leakage = deviceType->Vdd *
483  (4 * cmos_Ig_leakage(g_tp.min_w_nmos_, min_w_pmos, 2, nand) +
484  4 * cmos_Ig_leakage(g_tp.min_w_nmos_, min_w_pmos, 1, inv));
485 
486  inputrise = delay / deviceType->Vth;
487 
488  /* nmos delay + wire delay */
489  cap_eq = cwire + drain_C_(nsize, NCH, 1, 1, g_tp.cell_h_def)*2 +
490  nsense * sense_amp_input_cap(); //+receiver cap
491  /*
492  * NOTE: nmos is used as both pull up and pull down transistor
493  * in the transmitter. This is because for low voltage swing, drive
494  * resistance of nmos is less than pmos
495  * (for a detailed graph ref: On-Chip Wires: Scaling and Efficiency)
496  */
497  timeconst = (tr_R_on(nsize, NCH, 1)*RES_ADJ) * (cwire +
498  drain_C_(nsize, NCH, 1, 1, g_tp.cell_h_def)*2) +
499  rwire*cwire/2 +
500  (tr_R_on(nsize, NCH, 1)*RES_ADJ + rwire) *
501  nsense * sense_amp_input_cap();
502 
503  /*
504  * since we are pre-equalizing and overdriving the low
505  * swing wires, the net time constant is less
506  * than the actual value
507  */
508  delay += horowitz(inputrise, timeconst, deviceType->Vth/deviceType->Vdd, .25, 0);
509 #define VOL_SWING .1
510  temp_power += cap_eq*VOL_SWING*.400; /* .4v is the over drive voltage */
511  temp_power *= 2; /* differential wire */
512 
513  l_wire.delay = delay - transmitter.delay;
514  l_wire.power.readOp.dynamic = temp_power - transmitter.power.readOp.dynamic;
515  l_wire.power.readOp.leakage = deviceType->Vdd*
516  (4* cmos_Isub_leakage(nsize, 0, 1, nmos));
517 
518  l_wire.power.readOp.gate_leakage = deviceType->Vdd*
519  (4* cmos_Ig_leakage(nsize, 0, 1, nmos));
520 
521  //double rt = horowitz(inputrise, timeconst, deviceType->Vth/deviceType->Vdd,
522  // deviceType->Vth/deviceType->Vdd, RISE)/deviceType->Vth;
523 
524  delay += g_tp.sense_delay;
525 
526  sense_amp.delay = g_tp.sense_delay;
527  out_rise_time = g_tp.sense_delay/(deviceType->Vth);
528  sense_amp.power.readOp.dynamic = g_tp.sense_dy_power;
529  sense_amp.power.readOp.leakage = 0; //FIXME
530  sense_amp.power.readOp.gate_leakage = 0;
531 
532  power.readOp.dynamic = temp_power + sense_amp.power.readOp.dynamic;
533  power.readOp.leakage = transmitter.power.readOp.leakage +
534  l_wire.power.readOp.leakage +
535  sense_amp.power.readOp.leakage;
536  power.readOp.gate_leakage = transmitter.power.readOp.gate_leakage +
537  l_wire.power.readOp.gate_leakage +
538  sense_amp.power.readOp.gate_leakage;
539 }

551 void Wire::delay_optimal_wire ()
552 {
553  double len = wire_length;
554  //double min_wire_width = wire_width; //m
555  double beta = pmos_to_nmos_sz_ratio();
556  double switching = 0; // switching energy
557  double short_ckt = 0; // short-circuit energy
558  double tc = 0; // time constant
559  // input cap of min sized driver
560  double input_cap = gate_C(g_tp.min_w_nmos_ + min_w_pmos, 0);
561 
562  // output parasitic capacitance of
563  // the min. sized driver
564  double out_cap = drain_C_(min_w_pmos, PCH, 1, 1, g_tp.cell_h_def) +
565  drain_C_(g_tp.min_w_nmos_, NCH, 1, 1, g_tp.cell_h_def);
566  // drive resistance
567  double out_res = (tr_R_on(g_tp.min_w_nmos_, NCH, 1) +
568  tr_R_on(min_w_pmos, PCH, 1))/2;
569  double wr = wire_res(len); //ohm
570 
571  // wire cap /m
572  double wc = wire_cap(len);
573 
574  // size the repeater such that the delay of the wire is minimum
575  double repeater_scaling = sqrt(out_res*wc/(wr*input_cap)); // len will cancel
576 
577  // calc the optimum spacing between the repeaters (m)
578 
579  repeater_spacing = sqrt(2 * out_res * (out_cap + input_cap)/
580  ((wr/len)*(wc/len)));
581  repeater_size = repeater_scaling;
582 
583  switching = (repeater_scaling * (input_cap + out_cap) +
584  repeater_spacing * (wc/len)) * deviceType->Vdd * deviceType->Vdd;
585 
586  tc = out_res * (input_cap + out_cap) +
587  out_res * wc/len * repeater_spacing/repeater_scaling +
588  wr/len * repeater_spacing * input_cap * repeater_scaling +
589  0.5 * (wr/len) * (wc/len)* repeater_spacing * repeater_spacing;
590 
591  delay = 0.693 * tc * len/repeater_spacing;
592 
593 #define Ishort_ckt 65e-6 /* across all tech Ref:Banerjee et al. {IEEE TED} */
594  short_ckt = deviceType->Vdd * g_tp.min_w_nmos_ * Ishort_ckt * 1.0986 *
595  repeater_scaling * tc;
596 
597  area.set_area((len/repeater_spacing) *
598  compute_gate_area(INV, 1, min_w_pmos * repeater_scaling,
599  g_tp.min_w_nmos_ * repeater_scaling, g_tp.cell_h_def));
600  power.readOp.dynamic = ((len/repeater_spacing)*(switching + short_ckt));
601  power.readOp.leakage = ((len/repeater_spacing)*
602  deviceType->Vdd*
603  cmos_Isub_leakage(g_tp.min_w_nmos_*repeater_scaling, beta*g_tp.min_w_nmos_*repeater_scaling, 1, inv));
604  power.readOp.gate_leakage = ((len/repeater_spacing)*
605  deviceType->Vdd*
606  cmos_Ig_leakage(g_tp.min_w_nmos_*repeater_scaling, beta*g_tp.min_w_nmos_*repeater_scaling, 1, inv));
607 }
