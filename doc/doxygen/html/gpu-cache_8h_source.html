144  void init(char * config, FuncCache status)
145  {
146  cache_status= status;
147  assert( config );
148  char rp, wp, ap, mshr_type, wap, sif;
149 
150 
151  int ntok = sscanf(config,"%u:%u:%u,%c:%c:%c:%c:%c,%c:%u:%u,%u:%u,%u",
152  &m_nset, &m_line_sz, &m_assoc, &rp, &wp, &ap, &wap,
153  &sif,&mshr_type,&m_mshr_entries,&m_mshr_max_merge,
154  &m_miss_queue_size, &m_result_fifo_entries,
155  &m_data_port_width);
156 
157  if ( ntok < 11 ) {
158  if ( !strcmp(config,"none") ) {
159  m_disabled = true;
160  return;
161  }
162  exit_parse_error();
163  }
164  switch (rp) {
165  case 'L': m_replacement_policy = LRU; break;
166  case 'F': m_replacement_policy = FIFO; break;
167  default: exit_parse_error();
168  }
169  switch (wp) {
170  case 'R': m_write_policy = READ_ONLY; break;
171  case 'B': m_write_policy = WRITE_BACK; break;
172  case 'T': m_write_policy = WRITE_THROUGH; break;
173  case 'E': m_write_policy = WRITE_EVICT; break;
174  case 'L': m_write_policy = LOCAL_WB_GLOBAL_WT; break;
175  default: exit_parse_error();
176  }
177  switch (ap) {
178  case 'm': m_alloc_policy = ON_MISS; break;
179  case 'f': m_alloc_policy = ON_FILL; break;
180  default: exit_parse_error();
181  }
182  switch (mshr_type) {
183  case 'F': m_mshr_type = TEX_FIFO; assert(ntok==13); break;
184  case 'A': m_mshr_type = ASSOC; break;
185  default: exit_parse_error();
186  }
187  m_line_sz_log2 = LOGB2(m_line_sz);
188  m_nset_log2 = LOGB2(m_nset);
189  m_valid = true;
190 
191  switch(wap){
192  case 'W': m_write_alloc_policy = WRITE_ALLOCATE; break;
193  case 'N': m_write_alloc_policy = NO_WRITE_ALLOCATE; break;
194  default: exit_parse_error();
195  }
196 
197  // detect invalid configuration
198  if (m_alloc_policy == ON_FILL and m_write_policy == WRITE_BACK) {
199  // A writeback cache with allocate-on-fill policy will inevitably lead to deadlock:
200  // The deadlock happens when an incoming cache-fill evicts a dirty
201  // line, generating a writeback request. If the memory subsystem
202  // is congested, the interconnection network may not have
203  // sufficient buffer for the writeback request. This stalls the
204  // incoming cache-fill. The stall may propagate through the memory
205  // subsystem back to the output port of the same core, creating a
206  // deadlock where the wrtieback request and the incoming cache-fill
207  // are stalling each other.
208  assert(0 && "Invalid cache configuration: Writeback cache cannot allocate new line on fill. ");
209  }
210 
211  // default: port to data array width and granularity = line size
212  if (m_data_port_width == 0) {
213  m_data_port_width = m_line_sz;
214  }
215  assert(m_line_sz % m_data_port_width == 0);
216 
217  switch(sif){
218  case 'H': m_set_index_function = FERMI_HASH_SET_FUNCTION; break;
219  case 'C': m_set_index_function = CUSTOM_SET_FUNCTION; break;
220  case 'L': m_set_index_function = LINEAR_SET_FUNCTION; break;
221  default: exit_parse_error();
222  }
223  }
224  bool disabled() const { return m_disabled;}
225  unsigned get_line_sz() const
226  {
227  assert( m_valid );
228  return m_line_sz;
229  }
230  unsigned get_num_lines() const
231  {
232  assert( m_valid );
233  return m_nset * m_assoc;
234  }

240  m_nset, m_assoc, m_line_sz );
241  }
