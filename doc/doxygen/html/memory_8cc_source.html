47 template<unsigned BSIZE> void memory_space_impl<BSIZE>::write( mem_addr_t addr, size_t length, const void *data, class ptx_thread_info *thd, const ptx_instruction *pI)
48 {
49  mem_addr_t index = addr >> m_log2_block_size;
50  if ( (addr+length) <= (index+1)*BSIZE ) {
51  // fast route for intra-block access
52  unsigned offset = addr & (BSIZE-1);
53  unsigned nbytes = length;
54  m_data[index].write(offset,nbytes,(const unsigned char*)data);
55  } else {
56  // slow route for inter-block access
57  unsigned nbytes_remain = length;
58  unsigned src_offset = 0;
59  mem_addr_t current_addr = addr;
60 
61  while (nbytes_remain > 0) {
62  unsigned offset = current_addr & (BSIZE-1);
63  mem_addr_t page = current_addr >> m_log2_block_size;
64  mem_addr_t access_limit = offset + nbytes_remain;
65  if (access_limit > BSIZE) {
66  access_limit = BSIZE;
67  }
68 
69  size_t tx_bytes = access_limit - offset;
70  m_data[page].write(offset, tx_bytes, &((const unsigned char*)data)[src_offset]);
71 
72  // advance pointers
73  src_offset += tx_bytes;
74  current_addr += tx_bytes;
75  nbytes_remain -= tx_bytes;
76  }
77  assert(nbytes_remain == 0);
78  }
79  if( !m_watchpoints.empty() ) {
80  std::map<unsigned,mem_addr_t>::iterator i;
81  for( i=m_watchpoints.begin(); i!=m_watchpoints.end(); i++ ) {
82  mem_addr_t wa = i->second;
83  if( ((addr<=wa) && ((addr+length)>wa)) || ((addr>wa) && (addr < (wa+4))) )
84  hit_watchpoint(i->first,thd,pI);
85  }
86  }

110 template<unsigned BSIZE> void memory_space_impl<BSIZE>::read( mem_addr_t addr, size_t length, void *data ) const
111 {
112  mem_addr_t index = addr >> m_log2_block_size;
113  if ((addr+length) <= (index+1)*BSIZE ) {
114  // fast route for intra-block access
115  read_single_block(index, addr, length, data);
116  } else {
117  // slow route for inter-block access
118  unsigned nbytes_remain = length;
119  unsigned dst_offset = 0;
120  mem_addr_t current_addr = addr;
121 
122  while (nbytes_remain > 0) {
123  unsigned offset = current_addr & (BSIZE-1);
124  mem_addr_t page = current_addr >> m_log2_block_size;
125  mem_addr_t access_limit = offset + nbytes_remain;
126  if (access_limit > BSIZE) {
127  access_limit = BSIZE;
128  }
129 
130  size_t tx_bytes = access_limit - offset;
131  read_single_block(page, current_addr, tx_bytes, &((unsigned char*)data)[dst_offset]);
132 
133  // advance pointers
134  dst_offset += tx_bytes;
135  current_addr += tx_bytes;
136  nbytes_remain -= tx_bytes;
137  }
