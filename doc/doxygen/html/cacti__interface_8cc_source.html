51 bool mem_array::lt(const mem_array * m1, const mem_array * m2)
52 {
53  if (m1->Nspd < m2->Nspd) return true;
54  else if (m1->Nspd > m2->Nspd) return false;
55  else if (m1->Ndwl < m2->Ndwl) return true;
56  else if (m1->Ndwl > m2->Ndwl) return false;
57  else if (m1->Ndbl < m2->Ndbl) return true;
58  else if (m1->Ndbl > m2->Ndbl) return false;
59  else if (m1->deg_bl_muxing < m2->deg_bl_muxing) return true;
60  else if (m1->deg_bl_muxing > m2->deg_bl_muxing) return false;
61  else if (m1->Ndsam_lev_1 < m2->Ndsam_lev_1) return true;
62  else if (m1->Ndsam_lev_1 > m2->Ndsam_lev_1) return false;
63  else if (m1->Ndsam_lev_2 < m2->Ndsam_lev_2) return true;
64  else return false;
65 }

69 void uca_org_t::find_delay()
70 {
71  mem_array * data_arr = data_array2;
72  mem_array * tag_arr = tag_array2;
73 
74  // check whether it is a regular cache or scratch ram
75  if (g_ip->pure_ram|| g_ip->pure_cam || g_ip->fully_assoc)
76  {
77  access_time = data_arr->access_time;
78  }
79  // Both tag and data lookup happen in parallel
80  // and the entire set is sent over the data array h-tree without
81  // waiting for the way-select signal --TODO add the corresponding
82  // power overhead Nav
83  else if (g_ip->fast_access == true)
84  {
85  access_time = MAX(tag_arr->access_time, data_arr->access_time);
86  }
87  // Tag is accessed first. On a hit, way-select signal along with the
88  // address is sent to read/write the appropriate block in the data
89  // array
90  else if (g_ip->is_seq_acc == true)
91  {
92  access_time = tag_arr->access_time + data_arr->access_time;
93  }
94  // Normal access: tag array access and data array access happen in parallel.
95  // But, the data array will wait for the way-select and transfer only the
96  // appropriate block over the h-tree.
97  else
98  {
99  access_time = MAX(tag_arr->access_time + data_arr->delay_senseamp_mux_decoder,
100  data_arr->delay_before_subarray_output_driver) +
101  data_arr->delay_from_subarray_output_driver_to_output;
102  }
103 }
