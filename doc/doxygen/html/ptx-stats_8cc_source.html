52 // defining a PTX source line = filename + line number
53 class ptx_file_line
54 {
55 public:
56  ptx_file_line(const char* s, int l) {
57  if( s == NULL )
58  st = "NULL_NAME";
59  else
60  st = s;
61  line = l;
62  }
63 
64  bool operator<(const ptx_file_line &other) const {
65  if( st == other.st ) {
66  if( line < other.line )
67  return true;
68  else
69  return false;
70  } else {
71  return st < other.st;
72  }
73  }
74 
75  bool operator==(const ptx_file_line &other) const {
76  return (line==other.line) && (st==other.st);
77  }
78 
79  std::string st;
80  unsigned line;
81 };

120 // output statistics to a file
121 void ptx_file_line_stats_write_file()
122 {
123  // check if stat collection is turned on
124  if (enable_ptx_file_line_stats == 0) return;
125 
126  ptx_file_line_stats_map_t::iterator it;
127  FILE * pfile;
128 
129  pfile = fopen(ptx_line_stats_filename, "w");
130  fprintf(pfile,"kernel line : count latency dram_traffic smem_bk_conflicts smem_warp gmem_access_generated gmem_warp exposed_latency warp_divergence\n");
131  for( it=ptx_file_line_stats_tracker.begin(); it != ptx_file_line_stats_tracker.end(); it++ ) {
132  fprintf(pfile, "%s %i : ", it->first.st.c_str(), it->first.line);
133  fprintf(pfile, "%lu ", it->second.exec_count);
134  fprintf(pfile, "%llu ", it->second.latency);
135  fprintf(pfile, "%llu ", it->second.dram_traffic);
136  fprintf(pfile, "%llu ", it->second.smem_n_way_bank_conflict_total);
137  fprintf(pfile, "%lu ", it->second.smem_warp_count);
138  fprintf(pfile, "%llu ", it->second.gmem_n_access_total);
139  fprintf(pfile, "%lu ", it->second.gmem_warp_count);
140  fprintf(pfile, "%llu ", it->second.exposed_latency);
141  fprintf(pfile, "%llu ", it->second.warp_divergence);
142  fprintf(pfile, "\n");
143  }
144  fflush(pfile);
145  fclose(pfile);
146 }

195 // a class that tracks the inflight memory instructions of a shader core
196 // and attributes exposed latency to those instructions when signaled to do so
197 class ptx_inflight_memory_insn_tracker
198 {
199 public:
200  typedef std::map<const ptx_instruction *, int> insn_count_map;
201 
202  void add_count(const ptx_instruction * pInsn, int count = 1)
203  {
204  ptx_inflight_memory_insns[pInsn] += count;
205  }
206 
207  void sub_count(const ptx_instruction * pInsn, int count = 1)
208  {
209  insn_count_map::iterator i_insncount;
210  i_insncount = ptx_inflight_memory_insns.find(pInsn);
211 
212  assert(i_insncount != ptx_inflight_memory_insns.end());
213 
214  i_insncount->second -= count;
215 
216  if (i_insncount->second <= 0) {
217  ptx_inflight_memory_insns.erase(i_insncount);
218  }
219  }
220 
221  void attribute_exposed_latency(int count = 1)
222  {
223  insn_count_map &exlat_insnmap = ptx_inflight_memory_insns;
224  insn_count_map::const_iterator i_exlatinsn;
225 
226  i_exlatinsn = exlat_insnmap.begin();
227  for (; i_exlatinsn != exlat_insnmap.end(); ++i_exlatinsn) {
228  const ptx_instruction *pInsn = i_exlatinsn->first;
229  ptx_file_line_stats& line_stats = ptx_file_line_stats_tracker[ptx_file_line(pInsn->source_file(), pInsn->source_line())];
230  line_stats.exposed_latency += count;
231  }
232  }
233 
234  insn_count_map ptx_inflight_memory_insns;
235 };
