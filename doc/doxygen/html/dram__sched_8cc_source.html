63 void frfcfs_scheduler::data_collection(unsigned int bank)
64 {
65  if (gpu_sim_cycle > row_service_timestamp[bank]) {
66  curr_row_service_time[bank] = gpu_sim_cycle - row_service_timestamp[bank];
67  if (curr_row_service_time[bank] > m_stats->max_servicetime2samerow[m_dram->id][bank])
68  m_stats->max_servicetime2samerow[m_dram->id][bank] = curr_row_service_time[bank];
69  }
70  curr_row_service_time[bank] = 0;
71  row_service_timestamp[bank] = gpu_sim_cycle;
72  if (m_stats->concurrent_row_access[m_dram->id][bank] > m_stats->max_conc_access2samerow[m_dram->id][bank]) {
73  m_stats->max_conc_access2samerow[m_dram->id][bank] = m_stats->concurrent_row_access[m_dram->id][bank];
74  }
75  m_stats->concurrent_row_access[m_dram->id][bank] = 0;
76  m_stats->num_activates[m_dram->id][bank]++;
77 }

79 dram_req_t *frfcfs_scheduler::schedule( unsigned bank, unsigned curr_row )
80 {
81  if ( m_last_row[bank] == NULL ) {
82  if ( m_queue[bank].empty() )
83  return NULL;
84 
85  std::map<unsigned,std::list<std::list<dram_req_t*>::iterator> >::iterator bin_ptr = m_bins[bank].find( curr_row );
86  if ( bin_ptr == m_bins[bank].end()) {
87  dram_req_t *req = m_queue[bank].back();
88  bin_ptr = m_bins[bank].find( req->row );
89  assert( bin_ptr != m_bins[bank].end() ); // where did the request go???
90  m_last_row[bank] = &(bin_ptr->second);
91  data_collection(bank);
92  } else {
93  m_last_row[bank] = &(bin_ptr->second);
94 
95  }
96  }
97  std::list<dram_req_t*>::iterator next = m_last_row[bank]->back();
98  dram_req_t *req = (*next);
99 
100  m_stats->concurrent_row_access[m_dram->id][bank]++;
101  m_stats->row_access[m_dram->id][bank]++;
102  m_last_row[bank]->pop_back();
103 
104  m_queue[bank].erase(next);
105  if ( m_last_row[bank]->empty() ) {
106  m_bins[bank].erase( req->row );
107  m_last_row[bank] = NULL;
108  }
109 #ifdef DEBUG_FAST_IDEAL_SCHED
110  if ( req )
111  printf("%08u : DRAM(%u) scheduling memory request to bank=%u, row=%u\n",
112  (unsigned)gpu_sim_cycle, m_dram->id, req->bk, req->row );
113 #endif
114  assert( req != NULL && m_num_pending != 0 );
115  m_num_pending--;
116 
117  return req;
118 }

128 void dram_t::scheduler_frfcfs()
129 {
130  unsigned mrq_latency;
131  frfcfs_scheduler *sched = m_frfcfs_scheduler;
132  while ( !mrqq->empty() && (!m_config->gpgpu_frfcfs_dram_sched_queue_size || sched->num_pending() < m_config->gpgpu_frfcfs_dram_sched_queue_size)) {
133  dram_req_t *req = mrqq->pop();
134 
135  // Power stats
136  //if(req->data->get_type() != READ_REPLY && req->data->get_type() != WRITE_ACK)
137  m_stats->total_n_access++;
138 
139  if(req->data->get_type() == WRITE_REQUEST){
140  m_stats->total_n_writes++;
141  }else if(req->data->get_type() == READ_REQUEST){
142  m_stats->total_n_reads++;
143  }
144 
145  req->data->set_status(IN_PARTITION_MC_INPUT_QUEUE,gpu_sim_cycle+gpu_tot_sim_cycle);
146  sched->add_req(req);
147  }
148 
149  dram_req_t *req;
150  unsigned i;
151  for ( i=0; i < m_config->nbk; i++ ) {
152  unsigned b = (i+prio)%m_config->nbk;
153  if ( !bk[b]->mrq ) {
154 
155  req = sched->schedule(b, bk[b]->curr_row);
156 
157  if ( req ) {
158  req->data->set_status(IN_PARTITION_MC_BANK_ARB_QUEUE,gpu_sim_cycle+gpu_tot_sim_cycle);
159  prio = (prio+1)%m_config->nbk;
160  bk[b]->mrq = req;
161  if (m_config->gpgpu_memlatency_stat) {
162  mrq_latency = gpu_sim_cycle + gpu_tot_sim_cycle - bk[b]->mrq->timestamp;
163  bk[b]->mrq->timestamp = gpu_tot_sim_cycle + gpu_sim_cycle;
164  m_stats->mrq_lat_table[LOGB2(mrq_latency)]++;
165  if (mrq_latency > m_stats->max_mrq_latency) {
166  m_stats->max_mrq_latency = mrq_latency;
167  }
168  }
169 
170  break;
171  }
172  }
173  }
