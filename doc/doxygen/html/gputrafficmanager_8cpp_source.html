193 {
194  assert(stype!=0);
195 
196  // Flit::FlitType packet_type = Flit::ANY_TYPE;
197  int size = packet_size; //input size
198  int pid = _cur_pid++;
199  assert(_cur_pid);
200  int packet_destination = dest;
201  bool record = false;
202  bool watch = gWatchOut && (_packets_to_watch.count(pid) > 0);
203 
204  // In GPGPUSim, the core specified the packet_type and size
205 
206 #if 0
207  if(_use_read_write[cl]){
208  if(stype > 0) {
209  if (stype == 1) {
210  packet_type = Flit::READ_REQUEST;
211  size = _read_request_size[cl];
212  } else if (stype == 2) {
213  packet_type = Flit::WRITE_REQUEST;
214  size = _write_request_size[cl];
215  } else {
216  ostringstream err;
217  err << "Invalid packet type: " << packet_type;
218  Error( err.str( ) );
219  }
220  } else {
221  PacketReplyInfo* rinfo = _repliesPending[source].front();
222  if (rinfo->type == Flit::READ_REQUEST) {//read reply
223  size = _read_reply_size[cl];
224  packet_type = Flit::READ_REPLY;
225  } else if(rinfo->type == Flit::WRITE_REQUEST) { //write reply
226  size = _write_reply_size[cl];
227  packet_type = Flit::WRITE_REPLY;
228  } else {
229  ostringstream err;
230  err << "Invalid packet type: " << rinfo->type;
231  Error( err.str( ) );

235  record = rinfo->record;
236  _repliesPending[source].pop_front();
237  rinfo->Free();
238  }
239  }
240 #endif
241 
242  if ((packet_destination <0) || (packet_destination >= _nodes)) {
243  ostringstream err;
244  err << "Incorrect packet destination " << packet_destination
245  << " for stype " << packet_type;
246  Error( err.str( ) );
247  }
248 
249  if ( ( _sim_state == running ) ||
250  ( ( _sim_state == draining ) && ( time < _drain_time ) ) ) {
251  record = _measure_stats[cl];
252  }
253 
254  int subnetwork = subnet;
255  // ((packet_type == Flit::ANY_TYPE) ?
256  // RandomInt(_subnets-1) :
257  // _subnet[packet_type]);
258 
259  if ( watch ) {
260  *gWatchOut << GetSimTime() << " | "
261  << "node" << source << " | "
262  << "Enqueuing packet " << pid
263  << " at time " << time
264  << "." << endl;
265  }
266 
267  for ( int i = 0; i < size; ++i ) {
268  Flit * f = Flit::New();
269  f->id = _cur_id++;
270  assert(_cur_id);
271  f->pid = pid;
272  f->watch = watch | (gWatchOut && (_flits_to_watch.count(f->id) > 0));
273  f->subnetwork = subnetwork;
274  f->src = source;
275  f->ctime = time;
276  f->record = record;
277  f->cl = cl;
278  f->data = data;
279 
280  _total_in_flight_flits[f->cl].insert(make_pair(f->id, f));
281  if(record) {
282  _measured_in_flight_flits[f->cl].insert(make_pair(f->id, f));
283  }
284 
285  if(gTrace){
286  cout<<"New Flit "<<f->src<<endl;
287  }
288  f->type = packet_type;
289 
290  if ( i == 0 ) { // Head flit
291  f->head = true;
292  //packets are only generated to nodes smaller or equal to limit
293  f->dest = packet_destination;
294  } else {
295  f->head = false;
296  f->dest = -1;
297  }
298  switch( _pri_type ) {
299  case class_based:
300  f->pri = _class_priority[cl];
301  assert(f->pri >= 0);
302  break;
303  case age_based:
304  f->pri = numeric_limits<int>::max() - time;
305  assert(f->pri >= 0);
306  break;
307  case sequence_based:
308  f->pri = numeric_limits<int>::max() - _packet_seq_no[source];
309  assert(f->pri >= 0);
310  break;
311  default:
312  f->pri = 0;
313  }
314  if ( i == ( size - 1 ) ) { // Tail flit
315  f->tail = true;
316  } else {
317  f->tail = false;
318  }
319 
320  f->vc = -1;
321 
322  if ( f->watch ) {
323  *gWatchOut << GetSimTime() << " | "
324  << "node" << source << " | "
325  << "Enqueuing flit " << f->id
326  << " (packet " << f->pid
327  << ") at time " << time
328  << "." << endl;
329  }
330 
331  _input_queue[subnet][source][cl].push_back( f );
332  }

335 void GPUTrafficManager::_Step()
336 {
337  bool flits_in_flight = false;
338  for(int c = 0; c < _classes; ++c) {
339  flits_in_flight |= !_total_in_flight_flits[c].empty();
340  }
341  if(flits_in_flight && (_deadlock_timer++ >= _deadlock_warn_timeout)){
342  _deadlock_timer = 0;
343  cout << "WARNING: Possible network deadlock.\n";
344  }
345 
346  vector<map<int, Flit *> > flits(_subnets);
347 
348  for ( int subnet = 0; subnet < _subnets; ++subnet ) {
349  for ( int n = 0; n < _nodes; ++n ) {
350  Flit * const f = _net[subnet]->ReadFlit( n );
351  if ( f ) {
352  if(f->watch) {
353  *gWatchOut << GetSimTime() << " | "
354  << "node" << n << " | "
355  << "Ejecting flit " << f->id
356  << " (packet " << f->pid << ")"
357  << " from VC " << f->vc
358  << "." << endl;
359  }
360  g_icnt_interface->WriteOutBuffer(subnet, n, f);
361  }
362 
363  g_icnt_interface->Transfer2BoundaryBuffer(subnet, n);
364  Flit* const ejected_flit = g_icnt_interface->GetEjectedFlit(subnet, n);
365  if (ejected_flit) {
366  if(ejected_flit->head)
367  assert(ejected_flit->dest == n);
368  if(ejected_flit->watch) {
369  *gWatchOut << GetSimTime() << " | "
370  << "node" << n << " | "
371  << "Ejected flit " << ejected_flit->id
372  << " (packet " << ejected_flit->pid
373  << " VC " << ejected_flit->vc << ")"
374  << "from ejection buffer." << endl;
375  }
376  flits[subnet].insert(make_pair(n, ejected_flit));
377  if((_sim_state == warming_up) || (_sim_state == running)) {
378  ++_accepted_flits[ejected_flit->cl][n];
379  if(ejected_flit->tail) {
380  ++_accepted_packets[ejected_flit->cl][n];
381  }
382  }
383  }
384 
385  // Processing the credit From the network
386  Credit * const c = _net[subnet]->ReadCredit( n );
387  if ( c ) {
388 #ifdef TRACK_FLOWS
389  for(set<int>::const_iterator iter = c->vc.begin(); iter != c->vc.end(); ++iter) {
390  int const vc = *iter;
391  assert(!_outstanding_classes[n][subnet][vc].empty());
392  int cl = _outstanding_classes[n][subnet][vc].front();
393  _outstanding_classes[n][subnet][vc].pop();
394  assert(_outstanding_credits[cl][subnet][n] > 0);
395  --_outstanding_credits[cl][subnet][n];
396  }
397 #endif
398  _buf_states[n][subnet]->ProcessCredit(c);
399  c->Free();
400  }
401  }
402  _net[subnet]->ReadInputs( );
403  }
404 
405 // GPGPUSim will generate/inject packets from interconnection interface
406 #if 0
407  if ( !_empty_network ) {
408  _Inject();
409  }
410 #endif
411 
412  for(int subnet = 0; subnet < _subnets; ++subnet) {
413 
414  for(int n = 0; n < _nodes; ++n) {
415 
416  Flit * f = NULL;
417 
418  BufferState * const dest_buf = _buf_states[n][subnet];
419 
420  int const last_class = _last_class[n][subnet];
421 
422  int class_limit = _classes;
423 
424  if(_hold_switch_for_packet) {
425  list<Flit *> const & pp = _input_queue[subnet][n][last_class];
426  if(!pp.empty() && !pp.front()->head &&
427  !dest_buf->IsFullFor(pp.front()->vc)) {
428  f = pp.front();
429  assert(f->vc == _last_vc[n][subnet][last_class]);
430 
431  // if we're holding the connection, we don't need to check that class
432  // again in the for loop
433  --class_limit;
434  }
435  }
436 
437  for(int i = 1; i <= class_limit; ++i) {
438 
439  int const c = (last_class + i) % _classes;
440 
441  list<Flit *> const & pp = _input_queue[subnet][n][c];
442 
443  if(pp.empty()) {
444  continue;
445  }
446 
447  Flit * const cf = pp.front();
448  assert(cf);
449  assert(cf->cl == c);
450 
451  assert(cf->subnetwork == subnet);
452 
453  if(f && (f->pri >= cf->pri)) {
454  continue;
455  }
456 
457  if(cf->head && cf->vc == -1) { // Find first available VC
458 
459  OutputSet route_set;
460  _rf(NULL, cf, -1, &route_set, true);
461  set<OutputSet::sSetElement> const & os = route_set.GetSet();
462  assert(os.size() == 1);
463  OutputSet::sSetElement const & se = *os.begin();
464  assert(se.output_port == -1);
465  int vc_start = se.vc_start;
466  int vc_end = se.vc_end;
467  int vc_count = vc_end - vc_start + 1;
468  if(_noq) {
469  assert(_lookahead_routing);
470  const FlitChannel * inject = _net[subnet]->GetInject(n);
471  const Router * router = inject->GetSink();
472  assert(router);
473  int in_channel = inject->GetSinkPort();
474 
475  // NOTE: Because the lookahead is not for injection, but for the
476  // first hop, we have to temporarily set cf's VC to be non-negative
477  // in order to avoid seting of an assertion in the routing function.
478  cf->vc = vc_start;
479  _rf(router, cf, in_channel, &cf->la_route_set, false);
480  cf->vc = -1;
481 
482  if(cf->watch) {
483  *gWatchOut << GetSimTime() << " | "
484  << "node" << n << " | "
485  << "Generating lookahead routing info for flit " << cf->id
486  << " (NOQ)." << endl;
487  }
488  set<OutputSet::sSetElement> const sl = cf->la_route_set.GetSet();
489  assert(sl.size() == 1);
490  int next_output = sl.begin()->output_port;
491  vc_count /= router->NumOutputs();
492  vc_start += next_output * vc_count;
493  vc_end = vc_start + vc_count - 1;
494  assert(vc_start >= se.vc_start && vc_start <= se.vc_end);
495  assert(vc_end >= se.vc_start && vc_end <= se.vc_end);
496  assert(vc_start <= vc_end);
497  }
498  if(cf->watch) {
499  *gWatchOut << GetSimTime() << " | " << FullName() << " | "
500  << "Finding output VC for flit " << cf->id
501  << ":" << endl;
502  }
503  for(int i = 1; i <= vc_count; ++i) {
504  int const lvc = _last_vc[n][subnet][c];
505  int const vc =
506  (lvc < vc_start || lvc > vc_end) ?
507  vc_start :
508  (vc_start + (lvc - vc_start + i) % vc_count);
509  assert((vc >= vc_start) && (vc <= vc_end));
510  if(!dest_buf->IsAvailableFor(vc)) {
511  if(cf->watch) {
512  *gWatchOut << GetSimTime() << " | " << FullName() << " | "
513  << " Output VC " << vc << " is busy." << endl;
514  }
515  } else {
516  if(dest_buf->IsFullFor(vc)) {
517  if(cf->watch) {
518  *gWatchOut << GetSimTime() << " | " << FullName() << " | "
519  << " Output VC " << vc << " is full." << endl;
520  }
521  } else {
522  if(cf->watch) {
523  *gWatchOut << GetSimTime() << " | " << FullName() << " | "
524  << " Selected output VC " << vc << "." << endl;
525  }
526  cf->vc = vc;
527  break;
528  }
529  }
530  }
531  }
532 
533  if(cf->vc == -1) {
534  if(cf->watch) {
535  *gWatchOut << GetSimTime() << " | " << FullName() << " | "
536  << "No output VC found for flit " << cf->id
537  << "." << endl;
538  }
539  } else {
540  if(dest_buf->IsFullFor(cf->vc)) {
541  if(cf->watch) {
542  *gWatchOut << GetSimTime() << " | " << FullName() << " | "
543  << "Selected output VC " << cf->vc
544  << " is full for flit " << cf->id
545  << "." << endl;
546  }
547  } else {
548  f = cf;
549  }
550  }
551  }
552 
553  if(f) {
554 
555  assert(f->subnetwork == subnet);
556 
557  int const c = f->cl;
558 
559  if(f->head) {
560 
561  if (_lookahead_routing) {
562  if(!_noq) {
563  const FlitChannel * inject = _net[subnet]->GetInject(n);
564  const Router * router = inject->GetSink();
565  assert(router);
566  int in_channel = inject->GetSinkPort();
567  _rf(router, f, in_channel, &f->la_route_set, false);
568  if(f->watch) {
569  *gWatchOut << GetSimTime() << " | "
570  << "node" << n << " | "
571  << "Generating lookahead routing info for flit " << f->id
572  << "." << endl;
573  }
574  } else if(f->watch) {
575  *gWatchOut << GetSimTime() << " | "
576  << "node" << n << " | "
577  << "Already generated lookahead routing info for flit " << f->id
578  << " (NOQ)." << endl;
579  }
580  } else {
581  f->la_route_set.Clear();
582  }
583 
584  dest_buf->TakeBuffer(f->vc);
585  _last_vc[n][subnet][c] = f->vc;
586  }
587 
588  _last_class[n][subnet] = c;
589 
590  _input_queue[subnet][n][c].pop_front();
591 
592 #ifdef TRACK_FLOWS
593  ++_outstanding_credits[c][subnet][n];
594  _outstanding_classes[n][subnet][f->vc].push(c);
595 #endif
596 
597  dest_buf->SendingFlit(f);
598 
599  if(_pri_type == network_age_based) {
600  f->pri = numeric_limits<int>::max() - _time;
601  assert(f->pri >= 0);
602  }
603 
604  if(f->watch) {
605  *gWatchOut << GetSimTime() << " | "
606  << "node" << n << " | "
607  << "Injecting flit " << f->id
608  << " into subnet " << subnet
609  << " at time " << _time
610  << " with priority " << f->pri
611  << "." << endl;
612  }
613  f->itime = _time;
614 
615  // Pass VC "back"
616  if(!_input_queue[subnet][n][c].empty() && !f->tail) {
617  Flit * const nf = _input_queue[subnet][n][c].front();
618  nf->vc = f->vc;
619  }
620 
621  if((_sim_state == warming_up) || (_sim_state == running)) {
622  ++_sent_flits[c][n];
623  if(f->head) {
624  ++_sent_packets[c][n];
625  }
626  }
627 
628 #ifdef TRACK_FLOWS
629  ++_injected_flits[c][n];
630 #endif
631 
632  _net[subnet]->WriteFlit(f, n);
633 
634  }
635  }
636  }
637  //Send the credit To the network
638  for(int subnet = 0; subnet < _subnets; ++subnet) {
