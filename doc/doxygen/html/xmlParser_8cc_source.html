726  // sequences &amp;, &quot;, &apos;, &lt;, &gt; and replace them by the characters
727  // &,",',<,>. This function is used internally by the XML Parser. All the calls to

738  while ((lo>0)&&(*s))

760  } while(entity->s);

766  }

871 // Find the next token in a string.
872 // pcbToken contains the number of characters that have been read.
873 static NextToken GetNextToken(XML *pXML, int *pcbToken, enum XMLTokenTypeTag *pType)
874 {
875  NextToken result;
876  XMLCHAR ch;
877  XMLCHAR chTemp;
878  int indexStart,nFoundMatch,nIsText=FALSE;
879  result.pClr=NULL; // prevent warning
880 
881  // Find next non-white space character
882  do { indexStart=pXML->nIndex; ch=getNextChar(pXML); } while XML_isSPACECHAR(ch);
883 
884  if (ch)
885  {
886  // Cache the current string pointer
887  result.pStr = &pXML->lpXML[indexStart];
888 
889  // First check whether the token is in the clear tag list (meaning it
890  // does not need formatting).
891  ALLXMLClearTag *ctag=XMLClearTags;
892  do
893  {
894  if (xstrncmp(ctag->lpszOpen, result.pStr, ctag->openTagLen)==0)
895  {
896  result.pClr=ctag;
897  pXML->nIndex+=ctag->openTagLen-1;
898  *pType=eTokenClear;
899  return result;
900  }
901  ctag++;
902  } while(ctag->lpszOpen);
903 
904  // If we didn't find a clear tag then check for standard tokens
905  switch(ch)
906  {
907  // Check for quotes
908  case _CXML('\''):
909  case _CXML('\"'):
910  // Type of token
911  *pType = eTokenQuotedText;
912  chTemp = ch;
913 
914  // Set the size
915  nFoundMatch = FALSE;
916 
917  // Search through the string to find a matching quote
918  while((ch = getNextChar(pXML)))
919  {
920  if (ch==chTemp) { nFoundMatch = TRUE; break; }
921  if (ch==_CXML('<')) break;
922  }
923 
924  // If we failed to find a matching quote
925  if (nFoundMatch == FALSE)
926  {
927  pXML->nIndex=indexStart+1;
928  nIsText=TRUE;
929  break;
930  }
931 
932 // 4.02.2002
933 // if (FindNonWhiteSpace(pXML)) pXML->nIndex--;
934 
935  break;
936 
937  // Equals (used with attribute values)
938  case _CXML('='):
939  *pType = eTokenEquals;
940  break;
941 
942  // Close tag
943  case _CXML('>'):
944  *pType = eTokenCloseTag;
945  break;
946 
947  // Check for tag start and tag end
948  case _CXML('<'):
949 
950  // Peek at the next character to see if we have an end tag '</',
951  // or an xml declaration '<?'
952  chTemp = pXML->lpXML[pXML->nIndex];
953 
954  // If we have a tag end...
955  if (chTemp == _CXML('/'))
956  {
957  // Set the type and ensure we point at the next character
958  getNextChar(pXML);
959  *pType = eTokenTagEnd;
960  }
961 
962  // If we have an XML declaration tag
963  else if (chTemp == _CXML('?'))
964  {
965 
966  // Set the type and ensure we point at the next character
967  getNextChar(pXML);
968  *pType = eTokenDeclaration;
969  }
970 
971  // Otherwise we must have a start tag
972  else
973  {
974  *pType = eTokenTagStart;
975  }
976  break;
977 
978  // Check to see if we have a short hand type end tag ('/>').
979  case _CXML('/'):
980 
981  // Peek at the next character to see if we have a short end tag '/>'
982  chTemp = pXML->lpXML[pXML->nIndex];
983 
984  // If we have a short hand end tag...
985  if (chTemp == _CXML('>'))
986  {
987  // Set the type and ensure we point at the next character
988  getNextChar(pXML);
989  *pType = eTokenShortHandClose;
990  break;
991  }
992 
993  // If we haven't found a short hand closing tag then drop into the
994  // text process
995 
996  // Other characters
997  default:
998  nIsText = TRUE;
999  }
1000 
1001  // If this is a TEXT node
1002  if (nIsText)
1003  {
1004  // Indicate we are dealing with text
1005  *pType = eTokenText;
1006  while((ch = getNextChar(pXML)))
1007  {
1008  if XML_isSPACECHAR(ch)
1009  {
1010  indexStart++; break;
1011 
1012  } else if (ch==_CXML('/'))
1013  {
1014  // If we find a slash then this maybe text or a short hand end tag
1015  // Peek at the next character to see it we have short hand end tag
1016  ch=pXML->lpXML[pXML->nIndex];
1017  // If we found a short hand end tag then we need to exit the loop
1018  if (ch==_CXML('>')) { pXML->nIndex--; break; }
1019 
1020  } else if ((ch==_CXML('<'))||(ch==_CXML('>'))||(ch==_CXML('=')))
1021  {
1022  pXML->nIndex--; break;
1023  }
1024  }
1025  }
1026  *pcbToken = pXML->nIndex-indexStart;
1027  } else
1028  {
1029  // If we failed to obtain a valid character
1030  *pcbToken = 0;
1031  *pType = eTokenError;
1032  result.pStr=NULL;
1033  }
1034 
1035  return result;
1036 }

1296

1298  if (pXML->nFirst)
1299  {
1300  // Assume we are outside of a tag definition
1301  pXML->nFirst = FALSE;
1302  status = eOutsideTag;
1303  } else
1304  {
1305  // If this is not the first call then we should only be called when inside a tag.
1306  status = eInsideTag;

1328  case eTokenText:

1330  case eTokenShortHandClose: /* '/>' */
1331  case eTokenQuotedText:
1332  case eTokenEquals:

1386  // evented in a end tag specified in XML then

1408  }

1438  }

1444

1446  if (maybeAddTxT(pXML,token.pStr)) return FALSE;

1450
