33 void my_cuda_printf(const char *fmtstr,const char *arg_list)
34 {
35  FILE *fp = stdout;
36  unsigned i=0,j=0;
37  unsigned arg_offset=0;
38  char buf[64];
39  bool in_fmt=false;
40  while( fmtstr[i] ) {
41  char c = fmtstr[i++];
42  if( !in_fmt ) {
43  if( c != '%' ) {
44  fprintf(fp,"%c",c);
45  } else {
46  in_fmt=true;
47  buf[0] = c;
48  j=1;
49  }
50  } else {
51  if(!( c == 'u' || c == 'f' || c == 'd' )) {
52  printf("GPGPU-Sim PTX: ERROR ** printf parsing support is limited to %%u, %%f, %%d at present");
53  abort();
54  }
55  buf[j] = c;
56  buf[j+1] = 0;
57  void* ptr = (void*)&arg_list[arg_offset];
58  //unsigned long long value = ((unsigned long long*)arg_list)[arg_offset];
59  if( c == 'u' || c == 'd' ) {
60  fprintf(fp,buf,*((unsigned long long*)ptr));
61  } else if( c == 'f' ) {
62  double tmp = *((double*)ptr);
63  fprintf(fp,buf,tmp);
64  }
65  arg_offset++;
66  in_fmt=false;
67  }
68  }
69 }

71 void gpgpusim_cuda_vprintf(const ptx_instruction * pI, ptx_thread_info * thread, const function_info * target_func )
72 {
73  char *fmtstr = NULL;
74  char *arg_list = NULL;
75  unsigned n_return = target_func->has_return();
76  unsigned n_args = target_func->num_args();
77  assert( n_args == 2 );
78  for( unsigned arg=0; arg < n_args; arg ++ ) {
79  const operand_info &actual_param_op = pI->operand_lookup(n_return+1+arg);
80  const symbol *formal_param = target_func->get_arg(arg);
81  unsigned size=formal_param->get_size_in_bytes();
82  assert( formal_param->is_param_local() );
83  assert( actual_param_op.is_param_local() );
84  addr_t from_addr = actual_param_op.get_symbol()->get_address();
85  unsigned long long buffer[1024];
86  assert(size<1024*sizeof(unsigned long long));
87  thread->m_local_mem->read(from_addr,size,buffer);
88  addr_t addr = (addr_t)buffer[0]; // should be pointer to generic memory location
89  memory_space *mem=NULL;
90  memory_space_t space = generic_space;
91  decode_space(space,thread,actual_param_op,mem,addr); // figure out which space
92  if( arg == 0 ) {
93  unsigned len = 0;
94  char b = 0;
95  do { // figure out length
96  mem->read(addr+len,1,&b);
97  len++;
98  } while(b);
99  fmtstr = (char*)malloc(len+64);
100  for( int i=0; i < len; i++ )
101  mem->read(addr+i,1,fmtstr+i);
102  //mem->read(addr,len,fmtstr);
103  } else {
104  unsigned len = thread->get_finfo()->local_mem_framesize();
105  arg_list = (char*)malloc(len+64);
106  for( int i=0; i < len; i++ )
107  mem->read(addr+i,1,arg_list+i);
108  //mem->read(addr,len,arg_list);
109  }
110  }
111  my_cuda_printf(fmtstr,arg_list);
112  free(fmtstr);
113  free(arg_list);
114 }
