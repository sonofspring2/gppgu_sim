302 _cl_mem::_cl_mem(
303  cl_mem_flags flags,
304  size_t size ,
305  void * host_ptr,
306  cl_int * errcode_ret,
307  cl_device_id gpu )
308 {
309  setErrCode( errcode_ret, CL_SUCCESS );
310 
311  m_is_on_host = false;
312  m_flags = flags;
313  m_size = size;
314  m_host_ptr = host_ptr;
315  m_device_ptr = 0;
316 
317  if( (flags & (CL_MEM_USE_HOST_PTR|CL_MEM_COPY_HOST_PTR)) && host_ptr == NULL ) {
318  setErrCode( errcode_ret, CL_INVALID_HOST_PTR );
319  return;
320  }
321  if( (flags & CL_MEM_COPY_HOST_PTR) && (flags & CL_MEM_USE_HOST_PTR) ) {
322  setErrCode( errcode_ret, CL_INVALID_VALUE );
323  return;
324  }
325  if( flags & CL_MEM_ALLOC_HOST_PTR ) {
326  if( host_ptr )
327  gpgpusim_opencl_error(__my_func__,__LINE__," CL_MEM_ALLOC_HOST_PTR -- not yet supported/tested.\n");
328  m_host_ptr = malloc(size);
329  }
330 
331  if( flags & (CL_MEM_USE_HOST_PTR|CL_MEM_ALLOC_HOST_PTR) ) {
332  m_is_on_host = true;
333  } else {
334  m_is_on_host = false;
335  }
336  if( !(flags & (CL_MEM_USE_HOST_PTR|CL_MEM_ALLOC_HOST_PTR)) ) {
337  // if not allocating on host, then allocate GPU memory and make a copy
338  m_device_ptr = (size_t) gpu->the_device()->gpu_malloc(size);
339  if( host_ptr )
340  gpu->the_device()->memcpy_to_gpu( m_device_ptr, host_ptr, size );
341  }
342 }

561  fread(tmp,1,len,fp);

564  // clean up files...
565  char commandline[1024];
566  snprintf(commandline,1024,"rm -f %s", ptx_fname );
567  int result = system(commandline);
568  if( result != 0 )
569  printf("GPGPU-Sim OpenCL API: could not remove temporary files generated while generating PTX\n");
570  // remove any trailing characters from string
571  while( len > 0 && tmp[len] != '}' ) {
572  tmp[len] = 0;
573  len--;
574  }

862 extern CL_API_ENTRY cl_int CL_API_CALL
863 clEnqueueNDRangeKernel(cl_command_queue command_queue,
864  cl_kernel kernel,
865  cl_uint work_dim,
866  const size_t * global_work_offset,
867  const size_t * global_work_size,
868  const size_t * local_work_size,
869  cl_uint num_events_in_wait_list,
870  const cl_event * event_wait_list,
871  cl_event * event) CL_API_SUFFIX__VERSION_1_0
872 {
873  int _global_size[3];
874  int zeros[3] = { 0, 0, 0};
875  printf("\n\n\n");
876  char *mode = getenv("PTX_SIM_MODE_FUNC");
877  if ( mode )
878  sscanf(mode,"%u", &g_ptx_sim_mode);
879  printf("GPGPU-Sim OpenCL API: clEnqueueNDRangeKernel '%s' (mode=%s)\n", kernel->name().c_str(),
880  g_ptx_sim_mode?"functional simulation":"performance simulation");
881  if ( !work_dim || work_dim > 3 ) return CL_INVALID_WORK_DIMENSION;
882  size_t _local_size[3];
883  if( local_work_size != NULL ) {
884  for ( unsigned d=0; d < work_dim; d++ )
885  _local_size[d]=local_work_size[d];
886  } else {
887  printf("GPGPU-Sim OpenCL API: clEnqueueNDRangeKernel automatic local work size selection:\n");
888  for ( unsigned d=0; d < work_dim; d++ ) {
889  if( d==0 ) {
890  if( global_work_size[d] <= command_queue->get_device()->the_device()->threads_per_core() ) {
891  _local_size[d] = global_work_size[d];
892  } else {
893  // start with the maximum number of thread that a core may hold,
894  // and decrement by 64 threadsuntil there is a local_work_size
895  // that can perfectly divide the global_work_size.
896  unsigned n_thread_per_core = command_queue->get_device()->the_device()->threads_per_core();
897  size_t local_size_attempt = n_thread_per_core;
898  while (local_size_attempt > 1 and (n_thread_per_core % 64 == 0)) {
899  if (global_work_size[d] % local_size_attempt == 0) {
900  break;
901  }
902  local_size_attempt -= 64;
903  }
904  if (local_size_attempt == 0) local_size_attempt = 1;
905  _local_size[d] = local_size_attempt;
906  }
907  } else {
908  _local_size[d] = 1;
909  }
910  printf("GPGPU-Sim OpenCL API: clEnqueueNDRangeKernel global_work_size[%u] = %zu\n", d, global_work_size[d] );
911  printf("GPGPU-Sim OpenCL API: clEnqueueNDRangeKernel local_work_size[%u] = %zu\n", d, _local_size[d] );
912  }
913  }
914  for ( unsigned d=0; d < work_dim; d++ ) {
915  _global_size[d] = (int)global_work_size[d];
916  if ( (global_work_size[d] % _local_size[d]) != 0 )
917  return CL_INVALID_WORK_GROUP_SIZE;
918  }
919  if (global_work_offset != NULL){
920  for ( unsigned d=0; d < work_dim; d++ ) {
921  if (global_work_offset[d] != 0){
922  printf("GPGPU-Sim: global id offset is not supported\n");
923  abort();
924  }
925  }
926  }
927  assert( global_work_size[0] == _local_size[0] * (global_work_size[0]/_local_size[0]) ); // i.e., we can divide into equal CTAs
928  dim3 GridDim;
929  GridDim.x = global_work_size[0]/_local_size[0];
930  GridDim.y = (work_dim < 2)?1:(global_work_size[1]/_local_size[1]);
931  GridDim.z = (work_dim < 3)?1:(global_work_size[2]/_local_size[2]);
932  dim3 BlockDim;
933  BlockDim.x = _local_size[0];
934  BlockDim.y = (work_dim < 2)?1:_local_size[1];
935  BlockDim.z = (work_dim < 3)?1:_local_size[2];
936 
937  gpgpu_ptx_sim_arg_list_t params;
938  cl_int err_val = kernel->bind_args(params);
939  if ( err_val != CL_SUCCESS )
940  return err_val;
941 
942  gpgpu_t *gpu = command_queue->get_device()->the_device();
943  if (kernel->get_implementation()->get_ptx_version().ver() <3.0){
944  gpgpu_ptx_sim_memcpy_symbol( "%_global_size", _global_size, 3 * sizeof(int), 0, 1, gpu );
945  gpgpu_ptx_sim_memcpy_symbol( "%_work_dim", &work_dim, 1 * sizeof(int), 0, 1, gpu );
946  gpgpu_ptx_sim_memcpy_symbol( "%_global_num_groups", &GridDim, 3 * sizeof(int), 0, 1, gpu );
947  gpgpu_ptx_sim_memcpy_symbol( "%_global_launch_offset", zeros, 3 * sizeof(int), 0, 1, gpu );
948  gpgpu_ptx_sim_memcpy_symbol( "%_global_block_offset", zeros, 3 * sizeof(int), 0, 1, gpu );
949  }
950  kernel_info_t *grid = gpgpu_opencl_ptx_sim_init_grid(kernel->get_implementation(),params,GridDim,BlockDim,gpu);
951  if ( g_ptx_sim_mode )
952  gpgpu_opencl_ptx_sim_main_func( grid );
953  else
954  gpgpu_opencl_ptx_sim_main_perf( grid );
