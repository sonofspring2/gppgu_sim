93 void FatTree::_BuildNet( const Configuration& config )
94 {
95  cout << "Fat Tree" << endl;
96  cout << " k = " << _k << " levels = " << _n << endl;
97  cout << " each switch - total radix = "<< 2*_k << endl;
98  cout << " # of switches = "<< _size << endl;
99  cout << " # of channels = "<< _channels << endl;
100  cout << " # of nodes ( size of network ) = " << _nodes << endl;
101 
102 
103  // Number of router positions at each depth of the network
104  const int nPos = powi( _k, _n-1);
105 
106  //
107  // Allocate Routers
108  //
109  ostringstream name;
110  int level, pos, id, degree, port;
111  for ( level = 0 ; level < _n ; ++level ) {
112  for ( pos = 0 ; pos < nPos ; ++pos ) {
113 
114  if ( level == 0 ) //top routers is zero
115  degree = _k;
116  else
117  degree = 2 * _k;
118 
119  id = level * nPos + pos;
120 
121  name.str("");
122  name << "router_level" << level << "_" << pos;
123  Router * r = Router::NewRouter( config, this, name.str( ), id,
124  degree, degree );
125  _Router( level, pos ) = r;
126  _timed_modules.push_back(r);
127  }
128  }
129 
130  //
131  // Connect Channels to Routers
132  //
133 
134  //
135  // Router Connection Rule: Output Ports <gK Move DOWN Network
136  // Output Ports >=gK Move UP Network
137  // Input Ports <gK from DOWN Network
138  // Input Ports >=gK from up Network
139 
140  // Connecting Injection & Ejection Channels
141  for ( pos = 0 ; pos < nPos ; ++pos ) {
142  for(int index = 0; index<_k; index++){
143  int link = pos*_k + index;
144  _Router( _n-1, pos)->AddInputChannel( _inject[link],
145  _inject_cred[link]);
146  _Router( _n-1, pos)->AddOutputChannel( _eject[link],
147  _eject_cred[link]);
148  _inject[link]->SetLatency( 1 );
149  _inject_cred[link]->SetLatency( 1 );
150  _eject[link]->SetLatency( 1 );
151  _eject_cred[link]->SetLatency( 1 );
152  }
153  }
154 
155 #ifdef FATTREE_DEBUG
156  cout<<"\nAssigning output\n";
157 #endif
158 
159  //channels are numbered sequentially from an output channel perspective
160  int chan_per_direction = (_k * powi( _k , _n-1 )); //up or down
161  int chan_per_level = 2*(_k * powi( _k , _n-1 )); //up+down
162 
163  //connect all down output channels
164  //level n-1's down channel are injection channels
165  for (level = 0; level<_n-1; level++){
166  for ( pos = 0; pos < nPos; ++pos ) {
167  for ( port = 0; port < _k; ++port ) {
168  int link = (level*chan_per_level) + pos*_k + port;
169  _Router(level, pos)->AddOutputChannel( _chan[link],
170  _chan_cred[link] );
171  _chan[link]->SetLatency( 1 );
172  _chan_cred[link]->SetLatency( 1 );
173 #ifdef FATTREE_DEBUG
174  cout<<_Router(level, pos)->Name()<<" "
175  <<"down output "<<port<<" "
176  <<"channel_id "<<link<<endl;
177 #endif
178 
179  }
180  }
181  }
182  //connect all up output channels
183  //level 0 has no up chnanels
184  for (level = 1; level<_n; level++){
185  for ( pos = 0; pos < nPos; ++pos ) {
186  for ( port = 0; port < _k; ++port ) {
187  int link = (level*chan_per_level - chan_per_direction) + pos*_k + port ;
188  _Router(level, pos)->AddOutputChannel( _chan[link],
189  _chan_cred[link] );
190  _chan[link]->SetLatency( 1 );
191  _chan_cred[link]->SetLatency( 1 );
192 #ifdef FATTREE_DEBUG
193  cout<<_Router(level, pos)->Name()<<" "
194  <<"up output "<<port<<" "
195  <<"channel_id "<<link<<endl;
196 #endif
197  }
198  }
199  }
200 
201 #ifdef FATTREE_DEBUG
202  cout<<"\nAssigning Input\n";
203 #endif
204 
205  //connect all down input channels
206  for (level = 0; level<_n-1; level++){
207  //input channel are numbered interleavely, the interleaev depends on level
208  int routers_per_neighborhood = powi(_k,_n-1-(level));
209  int routers_per_branch = powi(_k,_n-1-(level+1));
210  int level_offset = routers_per_neighborhood*_k;
211  for ( pos = 0; pos < nPos; ++pos ) {
212  int neighborhood = pos/routers_per_neighborhood;
213  int neighborhood_pos = pos%routers_per_neighborhood;
214  for ( port = 0; port < _k; ++port ) {
215  int link =
216  ((level+1)*chan_per_level - chan_per_direction) //which levellevel
217  +neighborhood*level_offset //region in level
218  +port*routers_per_branch*gK //sub region in region
219  +(neighborhood_pos)%routers_per_branch*gK //router in subregion
220  +(neighborhood_pos)/routers_per_branch; //port on router
221 
222  _Router(level, pos)->AddInputChannel( _chan[link],
223  _chan_cred[link] );
224 #ifdef FATTREE_DEBUG
225  cout<<_Router(level, pos)->Name()<<" "
226  <<"down input "<<port<<" "
227  <<"channel_id "<<link<<endl;
228 #endif
229  }
230  }
231  }
232 
233 
234  //connect all up input channels
235  for (level = 1; level<_n; level++){
236  //input channel are numbered interleavely, the interleaev depends on level
237  int routers_per_neighborhood = powi(_k,_n-1-(level-1));
238  int routers_per_branch = powi(_k,_n-1-(level));
239  int level_offset = routers_per_neighborhood*_k;
240  for ( pos = 0; pos < nPos; ++pos ) {
241  int neighborhood = pos/routers_per_neighborhood;
242  int neighborhood_pos = pos%routers_per_neighborhood;
243  for ( port = 0; port < _k; ++port ) {
244  int link =
245  ((level-1)*chan_per_level) //which levellevel
246  +neighborhood*level_offset //region in level
247  +port*routers_per_branch*gK //sub region in region
248  +(neighborhood_pos)%routers_per_branch*gK //router in subregion
249  +(neighborhood_pos)/routers_per_branch; //port on router
250 
251  _Router(level, pos)->AddInputChannel( _chan[link],
252  _chan_cred[link] );
253 #ifdef FATTREE_DEBUG
254  cout<<_Router(level, pos)->Name()<<" "
255  <<"up input "<<port<<" "
256  <<"channel_id "<<link<<endl;
257 #endif
258  }
259  }
260  }
261 #ifdef FATTREE_DEBUG
262  cout<<"\nChannel assigned\n";
263 #endif
264 }
