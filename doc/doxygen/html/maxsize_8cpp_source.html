99 bool MaxSizeMatch::_ShortestAugmenting( )
100 {
101  int i, j, jn;
102  int slen, nslen;
103 
104  // start with empty stack
105  slen = 0;
106 
107  // push all unassigned inputs to the stack
108  for ( i = 0; i < _inputs; ++i ) {
109  j = (i + _prio) % _inputs;
110  if ( _inmatch[j] == -1 ) { // start with unmatched left nodes
111  _s[slen++] = j;
112  }
113  }
114 
115  _from.assign(_inputs, -1);
116 
117  for ( int iter = 0; iter < _inputs; iter++ ) {
118  nslen = 0;
119 
120  for ( int e = 0; e < slen; ++e ) {
121  i = _s[e];
122 
123  for ( j = 0; j < _outputs; ++j ) {
124  if ( ( _request[i][j].label != -1 ) && // edge (i,j) exists
125  ( _inmatch[i] != j ) && // (i,j) is not contained in the current matching
126  ( _from[j] == -1 ) ) { // no shorter path to j exists
127 
128  _from[j] = i; // how did we get to j?
129 
130 #ifdef DEBUG_MAXSIZE
131  cout << " got to " << j << " from " << i << endl;
132 #endif
133  if ( _outmatch[j] == -1 ) { // j is unmatched -- augmenting path found
134  goto found_augmenting;
135  } else { // j is matched
136  _ns[nslen] = _outmatch[j]; // add the destination of this edge to the leaf nodes
137  nslen++;
138 
139 #ifdef DEBUG_MAXSIZE
140  cout << " adding " << _outmatch[j] << endl;
141 #endif
142  }
143  }
144  }
145  }
146 
147  // no augmenting path found yet, swap stacks
148  int * t = _s;
149  _s = _ns;
150  _ns = t;
151  slen = nslen;
152  }
153 
154  return false; // no augmenting paths
155 
156  found_augmenting:
157 
158  // the augmenting path ends at node j on the right
159

161  cout << "Found path: " << j << "c <- ";
162 #endif
