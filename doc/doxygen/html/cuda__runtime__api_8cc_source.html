306  gpgpu_sim *the_gpu = gpgpu_ptx_sim_init_perf();

308  cudaDeviceProp *prop = (cudaDeviceProp *) calloc(sizeof(cudaDeviceProp),1);
309  snprintf(prop->name,256,"GPGPU-Sim_v%s", g_gpgpusim_version_string );
310  prop->major = 2;
311  prop->minor = 0;
312  prop->totalGlobalMem = 0x40000000 /* 1 GB */;
313  prop->memPitch = 0;
314  prop->maxThreadsPerBlock = 512;
315  prop->maxThreadsDim[0] = 512;
316  prop->maxThreadsDim[1] = 512;
317  prop->maxThreadsDim[2] = 512;
318  prop->maxGridSize[0] = 0x40000000;
319  prop->maxGridSize[1] = 0x40000000;
320  prop->maxGridSize[2] = 0x40000000;
321  prop->totalConstMem = 0x40000000;
322  prop->textureAlignment = 0;
323  prop->sharedMemPerBlock = the_gpu->shared_mem_size();
324  prop->regsPerBlock = the_gpu->num_registers_per_core();
325  prop->warpSize = the_gpu->wrp_size();
326  prop->clockRate = the_gpu->shader_clock();
327 #if (CUDART_VERSION >= 2010)
328  prop->multiProcessorCount = the_gpu->get_config().num_shader();
329 #endif
330  the_gpu->set_prop(prop);
331  the_device = new _cuda_device_id(the_gpu);
332  }
333  start_sim_thread(1);
334  return the_device;
