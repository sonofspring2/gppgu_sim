135  st_time_dist.resize(st_size);

141  void update_ld(unsigned int uid,unsigned int slot, long int time) {
142  if ( ld_time_map.find( uid )!=ld_time_map.end() ) {
143  ld_time_map[uid][slot]=time;
144  } else if (slot < NUM_MEM_REQ_STAT ) {
145  std::vector<long int> time_vec;
146  time_vec.resize(ld_vector_size);
147  time_vec[slot] = time;
148  ld_time_map[uid] = time_vec;
149  } else {
150  //It's a merged mshr! forget it
151  }
152  }
153  void update_st(unsigned int uid,unsigned int slot, long int time) {
154  if ( st_time_map.find( uid )!=st_time_map.end() ) {
155  st_time_map[uid][slot]=time;
156  } else {
157  std::vector<long int> time_vec;
158  time_vec.resize(st_vector_size);
159  time_vec[slot] = time;
160  st_time_map[uid] = time_vec;
161  }
162  }
163  void check_ld_update(unsigned int uid,unsigned int slot, long int latency) {
164  if ( ld_time_map.find( uid )!=ld_time_map.end() ) {
165  int our_latency = ld_time_map[uid][slot] - ld_time_map[uid][IN_ICNT_TO_MEM];
166  assert( our_latency == latency);
167  } else if (slot < NUM_MEM_REQ_STAT ) {
168  abort();
169  }
170  }
171  void check_st_update(unsigned int uid,unsigned int slot, long int latency) {
172  if ( st_time_map.find( uid )!=st_time_map.end() ) {
173  int our_latency = st_time_map[uid][slot] - st_time_map[uid][IN_ICNT_TO_MEM];
174  assert( our_latency == latency);
175  } else {
176  abort();
177  }
178  }
179 private:
180  void calculate_ld_dist(void) {
181  unsigned i,first;
182  long int last_update,diff;
183  int finished_count=0;
184  ld_time_dist.clear();
185  ld_time_dist.resize(ld_vector_size);
186  std::map< unsigned int, std::vector<long int> >::iterator iter, iter_temp;
187  iter =ld_time_map.begin() ;
188  while (iter != ld_time_map.end()) {
189  last_update=0;
190  first=-1;
191  if (!iter->second[IN_SHADER_FETCHED]) {
192  //this request is not done yet skip it!
193  ++iter;
194  continue;
195  }
196  while ( !last_update ) {
197  first++;
198  assert( first < iter->second.size() );
199  last_update = iter->second[first];
200  }
201 
202  for ( i=first;i<ld_vector_size;i++ ) {
203  diff = iter->second[i] - last_update;
204  if ( diff>0 ) {
205  ld_time_dist[i]+=diff;
206  last_update = iter->second[i];
207  }
208  }
209  iter_temp = iter;
210  iter++;
211  ld_time_map.erase(iter_temp);
212  finished_count++;
213  }
214  if ( finished_count ) {
215  for ( i=0;i<ld_vector_size;i++ ) {
216  overal_ld_time_dist[i] = (overal_ld_time_dist[i]*overal_ld_count + ld_time_dist[i]) / (overal_ld_count + finished_count);
217  }
218  overal_ld_count += finished_count;
219  for ( i=0;i<ld_vector_size;i++ ) {
220  ld_time_dist[i]/=finished_count;
221  }
222  }
223  }
