52 /* overview of MC models:
53  * McPAT memory controllers are modeled according to large number of industrial data points.
54  * The Basic memory controller architecture is base on the Synopsis designs
55  * (DesignWare DDR2/DDR3-Lite memory controllers and DDR2/DDR3-Lite protocol controllers)
56  * as in Cadence ChipEstimator Tool
57  *
58  * An MC has 3 parts as shown in this design. McPAT models both high performance MC
59  * based on Niagara processor designs and curving and low power MC based on data points in
60  * Cadence ChipEstimator Tool.
61  *
62  * The frontend is modeled analytically, the backend is modeled empirically according to
63  * DDR2/DDR3-Lite protocol controllers in Cadence ChipEstimator Tool
64  * The PHY is modeled based on
65  * "A 100mW 9.6Gb/s Transceiver in 90nm CMOS for next-generation memory interfaces ," ISSCC 2006,
66  * and A 14mW 6.25Gb/s Transceiver in 90nm CMOS for Serial Chip-to-Chip Communication," ISSCC 2007
67  *
68  * In Cadence ChipEstimator Tool there are two types of memory controllers: the full memory controllers
69  * that includes the frontend as the DesignWare DDR2/DDR3-Lite memory controllers and the backend only
70  * memory controllers as the DDR2/DDR3-Lite protocol controllers (except DesignWare DDR2/DDR3-Lite memory
71  * controllers, all memory controller IP in Cadence ChipEstimator Tool are backend memory controllers such as
72  * DDRC 1600A and DDRC 800A). Thus,to some extend the area and power difference between DesignWare
73  * DDR2/DDR3-Lite memory controllers and DDR2/DDR3-Lite protocol controllers can be an estimation to the
74  * frontend power and area, which is very close the analitically modeled results of the frontend for Niagara2@65nm
75  *
76  */

191 void MCPHY::compute()
192 {
193  //PHY uses internal data buswidth but the actuall off-chip datawidth is 64bits + ecc
194  double pmos_to_nmos_sizing_r = pmos_to_nmos_sz_ratio() ;
195  /*
196  * according to "A 100mW 9.6Gb/s Transceiver in 90nm CMOS for next-generation memory interfaces ," ISSCC 2006;
197  * From Cadence ChipEstimator for normal I/O around 0.4~0.8 mW/Gb/s
198  */
199  double power_per_gb_per_s,phy_gates, NMOS_sizing, PMOS_sizing;
200 
201  if (mc_type == MC)
202  {
203  if (mcp.type == 0)
204  {
205  power_per_gb_per_s = mcp.LVDS? 0.01:0.04;
206  //Based on die photos from Niagara 1 and 2.
207  //TODO merge this into undifferentiated core.PHY only achieves square root of the ideal scaling.
208  //area = (6.4323*log(peakDataTransferRate)-34.76)*memDataWidth/128.0*(l_ip.F_sz_um/0.09);
209  area.set_area((6.4323*log(mcp.peakDataTransferRate*2)-48.134)*mcp.dataBusWidth/128.0*(l_ip.F_sz_um/0.09)*mcp.num_channels*1e6/2);//TODO:/2
210  //This is from curve fitting based on Niagara 1 and 2's PHY die photo.
211  //This is power not energy, 10mw/Gb/s @90nm for each channel and scaling down
212  //power.readOp.dynamic = 0.02*memAccesses*llcBlocksize*8;//change from Bytes to bits.
213  power_t.readOp.dynamic = power_per_gb_per_s*sqrt(l_ip.F_sz_um/0.09)*g_tp.peri_global.Vdd/1.2*g_tp.peri_global.Vdd/1.2;
214  power_t.readOp.leakage = area.get_area()/2 *(g_tp.scaling_factor.core_tx_density)*cmos_Isub_leakage(g_tp.min_w_nmos_, g_tp.min_w_nmos_*pmos_to_nmos_sizing_r, 1, inv)*g_tp.peri_global.Vdd;//unit W
215  power_t.readOp.gate_leakage = area.get_area()/2 *(g_tp.scaling_factor.core_tx_density)*cmos_Ig_leakage(g_tp.min_w_nmos_, g_tp.min_w_nmos_*pmos_to_nmos_sizing_r, 1, inv)*g_tp.peri_global.Vdd;//unit W
216 
217  }
218  else
219  {
220  NMOS_sizing = g_tp.min_w_nmos_;
221  PMOS_sizing = g_tp.min_w_nmos_*pmos_to_nmos_sizing_r;
222  //Designware/synopsis 16bit DDR3 PHY is 1.3mm (WITH IOs) at 40nm for upto DDR3 2133 (PC3 17066)
223  double non_IO_percentage = 0.2;
224  area.set_area(1.3*non_IO_percentage/2133.0e6*mcp.clockRate/17066*mcp.peakDataTransferRate*mcp.dataBusWidth/16.0*(l_ip.F_sz_um/0.040)* (l_ip.F_sz_um/0.040)*mcp.num_channels*1e6);//um^2
225  phy_gates = 200000*mcp.dataBusWidth/64.0;
226  power_per_gb_per_s = 0.01;
227  //This is power not energy, 10mw/Gb/s @90nm for each channel and scaling down
228  power_t.readOp.dynamic = power_per_gb_per_s*(l_ip.F_sz_um/0.09)*g_tp.peri_global.Vdd/1.2*g_tp.peri_global.Vdd/1.2;
229  power_t.readOp.leakage = (mcp.withPHY? phy_gates:0)*cmos_Isub_leakage(NMOS_sizing, PMOS_sizing, 2, nand)*g_tp.peri_global.Vdd;//unit W
230  power_t.readOp.gate_leakage = (mcp.withPHY? phy_gates:0)*cmos_Ig_leakage(NMOS_sizing, PMOS_sizing, 2, nand)*g_tp.peri_global.Vdd;//unit W
231  }
232 
233  }
234  else

236  area.set_area(0.4e6/2*mcp.dataBusWidth/8);//area based on Cadence ChipEstimator for 8bit bus
237  }

241 // power_t.readOp.leakage *= phy_factor;
242 // power_t.readOp.gate_leakage *= phy_factor;
243

245  power_t.readOp.longer_channel_leakage = power_t.readOp.leakage * long_channel_device_reduction;
246 }
