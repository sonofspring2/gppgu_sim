74 void print_ptx_file( const char *p, unsigned source_num, const char *filename )
75 {
76  printf("\nGPGPU-Sim PTX: file _%u.ptx contents:\n\n", source_num );
77  char *s = strdup(p);
78  char *t = s;
79  unsigned n=1;
80  while ( *t != '\0' ) {
81  char *u = t;
82  while ( (*u != '\n') && (*u != '\0') ) u++;
83  unsigned last = (*u == '\0');
84  *u = '\0';
85  const ptx_instruction *pI = ptx_instruction_lookup(filename,n);
86  char pc[64];
87  if( pI && pI->get_PC() )
88  snprintf(pc,64,"%4u", pI->get_PC() );
89  else
90  snprintf(pc,64," ");
91  printf(" _%u.ptx %4u (pc=%s): %s\n", source_num, n, pc, t );
92  if ( last ) break;
93  t = u+1;
94  n++;
95  }
96  free(s);
97  fflush(stdout);
98 }

153 symbol_table *gpgpu_ptx_sim_load_ptx_from_string( const char *p, unsigned source_num )
154 {
155  char buf[1024];
156  snprintf(buf,1024,"_%u.ptx", source_num );
157  if( g_save_embedded_ptx ) {
158  FILE *fp = fopen(buf,"w");
159  fprintf(fp,"%s",p);
160  fclose(fp);
161  }
162  symbol_table *symtab=init_parser(buf);
163  ptx__scan_string(p);
164  int errors = ptx_parse ();
165  if ( errors ) {
166  char fname[1024];
167  snprintf(fname,1024,"_ptx_errors_XXXXXX");
168  int fd=mkstemp(fname);
169  close(fd);
170  printf("GPGPU-Sim PTX: parser error detected, exiting... but first extracting .ptx to \"%s\"\n", fname);
171  FILE *ptxfile = fopen(fname,"w");
172  fprintf(ptxfile,"%s", p );
173  fclose(ptxfile);
174  abort();
175  exit(40);
176  }
177 
178  if ( g_debug_execution >= 100 )
179  print_ptx_file(p,source_num,buf);
180 
181  printf("GPGPU-Sim PTX: finished parsing EMBEDDED .ptx file %s\n",buf);
182  return symtab;
183 }
