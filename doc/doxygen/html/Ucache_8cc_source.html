98 void * calc_time_mt_wrapper(void * void_obj)
99 {
100  calc_time_mt_wrapper_struct * calc_obj = (calc_time_mt_wrapper_struct *) void_obj;
101  uint32_t tid = calc_obj->tid;
102  list<mem_array *> & data_arr = calc_obj->data_arr;
103  list<mem_array *> & tag_arr = calc_obj->tag_arr;
104  bool is_tag = calc_obj->is_tag;
105  bool pure_ram = calc_obj->pure_ram;
106  bool pure_cam = calc_obj->pure_cam;
107  bool is_main_mem = calc_obj->is_main_mem;
108  double Nspd_min = calc_obj->Nspd_min;
109  min_values_t * data_res = calc_obj->data_res;
110  min_values_t * tag_res = calc_obj->tag_res;
111 
112  data_arr.clear();
113  data_arr.push_back(new mem_array);
114  tag_arr.clear();
115  tag_arr.push_back(new mem_array);
116 
117  uint32_t Ndwl_niter = _log2(MAXDATAN) + 1;
118  uint32_t Ndbl_niter = _log2(MAXDATAN) + 1;
119  uint32_t Ndcm_niter = _log2(MAX_COL_MUX) + 1;
120  uint32_t niter = Ndwl_niter * Ndbl_niter * Ndcm_niter;
121 
122 
123  bool is_valid_partition;
124  int wt_min, wt_max;
125 
126  if (g_ip->force_wiretype) {
127  if (g_ip->wt == 0) {
128  wt_min = Low_swing;
129  wt_max = Low_swing;
130  }
131  else {
132  wt_min = Global;
133  wt_max = Low_swing-1;
134  }
135  }
136  else {
137  wt_min = Global;
138  wt_max = Low_swing;
139  }
140 
141  for (double Nspd = Nspd_min; Nspd <= MAXDATASPD; Nspd *= 2)
142  {
143  for (int wr = wt_min; wr <= wt_max; wr++)
144  {
145  for (uint32_t iter = tid; iter < niter; iter += nthreads)
146  {
147  // reconstruct Ndwl, Ndbl, Ndcm
148  unsigned int Ndwl = 1 << (iter / (Ndbl_niter * Ndcm_niter));
149  unsigned int Ndbl = 1 << ((iter / (Ndcm_niter))%Ndbl_niter);
150  unsigned int Ndcm = 1 << (iter % Ndcm_niter);
151  for(unsigned int Ndsam_lev_1 = 1; Ndsam_lev_1 <= MAX_COL_MUX; Ndsam_lev_1 *= 2)
152  {
153  for(unsigned int Ndsam_lev_2 = 1; Ndsam_lev_2 <= MAX_COL_MUX; Ndsam_lev_2 *= 2)
154  {
155  //for debuging
156  if (g_ip->force_cache_config && is_tag == false)
157  {
158  wr = g_ip->wt;
159  Ndwl = g_ip->ndwl;
160  Ndbl = g_ip->ndbl;
161  Ndcm = g_ip->ndcm;
162  if(g_ip->nspd != 0) {
163  Nspd = g_ip->nspd;
164  }
165  if(g_ip->ndsam1 != 0) {
166  Ndsam_lev_1 = g_ip->ndsam1;
167  Ndsam_lev_2 = g_ip->ndsam2;
168  }
169  }
170 
171  if (is_tag == true)
172  {
173  is_valid_partition = calculate_time(is_tag, pure_ram, pure_cam, Nspd, Ndwl,
174  Ndbl, Ndcm, Ndsam_lev_1, Ndsam_lev_2,
175  tag_arr.back(), 0, NULL, NULL,
176  is_main_mem);
177  }
178  // If it's a fully-associative cache, the data array partition parameters are identical to that of
179  // the tag array, so compute data array partition properties also here.
180  if (is_tag == false || g_ip->fully_assoc)
181  {
182  is_valid_partition = calculate_time(is_tag/*false*/, pure_ram, pure_cam, Nspd, Ndwl,
183  Ndbl, Ndcm, Ndsam_lev_1, Ndsam_lev_2,
184  data_arr.back(), 0, NULL, NULL,
185  is_main_mem);
186  }
187 
188  if (is_valid_partition)
189  {
190  if (is_tag == true)
191  {
192  tag_arr.back()->wt = (enum Wire_type) wr;
193  tag_res->update_min_values(tag_arr.back());
194  tag_arr.push_back(new mem_array);
195  }
196  if (is_tag == false || g_ip->fully_assoc)
197  {
198  data_arr.back()->wt = (enum Wire_type) wr;
199  data_res->update_min_values(data_arr.back());
200  data_arr.push_back(new mem_array);
201  }
202  }
203 
204  if (g_ip->force_cache_config && is_tag == false)
205  {
206  wr = wt_max;
207  iter = niter;
208  if(g_ip->nspd != 0) {
209  Nspd = MAXDATASPD;
210  }
211  if (g_ip->ndsam1 != 0) {
212  Ndsam_lev_1 = MAX_COL_MUX+1;
213  Ndsam_lev_2 = MAX_COL_MUX+1;
214  }
215  }
216  }
217  }
218  }
219  }
220  }
221 
222  delete data_arr.back();
223  delete tag_arr.back();
224  data_arr.pop_back();
225  tag_arr.pop_back();
226 
227  pthread_exit(NULL);
228 }
