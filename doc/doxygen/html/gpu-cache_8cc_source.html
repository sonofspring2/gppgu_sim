50 unsigned l1d_cache_config::set_index(new_addr_type addr) const{
51  unsigned set_index = m_nset; // Default to linear set index function
52  unsigned lower_xor = 0;
53  unsigned upper_xor = 0;
54 
55  switch(m_set_index_function){
56  case FERMI_HASH_SET_FUNCTION:
57  /*
58  * Set Indexing function from "A Detailed GPU Cache Model Based on Reuse Distance Theory"
59  * Cedric Nugteren et al.
60  * ISCA 2014
61  */
62  if(m_nset == 32 || m_nset == 64){
63  // Lower xor value is bits 7-11
64  lower_xor = (addr >> m_line_sz_log2) & 0x1F;
65 
66  // Upper xor value is bits 13, 14, 15, 17, and 19
67  upper_xor = (addr & 0xE000) >> 13; // Bits 13, 14, 15
68  upper_xor |= (addr & 0x20000) >> 14; // Bit 17
69  upper_xor |= (addr & 0x80000) >> 15; // Bit 19
70 
71  set_index = (lower_xor ^ upper_xor);
72 
73  // 48KB cache prepends the set_index with bit 12
74  if(m_nset == 64)
75  set_index |= (addr & 0x1000) >> 7;
76 
77  }else{ /* Else incorrect number of sets for the hashing function */
78  assert("\nGPGPU-Sim cache configuration error: The number of sets should be "
79  "32 or 64 for the hashing set index function.\n" && 0);
80  }
81  break;
82 
83  case CUSTOM_SET_FUNCTION:
84  /* No custom set function implemented */
85  break;
86 
87  case LINEAR_SET_FUNCTION:
88  set_index = (addr >> m_line_sz_log2) & (m_nset-1);
89  break;
90  }
91 
92  // Linear function selected or custom set index function not implemented
93  assert((set_index < m_nset) && "\nError: Set index out of bounds. This is caused by "
94  "an incorrect or unimplemented custom set index function.\n");
95 
96  return set_index;
97 }

158 enum cache_request_status tag_array::probe( new_addr_type addr, unsigned &idx ) const {
159  //assert( m_config.m_write_policy == READ_ONLY );
160  unsigned set_index = m_config.set_index(addr);
161  new_addr_type tag = m_config.tag(addr);
162 
163  unsigned invalid_line = (unsigned)-1;
164  unsigned valid_line = (unsigned)-1;
165  unsigned valid_timestamp = (unsigned)-1;
166 
167  bool all_reserved = true;
168 
169  // check for hit or pending hit
170  for (unsigned way=0; way<m_config.m_assoc; way++) {
171  unsigned index = set_index*m_config.m_assoc+way;
172  cache_block_t *line = &m_lines[index];
173  if (line->m_tag == tag) {
174  if ( line->m_status == RESERVED ) {
175  idx = index;
176  return HIT_RESERVED;
177  } else if ( line->m_status == VALID ) {
178  idx = index;
179  return HIT;
180  } else if ( line->m_status == MODIFIED ) {
181  idx = index;
182  return HIT;
183  } else {
184  assert( line->m_status == INVALID );
185  }
186  }
187  if (line->m_status != RESERVED) {
188  all_reserved = false;
189  if (line->m_status == INVALID) {
190  invalid_line = index;
191  } else {
192  // valid line : keep track of most appropriate replacement candidate
193  if ( m_config.m_replacement_policy == LRU ) {
194  if ( line->m_last_access_time < valid_timestamp ) {
195  valid_timestamp = line->m_last_access_time;
196  valid_line = index;
197  }
198  } else if ( m_config.m_replacement_policy == FIFO ) {
199  if ( line->m_alloc_time < valid_timestamp ) {
200  valid_timestamp = line->m_alloc_time;
201  valid_line = index;
202  }
203  }
204  }
205  }
206  }
207  if ( all_reserved ) {
208  assert( m_config.m_alloc_policy == ON_MISS );
209  return RESERVATION_FAIL; // miss and not enough space in cache to allocate on miss
210  }
211 
212  if ( invalid_line != (unsigned)-1 ) {
213  idx = invalid_line;
214  } else if ( valid_line != (unsigned)-1) {
215  idx = valid_line;
216  } else abort(); // if an unreserved block exists, it is either invalid or replaceable
217 
218  return MISS;
219 }

230 enum cache_request_status tag_array::access( new_addr_type addr, unsigned time, unsigned &idx, bool &wb, cache_block_t &evicted )
231 {
232  m_access++;
233  shader_cache_access_log(m_core_id, m_type_id, 0); // log accesses to cache
234  enum cache_request_status status = probe(addr,idx);
235  switch (status) {
236  case HIT_RESERVED:
237  m_pending_hit++;
238  case HIT:
239  m_lines[idx].m_last_access_time=time;
240  break;
241  case MISS:
242  m_miss++;
243  shader_cache_access_log(m_core_id, m_type_id, 1); // log cache misses
244  if ( m_config.m_alloc_policy == ON_MISS ) {
245  if( m_lines[idx].m_status == MODIFIED ) {
246  wb = true;
247  evicted = m_lines[idx];
248  }
249  m_lines[idx].allocate( m_config.tag(addr), m_config.block_addr(addr), time );
250  }
251  break;
252  case RESERVATION_FAIL:
253  m_res_fail++;
254  shader_cache_access_log(m_core_id, m_type_id, 1); // log cache misses
255  break;
256  default:
257  fprintf( stderr, "tag_array::access - Error: Unknown"
258  "cache_request_status %d\n", status );
259  abort();
260  }
261  return status;
262 }

623 void baseline_cache::bandwidth_management::use_data_port(mem_fetch *mf, enum cache_request_status outcome, const std::list<cache_event> &events)
624 {
625  unsigned data_size = mf->get_data_size();
626  unsigned port_width = m_config.m_data_port_width;
627  switch (outcome) {
628  case HIT: {
629  unsigned data_cycles = data_size / port_width + ((data_size % port_width > 0)? 1 : 0);
630  m_data_port_occupied_cycles += data_cycles;
631  } break;
632  case HIT_RESERVED:
633  case MISS: {
634  // the data array is accessed to read out the entire line for write-back
635  if (was_writeback_sent(events)) {
636  unsigned data_cycles = m_config.m_line_sz / port_width;
637  m_data_port_occupied_cycles += data_cycles;
