77 void FlatFlyOnChip::_ComputeSize( const Configuration &config )
78 {
79  _k = config.GetInt( "k" ); // # of routers per dimension
80  _n = config.GetInt( "n" ); // dimension
81  _c = config.GetInt( "c" ); //concentration, may be different from k
82  _r = _c + (_k-1)*_n ; // total radix of the switch ( # of inputs/outputs)
83 
84  //how many routers in the x or y direction
85  _xcount = config.GetInt("x");
86  _ycount = config.GetInt("y");
87  assert(_xcount == _ycount);
88  //configuration of hohw many clients in X and Y per router
89  _xrouter = config.GetInt("xr");
90  _yrouter = config.GetInt("yr");
91  assert(_xrouter == _yrouter);
92  gK = _k;
93  gN = _n;
94  gC = _c;
95 
96  assert(_c == _xrouter*_yrouter);
97 
98  _nodes = powi( _k, _n )*_c; //network size
99 
100  _num_of_switch = _nodes / _c;
101  _channels = _num_of_switch * (_r - _c);
102  _size = _num_of_switch;
103

221  int curr3 = node%(_k*_k);
222  int curr4 = (int)(node/(_k*_k));

224  int curr6 = (node%(_k*_k*_k));//mmm didn't mean to be racist
225

227  for ( int cnt = 0; cnt < (_k ); ++cnt ) {
228  int other=0; //the other router that we are trying to connect
229  int offset = 0; //silly ness when node< other or when node>other
230  //if xdimension
231  if(dim == 0){
232  other = ycurr * _k +cnt;
233  } else if (dim ==1){
234  other = cnt * _k + xcurr;
235  if(_n==3){
236  other+= curr4*_k*_k;
237  }
238  if(_n==4){
239  curr4=((int)(node/(_k*_k)))%_k;
240  other+= curr4*_k*_k+curr5*_k*_k*_k;
241  }
242  }else if (dim ==2){
243  other = cnt*_k*_k + curr3;
244  if(_n==4){
245  other+= curr5*_k*_k*_k;
246  }
247  }else if (dim ==3){
248  other = cnt*_k*_k*_k+curr6;
249  }
250  assert(dim < 4);

252 #ifdef DEBUG_FLATFLY

260  int twod = abs(node/_xcount-other/_xcount);
261  length = _xrouter*oned + _yrouter *twod;
262  //oh the node<other silly ness
263  if(node<other){
264  offset = -1;
265  }
266  //node, dimension, router within dimension. Good luck understanding this
267  _output = (_k-1) * _n * node + (_k-1) * dim + cnt+offset;
268 
269 
270 #ifdef DEBUG_FLATFLY
271  cout << "Adding channel : " << _output << " to node " << node <<" and "<<other<<" with length "<<length<<endl;
272 #endif

287 
288  }
289  }

292  cout<<"Setup Finished Link"<<endl;

332 void adaptive_xyyx_flatfly( const Router *r, const Flit *f, int in_channel,
333  OutputSet *outputs, bool inject )
334 {
335  // ( Traffic Class , Routing Order ) -> Virtual Channel Range
336  int vcBegin = 0, vcEnd = gNumVCs-1;
337  if ( f->type == Flit::READ_REQUEST ) {
338  vcBegin = gReadReqBeginVC;
339  vcEnd = gReadReqEndVC;
340  } else if ( f->type == Flit::WRITE_REQUEST ) {
341  vcBegin = gWriteReqBeginVC;
342  vcEnd = gWriteReqEndVC;
343  } else if ( f->type == Flit::READ_REPLY ) {
344  vcBegin = gReadReplyBeginVC;
345  vcEnd = gReadReplyEndVC;
346  } else if ( f->type == Flit::WRITE_REPLY ) {
347  vcBegin = gWriteReplyBeginVC;
348  vcEnd = gWriteReplyEndVC;
349  }
350  assert(((f->vc >= vcBegin) && (f->vc <= vcEnd)) || (inject && (f->vc < 0)));
351 
352  int out_port;
353 
354  if(inject) {
355 
356  out_port = -1;
357 
358  } else {
359 
360  int dest = flatfly_transformation(f->dest);
361  int targetr = (int)(dest/gC);
362 
363  if(targetr==r->GetID()){ //if we are at the final router, yay, output to client
364  out_port = dest % gC;
365 
366  } else {
367 
368  //each class must have at least 2 vcs assigned or else xy_yx will deadlock
369  int const available_vcs = (vcEnd - vcBegin + 1) / 2;
370  assert(available_vcs > 0);
371 
372  int out_port_xy = flatfly_outport(dest, r->GetID());
373  int out_port_yx = flatfly_outport_yx(dest, r->GetID());
374 
375  // Route order (XY or YX) determined when packet is injected
376  // into the network, adaptively
377  bool x_then_y;
378  if(in_channel < gC){
379  int credit_xy = r->GetUsedCredit(out_port_xy);
380  int credit_yx = r->GetUsedCredit(out_port_yx);
381  if(credit_xy > credit_yx) {
382  x_then_y = false;
383  } else if(credit_xy < credit_yx) {
384  x_then_y = true;
385  } else {
386  x_then_y = (RandomInt(1) > 0);
387  }
388  } else {
389  x_then_y = (f->vc < (vcBegin + available_vcs));
390  }
391 
392  if(x_then_y) {
393  out_port = out_port_xy;
394  vcEnd -= available_vcs;
395  } else {
396  out_port = out_port_yx;
397  vcBegin += available_vcs;
398  }
399  }
400 
401  }
402 
403  outputs->Clear( );
404 
405  outputs->AddRange( out_port , vcBegin, vcEnd );
406 }

470 int flatfly_outport_yx(int dest, int rID) {
471  int dest_rID = (int) (dest / gC);
472  int _dim = gN;
473  int output = -1, dID, sID;
474 
475  if(dest_rID==rID){
476  return dest % gC;
477  }
478 
479  for (int d=_dim-1;d >= 0; d--) {
480  int power = powi(gK,d);
481  dID = int(dest_rID / power);
482  sID = int(rID / power);
483  if ( dID != sID ) {
484  output = gC + ((gK-1)*d) - 1;
485  if (dID > sID) {
486  output += dID;
487  } else {
488  output += dID + 1;
489  }
490  return output;
491  }
492  dest_rID = (int) (dest_rID %power);
493  rID = (int) (rID %power);
494  }
495  if (output == -1) {
496  cout << " ERROR ---- FLATFLY_OUTPORT function : output not found yx" << endl;
497  exit(-1);
498  }
499  return -1;
